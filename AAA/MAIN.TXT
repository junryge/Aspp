    class Character {
      constructor(frames, x, y, speed, scale, isStatic = false, bounds = null, pauseBehavior = 'normal', name = '') {
        this.frames = frames;
        this.x = x;
        this.y = y;
        this.speed = speed;
        this._originalSpeed = speed;
        this.scale = scale;
        this.frameIndex = 0;
        this.frameTimer = 0;
        this.frameDuration = 100 + Math.random() * 50;
        this.isStatic = isStatic;
        this.bounds = bounds || {minX: 150, maxX: world.width - 150, minY: 150, maxY: world.height - 150};

      setNewTarget() {
        this.targetX = this.bounds.minX + Math.random() * (this.bounds.maxX - this.bounds.minX);
        this.targetY = this.bounds.minY + Math.random() * (this.bounds.maxY - this.bounds.minY);
        this.isMoving = true;
        this.direction = this.targetX > this.x ? 1 : -1;
      }

      update(dt) {
        if(this.isStatic) return;

        // Lineup freeze: 줄서기/보고 중이면 완전 정지 (단, _reporting 보고 이동 중이면 허용)
        if (!this._reporting && (this._reportPhase === 'lineUp' || this._reportPhase === 'reporting')) {
          this.frameIndex = 0;
          return;
        }

        if (!this.isMoving) {
          // 보고 이동 중 멈춤 대기 (CEO 앞 도착 후 말풍선)
          if (this._reporting) {
            this.frameIndex = 0;
            return;
          }
          // 집으로 돌아간 직후 lineUpActive 중이면 돌아다니지 않음
          if (lineUpActive && !this._reportPhase && !this._reporting) {
            this.frameIndex = 0;
            return;
          }
          this.idleTimer -= dt * 1000;
          if (this.idleTimer <= 0) {
            this.setNewTarget();
          }
          this.frameIndex = 0; // Stand still pose
          return;
        }

        // Move towards target in 2D space
        const dx = this.targetX - this.x;
        const dy = this.targetY - this.y;
        const dist = Math.sqrt(dx*dx + dy*dy);
        
        if (dist < 5) { // Reached destination
          this.isMoving = false;

          // Fire onArrive callback if set
          if (this._onArrive) {
            const cb = this._onArrive;
            this._onArrive = null;
            cb(this);
          }

          if (this.pauseBehavior === 'strike') {
             this.idleTimer = 300 + Math.random() * 600; // jitter in place
          } else if (this.pauseBehavior === 'alwaysMove') {
             this.idleTimer = Math.random() * 500;
          } else if (this.pauseBehavior === 'oftenPause') {
             this.idleTimer = 2000 + Math.random() * 4000; // 2~6 sec pause
          } else {
             this.idleTimer = 500 + Math.random() * 2000; // normal
          }
        } else {
          // Adjust distance by standard 60fps movement delta
          const moveDist = this.speed * (dt * 1000) / 16;
          this.x += (dx / dist) * moveDist;
          this.y += (dy / dist) * moveDist;

          // Animate walking
          this.frameTimer += dt * 1000;
          if(this.frameTimer >= this.frameDuration) {
            this.frameTimer = 0;
            this.frameIndex = (this.frameIndex + 1) % this.frames.length;
          }
        }
      }

      draw(ctx) {
        const img = this.isStatic ? this.frames : this.frames[this.frameIndex];
        if(!img || !img.complete || img.naturalWidth === 0) return; // Safely return if image is not loaded
        
        ctx.save();
        ctx.translate(this.x, this.y);
        if(this.direction === -1 && !this.isStatic) {
          ctx.scale(-1, 1); // Flip horizontally if walking left
        }
        
        const w = img.width * this.scale;
        const h = img.height * this.scale;
        
        ctx.drawImage(img, -w/2, -h, w, h);

        ctx.restore();

        // Name label
        if (this.name) {
          ctx.save();
          ctx.font = 'bold 18px "Segoe UI", "Malgun Gothic", sans-serif';
          ctx.textAlign = 'center';
          const tw = ctx.measureText(this.name).width + 14;
          const lx = this.x, ly = this.y + 10;
          ctx.fillStyle = 'rgba(0,0,0,0.72)';
          ctx.beginPath();
          ctx.roundRect(lx - tw/2, ly - 14, tw, 22, 5);
          ctx.fill();
          ctx.fillStyle = '#fff';
          ctx.fillText(this.name, lx, ly + 3);
          ctx.restore();
        }
      }
    }

    function drawFireFlash(ctx) {
      if (!fireFlashText) return;
      const elapsed = Date.now() - fireFlashText.startTime;
      if (elapsed > 3000) { fireFlashText = null; return; }

      const progress = elapsed / 3000;
      const alpha = progress < 0.1 ? progress / 0.1 : (1 - progress);
      const scale = 1 + progress * 0.5;

      ctx.save();
      ctx.globalAlpha = Math.max(0, alpha);
      ctx.translate(fireFlashText.x, fireFlashText.y - progress * 80);

      // Big red "해고!!" text
      ctx.font = `bold ${Math.round(72 * scale)}px "Segoe UI", "Malgun Gothic", sans-serif`;
      ctx.textAlign = 'center';

      // Red glow shadow
      ctx.shadowColor = '#ff0000';
      ctx.shadowBlur = 30;

      // Outline
      ctx.strokeStyle = '#8b0000';
      ctx.lineWidth = 6;
      ctx.strokeText('해 고 !!', 0, 0);

      // Fill
      ctx.fillStyle = '#ff1111';
      ctx.fillText('해 고 !!', 0, 0);

      // Name below
      ctx.font = `bold ${Math.round(32 * scale)}px "Segoe UI", "Malgun Gothic", sans-serif`;
      ctx.fillStyle = '#ffffff';
      ctx.strokeStyle = '#333';
      ctx.lineWidth = 4;
      ctx.strokeText(fireFlashText.name, 0, 50);
      ctx.fillText(fireFlashText.name, 0, 50);

      ctx.restore();
    }

      if (!ceoChar) return;

    // ── 복도 시스템 (책상 절대 안 넘음) ──
    // 안전 복도 중심점 (모든 이동은 여기를 경유)
    const HUB = { x: 710, y: 550 };

    // 줄서기 위치: 복도 위에 한 줄로 (책상 사이 안전 통로)
    function getLineUpPositions(count) {
      const positions = [];
      const gap = 70;
      // 세로 복도(x=710) 위에 한 줄로 CEO 방향(위쪽)부터
      for (let i = 0; i < count; i++) {
        positions.push({ x: HUB.x, y: HUB.y + i * gap });
      }
      return positions;
    }

    // 웨이포인트 순서대로 이동
    function moveViaWaypoints(char, waypoints, finalCallback) {
      if (waypoints.length === 0) {
        if (finalCallback) finalCallback();
        return;
      }
      const wp = waypoints.shift();
      char.targetX = wp.x;
      char.targetY = wp.y;
      char.isMoving = true;
      char.direction = char.targetX > char.x ? 1 : -1;
      char._onArrive = () => {
        moveViaWaypoints(char, waypoints, finalCallback);
      };
    }

    // 경로 생성: 무조건 HUB(복도 교차점) 경유 → 직각 이동만
    function buildCorridorPath(fromX, fromY, toX, toY) {
      const path = [];

      // Step 1: 현재 위치에서 수직으로 복도 y까지 (같은 x에서 아래/위로)
      if (Math.abs(fromY - HUB.y) > 20) {
        path.push({ x: fromX, y: HUB.y });
      }

      // Step 2: 복도 y에서 수평으로 HUB x까지
      if (Math.abs(fromX - HUB.x) > 20) {
        path.push({ x: HUB.x, y: HUB.y });
      }

      // Step 3: HUB에서 목적지 x까지 수평 이동 (복도 y 유지)
      if (Math.abs(toX - HUB.x) > 20) {
        path.push({ x: toX, y: HUB.y });
      }

      // Step 4: 목적지 x에서 수직으로 목적지 y까지
      path.push({ x: toX, y: toY });

      return path;
    }

    function startLineUpReport() {
      if (lineUpActive || isOnStrike) return;
      lineUpActive = true;

      const ceoChar = characters.find(c => c.isStatic);
      const movingChars = characters.filter(c => !c.isStatic && !c._fired);
      if (movingChars.length === 0 || !ceoChar) { lineUpActive = false; return; }

      // Save original states
      const origStates = movingChars.map(c => ({
        char: c,
        bounds: { ...c.bounds },
        speed: c.speed,
        pause: c.pauseBehavior,
      }));

      // ㄱ자 줄서기 위치
      const positions = getLineUpPositions(movingChars.length);

      // Phase 1: All employees run to line-up via corridor waypoints
      let arrivedCount = 0;
      movingChars.forEach((c, i) => {
        c._reportPhase = 'running';
        c.speed = 4.5;
        c.bounds = { minX: -100, maxX: world.width + 100, minY: -100, maxY: world.height + 100 };
        c.pauseBehavior = 'alwaysMove';

        const dest = positions[i];
        const waypoints = buildCorridorPath(c.x, c.y, dest.x, dest.y);

        moveViaWaypoints(c, waypoints, () => {
          c._reportPhase = 'lineUp';
          c.direction = -1; // face CEO
          c.isMoving = false;
          c.idleTimer = 99999;
          arrivedCount++;
          if (arrivedCount === movingChars.length) {
            processLineUp(movingChars, ceoChar, origStates, 0);
          }
        });
      });
    }

    function sendCharHome(char, origState) {
      char._reportPhase = 'goingHome';
      char.speed = 3.5;
      char.bounds = { minX: -100, maxX: world.width + 100, minY: -100, maxY: world.height + 100 };

      const homeX = (origState.bounds.minX + origState.bounds.maxX) / 2;
      const homeY = (origState.bounds.minY + origState.bounds.maxY) / 2;

      // CEO 앞 → 복도로 → 복도 따라 → 집 (직각 이동만)
      const homePath = [
        { x: HUB.x, y: char.y },  // 수평으로 복도 x까지
        { x: HUB.x, y: HUB.y },   // 복도 중심으로
        { x: HUB.x, y: homeY },   // 복도에서 집 y까지 수직
        { x: homeX, y: homeY },    // 집으로 수평
      ];
      moveViaWaypoints(char, homePath, () => {
        char._reportPhase = null;
        char.speed = origState.speed;
        char.bounds = origState.bounds;
        char.isMoving = false;
        char.idleTimer = 99999;
      });
    }

    function drawSpeechBubble(ctx, char, text, textColor) {
      const bx = char.x;
      const by = char.y - (char.isStatic ? 180 : 130);
      const maxW = 360;
      const pad = 14;
      const lineH = 28;

      ctx.save();
      ctx.font = 'bold 22px "Segoe UI", "Malgun Gothic", sans-serif';

      // Word wrap
      const words = text.split('');
      const lines = [];
      let line = '';
      for (const ch of words) {
        const test = line + ch;
        if (ctx.measureText(test).width > maxW - pad * 2) {
          lines.push(line);
          line = ch;
        } else {
          line = test;
        }
      }
      if (line) lines.push(line);

      const boxW = Math.min(maxW, Math.max(...lines.map(l => ctx.measureText(l).width)) + pad * 2 + 10);
      const boxH = lines.length * lineH + pad * 2;
      const boxX = bx - boxW / 2;
      const boxY = by - boxH;

      // Bubble background
      ctx.fillStyle = '#fff';
      ctx.strokeStyle = '#2f3440';
      ctx.lineWidth = 3;
      ctx.beginPath();
      ctx.roundRect(boxX, boxY, boxW, boxH, 8);
      ctx.fill();
      ctx.stroke();

      // Tail triangle
      ctx.fillStyle = '#fff';
      ctx.strokeStyle = '#2f3440';
      ctx.beginPath();
      ctx.moveTo(bx - 10, boxY + boxH);
      ctx.lineTo(bx, boxY + boxH + 14);
      ctx.lineTo(bx + 10, boxY + boxH);
      ctx.closePath();
      ctx.fill();
      ctx.stroke();
      // Cover the tail top border
      ctx.fillStyle = '#fff';
      ctx.fillRect(bx - 9, boxY + boxH - 2, 18, 4);

      // Text
      ctx.fillStyle = textColor || '#111';
      ctx.textAlign = 'left';
      lines.forEach((l, i) => {
        ctx.fillText(l, boxX + pad, boxY + pad + lineH * (i + 0.78));
      });

      ctx.restore();
    }

    // Sort by Y
    function zSort() {
      characters.sort((a,b) => a.y - b.y);
    }

    let lastTime = performance.now();
    function loop(now) {
      const dt = Math.min((now - lastTime) / 1000, 0.033);
      lastTime = now;

      // Draw office background
      if (map.complete && map.naturalWidth > 0) {
        ctx.drawImage(map, 0, 0, world.width, world.height);
      } else {
        ctx.fillStyle = "#101725";
        ctx.fillRect(0, 0, world.width, world.height);
      }

      // Update & Draw chars
      characters.forEach(c => {
        // Skip drawing fired chars that walked off screen
        if (c._fired && c.x > world.width + 100) { c._fireWalkOut = false; return; }
        c.update(dt);
      });
      zSort();
      characters.forEach(c => {
        if (c._fired && c.x > world.width + 100) return;
        c.draw(ctx);
      });

      // Draw fire flash text
      drawFireFlash(ctx);

      // Draw report running indicator (❗ above running character)
      characters.forEach(c => {
        if (c._reportPhase === 'running') {
          ctx.save();
          const bounce = Math.sin(Date.now() / 100) * 4;
          ctx.font = 'bold 28px sans-serif';
          ctx.textAlign = 'center';
          ctx.fillStyle = '#ff4444';
          ctx.shadowColor = '#ff0000';
          ctx.shadowBlur = 8;
          ctx.fillText('❗', c.x, c.y - 100 + bounce);
          // Speed lines behind character
          ctx.globalAlpha = 0.4;
          ctx.strokeStyle = '#aaa';
          ctx.lineWidth = 2;
          for (let i = 0; i < 3; i++) {
            const lx = c.x - c.direction * (30 + i * 15);
            const ly = c.y - 50 + (i - 1) * 20;
            ctx.beginPath();
            ctx.moveTo(lx, ly);
            ctx.lineTo(lx - c.direction * 20, ly);
            ctx.stroke();
          }
          ctx.restore();
        }
      });

      // Slogan signs: 보고용은 말풍선, 파업용은 피켓
      characters.filter(c => !c.isStatic && c._slogan && !c._fired).forEach(c => {
        if (c._sloganColor && !isOnStrike) {
          drawSpeechBubble(ctx, c, c._slogan, c._sloganColor);
        } else {
          drawProtestSign(ctx, c);

      });

      // CEO cry (파업 전용)
      if (isOnStrike) {

        // CEO "살려줘" bubble

        if (ceoChar) {

          ctx.save();
          ctx.font = 'bold 22px "Segoe UI", "Malgun Gothic", sans-serif';
          const tw = ctx.measureText(msg).width + 24;
          // Shaking effect
          const shake = Math.sin(Date.now() / 80) * 3;
          const bx = cx + shake, by = cy;
          // Bubble
          ctx.fillStyle = '#e8f0fe';
          ctx.strokeStyle = '#1a56db';
          ctx.lineWidth = 3;
          ctx.beginPath();
          ctx.roundRect(bx - tw/2, by - 32, tw, 40, 8);
          ctx.fill();
          ctx.stroke();
          // Tail
          ctx.fillStyle = '#e8f0fe';
          ctx.beginPath();
          ctx.moveTo(bx - 8, by + 8);
          ctx.lineTo(bx, by + 22);
          ctx.lineTo(bx + 8, by + 8);
          ctx.closePath();
          ctx.fill();
          ctx.strokeStyle = '#1a56db';
          ctx.stroke();
          ctx.fillStyle = '#e8f0fe';
          ctx.fillRect(bx - 7, by + 6, 14, 4);
          // Text
          ctx.fillStyle = '#1a3a8a';

    (function initResizer() {
      const resizer = document.getElementById('panelResizer');
      const mainArea = document.querySelector('.main-area');
      let dragging = false;

      resizer.addEventListener('mousedown', (e) => {
        e.preventDefault();
        dragging = true;
        resizer.classList.add('dragging');
        document.body.style.cursor = 'col-resize';
        document.body.style.userSelect = 'none';
      });

      document.addEventListener('mousemove', (e) => {
        if (!dragging) return;
        const rect = mainArea.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const total = rect.width;
        const pct = Math.max(20, Math.min(80, (x / total) * 100));
        mainArea.style.gridTemplateColumns = `${pct}fr 8px ${100 - pct}fr`;
      });

      document.addEventListener('mouseup', () => {
        if (!dragging) return;
        dragging = false;
        resizer.classList.remove('dragging');
        document.body.style.cursor = '';
        document.body.style.userSelect = '';
      });
    })();

    // ── File upload to backend ──
    async function uploadFiles(files) {
      for (const f of files) {
        const ext = f.name.split('.').pop().toLowerCase();

        // CSV/TSV/TXT → 전용 CSV 파서도 호출 (행/열 분석용)
        if (['csv', 'tsv', 'txt'].includes(ext)) {
          const csvFd = new FormData();
          csvFd.append('file', f);
          await fetch('/api/upload_csv', { method: 'POST', body: csvFd }).catch(() => {});
        }

        // 범용 업로드 (모든 파일)
        const fd = new FormData();
        fd.append('file', f);
        const resp = await fetch('/api/upload_file', { method: 'POST', body: fd });
        if (!resp.ok) throw new Error(`${f.name} 업로드 실패`);
      }
    }

    function u2b(str) { return btoa(unescape(encodeURIComponent(str))); }

    function renderChartBlock(canvasId, jsonStr) {
      let raw = jsonStr.replace(/\/\/[^\n]*/g, '');
      // function 콜백 문자열 제거 (JSON.parse 호환)
      raw = raw.replace(/"function\(.*?\)\s*\{[\s\S]*?\}"/g, 'null');
      let config;
      try { config = JSON.parse(raw); } catch(e) {
        raw = raw.replace(/,\s*([\]}])/g, '$1').replace(/'/g, '"');
        try { config = JSON.parse(raw); } catch(e2) { return; }
      }
      if (!config.type) config.type = 'bar';
      if (!config.options) config.options = {};
      config.options.responsive = true;
      config.options.maintainAspectRatio = true;
      if (!config.options.plugins) config.options.plugins = {};

      const canvas = document.getElementById(canvasId);
      if (!canvas) return;
      if (_chartInstances[canvasId]) _chartInstances[canvasId].destroy();
      _chartInstances[canvasId] = new Chart(canvas.getContext('2d'), config);
    }

    // ── Render markdown + chart/drawio blocks ──
    function renderMd(text) {
      // 1) ```chart 블록 → Chart.js 캔버스로 변환 (marked 처리 전)
      let s = text.replace(/```chart\s*([\s\S]*?)```/g, (_, code) => {
        const trimmed = code.trim();
        const chartId = 'chart_' + Math.random().toString(36).substr(2, 9);
        const b64 = u2b(trimmed);
        return `<div class="chart-block">
          <div class="chart-header">
            <span>📊 인터랙티브 차트</span>
            <div class="chart-actions">
              <button class="chart-btn" onclick="downloadChartPng('${chartId}')">📥 PNG</button>
              <button class="chart-btn" onclick="copyChartJson(this)" data-chart="${b64}">📋 JSON</button>
            </div>
          </div>
          <div class="chart-canvas-wrap"><canvas id="${chartId}" data-chart-json="${b64}"></canvas></div>
          <details class="chart-raw-toggle"><summary>📝 원본 JSON</summary><pre>${escapeHtml(trimmed)}</pre></details>
        </div>`;
      });

      // 2) CSV 다운로드 마커 → 다운로드 버튼으로 변환
      s = s.replace(/\[CSV_DOWNLOAD:(\w+):(\d+)\]/g, (_, qid, cnt) =>
        `<button style="display:inline-flex;align-items:center;gap:6px;padding:10px 20px;background:#6366f1;color:#fff;border:none;border-radius:8px;cursor:pointer;font-size:14px;font-weight:600;margin:8px 0;" onmouseover="this.style.background='#4f46e5'" onmouseout="this.style.background='#6366f1'" onclick="downloadLogpressoCsv('${qid}')">📥 CSV 다운로드 (${cnt}건)</button>`
      );

      // 3) 나머지 마크다운 렌더링
      if (typeof marked !== 'undefined') {
        s = marked.parse(s);
      } else {
        s = escapeHtml(s);
        s = s.replace(/```(\w*)\n([\s\S]*?)```/g, '<pre><code class="lang-$1">$2</code></pre>');
        s = s.replace(/`([^`]+)`/g, '<code>$1</code>');
        s = s.replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>');
        s = s.replace(/\n/g, '<br>');
      }
      return s;
    }

    /* ── KPI 실시간 랜덤 업데이트 ── */
    const kpiState = {
      revenue: 2847, profit: 412, output: 98.2, yield: 96.7,
    };
    function updateKpi() {
      // 해고된 인원 수
      const firedCount = typeof firedCharacters !== 'undefined' ? firedCharacters.length : 0;
      const activeCount = 6 - firedCount;

      if (isOnStrike) {
        // 파업 중: 매 틱마다 급락
        kpiState.revenue -= (20 + Math.random() * 30);
        kpiState.profit -= (8 + Math.random() * 15);
        kpiState.output -= (0.5 + Math.random() * 1.2);
        kpiState.yield = Math.max(50, kpiState.yield - (0.3 + Math.random() * 0.8));
      } else if (firedCount > 0) {
        // 해고 상태: 인원 부족으로 하락
        const penalty = firedCount * 0.5;
        kpiState.revenue -= (5 + Math.random() * 10) * penalty;
        kpiState.profit -= (3 + Math.random() * 6) * penalty;
        kpiState.output -= (0.2 + Math.random() * 0.5) * penalty;
        kpiState.yield = Math.max(60, kpiState.yield - (0.1 + Math.random() * 0.3) * penalty);
      } else {
        // 정상: 소폭 랜덤 변동 (상승 편향)
        kpiState.revenue += (Math.random() - 0.48) * 15;
        kpiState.profit += (Math.random() - 0.48) * 5;
        kpiState.output += (Math.random() - 0.5) * 0.8;
        kpiState.yield = Math.min(99.9, Math.max(90, kpiState.yield + (Math.random() - 0.48) * 0.3));
      }

      // 최소값 클램프
      kpiState.revenue = Math.max(0, kpiState.revenue);
      kpiState.profit = Math.max(-500, kpiState.profit);
      kpiState.output = Math.max(0, kpiState.output);

      const rChg = ((kpiState.revenue - 2847) / 2847 * 100).toFixed(1);
      const pChg = ((kpiState.profit - 412) / 412 * 100).toFixed(1);
      const oChg = ((kpiState.output - 98.2) / 98.2 * 100).toFixed(1);
      const yChg = (kpiState.yield - 96.7).toFixed(1);

      setKpi('kpiRevenue', `₩${Math.round(kpiState.revenue).toLocaleString()}억`, rChg);
      setKpi('kpiProfit', kpiState.profit >= 0 ? `₩${Math.round(kpiState.profit).toLocaleString()}억` : `▼₩${Math.abs(Math.round(kpiState.profit)).toLocaleString()}억`, pChg);
      setKpi('kpiOutput', `${kpiState.output.toFixed(1)}K`, oChg);
      setKpi('kpiYield', `${kpiState.yield.toFixed(1)}%`, yChg);

      // 에이전트 수 업데이트
      if (isOnStrike) {
        document.getElementById('kpiAgents').textContent = '0/6';
        document.getElementById('kpiAgents').className = 'kpi-value down';
        document.getElementById('kpiAgents').nextElementSibling.textContent = '🪧 파업중';
        document.getElementById('kpiAgents').nextElementSibling.className = 'kpi-change down';
      } else {
        document.getElementById('kpiAgents').textContent = `${activeCount}/6`;
        document.getElementById('kpiAgents').className = activeCount >= 5 ? 'kpi-value up' : activeCount >= 3 ? 'kpi-value neutral' : 'kpi-value down';
        const agChgEl = document.getElementById('kpiAgents').nextElementSibling;
        if (firedCount > 0) {
          agChgEl.textContent = `🔥 ${firedCount}명 해고`;
          agChgEl.className = 'kpi-change down';
        } else {
          agChgEl.textContent = '● 가동중';
          agChgEl.className = 'kpi-change up';
        }
      }
    }
    function setKpi(id, val, chg) {
      const el = document.getElementById(id);
      if (!el) return;
      const n = parseFloat(chg);
      const cls = n > 0 ? 'up' : n < 0 ? 'down' : 'neutral';
      const arrow = n > 0 ? '▲' : n < 0 ? '▼' : '●';
      el.textContent = val;
      el.className = 'kpi-value ' + cls;
      const chgEl = el.nextElementSibling;
      if (chgEl) {
        chgEl.textContent = `${arrow} ${Math.abs(n).toFixed(1)}%`;
        chgEl.className = 'kpi-change ' + cls;
      }
    }

    function getFileIcon(name) {
      const ext = name.split('.').pop().toLowerCase();
      const map = {
        pdf: '📄', doc: '📝', docx: '📝', xls: '📊', xlsx: '📊', csv: '📊',
        ppt: '📽️', pptx: '📽️', txt: '📃', md: '📃',
        png: '🖼️', jpg: '🖼️', jpeg: '🖼️', gif: '🖼️', webp: '🖼️', svg: '🖼️',
        mp4: '🎬', mov: '🎬', mp3: '🎵', wav: '🎵',
        zip: '📦', rar: '📦', '7z': '📦',
        py: '🐍', js: '⚡', html: '🌐', css: '🎨', json: '📋',
      };
      return map[ext] || '📎';
    }

    function formatSize(bytes) {
      if (bytes < 1024) return bytes + 'B';
      if (bytes < 1048576) return (bytes / 1024).toFixed(1) + 'KB';
      return (bytes / 1048576).toFixed(1) + 'MB';
    }

    function buildAttachmentHtml(files) {
      if (!files.length) return '';
      let html = '<div class="msg-attachments">';
      for (const f of files) {
        const isImage = /\.(png|jpe?g|gif|webp|svg)$/i.test(f.name);
        if (isImage) {
          const url = URL.createObjectURL(f);
          html += `<img class="msg-attach-thumb" src="${url}" alt="${escapeHtml(f.name)}" onclick="window.open(this.src)" />`;
        }
        html += `
          <div class="msg-attach-item">
            <span class="ma-icon">${getFileIcon(f.name)}</span>
            <span class="ma-name">${escapeHtml(f.name)}</span>
            <span class="ma-size">${formatSize(f.size)}</span>
          </div>`;
      }
      html += '</div>';
      return html;
    }

    // ===== 에이전트 상태 표시 (병렬 상태) =====
    function setAgentStatus(key, status){
      // status: 'idle'|'working'|'done'|'error'
      parallelStatus[key] = status;
      const card = document.querySelector(`.agent-card.${key}`);
      if(!card) return;
      const dot = card.querySelector('.status');
      if(!dot) return;
      const colors = {idle:'#2ecc71', working:'#f59e0b', done:'#3b82f6', error:'#ef4444'};
      const labels = {idle:'대기', working:'작업중', done:'완료', error:'오류'};
      dot.style.background = colors[status] || '#2ecc71';
      dot.title = labels[status] || '대기';
      // 작업중이면 빠른 펄스
      if(status === 'working'){
        dot.style.animation = 'pulse 0.6s infinite';
      } else {
        dot.style.animation = status === 'idle' ? 'pulse 2s infinite' : 'none';
      }
      // 캔버스 캐릭터 슬로건 업데이트
      const char = characters.find(c => c._key === key);
      if(char){
        if(status === 'working'){
          char._slogan = '🔄 분석 중...';
          char._sloganColor = '#f59e0b';
        } else if(status === 'done'){
          char._slogan = '✅ 완료!';
          char._sloganColor = '#22c55e';
          setTimeout(()=>{ char._slogan = ''; }, 3000);
        } else if(status === 'error'){
          char._slogan = '❌ 오류';
          char._sloganColor = '#ef4444';
          setTimeout(()=>{ char._slogan = ''; }, 3000);
        } else {
          char._slogan = '';
        }
      }
    }

    function setParallelDone(results){
      // results: [{group, skills, error}]
      results.forEach(r => {
        const key = groupToAgent(r.group);

        Object.keys(parallelStatus).forEach(k => setAgentStatus(k, 'idle'));
      }, 5000);
    }

    async function executeResearch(mode){
      const taskInput = document.getElementById('researchTaskInput');
      const task = taskInput ? taskInput.value.trim() : '';
      if(!task){ alert('연구 과제를 입력해주세요.'); return; }

      const modeDef = RESEARCH_MODES[mode];
      const agentKeys = Object.keys(modeDef.agents);
      const chatArea = document.getElementById('chatArea');

      // 상태 초기화
      researchState = {
        mode, task, assignments: modeDef.agents,
        results: {}, status: 'running',
        completed: 0, total: agentKeys.length
      };
      saveResearchState();

      // CEO 채팅에 시작 메시지
      chatArea.innerHTML = `
        <div class="message user"><div class="msg-avatar">👔</div>
        <div class="msg-body"><b>[${modeDef.name}]</b><br>${escapeHtml(task)}</div></div>
        <div id="researchProgress" style="padding:12px;">
          <div style="color:#f59e0b;font-weight:700;">🔄 연구 진행 중...</div>
          <div id="workingImageArea" style="display:none;"></div>
          <div id="researchAgentStatus" style="margin-top:8px;"></div>
          <div id="researchProgressBar" style="margin-top:8px;background:var(--card);border-radius:4px;height:6px;overflow:hidden;">
            <div id="researchProgressFill" style="width:0%;height:100%;background:#6366f1;transition:width 0.5s;"></div>
          </div>
        </div>
      `;
      startWorkingAnimation();

      // 에이전트 상태 표시 업데이트
      function updateProgress(){
        const statusEl = document.getElementById('researchAgentStatus');
        if(!statusEl) return;
        let html = '';
        for(const [ak, info] of Object.entries(modeDef.agents)){
          const name = agentBaseData[ak]?.name || ak;
          const st = researchState.results[ak] ? '✅' : (parallelStatus[ak]==='working' ? '🟡' : '⏳');
          html += `<div style="font-size:12px;color:var(--muted);padding:2px 0;">${st} ${name}: ${info.role}</div>`;
        }
        statusEl.innerHTML = html;
        const fill = document.getElementById('researchProgressFill');
        if(fill) fill.style.width = `${(researchState.completed / researchState.total * 100)}%`;
      }
      updateProgress();

      // 에이전트 상태 → 작업중
      agentKeys.forEach(k => setAgentStatus(k, 'working'));

      // Promise.all로 병렬 실행
      const promises = agentKeys.map(async (ak) => {
        const info = modeDef.agents[ak];
        const agentName = agentBaseData[ak]?.name || ak;
        const subtaskPrompt = `[${modeDef.name}] 과제: "${task}"\n\n당신은 ${agentName}이고, 역할은 "${info.role}"입니다.\n당신의 전문 영역에서 이 과제를 분석하고 결과를 제출하세요.\n반드시 한국어로 답변하세요.`;

        try {
          const resp = await fetch('/api/chat', {
            method: 'POST',
            headers: {'Content-Type':'application/json'},
            body: JSON.stringify({
              env: [selectedEnv],
              messages: [{role:'user', content: subtaskPrompt}],
              skills: info.skills,
              effort: 2,
              format: 'auto',
              writing_style: 'auto',
              max_tokens: userMaxTokens,
              think_mode: false,
            })
          });
          const data = await resp.json();
          const result = data.error ? {error: data.error} : {content: data.content, model: data.model_used};
          researchState.results[ak] = result;
          researchState.completed++;
          setAgentStatus(ak, data.error ? 'error' : 'done');
          // 에이전트 히스토리에도 저장
          if(!agentHistory[ak]) agentHistory[ak] = [];
          agentHistory[ak].push({role:'user', content: `[${modeDef.name}] ${task}`});
          agentHistory[ak].push({role:'assistant', content: data.content || data.error || ''});
          updateProgress();
          saveResearchState();
          return {agent: ak, ...result};
        } catch(e) {
          researchState.results[ak] = {error: e.message};
          researchState.completed++;
          setAgentStatus(ak, 'error');
          updateProgress();
          return {agent: ak, error: e.message};
        }
      });

      const results = await Promise.all(promises);

      // 작업 이미지 중지
      stopWorkingAnimation();

      // CEO 합성
      const successes = results.filter(r => r.content && !r.error);
      if(successes.length === 0){
        chatArea.innerHTML += `<div class="message"><div class="msg-avatar" style="background:#ef4444;color:#fff;">!</div><div class="msg-body">❌ 모든 에이전트가 실패했습니다.</div></div>`;
        researchState.status = 'failed';
        saveResearchState();
        return;
      }

      // 합성 프롬프트 (에이전트 결과 길이 제한으로 컨텍스트 확보)
      const maxPerAgent = Math.min(4000, Math.floor(20000 / successes.length));
      let synthParts = successes.map(r => {
        const name = agentBaseData[r.agent]?.name || r.agent;
        const role = modeDef.agents[r.agent]?.role || '';
        const content = r.content.length > maxPerAgent ? r.content.substring(0, maxPerAgent) + '\n...(이하 생략)' : r.content;
        return `=== [${name}] ${role} ===\n${content}`;
      }).join('\n\n');

      chatArea.innerHTML += `<div style="padding:12px;color:#f59e0b;">🔄 CEO 통합 보고서 작성 중...</div>`;

      try {
        const synthResp = await fetch('/api/chat', {
          method: 'POST',
          headers: {'Content-Type':'application/json'},
          body: JSON.stringify({

    function saveToArchive(){
      if(!researchState) return;
      const archives = JSON.parse(localStorage.getItem('ceo_research_archives') || '[]');
      archives.unshift({
        id: Date.now(),
        date: new Date().toLocaleString('ko-KR'),

    // 보고서 다운로드
    function downloadResearchReport(){
      if(!researchState || !researchState.finalReport) return;
      const modeName = RESEARCH_MODES[researchState.mode]?.name || researchState.mode;
      let content = `# ${modeName} 보고서\n`;
      content += `과제: ${researchState.task}\n`;
      content += `날짜: ${new Date().toLocaleString('ko-KR')}\n`;
      content += `에이전트: ${Object.keys(researchState.assignments||{}).length}명\n\n`;
      content += `---\n\n`;
      content += `## CEO 통합 보고서\n\n${researchState.finalReport}\n\n`;
      content += `---\n\n## 에이전트별 상세 결과\n\n`;
      for(const [ak, r] of Object.entries(researchState.results || {})){
        const name = agentBaseData[ak]?.name || ak;
        const role = researchState.assignments[ak]?.role || '';
        content += `### ${name} (${role})\n${r.content || r.error || '결과 없음'}\n\n`;
      }
      const blob = new Blob([content], {type:'text/markdown;charset=utf-8'});
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `CEO_${modeName.replace(/[^가-힣a-zA-Z0-9]/g,'')}_${new Date().toISOString().slice(0,10)}.md`;
      a.click();
      URL.revokeObjectURL(url);
    }

    async function executeRevision(){
      const input = document.getElementById('revisionInput');
      const feedback = input ? input.value.trim() : '';
      if(!feedback){ alert('수정 요청 사항을 입력해주세요.'); return; }
      if(!researchState) return;

      const chatArea = document.getElementById('chatArea');
      chatArea.innerHTML += `<div class="message user"><div class="msg-avatar">👔</div><div class="msg-body">🔄 수정 요청: ${escapeHtml(feedback)}</div></div>`;

      // 어떤 에이전트를 재실행할지 판단 (이름 언급 감지)
      const mode = researchState.mode;
      const modeDef = RESEARCH_MODES[mode];
      let targetAgents = [];
      for(const ak of Object.keys(modeDef.agents)){
        const name = (agentBaseData[ak]?.name || '').toUpperCase();
        if(feedback.toUpperCase().includes(name)){
          targetAgents.push(ak);
        }
      }
      if(targetAgents.length === 0) targetAgents = Object.keys(modeDef.agents); // 특정 에이전트 안 지정하면 전체

      chatArea.innerHTML += `<div style="padding:12px;color:#f59e0b;">🔄 ${targetAgents.map(k=>agentBaseData[k]?.name||k).join(', ')} 재작업 중...</div>`;
      targetAgents.forEach(k => setAgentStatus(k, 'working'));

      // 해당 에이전트만 재실행 (피드백 포함)
      const promises = targetAgents.map(async (ak) => {
        const info = modeDef.agents[ak];
        const prevResult = researchState.results[ak]?.content || '';
        const prompt = `[수정 요청] CEO 피드백: "${feedback}"\n\n이전 분석 결과:\n${prevResult}\n\n위 피드백을 반영하여 다시 분석해주세요.\n과제: "${researchState.task}"\n역할: ${info.role}\n반드시 한국어로 답변하세요.`;

        try {
          const resp = await fetch('/api/chat', {
            method:'POST', headers:{'Content-Type':'application/json'},
            body: JSON.stringify({
              env:[selectedEnv], messages:[{role:'user',content:prompt}],
              skills:info.skills, effort:2, format:'auto', writing_style:'auto', max_tokens:userMaxTokens, think_mode:false,
            })
          });
          const data = await resp.json();
          researchState.results[ak] = data.error ? {error:data.error} : {content:data.content, model:data.model_used};
          setAgentStatus(ak, data.error ? 'error' : 'done');
          return {agent:ak, content:data.content, error:data.error};
        } catch(e){
          setAgentStatus(ak, 'error');
          return {agent:ak, error:e.message};
        }
      });

      await Promise.all(promises);

      // 다시 합성 (에이전트 결과 길이 제한)
      const successes = Object.entries(researchState.results).filter(([k,r])=>r.content && !r.error);
      const maxPerAgent2 = Math.min(4000, Math.floor(20000 / successes.length));
      let synthParts = successes.map(([ak,r]) => {
        const name = agentBaseData[ak]?.name || ak;
        const role = modeDef.agents[ak]?.role || '';
        const content = r.content.length > maxPerAgent2 ? r.content.substring(0, maxPerAgent2) + '\n...(이하 생략)' : r.content;
        return `=== [${name}] ${role} ===\n${content}`;
      }).join('\n\n');

      try {
        const synthResp = await fetch('/api/chat', {
          method:'POST', headers:{'Content-Type':'application/json'},
          body: JSON.stringify({
            env:[selectedEnv], messages:[{role:'user',content:researchState.task}],
            skills:[], effort:3, format:'report', writing_style:'auto', max_tokens:userSynthMaxTokens, n_ctx:userNCtx, think_mode:false,
            system_prompt: `CEO로서 수정된 분석 결과를 통합 보고서로 작성하세요.\n한국어로 작성.\n모든 내용을 빠짐없이 포함하여 완전한 보고서를 작성하세요.\n\n${synthParts}`
          })
        });
        const sd = await synthResp.json();
        chatArea.innerHTML += `
          <div class="message">
            <div class="msg-avatar" style="background:linear-gradient(135deg,#f59e0b,#ea580c);color:#fff;">📋</div>
            <div class="msg-body">
              <div style="font-weight:700;color:#f59e0b;margin-bottom:8px;">📋 수정된 CEO 통합 보고서</div>
              ${renderMd(sd.content||sd.error||'합성 실패')}
            </div>
          </div>
          <div id="researchActions" style="display:flex;gap:8px;padding:12px;">
            <button onclick="approveResearch()" style="flex:1;padding:10px;background:#22c55e;color:#fff;border:none;border-radius:8px;font-weight:700;cursor:pointer;">✅ 승인</button>
            <button onclick="requestRevision()" style="flex:1;padding:10px;background:#f59e0b;color:#fff;border:none;border-radius:8px;font-weight:700;cursor:pointer;">🔄 다시 수정</button>
          </div>
        `;
        setTimeout(() => initChartsInChat(), 100);
        researchState.finalReport = sd.content;
        saveResearchState();
      } catch(e){
        chatArea.innerHTML += `<div class="message"><div class="msg-body">❌ 재합성 실패: ${e.message}</div></div>`;
      }
      setTimeout(()=>{ targetAgents.forEach(k => setAgentStatus(k, 'idle')); }, 5000);
    }

    function selectAllCodeFiles(select){
      const checkboxes = document.querySelectorAll('#chatArea input[type="checkbox"]');
      selectedCodeFiles.clear();
      checkboxes.forEach(cb => {
        cb.checked = select;
        if(select) selectedCodeFiles.add(parseInt(cb.value));

    async function executeCodeAnalysis(){
      if(selectedCodeFiles.size === 0){ alert('분석할 파일을 선택해주세요.'); return; }
      const taskInput = document.getElementById('researchTaskInput');
      const task = taskInput ? taskInput.value.trim() : '';
      if(!task){ alert('분석/추가 요청을 입력해주세요.'); return; }

      // 파일 업로드
      try { await uploadFiles(pendingFiles); } catch(e){ alert('파일 업로드 실패: '+e.message); return; }

      // 선택된 파일 내용 수집
      const files = pendingFiles || [];
      let fileContext = '';
      selectedCodeFiles.forEach(idx => {
        const f = files[idx];
        if(f){
          const name = f.name || f.filename || '';
          const content = f.content_full || f.content_preview || f.content || '';
          fileContext += `\n=== 파일: ${name} ===\n${content.substring(0, 5000)}\n`;
        }
      });

      // researchTaskInput에 파일 컨텍스트 포함
      const fullTask = `${task}\n\n[분석 대상 파일 ${selectedCodeFiles.size}개]\n${fileContext}`;
      taskInput.value = fullTask;
      executeResearch('code-analysis');
    }

    async function resumeFailedOnly(){
      if(!researchState) return;
      const modeDef = RESEARCH_MODES[researchState.mode];
      if(!modeDef) return;
      const allAgents = Object.keys(researchState.assignments || {});
      const failedAgents = allAgents.filter(k => !researchState.results[k]?.content);
      if(failedAgents.length === 0){ alert('실패한 에이전트가 없습니다.'); return; }

      const chatArea = document.getElementById('chatArea');
      chatArea.innerHTML = `
        <div class="message user"><div class="msg-avatar">👔</div>
        <div class="msg-body">🔄 실패한 ${failedAgents.length}개 에이전트 재실행</div></div>
        <div id="researchProgress" style="padding:12px;">
          <div style="color:#f59e0b;font-weight:700;">🔄 재실행 중...</div>
          <div id="researchAgentStatus" style="margin-top:8px;"></div>
        </div>
      `;

      failedAgents.forEach(k => setAgentStatus(k, 'working'));

      const promises = failedAgents.map(async (ak) => {
        const info = researchState.assignments[ak];
        const agentName = agentBaseData[ak]?.name || ak;
        const prompt = `[${modeDef.name}] 과제: "${researchState.task}"\n\n당신은 ${agentName}이고, 역할은 "${info.role}"입니다.\n전문 영역에서 이 과제를 분석하고 결과를 제출하세요.\n반드시 한국어로 답변하세요.`;
        try {
          const resp = await fetch('/api/chat', {
            method:'POST', headers:{'Content-Type':'application/json'},
            body: JSON.stringify({
              env:[selectedEnv], messages:[{role:'user',content:prompt}],
              skills:info.skills, effort:2, format:'auto', writing_style:'auto', max_tokens:userMaxTokens, think_mode:false,
            })
          });
          const data = await resp.json();
          researchState.results[ak] = data.error ? {error:data.error} : {content:data.content, model:data.model_used};
          researchState.completed = Object.keys(researchState.results).filter(k=>researchState.results[k]?.content).length;
          setAgentStatus(ak, data.error ? 'error' : 'done');
          if(!agentHistory[ak]) agentHistory[ak] = [];
          agentHistory[ak].push({role:'user', content:`[${modeDef.name}] ${researchState.task}`});
          agentHistory[ak].push({role:'assistant', content:data.content||data.error||''});
          saveResearchState();
          return {agent:ak, content:data.content, error:data.error};
        } catch(e){
          researchState.results[ak] = {error:e.message};
          setAgentStatus(ak, 'error');
          return {agent:ak, error:e.message};
        }
      });

      await Promise.all(promises);

      // 전체 결과 확인
      const allResults = Object.entries(researchState.results);
      const successes = allResults.filter(([k,r])=>r.content && !r.error);
      const stillFailed = allResults.filter(([k,r])=>!r.content || r.error);

      const prog = document.getElementById('researchProgress');
      if(prog) prog.remove();

      if(successes.length === 0){
        chatArea.innerHTML += `<div class="message"><div class="msg-body">❌ 여전히 모든 에이전트 실패</div></div>`;
        return;
      }

      // 합성 (에이전트 결과 길이 제한)
      const maxPerAgent3 = Math.min(4000, Math.floor(20000 / successes.length));
      let synthParts = successes.map(([ak,r]) => {
        const name = agentBaseData[ak]?.name || ak;
        const role = researchState.assignments[ak]?.role || '';
        const content = r.content.length > maxPerAgent3 ? r.content.substring(0, maxPerAgent3) + '\n...(이하 생략)' : r.content;
        return `=== [${name}] ${role} ===\n${content}`;
      }).join('\n\n');

      chatArea.innerHTML += `<div style="padding:12px;color:#f59e0b;">🔄 CEO 통합 보고서 작성 중...</div>`;

      try {
        const synthResp = await fetch('/api/chat', {
          method:'POST', headers:{'Content-Type':'application/json'},
          body: JSON.stringify({
            env:[selectedEnv], messages:[{role:'user',content:researchState.task}],
            skills:[], effort:3, format:'report', writing_style:'auto', max_tokens:userSynthMaxTokens, n_ctx:userNCtx, think_mode:false,
            system_prompt:`CEO로서 ${successes.length}명 에이전트 분석을 통합 보고서로 작성.\n한국어.\n모든 내용을 빠짐없이 포함하여 완전한 보고서를 작성하세요.\n\n${synthParts}`
          })
        });
        const sd = await synthResp.json();
        chatArea.innerHTML += `
          <div class="message">
            <div class="msg-avatar" style="background:linear-gradient(135deg,#f59e0b,#ea580c);color:#fff;">📋</div>
            <div class="msg-body">
              <div style="font-weight:700;color:#f59e0b;margin-bottom:8px;">📋 CEO 통합 보고서</div>
              ${renderMd(sd.content||sd.error||'합성 실패')}
              ${stillFailed.length > 0 ? `<div style="color:#ef4444;font-size:11px;margin-top:8px;">⚠️ ${stillFailed.length}개 에이전트 여전히 실패</div>` : ''}
            </div>
          </div>
          <div id="researchActions" style="display:flex;gap:8px;padding:12px;">
            <button onclick="approveResearch()" style="flex:1;padding:10px;background:#22c55e;color:#fff;border:none;border-radius:8px;font-weight:700;cursor:pointer;">✅ 승인</button>
            <button onclick="requestRevision()" style="flex:1;padding:10px;background:#f59e0b;color:#fff;border:none;border-radius:8px;font-weight:700;cursor:pointer;">🔄 수정 요청</button>
          </div>
        `;
        setTimeout(() => initChartsInChat(), 100);
        researchState.status = 'review';
        researchState.finalReport = sd.content;
        saveResearchState();
        try { allAgentsReportToCeo(); } catch(e){}
      } catch(e){
        chatArea.innerHTML += `<div class="message"><div class="msg-body">❌ 합성 실패: ${e.message}</div></div>`;
      }
      setTimeout(()=>{ failedAgents.forEach(k => setAgentStatus(k, 'idle')); }, 5000);
    }

      for(const [ak, r] of doneResults){

        const role = researchState.assignments[ak]?.role || '';
        const agent = agentBaseData[ak] || {color1:'#6366f1',color2:'#8b5cf6',letter:'?'};
        html += `
          <div class="message">
            <div class="msg-avatar" style="background:linear-gradient(135deg,${agent.color1},${agent.color2});color:#fff;">${agent.letter}</div>
            <div class="msg-body">
              <div style="font-weight:700;color:${agent.color1};margin-bottom:4px;">${name} (${role})</div>
              ${renderMd(r.content)}
            </div>
          </div>`;
      }
      const failedAgents = Object.keys(researchState.assignments).filter(k=>!researchState.results[k]?.content);
      if(failedAgents.length > 0){
        html += `<div style="padding:12px;display:flex;gap:8px;">
          <button onclick="resumeFailedOnly()" style="padding:10px 16px;background:#f59e0b;color:#fff;border:none;border-radius:8px;cursor:pointer;font-weight:700;">🔄 실패한 ${failedAgents.length}개 재실행</button>
          <button onclick="openAgent('ceo')" style="padding:10px 16px;background:var(--card);color:var(--muted);border:1px solid var(--border);border-radius:8px;cursor:pointer;">← 돌아가기</button>
        </div>`;
      }
      chatArea.innerHTML = html;
      setTimeout(() => initChartsInChat(), 100);
    }

    // 처음부터 다시
    function redoResearchFull(){
      if(!researchState) return;
      if(!confirm('처음부터 다시 하시겠습니까? 기존 결과가 삭제됩니다.')) return;
      const mode = researchState.mode;
      const task = researchState.task;
      localStorage.removeItem('ceo_research_state');
      researchState = null;
      // 과제 입력 화면으로 가되 이전 과제 내용 유지
      startResearchMode(mode);
      setTimeout(()=>{
        const input = document.getElementById('researchTaskInput');
        if(input) input.value = task;
      }, 100);
    }

      localStorage.removeItem('ceo_research_state');
