function go(file){
  const r=new FileReader();
  r.onload=e=>{try{run(e.target.result,file.name,file.size);}catch(err){alert('오류: '+err.message);}};
  r.readAsText(file,'UTF-8');
}

function parseCSV(txt){
  const ls=txt.trim().split('\n');
  if(ls.length<2)throw new Error('최소 2행 필요');
  let sp=',';if(ls[0].split('\t').length>ls[0].split(',').length)sp='\t';
  const hd=ls[0].split(sp).map(h=>h.replace(/^"|"$/g,'').trim());
  const al={VHL_ID:['VHL_ID','VHLID','VHL','VEHICLE_ID'],FROM_HIDID:['FROM_HIDID','FROM_HID','FROMHID'],TO_HIDID:['TO_HIDID','TO_HID','TOHID'],TRANS_CNT:['TRANS_CNT','TRANSCNT','CNT'],EVENT_DT:['EVENT_DT','EVENT_DATE','_time','TIME','TIMESTAMP'],FAB_ID:['FAB_ID','FABID','FAB'],EQP_ID:['EQP_ID','EQPID'],MCP_NM:['MCP_NM','MCP']};
  const cm={};
  for(const[k,ns]of Object.entries(al)){for(const n of ns){const i=hd.findIndex(h=>h.toUpperCase()===n.toUpperCase());if(i>=0){cm[k]=i;break;}}}
  if(!('VHL_ID'in cm)||!('FROM_HIDID'in cm)||!('TO_HIDID'in cm))throw new Error('필수 컬럼 누락: VHL_ID, FROM_HIDID, TO_HIDID\n헤더: '+hd.join(', '));
  const rows=[];
  for(let i=1;i<ls.length;i++){
    const c=ls[i].split(sp).map(x=>x.replace(/^"|"$/g,'').trim());
    if(c.length<3)continue;
    rows.push({V:c[cm.VHL_ID]||'',F:c[cm.FROM_HIDID]||'',T:c[cm.TO_HIDID]||'',TC:cm.TRANS_CNT!==undefined?parseInt(c[cm.TRANS_CNT])||1:1,DT:cm.EVENT_DT!==undefined?c[cm.EVENT_DT]:'',FAB:cm.FAB_ID!==undefined?c[cm.FAB_ID]:'',EQP:cm.EQP_ID!==undefined?c[cm.EQP_ID]:'',MCP:cm.MCP_NM!==undefined?c[cm.MCP_NM]:''});
  }
  return rows;
}

function run(txt,fname,fsize){
  const rows=parseCSV(txt);

  document.getElementById('fnE').textContent=fname;
  document.getElementById('fmE').textContent=rows.length.toLocaleString()+' rows · '+(fsize/1024).toFixed(1)+' KB';
  document.getElementById('upW').style.display='none';
  document.getElementById('res').classList.add('on');

  // ===== 분석 시작 =====

  // 기본 집계
  const vhls=[...new Set(rows.map(r=>r.V))];
  const froms=new Set(rows.map(r=>r.F));
  const tos=new Set(rows.map(r=>r.T));
  const allHids=[...new Set([...froms,...tos])];
  const fabs=[...new Set(rows.map(r=>r.FAB).filter(Boolean))];
  const eqps=[...new Set(rows.map(r=>r.EQP).filter(Boolean))];
  const dts=[...new Set(rows.map(r=>r.DT).filter(Boolean))].sort();

  // VHL 접두어
  const vPrefix={};
  vhls.forEach(v=>{const p=v.replace(/[0-9]/g,'');vPrefix[p]=(vPrefix[p]||0)+1;});

  // 시간 범위
  const isSingleTime=dts.length<=1;
  let timeDesc='';
  if(dts.length===0) timeDesc='시간 정보 없음';
  else if(dts.length===1) timeDesc=dts[0]+' (단일 시점 스냅샷)';
  else timeDesc=dts[0]+' ~ '+dts[dts.length-1]+' ('+dts.length+'개 시점)';

  // HID 트래픽
  const ht={};
  rows.forEach(r=>{ht[r.F]=(ht[r.F]||0)+1;ht[r.T]=(ht[r.T]||0)+1;});
  const sH=Object.entries(ht).sort((a,b)=>b[1]-a[1]);
  const topHid=sH[0];
  const top5=sH.slice(0,5);
  const avgTraffic=(Object.values(ht).reduce((a,b)=>a+b,0)/allHids.length);

  // 이동 구간
  const pc={};
  rows.forEach(r=>{const k=r.F+'→'+r.T;pc[k]=(pc[k]||0)+1;});
  const topPairs=Object.entries(pc).sort((a,b)=>b[1]-a[1]);

  // 양방향
  const bidi=[];const bSeen=new Set();
  for(const[k,c]of Object.entries(pc)){
    const[a,b]=k.split('→');const rv=b+'→'+a;
    if(pc[rv]&&!bSeen.has(b+'↔'+a)){bidi.push({a,b,ab:c,ba:pc[rv],tot:c+pc[rv]});bSeen.add(a+'↔'+b);}
  }
  bidi.sort((a,b)=>b.tot-a.tot);

  // VHL별 이동 수
  const vg={};rows.forEach(r=>{if(!vg[r.V])vg[r.V]=[];vg[r.V].push(r);});
  const moveDist={};
  Object.values(vg).forEach(rs=>{const n=rs.length;moveDist[n]=(moveDist[n]||0)+1;});

  const multiVhls=Object.entries(vg).filter(([,rs])=>rs.length>=3).sort((a,b)=>b[1].length-a[1].length);

  // 반복 방문
  const circularVhls=[];
  for(const[vid,rs]of Object.entries(vg)){
    if(rs.length<2)continue;
    const fromCnt={};rs.forEach(r=>{fromCnt[r.F]=(fromCnt[r.F]||0)+1;});
    const reps=Object.entries(fromCnt).filter(([,c])=>c>1);
    if(reps.length)circularVhls.push({vid,reps,moves:rs.length});
  }
  circularVhls.sort((a,b)=>b.moves-a.moves);

  // 재시도
  const retries=rows.filter(r=>r.TC>=2);

  // 고빈도 연결 그룹
  const cTh=Math.max(3,Math.floor(rows.length/250));
  const adj={};
  for(const[k,c]of Object.entries(pc)){
    if(c<cTh)continue;
    const[a,b]=k.split('→');
    if(!adj[a])adj[a]=new Set();if(!adj[b])adj[b]=new Set();
    adj[a].add(b);adj[b].add(a);
  }
  const vis=new Set();const grps=[];
  for(const nd of Object.keys(adj)){
    if(vis.has(nd))continue;
    const cl=new Set();const stk=[nd];
    while(stk.length){const n=stk.pop();if(vis.has(n))continue;vis.add(n);cl.add(n);if(adj[n])for(const nb of adj[n])if(!vis.has(nb))stk.push(nb);}
    if(cl.size>=3)grps.push([...cl].sort((a,b)=>parseInt(a)-parseInt(b)));
  }
  grps.sort((a,b)=>{const ta=a.reduce((s,h)=>s+(ht[h]||0),0);const tb=b.reduce((s,h)=>s+(ht[h]||0),0);return tb-ta;});

  // ===== 리포트 생성 =====
  let html='';

  // 1. 데이터 개요
  html+=`<div class="section">
    <div class="stitle"><div class="num" style="background:var(--blue)">1</div>이 데이터는 뭔가?</div>
    <div class="fact">
      <div class="label blue">데이터 정체</div>
      <p>이 파일은 <span class="hl">${fabs.join(', ')||'(FAB 미상)'}</span> FAB의 <span class="hl">OHT HID INOUT</span> 로그입니다.</p>
      <p>OHT가 HID 존 A에서 HID 존 B로 <span class="hl">이동을 완료했을 때</span> 찍히는 기록입니다.</p>
      <p>총 <span class="hl">${rows.length.toLocaleString()}건</span>의 이동 완료 기록이 들어있습니다.</p>
    </div>
    <div class="fact">
      <div class="label blue">시간 범위</div>
      <p>${timeDesc}</p>
      ${isSingleTime?'<p>⚠️ <span class="hl">모든 레코드가 같은 시각</span>입니다. 시간 흐름에 따른 변화를 볼 수 없고, "이 시점의 이동 현황"만 확인 가능합니다.</p>':'<p>여러 시점의 데이터가 있어 시간에 따른 변화를 볼 수 있습니다.</p>'}
    </div>
    <div class="fact">
      <div class="label blue">등장하는 OHT</div>
      <p>총 <span class="hl">${vhls.length}대</span>의 OHT가 기록에 등장합니다.</p>
      <p>접두어별: ${Object.entries(vPrefix).map(([k,v])=>`<span class="code">${k}</span> ${v}대`).join(', ')}</p>
    </div>
    <div class="fact">
      <div class="label blue">사용된 HID 존</div>
      <p>총 <span class="hl">${allHids.length}개</span> HID 존이 사용되었습니다. (번호 범위: ${Math.min(...allHids.map(Number).filter(n=>!isNaN(n)))} ~ ${Math.max(...allHids.map(Number).filter(n=>!isNaN(n)))})</p>
    </div>
  </div>`;

  // 2. 트래픽 집중
  html+=`<div class="section">
    <div class="stitle"><div class="num" style="background:var(--cyan)">2</div>어떤 HID 존에 이동이 몰리는가?</div>
    <div class="fact">
      <div class="label cyan">트래픽 집중도</div>
      <p>HID 존별로 출발지(FROM) + 도착지(TO)로 사용된 총 건수를 합산했습니다.</p>
      <p>평균은 존당 <span class="hl">${avgTraffic.toFixed(1)}건</span>인데, 상위 존은 이보다 훨씬 높습니다:</p>
      <ul class="fact-list">
        ${top5.map(([h,v],i)=>`<li><span class="rank">${i+1}.</span> HID <span class="code">${h}</span> <span class="val">${v}건</span> — 평균의 ${(v/avgTraffic).toFixed(1)}배</li>`).join('')}
      </ul>
    </div>
    <div class="fact">
      <div class="label cyan">해석</div>
      <p>HID <span class="code">${topHid[0]}</span>이 <span class="hl">${topHid[1]}건</span>으로 가장 많습니다. 이 존을 출발하거나 도착하는 OHT가 다른 존보다 ${(topHid[1]/avgTraffic).toFixed(1)}배 많다는 뜻입니다.</p>
      <p>트래픽이 높다는 것은 그 존이 <span class="hl">많이 사용되는 구간</span>이라는 사실만 말해줍니다. 병목인지 정상인지는 이 데이터만으로 판단할 수 없습니다.</p>
    </div>
  </div>`;

  // 3. 자주 쓰이는 구간
  html+=`<div class="section">
    <div class="stitle"><div class="num" style="background:var(--green)">3</div>어떤 이동 구간이 자주 쓰이는가?</div>
    <div class="fact">
      <div class="label green">상위 이동 구간</div>
      <p>FROM → TO 조합별 건수입니다. 같은 경로를 여러 OHT가 이동했다는 뜻입니다.</p>
      <ul class="fact-list">
        ${topPairs.slice(0,10).map(([k,v],i)=>`<li><span class="rank">${i+1}.</span> <span class="code">${k}</span> <span class="val">${v}건</span></li>`).join('')}
      </ul>
    </div>
    <div class="fact">
      <div class="label green">이동 구간 다양성</div>
      <p>총 <span class="hl">${Object.keys(pc).length}가지</span> 서로 다른 이동 구간이 존재합니다.</p>
      <p>1회만 나타난 구간: <span class="hl">${Object.values(pc).filter(v=>v===1).length}개</span> (${(Object.values(pc).filter(v=>v===1).length/Object.keys(pc).length*100).toFixed(0)}%)</p>
      <p>3회 이상 나타난 구간: <span class="hl">${Object.values(pc).filter(v=>v>=3).length}개</span></p>
    </div>
  </div>`;

  // 4. 양방향
  html+=`<div class="section">
    <div class="stitle"><div class="num" style="background:var(--amber)">4</div>같은 구간에서 양방향 이동이 있는가?</div>`;
  if(bidi.length===0){
    html+=`<div class="fact"><div class="label amber">결과</div><p>양방향 이동 구간이 없습니다.</p></div>`;
  } else {
    html+=`<div class="fact">
      <div class="label amber">양방향 이동 쌍: ${bidi.length}개</div>
      <p>A→B 이동과 B→A 이동이 <span class="hl">모두 기록된</span> HID 쌍입니다.</p>
      <p>같은 구간에서 서로 반대 방향으로 OHT가 이동한 <span class="hl">사실</span>만 보여줍니다.</p>
      <ul class="fact-list">
        ${bidi.slice(0,10).map((p,i)=>`<li><span class="rank">${i+1}.</span> HID <span class="code">${p.a}</span> ↔ <span class="code">${p.b}</span> — ${p.a}→${p.b}: ${p.ab}건, ${p.b}→${p.a}: ${p.ba}건 (합계 <span class="val">${p.tot}건</span>)</li>`).join('')}
      </ul>
    </div>
    <div class="fact">
      <div class="label amber">이것이 의미하는 것과 아닌 것</div>
      <p>✅ 의미하는 것: 이 구간에서 양쪽 방향으로 OHT가 이동했다.</p>
      <p>❌ 의미하지 <b>않는</b> 것: 이 구간에서 충돌이나 데드락이 발생했다. (이 데이터로는 알 수 없음)</p>
    </div>`;
  }
  html+=`</div>`;

  // 5. VHL 이동 패턴
  html+=`<div class="section">
    <div class="stitle"><div class="num" style="background:var(--purple)">5</div>개별 OHT는 몇 번 이동했는가?</div>
    <div class="fact">
      <div class="label purple">이동 횟수 분포</div>
      <p>각 OHT가 이 데이터에서 몇 건의 이동 기록을 남겼는지:</p>
      <ul class="fact-list">
        ${Object.entries(moveDist).sort((a,b)=>parseInt(a[0])-parseInt(b[0])).map(([k,v])=>`<li><span class="code">${k}회</span> 이동한 OHT: <span class="val">${v}대</span></li>`).join('')}
      </ul>
    </div>`;
  if(multiVhls.length>0){
    html+=`<div class="fact">
      <div class="label purple">3회 이상 이동한 OHT: ${multiVhls.length}대</div>
      <p>같은 시간대에 여러 번 이동 기록을 남긴 OHT들의 실제 경로입니다:</p>
      <ul class="fact-list">
        ${multiVhls.slice(0,10).map(([vid,rs])=>{
          const route=rs.map(r=>r.F+'→'+r.T).join(' , ');
          return `<li><span class="code">${vid}</span> (${rs.length}회): <span style="color:var(--text2);font-size:11px">${route}</span></li>`;
        }).join('')}
      </ul>
    </div>`;
  }
  html+=`</div>`;

  // 6. 반복 방문
  if(circularVhls.length>0){
    html+=`<div class="section">
      <div class="stitle"><div class="num" style="background:var(--red)">6</div>같은 HID를 반복 방문한 OHT가 있는가?</div>
      <div class="fact">
        <div class="label red">반복 방문 OHT: ${circularVhls.length}대</div>
        <p>출발지(FROM)로 같은 HID를 2번 이상 사용한 OHT입니다.</p>
        <ul class="fact-list">
          ${circularVhls.slice(0,10).map(cv=>`<li><span class="code">${cv.vid}</span> — ${cv.reps.map(([h,c])=>`HID ${h}에서 ${c}회 출발`).join(', ')}</li>`).join('')}
        </ul>
      </div>
      <div class="fact">
        <div class="label red">이것이 의미하는 것과 아닌 것</div>
        <p>✅ 의미하는 것: 이 OHT가 같은 HID를 여러 번 출발지로 사용했다.</p>
        <p>❌ 의미하지 <b>않는</b> 것: 이 OHT가 제자리를 맴돌거나 문제가 있었다. (이유는 알 수 없음)</p>
      </div>
    </div>`;
  }

  // 7. TRANS_CNT
  html+=`<div class="section">
    <div class="stitle"><div class="num" style="background:var(--red)">7</div>TRANS_CNT가 2 이상인 기록이 있는가?</div>`;
  if(retries.length===0){
    html+=`<div class="fact"><div class="label green">결과</div><p>모든 레코드의 TRANS_CNT가 1입니다. 2 이상인 기록이 없습니다.</p></div>`;
  } else {
    html+=`<div class="fact">
      <div class="label red">TRANS_CNT ≥ 2: ${retries.length}건</div>
      <p>TRANS_CNT가 1이 아닌 기록입니다. 이 필드가 정확히 무엇을 의미하는지는 ATLAS 시스템 정의에 따릅니다.</p>
      <ul class="fact-list">
        ${retries.map(r=>`<li><span class="code">${r.V}</span> — HID ${r.F} → ${r.T}, TRANS_CNT=<span class="val">${r.TC}</span></li>`).join('')}
      </ul>
    </div>`;
  }
  html+=`</div>`;

  // 8. 연결 그룹
  if(grps.length>0){
    html+=`<div class="section">
      <div class="stitle"><div class="num" style="background:var(--cyan)">8</div>자주 함께 쓰이는 HID 그룹</div>
      <div class="fact">
        <div class="label cyan">고빈도 연결 그룹 (기준: 구간별 ${cTh}건 이상)</div>
        <p>이동 빈도가 ${cTh}건 이상인 구간끼리 연결하면 아래 HID 그룹이 만들어집니다. 같은 그룹 내 HID들은 서로 OHT가 자주 오가는 영역입니다.</p>
      </div>
      ${grps.slice(0,5).map((g,i)=>{
        const traffic=g.reduce((s,h)=>s+(ht[h]||0),0);
        return `<div class="fact">
          <div class="label cyan">그룹 ${i+1} — ${g.length}개 HID, 합산 트래픽 ${traffic}건</div>
          <p style="font-family:var(--mono);font-size:11px;color:var(--text2);word-break:break-all;">${g.join(', ')}</p>
        </div>`;
      }).join('')}
    </div>`;
  }

  // 알 수 없는 것
  html+=`<div class="section">
    <div class="stitle"><div class="num" style="background:var(--red)">⚠</div>이 데이터로 알 수 없는 것</div>
    <div class="cannot">
      <div class="label">확인 불가 항목</div>
      <p>아래 항목들은 이 HID INOUT 데이터에 해당 정보가 없어서 판단할 수 없습니다:</p>
      <p>• <b>데드락 발생 여부</b> — OHT 대기/차단 상태 정보가 없음</p>
      <p>• <b>OHT 상태</b> (RUN/WAIT/BLOCKED/DOWN) — 상태 필드가 없음</p>
      <p>• <b>이동 소요 시간</b> — 이동 시작 시각이 없고 완료 시각만 있음</p>
      <p>• <b>멈춰있는 OHT</b> — 이동 완료 기록이라 안 움직인 OHT는 기록에 안 남음</p>
      <p>• <b>이동 원인</b> — 정상 명령인지, 회피인지, 재시도인지 구분 불가</p>
      ${isSingleTime?'<p>• <b>시간에 따른 변화</b> — 모든 기록이 같은 시각이라 추세 분석 불가</p>':''}
    </div>
  </div>`;

  // 종합
  html+=`<div class="summary-box">
    <div class="stitle">📋 종합 요약</div>
    <p>이 파일은 <b>${fabs.join(', ')||'(FAB 미상)'}</b>의 <b>${timeDesc}</b> 기준, <b>${vhls.length}대</b> OHT가 <b>${allHids.length}개</b> HID 존 사이에서 <b>${rows.length.toLocaleString()}건</b> 이동한 기록입니다.</p>
    <p>트래픽이 가장 집중된 HID는 <b>${topHid[0]}</b> (${topHid[1]}건)이고, 양방향 이동이 확인된 구간은 <b>${bidi.length}쌍</b>, TRANS_CNT≥2인 기록은 <b>${retries.length}건</b>입니다.</p>
    ${multiVhls.length>0?`<p>3회 이상 이동한 OHT는 <b>${multiVhls.length}대</b>이고, 같은 HID를 반복 방문한 OHT는 <b>${circularVhls.length}대</b>입니다.</p>`:''}
    <p>데드락 판정, OHT 상태 분류, 가동률 계산은 이 데이터에 해당 정보가 없어서 할 수 없습니다.</p>
  </div>`;

  document.getElementById('report').innerHTML=html;
  window.scrollTo({top:0,behavior:'smooth'});
}
