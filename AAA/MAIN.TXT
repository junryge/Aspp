  // 3. HID OUT(퇴장) 밀도
  const outTop=sOut.slice(0,25);
  h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--green)">3</div><div class="st">어떤 HID에서 OHT가 많이 빠져나갔는가?</div></div>
    <div class="cc"><div class="ct">HID별 퇴장(OUT) 건수 — FROM_HIDID 기준, 상위 25</div>
      <div class="cb" style="height:260px;"><canvas id="c3"></canvas></div>
    </div>
    <div class="ex"><div class="el g">이 그래프는 뭘 보여주나?</div>
      <p>각 HID에서 <span class="hl">빠져나간(OUT) OHT 수</span>. 1위: HID <span class="cd">${sOut[0][0]}</span> = ${sOut[0][1]}대 퇴장.</p>
      <p>진입(IN) 차트와 비교하면 어떤 HID가 "통과 지점"이고 어떤 HID가 "종착 지점"인지 감을 잡을 수 있음.</p>
    </div></div>`;

  // 4. Bidirectional IN/OUT
  const b12=bidi.slice(0,12);
  h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--purple)">4</div><div class="st">양방향 IN/OUT 쌍</div></div>`;
  if(bidi.length===0){
    h+=`<div class="ex"><div class="el p">결과</div><p>양방향 IN/OUT 쌍 없음.</p></div>`;
  } else {
    h+=`<div class="cc"><div class="ct">A→B(A에서 OUT, B에 IN) 와 B→A(B에서 OUT, A에 IN) 가 모두 기록된 HID 쌍</div>
      <div class="cb" style="height:260px;"><canvas id="c4"></canvas></div>
      <div class="lgd"><span><span class="sq" style="background:var(--purple)"></span>A OUT → B IN</span><span><span class="sq" style="background:var(--pink)"></span>B OUT → A IN</span></div>
    </div>
    <div class="ex"><div class="el p">이 그래프는 뭘 보여주나?</div>
      <p>HID A에서 나와서 B에 들어간 OHT도 있고, B에서 나와서 A에 들어간 OHT도 있는 쌍 <span class="hl">${bidi.length}개</span>.</p>
    </div>
    <div class="ex"><div class="el p">알 수 있는 것 / 없는 것</div>
      <p class="yes">✅ 이 두 HID 사이에서 양방향으로 OHT가 오간 사실</p>
      <p class="no">❌ 동시에 오간 건지, 순차적인 건지 (같은 시각 스냅샷이라 구분 불가)</p>
    </div>`;
    h+=`<div class="tw"><table class="tb"><thead><tr><td>HID A</td><td>HID B</td><td>A OUT→B IN</td><td>B OUT→A IN</td><td>합계</td></tr></thead><tbody>`;
    b12.forEach(p=>{h+=`<tr><td class="mn">${p.a}</td><td class="mn">${p.b}</td><td>${p.ab}</td><td>${p.ba}</td><td style="font-weight:600">${p.tot}</td></tr>`;});
    h+=`</tbody></table></div>`;
  }
  h+=`</div>`;

  // 5. Multi-record VHL
  h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--pink)">5</div><div class="st">여러 HID를 거친 OHT</div></div>
    <div class="cc"><div class="ct">OHT별 기록 수 분포</div>
      <div class="cb" style="height:200px;"><canvas id="c5"></canvas></div>
    </div>`;
  const md={};Object.values(vg).forEach(rs=>{const n=rs.length;md[n]=(md[n]||0)+1;});
  const mdS=Object.entries(md).sort((a,b)=>parseInt(a[0])-parseInt(b[0]));
  h+=`<div class="ex"><div class="el pk">분포</div>
    <p>${mdS.map(([k,v])=>`<span class="hl">${k}건</span>: ${v}대`).join(' / ')}</p>
    <p>대부분 1건이고 2건 이상은 <span class="hl">${multiV.length}대</span>.</p>
  </div>`;
  if(multiV.length>0){
    h+=`<div class="ex"><div class="el pk">2건 이상 기록 OHT — 실제 HID 전환 경로</div>
      <p>이 OHT들은 이 시점에 여러 HID를 거친 기록이 있음:</p></div>
    <div class="tw"><table class="tb"><thead><tr><td>VHL</td><td>건수</td><td>OUT HID → IN HID (순서대로)</td></tr></thead><tbody>`;
    multiV.slice(0,15).forEach(([vid,rs])=>{
      const route=rs.map(r=>`${r.F}→${r.T}`).join(' , ');
      h+=`<tr><td class="mn" style="font-weight:600">${vid}</td><td>${rs.length}</td><td class="mn" style="color:var(--tx2);max-width:400px;word-break:break-all">${route}</td></tr>`;
    });
    h+=`</tbody></table></div>`;
  }
  h+=`</div>`;

  // 6. TRANS_CNT
  h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--red)">6</div><div class="st">TRANS_CNT ≥ 2</div></div>`;
  if(ret.length===0){
    h+=`<div class="ex"><div class="el g">결과</div><p>TRANS_CNT가 2 이상인 기록 없음.</p></div>`;
  } else {
    h+=`<div class="tw"><table class="tb"><thead><tr><td>VHL</td><td>OUT HID</td><td>IN HID</td><td>TRANS_CNT</td></tr></thead><tbody>`;
    ret.forEach(r=>{h+=`<tr><td class="mn">${r.V}</td><td class="mn">${r.F}</td><td class="mn">${r.T}</td><td style="color:var(--amber);font-weight:600">${r.TC}</td></tr>`;});
    h+=`</tbody></table></div>
    <div class="ex"><div class="el r">알 수 있는 것 / 없는 것</div>
      <p class="yes">✅ 이 VHL의 이 HID 전환에서 TRANS_CNT가 2로 기록됨</p>
      <p class="no">❌ 이게 뭘 의미하는지는 ATLAS 시스템의 TRANS_CNT 정의에 따라 다름</p>
    </div>`;
  }
  h+=`</div>`;

  // 알 수 없는 것
  h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--red)">⚠</div><div class="st">이 데이터로 알 수 없는 것</div></div>
    <div class="wb"><div class="el">확인 불가</div>
      <p>• <b>데드락 여부</b> — OHT가 실제로 막혀있는지, 대기 중인지 상태 정보 없음</p>
      <p>• <b>HID 체류 시간</b> — 언제 진입해서 얼마나 머물렀는지 (OUT 시각 없음)</p>
      <p>• <b>HID 수용 용량</b> — 몇 대가 들어갈 수 있는 HID인지 모르면 "꽉 찼는지" 판단 불가</p>
      <p>• <b>아직 안 나온 OHT</b> — IN만 있고 OUT이 없는 OHT (이 스냅샷만으로는 판단 불가)</p>
      ${single?'<p>• <b>시간 추세</b> — 전부 같은 시각이라 "점점 쌓이는 중인지" 확인 불가</p>':''}
    </div></div>`;

  // Summary
  h+=`<div class="sum"><h3>📋 종합 요약</h3>
    <p><b>${fabs.join(', ')||'(FAB 미상)'}</b>의 <b>${dts.length<=1?(dts[0]||'시간 미상'):dts[0]+' ~ '+dts[dts.length-1]}</b> 시점, <b>${vhls.length}대</b> OHT의 HID IN/OUT 기록 <b>${rows.length.toLocaleString()}건</b>.</p>
    <p>가장 많은 OHT가 진입한 HID: <b>${sIn[0][0]}</b> (${sIn[0][1]}대) / 가장 많이 빠져나간 HID: <b>${sOut[0][0]}</b> (${sOut[0][1]}대)</p>
    <p>IN>OUT 불균형 HID: <b>${piling.length}개</b> (최대 +${piling.length?piling[0][1].d:0}대) / OUT>IN: <b>${draining.length}개</b></p>
    <p>양방향 쌍: <b>${bidi.length}개</b> / TRANS_CNT≥2: <b>${ret.length}건</b></p>
  </div>`;

  rpt.innerHTML=h;

  // === Charts ===
  MC('c1',{type:'bar',data:{labels:inTop.map(([h])=>h),datasets:[{data:inTop.map(([,v])=>v),backgroundColor:inTop.map(([,v])=>v>=10?'#ef4456':v>=7?'#eda535':'#22c5d6'),borderRadius:3,barPercentage:.75}]},options:{...CO}});

  MC('c2',{type:'bar',data:{labels:imbTop.map(([h])=>'HID '+h),datasets:[{data:imbTop.map(([,v])=>v.d),backgroundColor:imbTop.map(([,v])=>v.d>0?'#ef4456':'#4a90ff'),borderRadius:3,barPercentage:.7}]},options:{...CO,scales:{...CO.scales,y:{...CO.scales.y,beginAtZero:false}}}});

  MC('c3',{type:'bar',data:{labels:outTop.map(([h])=>h),datasets:[{data:outTop.map(([,v])=>v),backgroundColor:'#3cc68a',borderRadius:3,barPercentage:.75}]},options:{...CO}});

  if(bidi.length>0)MC('c4',{type:'bar',data:{labels:b12.map(p=>p.a+'↔'+p.b),datasets:[{label:'A→B',data:b12.map(p=>p.ab),backgroundColor:'#a06cf0',borderRadius:3,barPercentage:.75},{label:'B→A',data:b12.map(p=>p.ba),backgroundColor:'#e868a4',borderRadius:3,barPercentage:.75}]},options:{...CO}});

  MC('c5',{type:'bar',data:{labels:mdS.map(([k])=>k+'건'),datasets:[{data:mdS.map(([,v])=>v),backgroundColor:'#e868a4',borderRadius:3,barPercentage:.6}]},options:{...CO}});

  window.scrollTo({top:0,behavior:'smooth'});
}
