    <div class="cc"><div class="ct">이동 횟수별 OHT 대수</div>
      <div class="cb" style="height:220px;"><canvas id="c4"></canvas></div>
    </div>
    <div class="ex"><div class="el p">이 그래프는 뭘 보여주나?</div>
      <p>${mdSorted.map(([k,v])=>`<span class="hl">${k}회</span> 이동: ${v}대`).join(' / ')}</p>
      <p>대부분(${md[1]||0}대)이 1회 이동이고, 3회 이상은 <span class="hl">${multi.length}대</span>.</p>
    </div>
    <div class="ex"><div class="el p">알 수 있는 것 / 없는 것</div>
      <p class="yes">✅ 이 데이터 범위에서 각 OHT가 몇 번 이동 기록을 남겼는지</p>
      <p class="no">❌ 1회만 찍힌 OHT가 한 번만 움직인 건지, 나머지가 다른 시간대에 찍힌 건지</p>
    </div></div>`;

  // 5. 다회 이동 VHL 경로
  if(multi.length>0){
    h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--purple)">5</div><div class="st">다회 이동 OHT 경로 상세 (3회+)</div></div>
      <div class="tw"><table class="tb"><thead><tr><td>VHL ID</td><td>이동 수</td><td>경로</td><td>반복 방문 HID</td></tr></thead><tbody>`;
    multi.slice(0,15).forEach(([vid,rs])=>{
      const route=rs.map(r=>r.F+'→'+r.T).join(' , ');
      const fc={};rs.forEach(r=>{fc[r.F]=(fc[r.F]||0)+1;});
      const rp=Object.entries(fc).filter(([,c])=>c>1);
      const rpStr=rp.length?rp.map(([h,c])=>`HID ${h} (${c}회)`).join(', '):'—';
      h+=`<tr><td class="mn" style="font-weight:600">${vid}</td><td>${rs.length}</td><td style="font-family:var(--mono);font-size:10px;color:var(--tx2);max-width:380px;word-break:break-all">${route}</td><td class="mn">${rpStr}</td></tr>`;
    });
    h+=`</tbody></table></div>
      <div class="ex"><div class="el p">알 수 있는 것 / 없는 것</div>
        <p class="yes">✅ 이 OHT들이 실제로 어떤 경로를 이동했는지, 같은 HID를 다시 방문했는지</p>
        <p class="no">❌ 반복 방문이 정상 운영인지, 문제 상황인지 (원인 정보 없음)</p>
      </div></div>`;
  }

  // 6. TRANS_CNT
  h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--red)">6</div><div class="st">TRANS_CNT ≥ 2 기록</div></div>`;
  if(ret.length===0){
    h+=`<div class="ex"><div class="el g">결과</div><p>모든 레코드의 TRANS_CNT가 1. 2 이상인 기록 없음.</p></div>`;
  } else {
    h+=`<div class="tw"><table class="tb"><thead><tr><td>VHL ID</td><td>FROM HID</td><td>TO HID</td><td>TRANS_CNT</td></tr></thead><tbody>`;
    ret.forEach(r=>{h+=`<tr><td class="mn">${r.V}</td><td class="mn">${r.F}</td><td class="mn">${r.T}</td><td style="color:var(--amber);font-weight:600">${r.TC}</td></tr>`;});
    h+=`</tbody></table></div>
      <div class="ex"><div class="el r">알 수 있는 것 / 없는 것</div>
        <p class="yes">✅ 이 VHL들의 이 구간에서 TRANS_CNT가 2로 기록됨</p>
        <p class="no">❌ 이게 재시도인지, 왕복인지, 시스템 중복 기록인지 (ATLAS 정의에 따라 다름)</p>
      </div>`;
  }
  h+=`</div>`;

  // 7. Groups
  if(grps.length>0){
    h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--cyan)">7</div><div class="st">자주 함께 쓰이는 HID 그룹</div></div>
      <div class="ex"><div class="el c">연결 기준</div><p>이동 빈도 <span class="hl">${cTh}건 이상</span>인 구간끼리 연결했을 때 묶이는 HID 그룹.</p></div>`;
    grps.slice(0,5).forEach((g,i)=>{
      const tr=g.reduce((s,h)=>s+(ht[h]||0),0);
      h+=`<div class="ex"><div class="el c">그룹 ${i+1} — ${g.length}개 HID, 합산 트래픽 ${tr}건</div>
        <p style="font-family:var(--mono);font-size:11px;color:var(--tx2);word-break:break-all">${g.join(', ')}</p></div>`;
    });
    h+=`<div class="ex"><div class="el c">알 수 있는 것 / 없는 것</div>
      <p class="yes">✅ 이 HID들 사이에서 OHT 이동이 빈번하다</p>
      <p class="no">❌ 이 영역이 병목인지, 핵심 운영 구간인지 (이 데이터로 구분 불가)</p>
    </div></div>`;
  }

  // 알 수 없는 것
  h+=`<div class="sec"><div class="sh"><div class="nm" style="background:var(--red)">⚠</div><div class="st">이 데이터로 알 수 없는 것</div></div>
    <div class="wb"><div class="el">확인 불가 항목</div>
      <p>• <b>데드락 발생 여부</b> — 대기/차단 상태 정보 없음</p>
      <p>• <b>OHT 상태 (RUN/WAIT/BLOCKED/DOWN)</b> — 상태 필드 없음</p>
      <p>• <b>이동 소요 시간</b> — 출발 시각 없이 완료 시각만 있음</p>
      <p>• <b>멈춰있는 OHT</b> — 이동 완료 기록이라 안 움직인 OHT는 안 찍힘</p>
      <p>• <b>이동 원인</b> — 정상 명령 / 회피 / 재시도 구분 불가</p>
      ${single?'<p>• <b>시간 추세</b> — 전부 같은 시각이라 변화 분석 불가</p>':''}
    </div></div>`;

  // Summary
  h+=`<div class="sum"><h3>📋 종합 요약</h3>
    <p><b>${fabs.join(', ')||'(FAB 미상)'}</b>의 <b>${dts.length<=1?(dts[0]||'시간 미상')+' 시점':dts[0]+' ~ '+dts[dts.length-1]}</b>, <b>${vhls.length}대</b> OHT가 <b>${allH.length}개</b> HID 존에서 <b>${rows.length.toLocaleString()}건</b> 이동한 기록.</p>
    <p>트래픽 1위: HID <b>${sH[0][0]}</b> (${sH[0][1]}건, 평균의 ${(sH[0][1]/avg).toFixed(1)}배) / 가장 많이 쓰인 구간: <b>${tP[0][0]}</b> (${tP[0][1]}건)</p>
    <p>양방향 이동 쌍: <b>${bidi.length}개</b> / TRANS_CNT≥2: <b>${ret.length}건</b> / 3회+ 이동 OHT: <b>${multi.length}대</b></p>
    <p>데드락 판정·OHT 상태 분류·가동률 계산은 이 데이터에 해당 정보가 없어서 <b>할 수 없습니다.</b></p>
  </div>`;

  rpt.innerHTML=h;

  // === Charts ===
  MC('c1',{type:'bar',data:{labels:t25.map(([h])=>h),datasets:[{data:t25.map(([,v])=>v),backgroundColor:t25.map(([,v],i)=>i<5?'#ef4456':i<10?'#eda535':'#4a90ff'),borderRadius:3,barPercentage:.75}]},options:{...CO}});
  MC('c2',{type:'bar',data:{labels:tp15.map(([k])=>k),datasets:[{data:tp15.map(([,v])=>v),backgroundColor:'#3cc68a',borderRadius:3,barPercentage:.7}]},options:{...CO}});
  if(bidi.length>0)MC('c3',{type:'bar',data:{labels:b12.map(p=>p.a+'↔'+p.b),datasets:[{label:'A→B',data:b12.map(p=>p.ab),backgroundColor:'#4a90ff',borderRadius:3,barPercentage:.75},{label:'B→A',data:b12.map(p=>p.ba),backgroundColor:'#eda535',borderRadius:3,barPercentage:.75}]},options:{...CO,plugins:{legend:{display:false}}}});
  MC('c4',{type:'bar',data:{labels:mdSorted.map(([k])=>k+'회'),datasets:[{data:mdSorted.map(([,v])=>v),backgroundColor:'#a06cf0',borderRadius:3,barPercentage:.6}]},options:{...CO}});

  window.scrollTo({top:0,behavior:'smooth'});
}
