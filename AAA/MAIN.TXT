function handleFile(file){

function parseCSV(text){
  const lines=text.trim().split('\n');
  if(lines.length<2)throw new Error('최소 2행 필요');
  let sep=',';
  if(lines[0].split('\t').length>lines[0].split(',').length)sep='\t';
  const hdr=lines[0].split(sep).map(h=>h.replace(/^"|"$/g,'').trim());
  const alias={
    VHL_ID:['VHL_ID','VHLID','VHL','VEHICLE_ID'],
    FROM_HIDID:['FROM_HIDID','FROM_HID','FROMHID'],
    TO_HIDID:['TO_HIDID','TO_HID','TOHID'],
    TRANS_CNT:['TRANS_CNT','TRANSCNT','CNT'],
    EVENT_DT:['EVENT_DT','EVENT_DATE','_time','TIME','TIMESTAMP'],
    FAB_ID:['FAB_ID','FABID','FAB']
  };
  const cm={};
  for(const[k,ns]of Object.entries(alias)){
    for(const n of ns){const i=hdr.findIndex(h=>h.toUpperCase()===n.toUpperCase());if(i>=0){cm[k]=i;break;}}
  }
  if(!('VHL_ID'in cm)||!('FROM_HIDID'in cm)||!('TO_HIDID'in cm))
    throw new Error('필수 컬럼 누락: VHL_ID, FROM_HIDID, TO_HIDID\n헤더: '+hdr.join(', '));
  const rows=[];
  for(let i=1;i<lines.length;i++){
    const c=lines[i].split(sep).map(x=>x.replace(/^"|"$/g,'').trim());
    if(c.length<3)continue;
    rows.push({
      VHL_ID:c[cm.VHL_ID]||'',
      FROM:c[cm.FROM_HIDID]||'',
      TO:c[cm.TO_HIDID]||'',
      TCNT:cm.TRANS_CNT!==undefined?parseInt(c[cm.TRANS_CNT])||1:1,
      DT:cm.EVENT_DT!==undefined?c[cm.EVENT_DT]:'',
      FAB:cm.FAB_ID!==undefined?c[cm.FAB_ID]:''
    });
  }
  return rows;
}

// State classification based on HID transitions
const STATE_COLORS={
  RUN:'#4caf50', IDLE:'#66bb6a', WAIT:'#ffc107', PLANNED_STOP:'#ff9800',
  BLOCKED:'#f44336', DEADLOCK_RISK:'#9c27b0', DEADLOCK_RISK2:'#7b1fa2',
  DOWN:'#78909c', MANUAL_RECOVERY:'#e91e63'
};

function classifyState(from,to,tcnt,bidiSet,hotHids,clusterHids){
  const pair=from+'↔'+to;
  const rpair=to+'↔'+from;
  // Deadlock risk: bidirectional pair + high traffic zone
  if(bidiSet.has(pair)||bidiSet.has(rpair)){
    if(clusterHids.has(from)||clusterHids.has(to))return 'DEADLOCK_RISK2';
    return 'DEADLOCK_RISK';
  }
  // Blocked: retry
  if(tcnt>=2)return 'BLOCKED';
  // Hot zone wait
  if(hotHids.has(from)&&hotHids.has(to))return 'WAIT';
  if(hotHids.has(from)||hotHids.has(to))return 'PLANNED_STOP';
  // Normal movement
  const dist=Math.abs(parseInt(from)-parseInt(to));
  if(dist<=2)return 'RUN';
  if(dist<=5)return 'IDLE';
  return 'RUN';
}

function analyze(text,fname,fsize){
  const rows=parseCSV(text);
  
  // Detect FAB
  const fabs=new Set(rows.map(r=>r.FAB).filter(Boolean));
  if(fabs.size)document.getElementById('fabTag').textContent=[...fabs].join(', ');

  document.getElementById('fnEl').textContent=fname;
  document.getElementById('fmEl').textContent=rows.length.toLocaleString()+' rows · '+(fsize/1024).toFixed(1)+' KB';
  document.getElementById('uploadWrap').style.display='none';
  document.getElementById('dash').classList.add('active');
  document.getElementById('kpiStrip').style.display='flex';

  // ============ CORE ANALYSIS ============
  const vhlSet=new Set(rows.map(r=>r.VHL_ID));
  const allHids=new Set([...rows.map(r=>r.FROM),...rows.map(r=>r.TO)]);

  // HID traffic
  const hidTraffic={};
  rows.forEach(r=>{hidTraffic[r.FROM]=(hidTraffic[r.FROM]||0)+1;hidTraffic[r.TO]=(hidTraffic[r.TO]||0)+1;});
  const sortedHids=Object.entries(hidTraffic).sort((a,b)=>b[1]-a[1]);
  const hotThreshold=sortedHids.length>10?sortedHids[9][1]:0;
  const hotHids=new Set(sortedHids.filter(([,v])=>v>=hotThreshold).map(([k])=>k));

  // Bidirectional
  const pairCnt={};
  rows.forEach(r=>{const k=r.FROM+'→'+r.TO;pairCnt[k]=(pairCnt[k]||0)+1;});
  const bidiSet=new Set();
  const bidiPairs=[];
  const bidiSeen=new Set();
  for(const[k,c]of Object.entries(pairCnt)){
    const[a,b]=k.split('→');
    const rv=b+'→'+a;
    if(pairCnt[rv]&&!bidiSeen.has(b+'↔'+a)){
      bidiPairs.push({a,b,ab:c,ba:pairCnt[rv],total:c+pairCnt[rv]});
      bidiSeen.add(a+'↔'+b);
      bidiSet.add(a+'↔'+b);
    }
  }
  bidiPairs.sort((a,b)=>b.total-a.total);

  // Clusters
  const cThresh=Math.max(3,Math.floor(rows.length/250));
  const adj={};
  for(const[k,c]of Object.entries(pairCnt)){
    if(c<cThresh)continue;
    const[a,b]=k.split('→');
    if(!adj[a])adj[a]=new Set();if(!adj[b])adj[b]=new Set();
    adj[a].add(b);adj[b].add(a);
  }
  const visited=new Set();const clusters=[];
  for(const node of Object.keys(adj)){
    if(visited.has(node))continue;
    const cl=new Set();const stk=[node];
    while(stk.length){const n=stk.pop();if(visited.has(n))continue;visited.add(n);cl.add(n);if(adj[n])for(const nb of adj[n])if(!visited.has(nb))stk.push(nb);}
    if(cl.size>=3)clusters.push([...cl].sort((a,b)=>parseInt(a)-parseInt(b)));
  }
  clusters.sort((a,b)=>{
    const ta=a.reduce((s,h)=>s+(hidTraffic[h]||0),0);
    const tb=b.reduce((s,h)=>s+(hidTraffic[h]||0),0);
    return tb-ta;
  });
  const clusterHids=new Set(clusters.length?clusters[0]:[]);

  // VHL grouped
  const vhlRows={};
  rows.forEach(r=>{if(!vhlRows[r.VHL_ID])vhlRows[r.VHL_ID]=[];vhlRows[r.VHL_ID].push(r);});

  // Classify each row's state
  rows.forEach(r=>{
    r.STATE=classifyState(r.FROM,r.TO,r.TCNT,bidiSet,hotHids,clusterHids);
  });

  // Per-VHL state summary
  const vhlStates={};
  for(const[vid,rs]of Object.entries(vhlRows)){
    const counts={RUN:0,IDLE:0,WAIT:0,PLANNED_STOP:0,BLOCKED:0,DEADLOCK_RISK:0,DEADLOCK_RISK2:0,DOWN:0,MANUAL_RECOVERY:0};
    rs.forEach(r=>counts[r.STATE]=(counts[r.STATE]||0)+1);
    const total=rs.length;
    const runRate=((counts.RUN+counts.IDLE)/total*100);
    vhlStates[vid]={counts,total,runRate,rows:rs};
  }

  // Stats
  const totalVhl=vhlSet.size;
  const dlSuspect=rows.filter(r=>r.STATE==='DEADLOCK_RISK'||r.STATE==='DEADLOCK_RISK2').length;
  const avgEff=Object.values(vhlStates).reduce((s,v)=>s+v.runRate,0)/totalVhl;
  const waitTimes=rows.filter(r=>r.STATE==='WAIT'||r.STATE==='PLANNED_STOP');
  const blockTimes=rows.filter(r=>r.STATE==='BLOCKED'||r.STATE==='DEADLOCK_RISK'||r.STATE==='DEADLOCK_RISK2');
  const retries=rows.filter(r=>r.TCNT>=2);

  // KPI
  const kpiStrip=document.getElementById('kpiStrip');
  kpiStrip.innerHTML=`
    <div class="kpi-item">
      <div class="kpi-icon" style="background:rgba(76,175,80,0.15);">📊</div>
      <div><div class="kpi-val">${Math.round(avgEff)}%</div><div class="kpi-label">Average OHT Efficiency</div></div>
    </div>
    <div class="kpi-item">
      <div class="kpi-icon" style="background:rgba(156,39,176,0.15);">🔒</div>
      <div><div class="kpi-val" style="color:var(--purple)">${dlSuspect}</div><div class="kpi-label">Deadlock Suspected</div></div>
    </div>
    <div class="kpi-item">
      <div class="kpi-icon" style="background:rgba(255,193,7,0.15);">⏱</div>
      <div><div class="kpi-val">${waitTimes.length}</div><div class="kpi-label">Wait Events</div></div>
    </div>
    <div class="kpi-item">
      <div class="kpi-icon" style="background:rgba(244,67,54,0.15);">🚫</div>
      <div><div class="kpi-val" style="color:var(--red)">${blockTimes.length}</div><div class="kpi-label">Blocked Events</div></div>
    </div>
    <div class="kpi-item">
      <div class="kpi-icon" style="background:rgba(0,188,212,0.15);">🏭</div>
      <div><div class="kpi-val">${totalVhl}</div><div class="kpi-label">Active OHTs</div></div>
    </div>
  `;

  // ============ TIMELINE ============
  const tlScroll=document.getElementById('tlScroll');
  // Sort VHLs by most interesting (low efficiency first)
  const sortedVhls=Object.entries(vhlStates)
    .sort((a,b)=>a[1].runRate-b[1].runRate)
    .slice(0,30); // Show top 30

  let tlHTML='';
  sortedVhls.forEach(([vid,st])=>{
    let segs='';
    st.rows.forEach(r=>{
      const w=Math.max(100/st.total,2);
      segs+=`<div class="tl-seg" style="width:${w}%;background:${STATE_COLORS[r.STATE]}" title="${vid}: ${r.FROM}→${r.TO} (${r.STATE})"></div>`;
    });
    tlHTML+=`<div class="tl-row"><div class="tl-label">${vid}</div><div class="tl-bar-area">${segs}</div></div>`;
  });
  tlScroll.innerHTML=tlHTML;

  // ============ EFFICIENCY CHART ============
  const effVhls=sortedVhls.slice(0,20);
  const effLabels=effVhls.map(([v])=>v);
  const effData=effVhls.map(([,s])=>Math.round(s.runRate));

  if(effChartInst)effChartInst.destroy();
  document.getElementById('avgLabel').textContent='Average '+Math.round(avgEff)+'%';
  effChartInst=new Chart(document.getElementById('effChart'),{
    type:'bar',
    data:{
      labels:effLabels,
      datasets:[{
        data:effData,
        backgroundColor:effData.map(v=>v>=70?'#4caf50':v>=50?'#ffc107':v>=30?'#ff9800':'#f44336'),
        borderRadius:3,barPercentage:0.7
      }]
    },
    options:{
      responsive:true,maintainAspectRatio:false,
      plugins:{
        legend:{display:false},
        tooltip:{callbacks:{label:c=>c.raw+'%'}}
      },
      scales:{
        x:{ticks:{color:'#6b6c82',font:{size:9,family:'JetBrains Mono'},maxRotation:45},grid:{display:false}},
        y:{min:0,max:100,ticks:{color:'#6b6c82',callback:v=>v+'%'},grid:{color:'rgba(255,255,255,0.04)'}}
      }
    }
  });

  // ============ LOSS CHART ============
  const lossVhls=sortedVhls.slice(0,15);
  const lossLabels=lossVhls.map(([v])=>v);
  const stateKeys=['PLANNED_STOP','BLOCKED','WAIT','DEADLOCK_RISK','DEADLOCK_RISK2','MANUAL_RECOVERY','DOWN'];
  const stateLabels=['Planned Stop','Blocked','Wait for Route','Deadlock Risk','Deadlock Risk+','Manual Recovery','Down'];
  const stateClrs=['#78909c','#f44336','#ffc107','#9c27b0','#7b1fa2','#e91e63','#546e7a'];
  
  const datasets=stateKeys.map((sk,i)=>({
    label:stateLabels[i],
    data:lossVhls.map(([,s])=>s.counts[sk]||0),
    backgroundColor:stateClrs[i],
    borderRadius:2
  }));

  if(lossChartInst)lossChartInst.destroy();
  lossChartInst=new Chart(document.getElementById('lossChart'),{
    type:'bar',
    data:{labels:lossLabels,datasets},
    options:{
      responsive:true,maintainAspectRatio:false,
      plugins:{legend:{display:false}},
      scales:{
        x:{stacked:true,ticks:{color:'#6b6c82',font:{size:9,family:'JetBrains Mono'},maxRotation:45},grid:{display:false}},
        y:{stacked:true,ticks:{color:'#6b6c82'},grid:{color:'rgba(255,255,255,0.04)'}}
      }
    }
  });

  // ============ CLUSTERS ============
  const clCards=document.getElementById('clCards');
  clCards.innerHTML='';
  clusters.slice(0,5).forEach((cl,i)=>{
    const traffic=cl.reduce((s,h)=>s+(hidTraffic[h]||0),0);
    const level=i===0?'crit':i===1?'high':'med';
    const levelTxt=i===0?'CRITICAL':i===1?'HIGH':'MEDIUM';
    clCards.innerHTML+=`
      <div class="cl-card">
        <div class="cl-head">
          <span class="cl-badge ${level}">${levelTxt}</span>
          <span class="cl-title">클러스터 ${i+1} — ${cl.length}개 존, 트래픽 ${traffic}건</span>
        </div>
        <div class="cl-hids">HID: ${cl.join(', ')}</div>
      </div>`;
  });

  // ============ COMPLEX VHL TABLE ============
  const cplxBody=document.querySelector('#cplxTbl tbody');
  cplxBody.innerHTML='';
  const complexVhls=Object.entries(vhlRows)
    .filter(([,rs])=>rs.length>=3)
    .sort((a,b)=>b[1].length-a[1].length)
    .slice(0,15);
  complexVhls.forEach(([vid,rs])=>{
    const route=rs.map(r=>r.FROM+'→'+r.TO).join(' → ');
    // Detect patterns
    const fromList=rs.map(r=>r.FROM);
    const cnt={};fromList.forEach(h=>{cnt[h]=(cnt[h]||0)+1;});
    const repeats=Object.entries(cnt).filter(([,c])=>c>1);
    let tag='<span class="tag b">복합이동</span>';
    if(repeats.length)tag=`<span class="tag r">HID ${repeats[0][0]} 순환</span>`;
    // Check bidirectional
    for(let i=0;i<rs.length-1;i++){
      if(rs[i].FROM===rs[i+1].TO&&rs[i].TO===rs[i+1].FROM){
        tag=`<span class="tag w">${rs[i].FROM}↔${rs[i].TO} 왕복</span>`;break;
      }
    }
    cplxBody.innerHTML+=`<tr>
      <td class="mono" style="font-weight:600">${vid}</td>
      <td>${rs.length}</td>
      <td class="route">${route}</td>
      <td>${tag}</td>
    </tr>`;
  });

  // ============ BIDI TABLE ============
  const bidiBody=document.querySelector('#bidiTbl tbody');
  bidiBody.innerHTML='';
  bidiPairs.slice(0,15).forEach(p=>{
    bidiBody.innerHTML+=`<tr>
      <td class="mono">${p.a}</td><td class="mono">${p.b}</td>
      <td>${p.ab}</td><td>${p.ba}</td>
      <td style="font-weight:600;color:${p.total>=10?'var(--red)':p.total>=7?'var(--orange)':'var(--text)'}">${p.total}</td>
    </tr>`;
  });

  // ============ RETRY TABLE ============
  const retryBody=document.querySelector('#retryTbl tbody');
  retryBody.innerHTML='';
  retries.forEach(r=>{
    retryBody.innerHTML+=`<tr>
      <td class="mono">${r.VHL_ID}</td><td class="mono">${r.FROM}</td>
      <td class="mono">${r.TO}</td>
      <td style="color:var(--orange);font-weight:600">${r.TCNT}</td>
    </tr>`;
  });
  if(!retries.length)retryBody.innerHTML='<tr><td colspan="4" style="text-align:center;color:var(--text3);padding:20px">재시도 없음</td></tr>';
}
