    // Load sprite frames
    function loadFrames(dir, count) {
      const frames = [];
      for(let i = 1; i <= count; i++) {
        const img = new Image();
        img.src = `img/${dir}/frame_0${i}_transparent.png`;
        frames.push(img);
      }
      return frames;

    const ceoImg = new Image();
    ceoImg.src = "img/CEO_PO_transparent.png";

    class Character {
      constructor(frames, x, y, speed, scale, isStatic = false, bounds = null, pauseBehavior = 'normal', name = '') {
        this.frames = frames;
        this.x = x;
        this.y = y;
        this.speed = speed;
        this.scale = scale;
        this.frameIndex = 0;
        this.frameTimer = 0;
        this.frameDuration = 100 + Math.random() * 50;
        this.isStatic = isStatic;
        this.bounds = bounds || {minX: 150, maxX: world.width - 150, minY: 150, maxY: world.height - 150};
        this.direction = Math.random() < 0.5 ? 1 : -1;
        this.pauseBehavior = pauseBehavior;
        this.name = name;

        // Roaming state
        this.isMoving = false;
        this.targetX = this.x;
        this.targetY = this.y;
        this.idleTimer = Math.random() * 1000; 
      }

      setNewTarget() {
        this.targetX = this.bounds.minX + Math.random() * (this.bounds.maxX - this.bounds.minX);
        this.targetY = this.bounds.minY + Math.random() * (this.bounds.maxY - this.bounds.minY);
        this.isMoving = true;
        this.direction = this.targetX > this.x ? 1 : -1;
      }

      update(dt) {
        if(this.isStatic) return;

        if (!this.isMoving) {
          this.idleTimer -= dt * 1000;
          if (this.idleTimer <= 0) {
            this.setNewTarget();
          }
          this.frameIndex = 0; // Stand still pose
          return;
        }

        // Move towards target in 2D space
        const dx = this.targetX - this.x;
        const dy = this.targetY - this.y;
        const dist = Math.sqrt(dx*dx + dy*dy);
        
        if (dist < 5) { // Reached destination
          this.isMoving = false;

          // Fire onArrive callback if set
          if (this._onArrive) {
            const cb = this._onArrive;
            this._onArrive = null;
            cb(this);
          }

          if (this.pauseBehavior === 'strike') {
             this.idleTimer = 300 + Math.random() * 600; // jitter in place
          } else if (this.pauseBehavior === 'alwaysMove') {
             this.idleTimer = Math.random() * 500;
          } else if (this.pauseBehavior === 'oftenPause') {
             this.idleTimer = 2000 + Math.random() * 4000; // 2~6 sec pause
          } else {
             this.idleTimer = 500 + Math.random() * 2000; // normal
          }
        } else {
          // Adjust distance by standard 60fps movement delta
          const moveDist = this.speed * (dt * 1000) / 16;
          this.x += (dx / dist) * moveDist;
          this.y += (dy / dist) * moveDist;

          // Animate walking
          this.frameTimer += dt * 1000;
          if(this.frameTimer >= this.frameDuration) {
            this.frameTimer = 0;
            this.frameIndex = (this.frameIndex + 1) % this.frames.length;
          }
        }
      }

      draw(ctx) {
        const img = this.isStatic ? this.frames : this.frames[this.frameIndex];
        if(!img || !img.complete || img.naturalWidth === 0) return; // Safely return if image is not loaded
        
        ctx.save();
        ctx.translate(this.x, this.y);
        if(this.direction === -1 && !this.isStatic) {
          ctx.scale(-1, 1); // Flip horizontally if walking left
        }
        
        const w = img.width * this.scale;
        const h = img.height * this.scale;
        
        ctx.drawImage(img, -w/2, -h, w, h);
        ctx.restore();

        // Name label
        if (this.name) {
          ctx.save();
          ctx.font = 'bold 18px "Segoe UI", "Malgun Gothic", sans-serif';
          ctx.textAlign = 'center';
          const tw = ctx.measureText(this.name).width + 14;
          const lx = this.x, ly = this.y + 10;
          ctx.fillStyle = 'rgba(0,0,0,0.72)';
          ctx.beginPath();
          ctx.roundRect(lx - tw/2, ly - 14, tw, 22, 5);
          ctx.fill();
          ctx.fillStyle = '#fff';
          ctx.fillText(this.name, lx, ly + 3);
          ctx.restore();
        }
      }
    }

    function playStrikeHorn() {
      // Protest horn / alarm sound
      const now = audioCtx.currentTime;

      // Horn 1
      const osc1 = audioCtx.createOscillator();
      const gain1 = audioCtx.createGain();
      osc1.type = 'sawtooth';

      osc1.frequency.linearRampToValueAtTime(330, now + 0.3);
      osc1.frequency.linearRampToValueAtTime(220, now + 0.6);
      gain1.gain.setValueAtTime(0.3, now);
      gain1.gain.linearRampToValueAtTime(0, now + 0.8);
      osc1.connect(gain1).connect(audioCtx.destination);
      osc1.start(now); osc1.stop(now + 0.8);

      // Horn 2 (delayed)
      const osc2 = audioCtx.createOscillator();
      const gain2 = audioCtx.createGain();
      osc2.type = 'sawtooth';
      osc2.frequency.setValueAtTime(280, now + 0.9);
      osc2.frequency.linearRampToValueAtTime(400, now + 1.2);
      osc2.frequency.linearRampToValueAtTime(280, now + 1.5);
      gain2.gain.setValueAtTime(0, now);
      gain2.gain.setValueAtTime(0.35, now + 0.9);
      gain2.gain.linearRampToValueAtTime(0, now + 1.7);
      osc2.connect(gain2).connect(audioCtx.destination);
      osc2.start(now + 0.9); osc2.stop(now + 1.7);

      // Big horn 3
      const osc3 = audioCtx.createOscillator();
      const gain3 = audioCtx.createGain();
      osc3.type = 'sawtooth';
      osc3.frequency.setValueAtTime(180, now + 1.8);
      osc3.frequency.linearRampToValueAtTime(440, now + 2.5);
      gain3.gain.setValueAtTime(0, now);
      gain3.gain.setValueAtTime(0.4, now + 1.8);
      gain3.gain.linearRampToValueAtTime(0, now + 3.0);
      osc3.connect(gain3).connect(audioCtx.destination);
      osc3.start(now + 1.8); osc3.stop(now + 3.0);
    }

    function speakSlogan(text) {
      if (!window.speechSynthesis) return;
      const utter = new SpeechSynthesisUtterance(text);
      utter.lang = 'ko-KR';
      utter.rate = 1.1;
      utter.pitch = 1.2;
      utter.volume = 1.0;
      // Try to find Korean voice
      const voices = speechSynthesis.getVoices();
      const koVoice = voices.find(v => v.lang.startsWith('ko'));
      if (koVoice) utter.voice = koVoice;
      speechSynthesis.speak(utter);
    }

    function startStrikeChant() {
      // Chant a slogan every 4 seconds
      strikeChantInterval = setInterval(() => {
        if (!isOnStrike) return;
        const slogan = strikeSlogans[Math.floor(Math.random() * strikeSlogans.length)];
        speakSlogan(slogan);
        // Also update random character's sign
        const movingChars = characters.filter(c => !c.isStatic && c._slogan);
        if (movingChars.length > 0) {
          const rc = movingChars[Math.floor(Math.random() * movingChars.length)];
          rc._slogan = slogan;
        }
      }, 4000);
    }

    function stopStrikeChant() {
      clearInterval(strikeChantInterval);
      strikeChantInterval = null;
      if (window.speechSynthesis) speechSynthesis.cancel();
    }

    // Strike line-up positions (below chalkboard, evenly spaced)
    function getStrikePositions() {
      const startX = 500, endX = 1350;
      const y = 420;
      const movingChars = characters.filter(c => !c.isStatic);
      const gap = (endX - startX) / (movingChars.length - 1 || 1);
      return movingChars.map((c, i) => ({
        x: startX + gap * i,
        y: y + (i % 2 === 0 ? 0 : 15) // slight zigzag
      }));
    }

    function toggleStrike() {
      isOnStrike = !isOnStrike;
      const btn = document.getElementById('strikeBtn');
      const movingChars = characters.filter(c => !c.isStatic);

      if (isOnStrike) {
        btn.textContent = '✅ 복귀';
        btn.classList.add('active');
        // Disable agent cards + red flash
        document.querySelectorAll('.agent-card').forEach(c => c.classList.add('on-strike'));
        if (currentAgent) closeAgent();
        audioCtx.resume(); // unlock audio
        playStrikeHorn();
        strikeBGM.currentTime = 0;
        strikeBGM.play().catch(() => {});
        setTimeout(() => speakSlogan('왜 이러는겨! 일해라! 파업은 없다!'), 2500);
        setTimeout(() => startStrikeChant(), 5000);

        // Start firing employees every 5 seconds
        fireInterval = setInterval(() => {
          if (!isOnStrike) return;
          const movingAlive = characters.filter(c => !c.isStatic && !c._fired);
          if (movingAlive.length === 0) {
            clearInterval(fireInterval);
            fireInterval = null;
            return;
          }
          const victim = movingAlive[Math.floor(Math.random() * movingAlive.length)];
          fireEmployee(victim);
        }, 5000);

        // Move all characters to strike line
        const positions = getStrikePositions();
        movingChars.forEach((c, i) => {
          c._savedBounds = { ...c.bounds };
          c._savedPause = c.pauseBehavior;
          c.targetX = positions[i].x;
          c.targetY = positions[i].y;
          c.isMoving = true;
          c.direction = c.targetX > c.x ? 1 : -1;
          c.pauseBehavior = 'strike';
          // Lock bounds to strike area
          c.bounds = { minX: positions[i].x - 30, maxX: positions[i].x + 30, minY: positions[i].y - 15, maxY: positions[i].y + 15 };
          // Assign slogan
          c._slogan = strikeSlogans[Math.floor(Math.random() * strikeSlogans.length)];
        });
      } else {
        btn.textContent = '🪧 파업';
        btn.classList.remove('active');
        // Re-enable agent cards
        document.querySelectorAll('.agent-card').forEach(c => c.classList.remove('on-strike'));
        stopStrikeChant();
        strikeBGM.pause();
        strikeBGM.currentTime = 0;

        // Stop firing timer
        if (fireInterval) { clearInterval(fireInterval); fireInterval = null; }

        // Restore fired employees
        firedCharacters.forEach(fc => {
          fc.char._fired = false;
          fc.char._fireWalkOut = false;
          fc.char.x = fc.origX;
          fc.char.y = fc.origY;
          if (fc.char._savedBounds) fc.char.bounds = fc.char._savedBounds;
          if (fc.char._savedPause) fc.char.pauseBehavior = fc.char._savedPause;
          fc.char._slogan = null;
          fc.char.isMoving = false;
          fc.char.idleTimer = 200 + Math.random() * 500;
        });
        firedCharacters = [];
        fireFlashText = null;

        // Return to normal (remaining chars)
        characters.filter(c => !c.isStatic).forEach(c => {
          if (c._savedBounds) c.bounds = c._savedBounds;
          if (c._savedPause) c.pauseBehavior = c._savedPause;
          c._slogan = null;
          c.isMoving = false;
          c.idleTimer = 200 + Math.random() * 500;
        });
      }
    }

    function fireEmployee(char) {
      char._fired = true;
      char._fireWalkOut = true;
      char._slogan = null;
      // Save original position for restore
      firedCharacters.push({ char, origX: char._savedBounds ? (char._savedBounds.minX + char._savedBounds.maxX) / 2 : char.x, origY: char._savedBounds ? (char._savedBounds.minY + char._savedBounds.maxY) / 2 : char.y });

      char.direction = 1;
      char.pauseBehavior = 'alwaysMove';
      char.bounds = { minX: -200, maxX: world.width + 400, minY: 0, maxY: world.height };
    }

    function drawFireFlash(ctx) {
      if (!fireFlashText) return;
      const elapsed = Date.now() - fireFlashText.startTime;
      if (elapsed > 3000) { fireFlashText = null; return; }

      const progress = elapsed / 3000;
      const alpha = progress < 0.1 ? progress / 0.1 : (1 - progress);
      const scale = 1 + progress * 0.5;

      ctx.save();
      ctx.globalAlpha = Math.max(0, alpha);
      ctx.translate(fireFlashText.x, fireFlashText.y - progress * 80);

      // Big red "해고!!" text
      ctx.font = `bold ${Math.round(72 * scale)}px "Segoe UI", "Malgun Gothic", sans-serif`;
      ctx.textAlign = 'center';

      // Red glow shadow
      ctx.shadowColor = '#ff0000';
      ctx.shadowBlur = 30;

      // Outline
      ctx.strokeStyle = '#8b0000';
      ctx.lineWidth = 6;
      ctx.strokeText('해 고 !!', 0, 0);

      // Fill
      ctx.fillStyle = '#ff1111';
      ctx.fillText('해 고 !!', 0, 0);

      // Name below
      ctx.font = `bold ${Math.round(32 * scale)}px "Segoe UI", "Malgun Gothic", sans-serif`;
      ctx.fillStyle = '#ffffff';
      ctx.strokeStyle = '#333';
      ctx.lineWidth = 4;
      ctx.strokeText(fireFlashText.name, 0, 50);
      ctx.fillText(fireFlashText.name, 0, 50);

      ctx.restore();
    }

    function startLineUpReport() {
      if (lineUpActive || isOnStrike) return;
      lineUpActive = true;

      const ceoChar = characters.find(c => c.isStatic);
      const movingChars = characters.filter(c => !c.isStatic && !c._fired);
      if (movingChars.length === 0 || !ceoChar) { lineUpActive = false; return; }

      // Save original states
      const origStates = movingChars.map(c => ({
        char: c,
        bounds: { ...c.bounds },
        speed: c.speed,
        pause: c.pauseBehavior,
      }));

      // Line-up positions: queue in front of CEO, going down
      const lineX = ceoChar.x + 120;
      const lineStartY = ceoChar.y + 30;
      const lineGap = 80;

      // Phase 1: All employees run to line-up positions
      let arrivedCount = 0;
      movingChars.forEach((c, i) => {
        c._reportPhase = 'running';
        c.speed = 4.5;
        c.bounds = { minX: -100, maxX: world.width + 100, minY: -100, maxY: world.height + 100 };
        c.pauseBehavior = 'alwaysMove';
        c.targetX = lineX;
        c.targetY = lineStartY + i * lineGap;
        c.isMoving = true;
        c.direction = c.targetX > c.x ? 1 : -1;

        c._onArrive = () => {
          c._reportPhase = 'lineUp';
          c.direction = -1; // face CEO
          c.isMoving = false;
          c.idleTimer = 99999;
          arrivedCount++;
          if (arrivedCount === movingChars.length) {
            // Everyone lined up → start individual reports
            processLineUp(movingChars, ceoChar, origStates, 0);
          }
        };
      });
    }

    function processLineUp(lineup, ceoChar, origStates, idx) {
      if (idx >= lineup.length) {
        // All done → CEO final comment, then everyone goes home
        const finalMsgs = ['수고했다! 해산!', '좋아, 다들 열심히!', '오늘도 파이팅!'];
        activeBubble = { char: ceoChar, text: finalMsgs[Math.floor(Math.random() * finalMsgs.length)], timer: 2500 };

        setTimeout(() => {
          // Send everyone back
          origStates.forEach(os => {
            const c = os.char;
            c._reportPhase = null;
            c.speed = os.speed;
            c.bounds = os.bounds;
            c.pauseBehavior = os.pause;
            c.targetX = (os.bounds.minX + os.bounds.maxX) / 2;
            c.targetY = (os.bounds.minY + os.bounds.maxY) / 2;
            c.isMoving = true;
            c.direction = c.targetX > c.x ? 1 : -1;
            c._onArrive = () => {
              c.isMoving = false;
              c.idleTimer = 1000 + Math.random() * 2000;
            };
          });
          pendingReports = {};
          lineUpActive = false;
        }, 2500);
        return;
      }

      const char = lineup[idx];
      // Step forward to CEO
      char._reportPhase = 'running';
      char.speed = 3.5;
      char.targetX = ceoChar.x + 80;
      char.targetY = ceoChar.y + 60;
      char.isMoving = true;
      char.direction = -1;

      char._onArrive = () => {
        char._reportPhase = 'reporting';
        char.direction = -1;
        char.isMoving = false;
        char.idleTimer = 99999;

        // Find agent key for this character
        const agentKey = Object.entries(agentCharMap).find(([k, v]) => v === char.name)?.[0];
        const hasReport = agentKey && pendingReports[agentKey];

        if (hasReport) {
          // Has something to report
          const reportText = pendingReports[agentKey];
          activeBubble = { char, text: '보고합니다!', timer: 1800 };

          setTimeout(() => {
            const shortText = reportText.length > 25 ? reportText.substring(0, 25) + '...' : reportText;
            activeBubble = { char, text: `📋 ${shortText}`, timer: 2500 };
          }, 1800);

          setTimeout(() => {
            const ceoResponses = ['좋아, 수고했어!', '확인했다!', '잘했어!', '알겠어, 고생!'];
            activeBubble = { char: ceoChar, text: ceoResponses[Math.floor(Math.random() * ceoResponses.length)], timer: 1500 };
          }, 3500);

          // After 5s → step back to line, next person
          setTimeout(() => {
            stepBackToLine(char, lineup, ceoChar, origStates, idx);
          }, 5000);
        } else {
          // Nothing to report
          activeBubble = { char, text: '보고할 내용 없습니다!', timer: 2000 };

          setTimeout(() => {
            activeBubble = { char: ceoChar, text: '알겠어, 수고!', timer: 1200 };
          }, 1500);

          // After 2.5s → step back, next
          setTimeout(() => {
            stepBackToLine(char, lineup, ceoChar, origStates, idx);
          }, 2500);
        }
      };
    }

    function stepBackToLine(char, lineup, ceoChar, origStates, idx) {
      // Step back to line position
      const lineX = ceoChar.x + 120;
      const lineStartY = ceoChar.y + 30;
      const lineGap = 80;

      char._reportPhase = 'lineUp';
      char.speed = 3.0;
      char.targetX = lineX;
      char.targetY = lineStartY + idx * lineGap;
      char.isMoving = true;
      char.direction = 1;

      char._onArrive = () => {
        char.direction = -1;
        char.isMoving = false;
        char.idleTimer = 99999;
        // Next person
        processLineUp(lineup, ceoChar, origStates, idx + 1);
      };
    }

    function drawSpeechBubble(ctx, char, text) {
      const bx = char.x;
      const by = char.y - (char.isStatic ? 180 : 130);
      const maxW = 280;
      const pad = 12;
      const lineH = 22;

      ctx.save();
      ctx.font = 'bold 16px "Segoe UI", "Malgun Gothic", sans-serif';

      // Word wrap
      const words = text.split('');
      const lines = [];
      let line = '';
      for (const ch of words) {
        const test = line + ch;
        if (ctx.measureText(test).width > maxW - pad * 2) {
          lines.push(line);
          line = ch;
        } else {
          line = test;
        }
      }
      if (line) lines.push(line);

      const boxW = Math.min(maxW, Math.max(...lines.map(l => ctx.measureText(l).width)) + pad * 2 + 10);
      const boxH = lines.length * lineH + pad * 2;
      const boxX = bx - boxW / 2;
      const boxY = by - boxH;

      // Bubble background
      ctx.fillStyle = '#fff';
      ctx.strokeStyle = '#2f3440';
      ctx.lineWidth = 3;
      ctx.beginPath();
      ctx.roundRect(boxX, boxY, boxW, boxH, 8);
      ctx.fill();
      ctx.stroke();

      // Tail triangle
      ctx.fillStyle = '#fff';
      ctx.strokeStyle = '#2f3440';
      ctx.beginPath();
      ctx.moveTo(bx - 10, boxY + boxH);
      ctx.lineTo(bx, boxY + boxH + 14);
      ctx.lineTo(bx + 10, boxY + boxH);
      ctx.closePath();
      ctx.fill();
      ctx.stroke();
      // Cover the tail top border
      ctx.fillStyle = '#fff';
      ctx.fillRect(bx - 9, boxY + boxH - 2, 18, 4);

      // Text
      ctx.fillStyle = '#2e3440';
      ctx.textAlign = 'left';
      lines.forEach((l, i) => {
        ctx.fillText(l, boxX + pad, boxY + pad + lineH * (i + 0.75));
      });

      ctx.restore();
    }

    // Sort by Y
    function zSort() {
      characters.sort((a,b) => a.y - b.y);
    }

    let lastTime = performance.now();
    function loop(now) {
      const dt = Math.min((now - lastTime) / 1000, 0.033);
      lastTime = now;

      // Draw office background
      if (map.complete && map.naturalWidth > 0) {
        ctx.drawImage(map, 0, 0, world.width, world.height);
      } else {
        ctx.fillStyle = "#101725";
        ctx.fillRect(0, 0, world.width, world.height);
      }

      // Update & Draw chars
      characters.forEach(c => {
        // Skip drawing fired chars that walked off screen
        if (c._fired && c.x > world.width + 100) { c._fireWalkOut = false; return; }
        c.update(dt);
      });
      zSort();
      characters.forEach(c => {
        if (c._fired && c.x > world.width + 100) return;
        c.draw(ctx);
      });

      // Draw fire flash text
      drawFireFlash(ctx);

      // Draw report running indicator (❗ above running character)
      characters.forEach(c => {
        if (c._reportPhase === 'running') {
          ctx.save();
          const bounce = Math.sin(Date.now() / 100) * 4;
          ctx.font = 'bold 28px sans-serif';
          ctx.textAlign = 'center';
          ctx.fillStyle = '#ff4444';
          ctx.shadowColor = '#ff0000';
          ctx.shadowBlur = 8;
          ctx.fillText('❗', c.x, c.y - 100 + bounce);
          // Speed lines behind character
          ctx.globalAlpha = 0.4;
          ctx.strokeStyle = '#aaa';
          ctx.lineWidth = 2;
          for (let i = 0; i < 3; i++) {
            const lx = c.x - c.direction * (30 + i * 15);
            const ly = c.y - 50 + (i - 1) * 20;
            ctx.beginPath();
            ctx.moveTo(lx, ly);
            ctx.lineTo(lx - c.direction * 20, ly);
            ctx.stroke();
          }
          ctx.restore();
        }
      });

      // Strike protest signs + CEO cry
      if (isOnStrike) {
        characters.filter(c => !c.isStatic && c._slogan && !c._fired).forEach(c => drawProtestSign(ctx, c));

        // CEO "살려줘" bubble
        const ceoChar = characters.find(c => c.isStatic);
        if (ceoChar) {
          const cx = ceoChar.x, cy = ceoChar.y - 200;
          const ceoMsgs = ['왜 이러는겨!!', '사장이 다 해줬잖아!', '일해라!!', '파업은 없다!', '머니가 권리다!!', '다시 일하라고!!', '내가 밥을 굶겼나!', '복귀 안 하면 각오해!'];

          ctx.save();
          ctx.font = 'bold 22px "Segoe UI", "Malgun Gothic", sans-serif';
          const tw = ctx.measureText(msg).width + 24;
          // Shaking effect
          const shake = Math.sin(Date.now() / 80) * 3;
          const bx = cx + shake, by = cy;
          // Bubble
          ctx.fillStyle = '#e8f0fe';
          ctx.strokeStyle = '#1a56db';
          ctx.lineWidth = 3;
          ctx.beginPath();
          ctx.roundRect(bx - tw/2, by - 32, tw, 40, 8);
          ctx.fill();
          ctx.stroke();
          // Tail
          ctx.fillStyle = '#e8f0fe';
          ctx.beginPath();
          ctx.moveTo(bx - 8, by + 8);
          ctx.lineTo(bx, by + 22);
          ctx.lineTo(bx + 8, by + 8);
          ctx.closePath();
          ctx.fill();
          ctx.strokeStyle = '#1a56db';
          ctx.stroke();
          ctx.fillStyle = '#e8f0fe';
          ctx.fillRect(bx - 7, by + 6, 14, 4);
          // Text
          ctx.fillStyle = '#1a3a8a';
          ctx.textAlign = 'center';
          ctx.fillText(msg, bx, by);
          ctx.restore();
        }
      }

      // Speech bubble (not during strike)
      if (activeBubble && !isOnStrike) {
        activeBubble.timer -= dt * 1000;
        if (activeBubble.timer > 0) {
          drawSpeechBubble(ctx, activeBubble.char, activeBubble.text);
        } else {
          activeBubble = null;
        }

    function openAgent(key) {
      if (isOnStrike) return;
      // 같은 에이전트 클릭 시 접힌 상태면 펼치기
      if (currentAgent === key && chatCollapsed) {
        toggleChat();
        return;
      }
      const agent = (key === 'ceo') ? agentBaseData[key] : getAgentData(key);
      if (!agent) return;
      currentAgent = key;
      chatCollapsed = false;

      // Highlight sidebar card
      document.querySelectorAll('.agent-card').forEach(c => c.classList.remove('active'));
      const card = document.querySelector(`.agent-card.${key}`);
      if (card) card.classList.add('active');

      // Header
      document.getElementById('chatHeader').innerHTML = `
        <div class="ch-avatar" style="background:linear-gradient(135deg,${agent.color1},${agent.color2})">${agent.letter}</div>
        <div class="ch-info">
          <h2>${agent.name}</h2>
          <p>${agent.title}</p>
        </div>
        <button class="back-btn" onclick="toggleChat()">← 접기</button>
      `;

      // Chat area with welcome
      const qbtns = agent.quickActions.map(q =>
        `<button class="quick-btn" onclick="sendQuick('${q.replace(/'/g,"\\\'")}')">${q}</button>`
      ).join('');
      document.getElementById('chatArea').innerHTML = `
        <div class="welcome-msg">
          <div class="big-icon">${agent.icon}</div>
          <h2>${agent.name} 에이전트</h2>
          <p>${agent.desc}</p>
          <div class="quick-actions">${qbtns}</div>
        </div>
      `;

      // Send button color
      document.getElementById('sendBtn').style.background = agent.color1;
      document.getElementById('userInput').placeholder = `${agent.name}에게 질문하세요...`;

      // Show chat alongside canvas (split view)
      document.querySelector('.main-area').classList.add('split');
      document.getElementById('stageEl').style.display = '';
      document.getElementById('chatPanel').style.display = 'flex';
      document.getElementById('userInput').focus();
    }

    let chatCollapsed = false;

    function toggleChat() {
      if (!currentAgent) return;
      chatCollapsed = !chatCollapsed;
      if (chatCollapsed) {
        // 접기: 채팅 숨기고 캔버스 전체로 복원
        document.getElementById('chatPanel').style.display = 'none';
        document.querySelector('.main-area').classList.remove('split');
      } else {
        // 펼치기: 채팅 다시 보이기 (캔버스와 나란히)
        document.getElementById('chatPanel').style.display = 'flex';
        document.querySelector('.main-area').classList.add('split');
        const backBtn = document.querySelector('.back-btn');
        if (backBtn) backBtn.textContent = '← 접기';
      }
    }

    function closeAgent() {
      currentAgent = null;
      chatCollapsed = false;
      document.querySelectorAll('.agent-card').forEach(c => c.classList.remove('active'));
      document.getElementById('chatPanel').style.display = 'none';
      document.querySelector('.main-area').classList.remove('split');
    }

    function sendMsg() {
      const input = document.getElementById('userInput');
      const text = input.value.trim();
      if (!text || !currentAgent) return;

      const chatArea = document.getElementById('chatArea');
      // Remove welcome message if present
      const welcome = chatArea.querySelector('.welcome-msg');
      if (welcome) welcome.remove();

      // User message
      chatArea.innerHTML += `
        <div class="message user">
          <div class="msg-avatar">👤</div>
          <div class="msg-body">${escapeHtml(text)}</div>
        </div>
      `;
      input.value = '';

      // Agent response (placeholder)
      const agent = (currentAgent === 'ceo') ? agentBaseData[currentAgent] : getAgentData(currentAgent);
      const reportAgent = currentAgent; // capture for closure
      const reportText = text; // capture for closure
      setTimeout(() => {
        const responseText = `요청을 접수했습니다: "${escapeHtml(reportText)}"`;
        chatArea.innerHTML += `
          <div class="message">
            <div class="msg-avatar" style="background:linear-gradient(135deg,${agent.color1},${agent.color2});color:#fff;">${agent.letter}</div>
            <div class="msg-body">${responseText}<br><br>⏳ 백엔드 연결 후 실제 응답이 표시됩니다.</div>
          </div>
        `;
        chatArea.scrollTop = chatArea.scrollHeight;

        // Trigger 2D character to run to CEO and report
        if (reportAgent !== 'ceo') {
          triggerReport(reportAgent, reportText);
        }
      }, 500);

      chatArea.scrollTop = chatArea.scrollHeight;
    }

    function sendQuick(text) {
      document.getElementById('userInput').value = text;
      sendMsg();
    }

    function escapeHtml(t) {
      const d = document.createElement('div');
      d.textContent = t;
      return d.innerHTML;
    }
