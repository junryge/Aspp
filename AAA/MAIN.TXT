function parseCSV(txt){
  const ls=txt.trim().split('\n');
  if(ls.length<2)throw new Error('최소 2행 필요');
  let sp=',';if(ls[0].split('\t').length>ls[0].split(',').length)sp='\t';
  const hd=ls[0].split(sp).map(h=>h.replace(/^"|"$/g,'').trim());
  const al={VHL_ID:['VHL_ID','VHLID','VHL','VEHICLE_ID'],FROM_HIDID:['FROM_HIDID','FROM_HID','FROMHID'],TO_HIDID:['TO_HIDID','TO_HID','TOHID'],TRANS_CNT:['TRANS_CNT','TRANSCNT','CNT'],EVENT_DT:['EVENT_DT','EVENT_DATE','_time','TIME'],FAB_ID:['FAB_ID','FABID','FAB']};
  const cm={};
  for(const[k,ns]of Object.entries(al)){for(const n of ns){const i=hd.findIndex(h=>h.toUpperCase()===n.toUpperCase());if(i>=0){cm[k]=i;break;}}}
  if(!('VHL_ID'in cm)||!('FROM_HIDID'in cm)||!('TO_HIDID'in cm))throw new Error('필수 컬럼 누락: VHL_ID, FROM_HIDID, TO_HIDID\n헤더: '+hd.join(', '));
  const rows=[];
  for(let i=1;i<ls.length;i++){
    const c=ls[i].split(sp).map(x=>x.replace(/^"|"$/g,'').trim());
    if(c.length<3)continue;
    rows.push({V:c[cm.VHL_ID]||'',F:c[cm.FROM_HIDID]||'',T:c[cm.TO_HIDID]||'',TC:cm.TRANS_CNT!==undefined?parseInt(c[cm.TRANS_CNT])||1:1,DT:cm.EVENT_DT!==undefined?c[cm.EVENT_DT]:'',FAB:cm.FAB_ID!==undefined?c[cm.FAB_ID]:''});
  }
  return rows;
}

const CH_OPT={responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false}},scales:{x:{ticks:{color:'#636580',font:{size:9,family:'JetBrains Mono'},maxRotation:50},grid:{display:false}},y:{ticks:{color:'#636580'},grid:{color:'rgba(255,255,255,.04)'},beginAtZero:true}}};

// Stable color from HID number
function hidColor(hid){
  const n=parseInt(hid)||0;
  const hue=(n*137.508)%360;
  return `hsl(${hue},55%,55%)`;
}

function run(txt,fname,fsize){
  const rows=parseCSV(txt);
  const fabs=new Set(rows.map(r=>r.FAB).filter(Boolean));
  if(fabs.size)document.getElementById('fabTag').textContent=[...fabs].join(', ');

  document.getElementById('fnE').textContent=fname;
  document.getElementById('fmE').textContent=rows.length.toLocaleString()+' rows · '+(fsize/1024).toFixed(1)+' KB';
  document.getElementById('upW').style.display='none';
  document.getElementById('dash').classList.add('on');

  const vhlSet=new Set(rows.map(r=>r.V));
  const allH=new Set([...rows.map(r=>r.F),...rows.map(r=>r.T)]);

  // HID traffic
  const ht={};
  rows.forEach(r=>{ht[r.F]=(ht[r.F]||0)+1;ht[r.T]=(ht[r.T]||0)+1;});
  const sH=Object.entries(ht).sort((a,b)=>b[1]-a[1]);

  // Pair counts
  const pc={};
  rows.forEach(r=>{const k=r.F+'→'+r.T;pc[k]=(pc[k]||0)+1;});
  const topPairs=Object.entries(pc).sort((a,b)=>b[1]-a[1]).slice(0,20);

  // Bidirectional
  const bidi=[];const bSeen=new Set();
  for(const[k,c]of Object.entries(pc)){
    const[a,b]=k.split('→');const rv=b+'→'+a;
    if(pc[rv]&&!bSeen.has(b+'↔'+a)){bidi.push({a,b,ab:c,ba:pc[rv],tot:c+pc[rv]});bSeen.add(a+'↔'+b);}
  }
  bidi.sort((a,b)=>b.tot-a.tot);

  // VHL grouped
  const vg={};rows.forEach(r=>{if(!vg[r.V])vg[r.V]=[];vg[r.V].push(r);});

  // Retries
  const retries=rows.filter(r=>r.TC>=2);

  // Connected groups
  const cTh=Math.max(3,Math.floor(rows.length/250));
  const adj={};
  for(const[k,c]of Object.entries(pc)){
    if(c<cTh)continue;
    const[a,b]=k.split('→');
    if(!adj[a])adj[a]=new Set();if(!adj[b])adj[b]=new Set();
    adj[a].add(b);adj[b].add(a);
  }
  const vis=new Set();const grps=[];
  for(const nd of Object.keys(adj)){
    if(vis.has(nd))continue;
    const cl=new Set();const stk=[nd];
    while(stk.length){const n=stk.pop();if(vis.has(n))continue;vis.add(n);cl.add(n);if(adj[n])for(const nb of adj[n])if(!vis.has(nb))stk.push(nb);}
    if(cl.size>=3)grps.push([...cl].sort((a,b)=>parseInt(a)-parseInt(b)));
  }
  grps.sort((a,b)=>{const ta=a.reduce((s,h)=>s+(ht[h]||0),0);const tb=b.reduce((s,h)=>s+(ht[h]||0),0);return tb-ta;});

  // === KPI ===
  document.getElementById('kpiR').innerHTML=`
    <div class="kpi"><div class="lb">총 레코드</div><div class="vl">${rows.length.toLocaleString()}</div></div>
    <div class="kpi"><div class="lb">OHT 대수 (VHL)</div><div class="vl">${vhlSet.size}</div></div>
    <div class="kpi"><div class="lb">HID 존 수</div><div class="vl">${allH.size}</div></div>
    <div class="kpi"><div class="lb">이동 구간 종류</div><div class="vl">${Object.keys(pc).length}</div></div>
    <div class="kpi"><div class="lb">양방향 이동 쌍</div><div class="vl" style="color:var(--amber)">${bidi.length}</div></div>
    <div class="kpi"><div class="lb">TRANS_CNT ≥ 2</div><div class="vl" style="color:var(--red)">${retries.length}</div></div>
  `;

  // === 1. Traffic Chart ===
  const t25=sH.slice(0,25);
  charts.push(new Chart(document.getElementById('trafC'),{type:'bar',data:{
    labels:t25.map(([h])=>h),
    datasets:[{data:t25.map(([,v])=>v),backgroundColor:t25.map(([,v],i)=>i<5?'#ef4456':i<10?'#f0a030':'#4a8eff'),borderRadius:3,barPercentage:.75}]
  },options:{...CH_OPT}}));

  // === 2. Pair Chart ===
  charts.push(new Chart(document.getElementById('pairC'),{type:'bar',data:{
    labels:topPairs.map(([k])=>k),
    datasets:[{data:topPairs.map(([,v])=>v),backgroundColor:'#22c5d6',borderRadius:3,barPercentage:.7}]
  },options:{...CH_OPT}}));

  // === 3. Bidirectional Chart + Table ===
  const b15=bidi.slice(0,15);
  charts.push(new Chart(document.getElementById('bidiC'),{type:'bar',data:{
    labels:b15.map(p=>p.a+'↔'+p.b),
    datasets:[
      {label:'A→B',data:b15.map(p=>p.ab),backgroundColor:'#4a8eff',borderRadius:3,barPercentage:.75},
      {label:'B→A',data:b15.map(p=>p.ba),backgroundColor:'#f0a030',borderRadius:3,barPercentage:.75}
    ]
  },options:{...CH_OPT,plugins:{legend:{display:true,labels:{color:'#9a9cb2',font:{size:10,family:'JetBrains Mono'},boxWidth:10,boxHeight:10,borderRadius:2}}}}}));

  const bBody=document.querySelector('#bidiT tbody');bBody.innerHTML='';
  b15.forEach(p=>{bBody.innerHTML+=`<tr><td class="mn">${p.a}</td><td class="mn">${p.b}</td><td>${p.ab}</td><td>${p.ba}</td><td style="font-weight:600">${p.tot}</td></tr>`;});
  if(!b15.length)bBody.innerHTML='<tr><td colspan=5 style="text-align:center;color:var(--text3);padding:20px">양방향 이동 없음</td></tr>';

  // === 4. VHL Distribution ===
  const moveCounts={};
  for(const[,rs]of Object.entries(vg)){const n=rs.length;moveCounts[n]=(moveCounts[n]||0)+1;}
  const mcSorted=Object.entries(moveCounts).sort((a,b)=>parseInt(a[0])-parseInt(b[0]));
  charts.push(new Chart(document.getElementById('vhlDistC'),{type:'bar',data:{
    labels:mcSorted.map(([k])=>k+'회'),
    datasets:[{data:mcSorted.map(([,v])=>v),backgroundColor:'#3cc68a',borderRadius:3,barPercentage:.7}]
  },options:{...CH_OPT,scales:{...CH_OPT.scales,y:{...CH_OPT.scales.y,title:{display:true,text:'VHL 수',color:'#636580',font:{size:10}}}}}}));

  // Complex VHL mini list
  const cplx=Object.entries(vg).filter(([,rs])=>rs.length>=3).sort((a,b)=>b[1].length-a[1].length);
  document.getElementById('cplxCnt').textContent=cplx.length+'건';
  const cplxS=document.getElementById('cplxScroll');
  cplxS.innerHTML='';
  cplx.slice(0,30).forEach(([vid,rs])=>{
    let segs='';
    rs.forEach(r=>{
      const w=Math.max(100/rs.length,4);
      segs+=`<div class="tl-s" style="width:${w}%;background:${hidColor(r.T)}" data-tip="${vid}: ${r.F}→${r.T}"></div>`;
    });
    cplxS.innerHTML+=`<div class="tl-r"><div class="tl-lb">${vid}</div><div class="tl-bars">${segs}</div></div>`;
  });

  // === 5. Route Table ===
  const rBody=document.querySelector('#routeT tbody');rBody.innerHTML='';
  cplx.slice(0,20).forEach(([vid,rs])=>{
    const route=rs.map(r=>r.F+'→'+r.T).join(' , ');
    const fromList=rs.map(r=>r.F);
    const cnt={};fromList.forEach(h=>{cnt[h]=(cnt[h]||0)+1;});
    const reps=Object.entries(cnt).filter(([,c])=>c>1);
    const repStr=reps.length?reps.map(([h,c])=>`HID ${h} (${c}회)`).join(', '):'—';
    rBody.innerHTML+=`<tr><td class="mn" style="font-weight:600">${vid}</td><td>${rs.length}</td><td class="rt-cell">${route}</td><td class="mn">${repStr}</td></tr>`;
  });
  if(!cplx.length)rBody.innerHTML='<tr><td colspan=4 style="text-align:center;color:var(--text3);padding:20px">3회 이상 이동 VHL 없음</td></tr>';

  // === 6. Retry Table ===
  const rtBody=document.querySelector('#retryT tbody');rtBody.innerHTML='';
  retries.forEach(r=>{rtBody.innerHTML+=`<tr><td class="mn">${r.V}</td><td class="mn">${r.F}</td><td class="mn">${r.T}</td><td style="color:var(--amber);font-weight:600">${r.TC}</td></tr>`;});
  if(!retries.length)rtBody.innerHTML='<tr><td colspan=4 style="text-align:center;color:var(--text3);padding:20px">TRANS_CNT ≥ 2 기록 없음</td></tr>';

  // === 7. Connected Groups ===
  const gc=document.getElementById('grpCards');gc.innerHTML='';
  if(!grps.length)gc.innerHTML='<div class="cd"><div class="cd-b">고빈도(≥'+cTh+'건) 연결 그룹 없음</div></div>';
  grps.slice(0,6).forEach((g,i)=>{
    const traffic=g.reduce((s,h)=>s+(ht[h]||0),0);
    gc.innerHTML+=`<div class="cd">
      <div class="cd-h"><span class="badge" style="background:${i===0?'var(--red-d)':i===1?'var(--amber-d)':'var(--blue-d)'};color:${i===0?'var(--red)':i===1?'var(--amber)':'var(--blue)'}">그룹 ${i+1}</span>
      <span class="cd-t">${g.length}개 HID · 트래픽 합산 ${traffic}건 · 연결 기준 ≥${cTh}건</span></div>
      <div class="cd-hids">${g.join(', ')}</div>
    </div>`;
  });

  // === 8. Timeline ===
  const tlA=document.getElementById('tlArea');tlA.innerHTML='';
  const tlVhls=Object.entries(vg).filter(([,rs])=>rs.length>=2).sort((a,b)=>b[1].length-a[1].length).slice(0,30);
  tlVhls.forEach(([vid,rs])=>{
    let segs='';
    rs.forEach(r=>{
      const w=Math.max(100/rs.length,3);
      segs+=`<div class="tl-s" style="width:${w}%;background:${hidColor(r.T)}" data-tip="${vid}: ${r.F}→${r.T} (TRANS_CNT=${r.TC})"></div>`;
    });
    tlA.innerHTML+=`<div class="tl-r"><div class="tl-lb">${vid}</div><div class="tl-bars">${segs}</div></div>`;
  });
}
