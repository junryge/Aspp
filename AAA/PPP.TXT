function fitView(){
  if(!t3Ctrl||!t3Cam||!S.nodes.length)return;
  var bx1=Infinity,bx2=-Infinity,bz1=Infinity,bz2=-Infinity;
  S.nodes.forEach(function(n){
    var hw=(n.w||40)/2+20, hh=(n.h||40)/2+20;
    if(n.x-hw<bx1)bx1=n.x-hw; if(n.x+hw>bx2)bx2=n.x+hw;
    if(n.y-hh<bz1)bz1=n.y-hh; if(n.y+hh>bz2)bz2=n.y+hh;
  });
  var cx=(bx1+bx2)/2, cz=(bz1+bz2)/2;
  var rng=Math.max(bx2-bx1, bz2-bz1, 100);
  t3Ctrl.target.set(cx,0,cz);
  t3Cam.position.set(cx+rng*.6, rng*.5, cz+rng*.6);
  t3Ctrl.update();
}

function dk3(hex,f){if(!hex||hex[0]!=='#')return 0x888888;const v=parseInt(hex.slice(1),16);const r=(v>>16&255)*f|0,g=(v>>8&255)*f|0,b=(v&255)*f|0;return(r<<16)|(g<<8)|b;}
function hexToInt(h){return h&&h[0]==='#'?parseInt(h.slice(1),16):0x888888;}

function dispose3D(){
  t3Objs.forEach(function(o){t3Scene.remove(o);o.traverse(function(c){if(c.geometry)c.geometry.dispose();if(c.material){if(c.material.map)c.material.map.dispose();c.material.dispose();}});});
  t3Objs=[];
  // GLB mixer 정리 (imported GLB 애니메이션 — 에디터 내부 build3D에서는 mixer 안 씀)
  if(window._glbMixers){
    window._glbMixers.forEach(function(m){m.stopAllAction();});
    window._glbMixers=[];
  }
}

function build3D(){
  dispose3D();
  t3Ren.setClearColor(hexToInt(S.bgColor||'#0c1017'));
  t3Scene.fog.color.set(hexToInt(S.bgColor||'#0c1017'));
  var _fc=hexToInt(S.floorColor||'#1a2540');
  var _gc=hexToInt(S.gridColor||'#2a3a5a');
  var gpGeo=new THREE.PlaneGeometry(1200,1200);var gpMat=new THREE.MeshBasicMaterial({color:_fc});
  var gp=new THREE.Mesh(gpGeo,gpMat);gp.rotation.x=-Math.PI/2;gp.position.y=-1;gp.receiveShadow=true;t3Scene.add(gp);t3Objs.push(gp);
  var grid=new THREE.GridHelper(1200,60,_gc,_gc);grid.position.y=-.5;grid.visible=S.grid;t3Scene.add(grid);t3Objs.push(grid);

  S.nodes.forEach(function(n){
    var g=new THREE.Group();g.userData={node:n,type:n.type};
    var px=n.x,pz=n.y;
    switch(n.type){
      case'floor':{
        var w=n.w||200,h=n.h||80,tk=3;
        var geo=new THREE.BoxGeometry(w,tk,h);
        var mat=new THREE.MeshStandardMaterial({color:hexToInt(n.color||'#c8dca8'),transparent:true,opacity:.7,roughness:.6});
        var m=new THREE.Mesh(geo,mat);m.position.set(0,tk/2,0);m.receiveShadow=true;m.castShadow=false;
        g.add(m);
        var edge=new THREE.LineSegments(new THREE.EdgesGeometry(geo),new THREE.LineBasicMaterial({color:hexToInt(n.border||'#90b060'),transparent:true,opacity:.6}));
        edge.position.copy(m.position);g.add(edge);
        break;}
      case'eq_grid':{
        var rows=n.rows||3,cols=n.cols||4,gap=18,sz=7;
        for(var r=0;r<rows;r++)for(var cc=0;cc<cols;cc++){

var _selPrevMats=[];
function updateSelHighlight(){
  _selPrevMats.forEach(function(s){
    if(s.mesh.material){
      s.mesh.material.emissive.copy(s.origEmissive);
      s.mesh.material.emissiveIntensity=s.origIntensity;
    }
  });
  _selPrevMats=[];
  var ids=S.sels&&S.sels.length>0?S.sels:(S.sel?[S.sel]:[]);
  if(ids.length===0)return;
  var idSet=new Set(ids);
  t3Objs.forEach(function(grp){
    if(!grp.userData||!grp.userData.node||!idSet.has(grp.userData.node.id))return;
    grp.traverse(function(child){
      if(child.isMesh&&child.material&&child.material.emissive){
        _selPrevMats.push({
          mesh:child,
          origEmissive:child.material.emissive.clone(),
          origIntensity:child.material.emissiveIntensity||0
        });
      }
    });
  });
}

var _prevAnimTs=0;
function anim3D(ts){
  t3RAF=requestAnimationFrame(anim3D);
  t3Ctrl.update();
  // GLB AnimationMixer 업데이트
  var delta=(ts-_prevAnimTs)/1000; // ms → sec
  _prevAnimTs=ts;
  if(delta>0&&delta<1&&window._glbMixers){
    window._glbMixers.forEach(function(mixer){mixer.update(delta);});
  }
  var glowI=Math.sin(ts/300)*.25+.55;
  _selPrevMats.forEach(function(s){
    if(s.mesh.material){
      s.mesh.material.emissive.set(0xff2020);
      s.mesh.material.emissiveIntensity=glowI;
    }
  });
  t3Scene.traverse(function(obj){
    var u=obj.userData;
    if(u.flow){var f=u.flow;var t=((ts%3000)/3000+f.idx/f.total)%1;obj.position.x=-f.len/2+f.len*t;}
    if(u.lift){var lt2=u.lift;obj.position.y=5+Math.abs(Math.sin(ts/2500))*lt2.h*.8;}
    if(u.bridgeCube){var bc2=u.bridgeCube;obj.position.y=5+Math.abs(Math.sin(ts/2000))*bc2.h*.85;}
    if(u.connFlow){var cf=u.connFlow;var t2=((ts%3500)/3500+cf.idx/cf.total)%1;var pt2=cf.curve.getPointAt(t2);obj.position.copy(pt2);obj.material.emissiveIntensity=Math.sin(t2*Math.PI)*.5+.3;}
    if(u.tunFlow){var tf=u.tunFlow;var t3=((ts%3000)/3000+tf.idx/tf.total)%1;var pt=tf.curve.getPointAt(t3);obj.position.copy(pt);obj.material.opacity=Math.sin(t3*Math.PI)*.7+.3;}
  });
  t3Ren.render(t3Scene,t3Cam);
}

// Blender 호환: geometry는 로컬 좌표, 노드에 월드 위치, 애니메이션은 절대 위치 키프레임

function export3DGLB(){
  if(!S.nodes.length){alert('노드가 없습니다.');return;}
  if(!t3Scene){init3D();build3D();}else{build3D();}
  stat('GLB 내보내기 준비중 (애니메이션 포함)...');

  var accessors=[],bufferViews=[];
  var gltfMeshes=[],gltfNodes=[],gltfAnimations=[];
  var byteOffset=0;
  var matMap={},matList=[];
  var bufParts=[];
  var sceneNodeIndices=[];
  var animTracks=[];

  function pad4(){var p=byteOffset%4;if(p>0){var n=4-p;bufParts.push(new Uint8Array(n));byteOffset+=n;}}
  function addBufView(data,target){
    var bvIdx=bufferViews.length;
    var raw=new Uint8Array(data.buffer,data.byteOffset,data.byteLength);
    bufferViews.push({buffer:0,byteOffset:byteOffset,byteLength:raw.byteLength,target:target||undefined});
    bufParts.push(raw);
    byteOffset+=raw.byteLength;
    return bvIdx;
  }
  function addAccessor(bvIdx,compType,count,type,min,max){
    var idx=accessors.length;
    var a={bufferView:bvIdx,componentType:compType,count:count,type:type};
    if(min)a.min=min;if(max)a.max=max;
    accessors.push(a);return idx;
  }
  function getMatIdx(mat){
    var r=0.5,g=0.5,b=0.5,a=1;
    if(mat&&mat.color){r=mat.color.r;g=mat.color.g;b=mat.color.b;}
    if(mat&&mat.opacity!==undefined)a=mat.opacity;
    var key=r.toFixed(3)+'_'+g.toFixed(3)+'_'+b.toFixed(3)+'_'+a.toFixed(3);
    if(matMap[key]!==undefined)return matMap[key];
    var idx=matList.length;matMap[key]=idx;
    var m={pbrMetallicRoughness:{baseColorFactor:[r,g,b,a],metallicFactor:0.1,roughnessFactor:0.6},name:'mat_'+idx};
    if(a<1)m.alphaMode='BLEND';
    matList.push(m);return idx;
  }

  // geometry를 로컬 좌표(부모 그룹 기준)로 변환
  function processLocalMesh(obj, parentWorldMatrix){
    var geo=obj.geometry;
    var pos=geo.getAttribute('position');
    if(!pos)return -1;
    var norm=geo.getAttribute('normal');
    obj.updateWorldMatrix(true,false);

    // 로컬 변환 = parentInverse * world
    var parentInv=parentWorldMatrix.clone().invert();
    var localMat=parentInv.clone().multiply(obj.matrixWorld);
    var nm3=new THREE.Matrix3().getNormalMatrix(localMat);

    var vCount=pos.count;
    var posArr=new Float32Array(vCount*3);
    var normArr=norm?new Float32Array(vCount*3):null;
    var v=new THREE.Vector3(),n=new THREE.Vector3();
    var pMin=[Infinity,Infinity,Infinity],pMax=[-Infinity,-Infinity,-Infinity];
    for(var i=0;i<vCount;i++){
      v.set(pos.getX(i),pos.getY(i),pos.getZ(i)).applyMatrix4(localMat);
      posArr[i*3]=v.x;posArr[i*3+1]=v.y;posArr[i*3+2]=v.z;
      if(v.x<pMin[0])pMin[0]=v.x;if(v.y<pMin[1])pMin[1]=v.y;if(v.z<pMin[2])pMin[2]=v.z;
      if(v.x>pMax[0])pMax[0]=v.x;if(v.y>pMax[1])pMax[1]=v.y;if(v.z>pMax[2])pMax[2]=v.z;
      if(norm){
        n.set(norm.getX(i),norm.getY(i),norm.getZ(i)).applyMatrix3(nm3).normalize();
        normArr[i*3]=n.x;normArr[i*3+1]=n.y;normArr[i*3+2]=n.z;
      }
    }
    var idxSrc=geo.getIndex();
    var idxCount=idxSrc?idxSrc.count:vCount;
    var use32=vCount>65535;
    var idxArr=use32?new Uint32Array(idxCount):new Uint16Array(idxCount);
    if(idxSrc){for(var i=0;i<idxCount;i++)idxArr[i]=idxSrc.getX(i);}
    else{for(var i=0;i<idxCount;i++)idxArr[i]=i;}

    var posBV=addBufView(posArr,34962);
    var posAcc=addAccessor(posBV,5126,vCount,'VEC3',pMin,pMax);
    var normAcc=-1;
    if(normArr){var normBV=addBufView(normArr,34962);normAcc=addAccessor(normBV,5126,vCount,'VEC3');}
    pad4();
    var idxBV=addBufView(idxArr,34963);
    var idxAcc=addAccessor(idxBV,use32?5125:5123,idxCount,'SCALAR');
    pad4();

    var prim={attributes:{POSITION:posAcc},indices:idxAcc,material:getMatIdx(obj.material)};
    if(normAcc>=0)prim.attributes.NORMAL=normAcc;
    var meshIdx=gltfMeshes.length;
    gltfMeshes.push({primitives:[prim],name:'mesh_'+meshIdx});
    return meshIdx;
  }

  // ── 애니메이션 키프레임 (절대 위치로 bake) ──
  var ANIM_DUR=5.0;
  var ANIM_SAMPLES=60;

  function bakeTranslationAnim(nodeIdx, posFunc){
    var times=new Float32Array(ANIM_SAMPLES);
    var values=new Float32Array(ANIM_SAMPLES*3);
    for(var i=0;i<ANIM_SAMPLES;i++){
      var t=i/(ANIM_SAMPLES-1);
      times[i]=t*ANIM_DUR;
      var p=posFunc(t);
      values[i*3]=p[0];values[i*3+1]=p[1];values[i*3+2]=p[2];
    }
    animTracks.push({nodeIdx:nodeIdx,path:'translation',times:times,values:values});
  }

  // ── 씬 순회: 그룹 단위로 처리 ──
  // t3Objs에는 각 노드의 Group 또는 ground plane이 들어있음
  t3Objs.forEach(function(grp){
    if(!grp)return;

    // 단일 메시 (ground plane 등) — 그룹이 아닌 경우
    if(grp.isMesh&&grp.geometry&&!grp.isSprite){
      grp.updateWorldMatrix(true,false);
      var wp=new THREE.Vector3();grp.getWorldPosition(wp);
      // identity parent → geometry를 로컬 좌표(원점 기준)로
      var identityParent=new THREE.Matrix4().makeTranslation(wp.x,wp.y,wp.z);
      var meshIdx=processLocalMesh(grp, identityParent);
      if(meshIdx>=0){
        var nodeIdx=gltfNodes.length;
        gltfNodes.push({mesh:meshIdx,name:grp.name||'ground_'+nodeIdx,translation:[wp.x,wp.y,wp.z]});
        sceneNodeIndices.push(nodeIdx);
      }
      return;
    }

    // GridHelper 등 LineSegments는 스킵
    if(!grp.isGroup&&!grp.isMesh)return;

    // 그룹 안의 메시들을 자식 노드로 처리
    grp.updateWorldMatrix(true,false);
    var grpWorld=new THREE.Vector3();grp.getWorldPosition(grpWorld);
    var parentNode={name:grp.name||(grp.userData&&grp.userData.node?(grp.userData.node.label||grp.userData.node.type):'group'),translation:[grpWorld.x,grpWorld.y,grpWorld.z],children:[]};
    var parentNodeIdx=gltfNodes.length;
    gltfNodes.push(parentNode);
    sceneNodeIndices.push(parentNodeIdx);

    // 부모 월드 매트릭스 (그룹의 위치)
    var parentWM=new THREE.Matrix4().makeTranslation(grpWorld.x,grpWorld.y,grpWorld.z);

    grp.traverse(function(child){
      if(child===grp)return;
      if(!child.isMesh||!child.geometry||child.isSprite)return;

      var u=child.userData;
      child.updateWorldMatrix(true,false);
      var cw=new THREE.Vector3();child.getWorldPosition(cw);
      // 부모 기준 로컬 위치
      var localX=cw.x-grpWorld.x, localY=cw.y-grpWorld.y, localZ=cw.z-grpWorld.z;

      // geometry를 해당 메시의 자체 로컬 좌표로 변환
      // 메시의 부모 = 이 child의 월드 위치를 기준으로 identity
      var childWM=new THREE.Matrix4().makeTranslation(cw.x,cw.y,cw.z);
      var meshIdx=processLocalMesh(child, childWM);
      if(meshIdx<0)return;

      var childNodeIdx=gltfNodes.length;
      var childNode={mesh:meshIdx,name:child.name||('child_'+childNodeIdx),translation:[localX,localY,localZ]};
      gltfNodes.push(childNode);
      parentNode.children.push(childNodeIdx);

      // ── 애니메이션: 절대 위치(부모 기준 로컬)로 키프레임 bake ──
      if(u.lift){
        var lh=u.lift.h||70;
        var baseLocalY=localY;
        bakeTranslationAnim(childNodeIdx, function(t){
          var dy=Math.abs(Math.sin(t*Math.PI*2))*lh*0.8;
          return [localX, baseLocalY+dy, localZ];
        });
      }
      else if(u.bridgeCube){
        var bh=u.bridgeCube.h||120;
        var baseLocalY2=localY;
        bakeTranslationAnim(childNodeIdx, function(t){
          var dy=Math.abs(Math.sin(t*Math.PI*2))*bh*0.85;
          return [localX, baseLocalY2+dy, localZ];
        });
      }
      else if(u.flow){
        var fl=u.flow;
        bakeTranslationAnim(childNodeIdx, function(t){
          var phase=((t+fl.idx/fl.total)%1);
          var dx=-fl.len/2+fl.len*phase;
          return [dx, localY, localZ];
        });
      }
      else if(u.connFlow){
        var cf=u.connFlow;
        bakeTranslationAnim(childNodeIdx, function(t){
          var phase=((t+cf.idx/cf.total)%1);
          var pt=cf.curve.getPointAt(phase);
          // 커브 좌표는 월드 좌표 → 부모 기준 로컬로 변환
          return [pt.x-grpWorld.x, pt.y-grpWorld.y, pt.z-grpWorld.z];
        });
      }
      else if(u.tunFlow){
        var tf=u.tunFlow;
        bakeTranslationAnim(childNodeIdx, function(t){
          var phase=((t+tf.idx/tf.total)%1);
          var pt=tf.curve.getPointAt(phase);
          return [pt.x-grpWorld.x, pt.y-grpWorld.y, pt.z-grpWorld.z];
        });
      }
    });
  });

  // ── Build animation data into buffer ──
  if(animTracks.length>0){
    var channels=[],samplers=[];
    animTracks.forEach(function(track){
      var tMin=track.times[0],tMax=track.times[track.times.length-1];
      pad4();
      var timeBV=addBufView(track.times);
      var timeAcc=addAccessor(timeBV,5126,track.times.length,'SCALAR',[tMin],[tMax]);
      pad4();
      var valBV=addBufView(track.values);
      var valAcc=addAccessor(valBV,5126,track.values.length/3,'VEC3');
      var samplerIdx=samplers.length;
      samplers.push({input:timeAcc,output:valAcc,interpolation:'LINEAR'});
      channels.push({sampler:samplerIdx,target:{node:track.nodeIdx,path:track.path}});
    });
    gltfAnimations.push({name:'FAB_Animation',channels:channels,samplers:samplers});
  }

  // ── Assemble glTF JSON ──
  var gltf={
    asset:{version:'2.0',generator:'FAB Layout Editor GLB Export (Animated)'},
    scene:0,
    scenes:[{nodes:sceneNodeIndices}],
    nodes:gltfNodes,
    meshes:gltfMeshes,
    accessors:accessors,
    bufferViews:bufferViews,
    buffers:[{byteLength:byteOffset}],
    materials:matList
  };
  if(gltfAnimations.length>0)gltf.animations=gltfAnimations;

  var jsonStr=JSON.stringify(gltf);
  while(jsonStr.length%4!==0)jsonStr+=' ';
  var jsonBytes=new TextEncoder().encode(jsonStr);

  var binBytes=new Uint8Array(byteOffset);
  var off=0;
  bufParts.forEach(function(part){binBytes.set(part,off);off+=part.length;});

  var totalLen=12+8+jsonBytes.length+8+binBytes.length;
  var glb=new ArrayBuffer(totalLen);
  var dv=new DataView(glb);
  var u8=new Uint8Array(glb);

  dv.setUint32(0,0x46546C67,true);
  dv.setUint32(4,2,true);
  dv.setUint32(8,totalLen,true);
  dv.setUint32(12,jsonBytes.length,true);
  dv.setUint32(16,0x4E4F534A,true);
  u8.set(jsonBytes,20);
  var binOff=20+jsonBytes.length;
  dv.setUint32(binOff,binBytes.length,true);
  dv.setUint32(binOff+4,0x004E4942,true);
  u8.set(binBytes,binOff+8);

  var blob=new Blob([glb],{type:'model/gltf-binary'});
  var a=document.createElement('a');
  a.href=URL.createObjectURL(blob);a.download='fab_layout_3d.glb';a.click();
  stat('GLB 내보내기 완료 (애니 '+animTracks.length+'트랙, 메시 '+gltfMeshes.length+'개, 노드 '+gltfNodes.length+'개)');
}

// GLB LOAD — glTF Binary → Three.js Scene

function loadGLB(){
  var inp=document.createElement('input');
  inp.type='file';inp.accept='.glb,.gltf';
  inp.onchange=function(ev){
    var file=ev.target.files[0];
    if(!file)return;
    stat('GLB 로딩중: '+file.name+'...');
    var reader=new FileReader();
    reader.onload=function(re){
      try{
        var buf=re.target.result;
        var parsed=parseGLB(buf);
        if(!parsed){stat('GLB 파싱 실패');return;}
        addGLBToScene(parsed, file.name);
      }catch(err){
        alert('GLB 로드 실패: '+err.message);
        stat('GLB 로드 실패');
        console.error(err);
      }
    };
    reader.readAsArrayBuffer(file);
  };
  inp.click();
}

function parseGLB(buffer){
  var dv=new DataView(buffer);
  // Header
  var magic=dv.getUint32(0,true);
  if(magic!==0x46546C67){alert('올바른 GLB 파일이 아닙니다.');return null;}
  var version=dv.getUint32(4,true);
  var totalLen=dv.getUint32(8,true);

  // Chunks
  var offset=12;
  var jsonData=null, binData=null;
  while(offset<totalLen){
    var chunkLen=dv.getUint32(offset,true);
    var chunkType=dv.getUint32(offset+4,true);
    var chunkData=buffer.slice(offset+8, offset+8+chunkLen);
    if(chunkType===0x4E4F534A){ // JSON
      jsonData=JSON.parse(new TextDecoder().decode(chunkData));
    }else if(chunkType===0x004E4942){ // BIN
      binData=chunkData;
    }
    offset+=8+chunkLen;
  }
  if(!jsonData){alert('GLB에 JSON 데이터가 없습니다.');return null;}
  return {json:jsonData, bin:binData};
}

function addGLBToScene(parsed, filename){
  var json=parsed.json;
  var bin=parsed.bin;

  // Parse accessors helper
  function getAccessorData(accIdx){
    var acc=json.accessors[accIdx];
    var bv=json.bufferViews[acc.bufferView];
    var byteOff=(bv.byteOffset||0)+(acc.byteOffset||0);
    var compType=acc.componentType;
    var count=acc.count;
    var typeSize={'SCALAR':1,'VEC2':2,'VEC3':3,'VEC4':4,'MAT4':16}[acc.type]||1;
    var totalElements=count*typeSize;

    // TypedArray는 byteOffset이 element size의 배수여야 함 → 복사로 안전하게 처리
    var slice=new Uint8Array(bin, byteOff, totalElements*(compType===5126?4:compType===5125?4:compType===5123?2:1));
    var arrBuf=slice.buffer.slice(slice.byteOffset, slice.byteOffset+slice.byteLength);
    if(compType===5126) return new Float32Array(arrBuf);
    if(compType===5123) return new Uint16Array(arrBuf);
    if(compType===5125) return new Uint32Array(arrBuf);
    if(compType===5121) return new Uint8Array(arrBuf);
    return new Float32Array(arrBuf);
  }

  // Parse materials
  function getMaterial(matIdx){
    if(matIdx==null||!json.materials||!json.materials[matIdx]){
      return new THREE.MeshStandardMaterial({color:0xaaaaaa,roughness:0.6});
    }
    var m=json.materials[matIdx];
    var pbr=m.pbrMetallicRoughness||{};
    var bc=pbr.baseColorFactor||[0.8,0.8,0.8,1];
    var mat=new THREE.MeshStandardMaterial({
      color:new THREE.Color(bc[0],bc[1],bc[2]),
      roughness:pbr.roughnessFactor!=null?pbr.roughnessFactor:0.6,
      metalness:pbr.metallicFactor!=null?pbr.metallicFactor:0.1,
    });
    if(bc[3]<1||m.alphaMode==='BLEND'){
      mat.transparent=true;
      mat.opacity=bc[3];
    }
    if(m.doubleSided)mat.side=THREE.DoubleSide;
    return mat;
  }

  // Build meshes
  var glbGroup=new THREE.Group();
  glbGroup.userData={glbImport:true, filename:filename};

  var meshDefs=json.meshes||[];
  var nodeDefs=json.nodes||[];
  var sceneDef=(json.scenes&&json.scenes[json.scene||0])||{nodes:[]};

  // nodeIdx → Three.js Object3D 매핑 (애니메이션 타겟용)
  var nodeMap={};

  // Build node hierarchy
  function buildNode(nodeIdx){
    var nd=nodeDefs[nodeIdx];
    if(!nd)return null;
    var obj;
    if(nd.mesh!=null&&meshDefs[nd.mesh]){
      var mDef=meshDefs[nd.mesh];
      obj=new THREE.Group();
      (mDef.primitives||[]).forEach(function(prim){
        var geo=new THREE.BufferGeometry();
        if(prim.attributes&&prim.attributes.POSITION!=null){
          var posData=getAccessorData(prim.attributes.POSITION);
          geo.setAttribute('position',new THREE.BufferAttribute(new Float32Array(posData),3));
        }
        if(prim.attributes&&prim.attributes.NORMAL!=null){
          var normData=getAccessorData(prim.attributes.NORMAL);
          geo.setAttribute('normal',new THREE.BufferAttribute(new Float32Array(normData),3));
        }else{
          geo.computeVertexNormals();
        }
        if(prim.attributes&&prim.attributes.TEXCOORD_0!=null){
          var uvData=getAccessorData(prim.attributes.TEXCOORD_0);
          geo.setAttribute('uv',new THREE.BufferAttribute(new Float32Array(uvData),2));
        }
        if(prim.indices!=null){
          var idxData=getAccessorData(prim.indices);
          geo.setIndex(new THREE.BufferAttribute(
            idxData instanceof Uint32Array?new Uint32Array(idxData):new Uint16Array(idxData),1));
        }
        var mat=getMaterial(prim.material);
        var mesh=new THREE.Mesh(geo,mat);
        mesh.castShadow=true;mesh.receiveShadow=true;
        obj.add(mesh);
      });
    }else{
      obj=new THREE.Group();
    }
    obj.name=nd.name||('node_'+nodeIdx);

    // Transform
    if(nd.matrix){
      var m4=new THREE.Matrix4();m4.fromArray(nd.matrix);
      obj.applyMatrix4(m4);
    }else{
      if(nd.translation)obj.position.fromArray(nd.translation);
      if(nd.rotation){var q=new THREE.Quaternion().fromArray(nd.rotation);obj.quaternion.copy(q);}
      if(nd.scale)obj.scale.fromArray(nd.scale);
    }

    // Children
    if(nd.children){
      nd.children.forEach(function(ci){
        var child=buildNode(ci);
        if(child)obj.add(child);
      });
    }

    nodeMap[nodeIdx]=obj;
    return obj;
  }

  // Build scene
  var sceneNodes=sceneDef.nodes||[];
  if(sceneNodes.length===0){
    for(var i=0;i<nodeDefs.length;i++) sceneNodes.push(i);
  }
  sceneNodes.forEach(function(ni){
    var n=buildNode(ni);
    if(n)glbGroup.add(n);
  });

  // ── Parse animations → Three.js AnimationClip ──
  var animClips=[];
  var animCount=0;
  if(json.animations&&json.animations.length>0){
    json.animations.forEach(function(animDef){
      var tracks=[];
      (animDef.channels||[]).forEach(function(ch){
        var sampler=animDef.samplers[ch.sampler];
        var targetNode=nodeMap[ch.target.node];
        if(!targetNode||!sampler)return;

        var inputData=getAccessorData(sampler.input);   // time keyframes
        var outputData=getAccessorData(sampler.output);  // value keyframes
        var times=new Float32Array(inputData);
        var values=new Float32Array(outputData);

        var path=ch.target.path; // translation, rotation, scale
        var trackName=targetNode.name;
        var interpolation=sampler.interpolation||'LINEAR';
        var interpType=THREE.InterpolateLinear;
        if(interpolation==='STEP')interpType=THREE.InterpolateDiscrete;
        if(interpolation==='CUBICSPLINE')interpType=THREE.InterpolateSmooth;

        var track=null;
        if(path==='translation'){
          track=new THREE.VectorKeyframeTrack(trackName+'.position',times,values,interpType);
        }else if(path==='rotation'){
          track=new THREE.QuaternionKeyframeTrack(trackName+'.quaternion',times,values,interpType);
        }else if(path==='scale'){
          track=new THREE.VectorKeyframeTrack(trackName+'.scale',times,values,interpType);
        }
        if(track){tracks.push(track);animCount++;}
      });
      if(tracks.length>0){
        var clip=new THREE.AnimationClip(animDef.name||'animation',undefined,tracks);
        animClips.push(clip);
      }
    });
  }

  // ── Add to scene ──
  t3Scene.add(glbGroup);
  t3Objs.push(glbGroup);

  // ── Setup AnimationMixer if animations exist ──
  if(animClips.length>0){
    var mixer=new THREE.AnimationMixer(glbGroup);
    animClips.forEach(function(clip){
      var action=mixer.clipAction(clip);
      action.setLoop(THREE.LoopRepeat);
      action.play();
    });
    // 글로벌 mixer 배열에 등록 (anim3D에서 업데이트)
    if(!window._glbMixers)window._glbMixers=[];
    window._glbMixers.push(mixer);
  }

  // Fit camera
  var bbox=new THREE.Box3().setFromObject(glbGroup);
  var center=new THREE.Vector3();bbox.getCenter(center);
  var size=new THREE.Vector3();bbox.getSize(size);
  var maxDim=Math.max(size.x,size.y,size.z);

  var rng=maxDim||200;
  t3Ctrl.target.copy(center);
  t3Cam.position.set(center.x+rng*.7, center.y+rng*.5, center.z+rng*.7);
  t3Ctrl.update();

  var meshCount=0;
  glbGroup.traverse(function(c){if(c.isMesh)meshCount++;});
  stat('GLB 로드 완료: '+filename+' (메시 '+meshCount+'개, 애니 '+animCount+'트랙'+(maxDim?', 크기 '+maxDim.toFixed(0):'')+')');
}

  t3Objs.forEach(function(o){
    if(o.userData&&o.userData.node)hitMeshes.push(o);
    if(o.userData&&o.userData.conn)connMeshes.push(o);
  });
  var intersects=_ray.intersectObjects(hitMeshes,true);
  var hitNode=null;
  if(intersects.length>0){
    var hit=intersects[0].object;
    while(hit&&!(hit.userData&&hit.userData.node)&&hit.parent)hit=hit.parent;
    if(hit&&hit.userData&&hit.userData.node) hitNode=hit.userData.node;
  }
  var hitConn=null;
  if(!hitNode){
    var cInts=_ray.intersectObjects(connMeshes,true);
    if(cInts.length>0){
      var cHit=cInts[0].object;
      while(cHit&&!(cHit.userData&&cHit.userData.conn)&&cHit.parent)cHit=cHit.parent;
      if(cHit&&cHit.userData&&cHit.userData.conn) hitConn=cHit.userData.conn;
    }
  }

  if(S.tool==='connect'){
    if(hitNode){
      if(!S.cFrom){
        S.cFrom=hitNode.id;
        S.sel=hitNode.id;S.sels=[hitNode.id];S.selC=null;
        updProp();updateSelHighlight();
        stat('연결 출발: '+(hitNode.label||hitNode.id)+' → 도착 노드를 클릭하세요');
      }else if(S.cFrom!==hitNode.id){
        push();
        var newConn={id:gid(),from:S.cFrom,to:hitNode.id,label:'',color:'#3b82f6',width:3,dashed:false};
        S.conns.push(newConn);
        S.selC=newConn;S.sel=null;S.sels=[];
        S.cFrom=null;
        updProp();cnts();update3DPositions();
        stat('연결 완료');
      }else{
        S.cFrom=null;
        stat('연결 취소 (같은 노드)');
      }
    }else{
      S.cFrom=null;
      stat('연결 취소');
    }
    _3dMouseDown=null;
    return;
  }

  if(hitNode){
    var isShift=e.shiftKey;
    if(isShift){
      var idx=S.sels.indexOf(hitNode.id);
      if(idx>=0){S.sels.splice(idx,1);}else{S.sels.push(hitNode.id);}
      S.sel=S.sels.length>0?S.sels[S.sels.length-1]:null;
      S.selC=null;
    }else{
      if(S.sels.length>1&&S.sels.indexOf(hitNode.id)>=0){
      }else{
        S.sel=hitNode.id;S.sels=[hitNode.id];S.selC=null;
      }
    }
    _3dDragNode=hitNode;hitNode._origZ=hitNode.z||0;
    var hitPt2=new THREE.Vector3();
    _ray.ray.intersectPlane(_gPlane,hitPt2);
    if(hitPt2){_3dDragOffset={x:hitNode.x-hitPt2.x,z:hitNode.y-hitPt2.z};}
    else{_3dDragOffset={x:0,z:0};}
    if(S.sels.length>1){
      _3dMultiStart={};
      S.sels.forEach(function(id){
        var nn=S.nodes.find(function(x){return x.id===id;});
        if(nn) _3dMultiStart[id]={x:nn.x,y:nn.y};
      });
      _3dMultiAnchor={x:hitNode.x,y:hitNode.y};
    }else{
      _3dMultiStart=null;_3dMultiAnchor=null;
    }
    updProp();cnts();updateSelHighlight();
    stat(S.sels.length>1?S.sels.length+'개 선택됨':'선택: '+(hitNode.label||hitNode.id));
  }else if(hitConn){
    S.sel=null;S.sels=[];S.selC=hitConn;_3dDragNode=null;
    updProp();cnts();updateSelHighlight();
    stat('연결선 선택: '+(hitConn.label||hitConn.id));
  }else{
    if(e.altKey){
      clearSel();_3dDragNode=null;
      _3dBoxSel={sx:e.clientX,sy:e.clientY,active:false};
      updProp();updateSelHighlight();
    }else{
      clearSel();_3dDragNode=null;updProp();updateSelHighlight();
    }
  }
});

var _tc3dEl=document.getElementById('tc3d');
document.addEventListener('mousemove',function(e){
  if(!_3dDrag&&!_3dLabelDrag&&t3Ren&&t3Cam){
    var isCtrl=e.ctrlKey||e.metaKey;
    if(isCtrl){
      var rect2=t3Ren.domElement.getBoundingClientRect();
      if(e.clientX>=rect2.left&&e.clientX<=rect2.right&&e.clientY>=rect2.top&&e.clientY<=rect2.bottom){
        _m2.x=((e.clientX-rect2.left)/rect2.width)*2-1;
        _m2.y=-((e.clientY-rect2.top)/rect2.height)*2+1;
        _ray.setFromCamera(_m2,t3Cam);
        var labelSprites=[];
        t3Objs.forEach(function(o){o.traverse(function(c){if(c.isSprite&&c.userData&&c.userData.isLabel)labelSprites.push(c);});});
        var lInts=_ray.intersectObjects(labelSprites,false);
        _tc3dEl.style.cursor=lInts.length>0?'grab':'';
      }
    }else{
      if(_tc3dEl.style.cursor==='grab')_tc3dEl.style.cursor='';
    }
  }

  if(_3dMouseDown&&_3dLabelNode&&e.buttons===1){
    var dx2=e.clientX-_3dMouseDown.x,dy2=e.clientY-_3dMouseDown.y;
    if(!_3dLabelDrag&&Math.abs(dx2)+Math.abs(dy2)>3){
      _3dLabelDrag=true;
      t3Ctrl.enabled=false;
      _tc3dEl.style.cursor='grabbing';
      push();
    }
    if(_3dLabelDrag&&_3dLabelSprite){
      if(e.shiftKey){
        var deltaY=_3dLabelStartY-e.clientY;
        _3dLabelSprite.position.y=_3dLabelOrigY+deltaY*0.5;
        _3dLabelNode.labelOY=_3dLabelSprite.position.y;
      }else if(_3dLabelPlane){
        var rect3=t3Ren.domElement.getBoundingClientRect();
        _m2.x=((e.clientX-rect3.left)/rect3.width)*2-1;
        _m2.y=-((e.clientY-rect3.top)/rect3.height)*2+1;
        _ray.setFromCamera(_m2,t3Cam);
        var tgt=new THREE.Vector3();
        _ray.ray.intersectPlane(_3dLabelPlane,tgt);
        if(tgt){
          var parentPos=_3dLabelSprite.parent.position;
          _3dLabelSprite.position.x=tgt.x+_3dLabelOffset.x-parentPos.x;
          _3dLabelSprite.position.z=tgt.z+_3dLabelOffset.z-parentPos.z;
          _3dLabelNode.labelOX=_3dLabelSprite.position.x;
          _3dLabelNode.labelOZ=_3dLabelSprite.position.z;
        }
      }
    }
    return;
  }

  if(_3dMouseDown&&_3dBoxSel&&!_3dDragNode&&e.buttons===1){
    var bdx=e.clientX-_3dBoxSel.sx,bdy=e.clientY-_3dBoxSel.sy;
    if(!_3dBoxSel.active&&Math.abs(bdx)+Math.abs(bdy)>5){
      _3dBoxSel.active=true;
      t3Ctrl.enabled=false;
      if(!document.getElementById('selBox3d')){
        var sb=document.createElement('div');sb.id='selBox3d';
        sb.style.cssText='position:fixed;border:1px solid #3b82f6;background:rgba(59,130,246,0.1);pointer-events:none;z-index:999;';
        document.body.appendChild(sb);
      }
    }
    if(_3dBoxSel.active){
      _3dBoxSel._ex=e.clientX;_3dBoxSel._ey=e.clientY;
      var sb2=document.getElementById('selBox3d');
      if(sb2){
        var bx1=Math.min(_3dBoxSel.sx,e.clientX),by1=Math.min(_3dBoxSel.sy,e.clientY);
        var bx2=Math.max(_3dBoxSel.sx,e.clientX),by2=Math.max(_3dBoxSel.sy,e.clientY);
        sb2.style.left=bx1+'px';sb2.style.top=by1+'px';sb2.style.width=(bx2-bx1)+'px';sb2.style.height=(by2-by1)+'px';sb2.style.display='block';
      }
    }
    return;
  }

  if(!_3dMouseDown||!_3dDragNode||e.buttons!==1)return;
  var dx=e.clientX-_3dMouseDown.x,dy=e.clientY-_3dMouseDown.y;
  if(!_3dDrag&&Math.abs(dx)+Math.abs(dy)>5){
    _3dDrag=true;
    _3dDragZ=e.shiftKey;
    t3Ctrl.enabled=false;
    _tc3dEl.style.cursor='grabbing';
    push();
  }
  if(_3dDrag) _3dDragZ=e.shiftKey;
  if(_3dDrag&&!_3dDragZ){
    var rect=t3Ren.domElement.getBoundingClientRect();
    _m2.x=((e.clientX-rect.left)/rect.width)*2-1;
    _m2.y=-((e.clientY-rect.top)/rect.height)*2+1;
    _ray.setFromCamera(_m2,t3Cam);
    var target=new THREE.Vector3();
    _ray.ray.intersectPlane(_gPlane,target);
    if(target){
      var newX=target.x+_3dDragOffset.x;
      var newY=target.z+_3dDragOffset.z;
      if(_3dMultiStart&&_3dMultiAnchor&&S.sels.length>1){
        var mdx=newX-_3dMultiAnchor.x,mdy=newY-_3dMultiAnchor.y;
        S.sels.forEach(function(id){
          var nn=S.nodes.find(function(x){return x.id===id;});
          var st=_3dMultiStart[id];
          if(nn&&st){
            nn.x=st.x+mdx;nn.y=st.y+mdy;
            var grp=t3Objs.find(function(o){return o.userData&&o.userData.node&&o.userData.node.id===id;});
            if(grp)grp.position.set(nn.x,nn.z||0,nn.y);
          }
        });
      }else{
        _3dDragNode.x=newX;_3dDragNode.y=newY;
        var grp=t3Objs.find(function(o){return o.userData&&o.userData.node===_3dDragNode;});
        if(grp)grp.position.set(_3dDragNode.x,_3dDragNode.z||0,_3dDragNode.y);
      }
      updProp();
    }
  }else if(_3dDrag&&_3dDragZ){
    var deltaY=e.clientY-_3dMouseDown.y;
    _3dDragNode.z=(_3dDragNode._origZ||0)-deltaY;
    var grp=t3Objs.find(function(o){return o.userData&&o.userData.node===_3dDragNode;});
    if(grp)grp.position.y=_3dDragNode.z||0;
    updProp();
  }
});

    var savedTarget2=t3Ctrl.target.clone();
    build3D();
    t3Cam.position.copy(savedPos2);
    t3Ctrl.target.copy(savedTarget2);
    t3Ctrl.update();
    stat('이동 완료');
  }
  if(_3dBoxSel&&_3dBoxSel.active){
    var sb3=document.getElementById('selBox3d');
    if(sb3)sb3.style.display='none';

      if(!o.userData||!o.userData.node)return;
      var pos=new THREE.Vector3();o.getWorldPosition(pos);
      pos.project(t3Cam);
      var rect=t3Ren.domElement.getBoundingClientRect();
      var sx2=(pos.x*.5+.5)*rect.width+rect.left;
      var sy2=(-.5*pos.y+.5)*rect.height+rect.top;
      if(sx2>=bx1&&sx2<=bx2&&sy2>=by1&&sy2<=by2) selected.push(o.userData.node.id);
    });

      S.sels=selected;S.sel=selected[0];S.selC=null;
      updProp();updateSelHighlight();
      stat(selected.length+'개 선택됨');
    }
    t3Ctrl.enabled=true;
  }
  _3dBoxSel=null;_3dMultiStart=null;_3dMultiAnchor=null;
  if(_3dMouseDown){
    _3dDrag=false;_3dDragNode=null;_3dMouseDown=null;_3dDragZ=false;
    t3Ctrl.enabled=true;
  }
});
