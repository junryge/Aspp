"""
Java 소스파일에서 package/import 문을 제거하고 코드 본문만 추출하여 새 폴더에 저장
사용법: python strip_java_headers.py <입력폴더> <출력폴더>
"""

import os
import sys
import re


def strip_java_headers(source: str) -> str:
    lines = source.splitlines()
    result = []
    header_done = False

    for line in lines:
        stripped = line.strip()

        if not header_done:
            # package / import 라인 스킵
            if stripped.startswith('package ') or stripped.startswith('import '):
                continue
            # 헤더 영역의 빈 줄도 스킵
            if stripped == '':
                continue
            # 그 외 실제 코드 시작
            header_done = True

        result.append(line)

    return '\n'.join(result)


def process_folder(input_dir: str, output_dir: str):
    if not os.path.isdir(input_dir):
        print(f"[ERROR] 입력 폴더가 없음: {input_dir}")
        sys.exit(1)

    os.makedirs(output_dir, exist_ok=True)

    count = 0
    for root, dirs, files in os.walk(input_dir):
        for fname in files:
            if not fname.endswith('.java'):
                continue

            src_path = os.path.join(root, fname)
            # 하위 폴더 구조 유지
            rel = os.path.relpath(src_path, input_dir)
            dst_path = os.path.join(output_dir, rel)
            os.makedirs(os.path.dirname(dst_path), exist_ok=True)

            with open(src_path, 'r', encoding='utf-8') as f:
                content = f.read()

            cleaned = strip_java_headers(content)

            with open(dst_path, 'w', encoding='utf-8') as f:
                f.write(cleaned)

            count += 1
            print(f"  [OK] {rel}")

    print(f"\n완료: {count}개 파일 처리 → {output_dir}")


if __name__ == '__main__':
    if len(sys.argv) < 3:
        print("사용법: python strip_java_headers.py <입력폴더> <출력폴더>")
        print("예시:   python strip_java_headers.py ./src ./src_clean")
        sys.exit(1)

    process_folder(sys.argv[1], sys.argv[2])
