import json, datetime

REPORTS = [
    r"C:\연구과제\CODE\데모스_분석툴\레포트1000_4000\레포트10030\reports_db.json",
    r"C:\연구과제\CODE\데모스_분석툴\레포트1000_4000\레포트10040\reports_db.json",
    r"C:\연구과제\CODE\데모스_분석툴\레포트1000_4000\레포트10050\reports_db.json",
    r"C:\연구과제\CODE\데모스_분석툴\레포트1000_4000\레포트10060\reports_db.json",
]
HTML_FILE = r"C:\연구과제\CODE\데모스_분석툴\RPA\html\LLM사건발생_{yesterday}.html"

with open(HTML_FILE, encoding="utf-8") as f:
    body = f.read()

entry = {
    "title": "{yesterday:%Y_%m_%d}_Incident_Report_System",
    "body": body,
    "author": "AMHS",
    "status": "draft",
    "updatedAt": datetime.datetime.now().isoformat() + "Z"
}
key = "{yesterday:%Y-%m-%d}"

done = 0
for path in REPORTS:
    try:
        with open(path, encoding="utf-8") as f:
            db = json.load(f)
        db[key] = dict(entry)
        with open(path, "w", encoding="utf-8") as f:
            json.dump(db, f, ensure_ascii=False, indent=2)
        done += 1
        print("registered:", path)
    except Exception as e:
        print("FAIL:", path, e)

print("total:", done, "->", key)
