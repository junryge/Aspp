function connectWS() {
  const proto = location.protocol === 'https:' ? 'wss' : 'ws';
  ws = new WebSocket(`${proto}://${location.host}/ws`);
  ws.onopen = () => console.log('WS connected');
  ws.onmessage = (e) => { try { updateUI(JSON.parse(e.data)); } catch(err) {} };
  ws.onclose = () => setTimeout(connectWS, 2000);
  ws.onerror = () => ws.close();
}
function sendCmd(action, extra) {
  if (ws && ws.readyState === 1) ws.send(JSON.stringify({ action, ...extra }));
}
function setSpeed(s) {
  currentSpeed = s;
  sendCmd('speed', { speed: s });
  document.querySelectorAll('.speed-btn').forEach(b => b.classList.toggle('selected', b.textContent === (s===0?'MAX':`x${s}`)));
}

let userSliderDragging = false;
let _sliderInputDebounce = null;
function onSliderInput(val) {
  // 드래그 중 표시 시각만 즉각 갱신 (서버 호출은 throttle)
  const frame = parseInt(val) || 0;
  const total = parseInt(document.getElementById('time-slider').max) || 0;
  const fd = document.getElementById('frame-display');
  if (fd) fd.textContent = `${frame}/${total}`;
  // 50ms 단위로 jump 명령 보내기 (스크럽 미리보기)
  if (_sliderInputDebounce) clearTimeout(_sliderInputDebounce);
  _sliderInputDebounce = setTimeout(() => jumpFrame(frame), 50);
}
function onSliderChange(val) {
  // 마우스 떼는 순간 — 최종 위치 확정
  if (_sliderInputDebounce) { clearTimeout(_sliderInputDebounce); _sliderInputDebounce = null; }
  jumpFrame(parseInt(val) || 0);
  // 약간 후 드래그 플래그 해제 (WS 응답이 도착하기 전 살짝 보호)
  setTimeout(() => { userSliderDragging = false; }, 250);
}

// 로그프레소 조회 → CSV → 리플레이 로드 (datetime-local + 테이블명)
// ============================================================
function _toLP(ts) {
  // 사용자가 직접 14자리로 적거나, 'yyyy-MM-dd HH:mm:ss' / 'yyyy-MM-ddTHH:mm:ss' 등 어떤
  // 포맷으로 적어도 숫자만 추출 후 14자리로 정규화.
  if (!ts) return '';
  const digits = String(ts).replace(/[^0-9]/g, '');
  if (digits.length < 8) return '';     // 최소 yyyyMMdd 는 있어야
  return digits.padEnd(14, '0').slice(0, 14);
}
// FAB/Layout 에 맞는 기본 테이블명 생성 — config.py 규칙과 동일.
// 예) M14A/A → oht_data_m14a,  M16A/BR → oht_data_m16br,  M16A/E → oht_data_m16e
function _logpressoTableFor(fab, prefix) {
  if (!fab) return '';
  const num = (String(fab).match(/\d+/) || [''])[0];
  return `oht_data_m${num}${(prefix || '').toLowerCase()}`;
}
// 현재 FAB/prefix 기준 테이블명을 input 에 주입. 사용자가 직접 수정했어도
// FAB 자체가 바뀌면 그에 맞는 새 기본값으로 교체 (덮어쓰기).
function syncLogpressoTable() {
  const el = document.getElementById('lp-table');
  if (!el) return;
  const next = _logpressoTableFor(currentFab, currentPrefix);
  if (next) {
    el.value = next;
    el.placeholder = next;
  }
}
async function logpressoLoad() {
  const fromEl = document.getElementById('lp-from');
  const toEl   = document.getElementById('lp-to');
  const tblEl  = document.getElementById('lp-table');
  const btn    = document.getElementById('lp-btn');
  const stat   = document.getElementById('lp-status');

  const from_dt = _toLP(fromEl.value);
  const to_dt   = _toLP(toEl.value);
  const table   = (tblEl.value || '').trim();
  if (!from_dt || !to_dt || !table) {
    alert('시작/끝 시각과 테이블명을 모두 입력하세요.\n예) oht_data_m16br');
    return;
  }
  if (from_dt >= to_dt) {
    alert('끝 시각이 시작 시각보다 늦어야 합니다.');
    return;
  }
  // 최대 조회 구간 30분 제한 — 서버 부하 방지
  // yyyyMMddHHmmss 파싱하여 분 차이 계산
  const _parse = s => new Date(
    +s.slice(0,4), +s.slice(4,6)-1, +s.slice(6,8),
    +s.slice(8,10), +s.slice(10,12), +s.slice(12,14));
  const _diffMin = (_parse(to_dt) - _parse(from_dt)) / 60000;
  if (_diffMin > 30) {
    alert(`조회 구간이 ${_diffMin.toFixed(1)}분입니다.\n30분 이내로 줄여주세요.`);
    return;
  }

  // 프론트 상태 리셋 (loadDate 와 동일 처리)
  obsHistory = []; jamEvents = []; _lastObsSampleTime = ''; _lastJamCount = 0;
  fullObsHistory = []; bottleneckCache = null;
  vehicleDisplay = {}; mapData.vehicles = [];
  ['spike-modal','bottleneck-modal','risk-modal'].forEach(id => {
    const el = document.getElementById(id);
    if (el) el.style.display = 'none';
  });

  btn.disabled = true;
  const t0 = performance.now();
  stat.textContent = '⏳ 조회 중...';
  document.getElementById('loading-msg').style.display = 'inline';
  try {
    const resp = await fetch('/api/logpresso/load', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({ from_dt, to_dt, table }),
    });
    const data = await resp.json();
    if (!resp.ok || data.error) {
      stat.textContent = `❌ ${data.error || resp.status}`;
      alert(`로그프레소 조회 실패: ${data.error || resp.status}`);
      return;
    }
    const lp = data.logpresso || {};
    stat.textContent = `✅ ${lp.rows?.toLocaleString?.() || '?'} 건  ${lp.elapsed_sec || '?'}s`;

    // 로드 결과 — loadDate 와 동일하게 후속 데이터 갱신
    document.getElementById('time-slider').max = data.total_frames || 100;
    updateDataSources(data.stats);
    if (!railGraph) await loadRailGraph();
    try {
      const starResp = await fetch('/api/star-history');
      starHistory = await starResp.json();
      drawQueueChart();
    } catch (e) {}
    try {
      const hidResp = await fetch('/api/hid-speeds');
      hidSpeeds = await hidResp.json();
      renderZoneList();
    } catch (e) {}
    try {
      const obsResp = await fetch('/api/obs-jam-history');
      fullObsHistory = await obsResp.json();
    } catch (e) {}
    console.log(`[로그프레소] 로드 완료 (${((performance.now()-t0)/1000).toFixed(1)}s):`, lp);
  } catch (e) {
    stat.textContent = `❌ ${e.message || e}`;
    alert(`요청 실패: ${e.message || e}`);
  } finally {
    btn.disabled = false;
    document.getElementById('loading-msg').style.display = 'none';
  }
}

      } else if (pct >= 70) {
        zoneAlerts.push(`<div style="color:#f59e0b;">&#9679; HID ${z.id} <b>주의</b> (${count}/${max})</div>`);
      }
    }
  }
  const alertEl = document.getElementById('p-zone-alerts');
  if (alertEl) {
    if (zoneAlerts.length > 0) {
      alertEl.innerHTML = `<div style="color:#9ca3af;margin-bottom:2px;">위험 Zone (${zoneAlerts.length}개):</div>` + zoneAlerts.join('');
    } else {
      alertEl.innerHTML = '';
    }
  }


  // ===== 레일 차단 이벤트 처리 (맵 표시 + 사이드바 로그 + 배너) =====
  const rcs = d.railCuts || [];
  mapData.railCuts = rcs;

  // 활성 차단 edge 집합 계산 (drawMap에서 재사용)
  const blockedEdges = {};  // "from_to" -> { rc, from, to }
  for (const rc of rcs) {
    const st = (rc.state || '').toUpperCase().trim();
    // 복구 이벤트 스킵 — 정확히 일치 (ABNORMAL이 NORMAL을 포함하는 버그 방지!)
    const isRestore = (st === 'NORMAL' || st === 'RESTORE' || st === 'CLEAR' || st === 'RECOVER' || st === 'OFF' || st === 'FREE');
    if (isRestore) continue;

    // affect_addrs 배열 전체 표시, 없으면 from/to만
    if (rc.affect_addrs && rc.affect_addrs.length > 0) {
      for (const aa of rc.affect_addrs) {
        blockedEdges[`${aa[0]}_${aa[1]}`] = { rc, from: aa[0], to: aa[1] };
      }
    } else if (rc.from_addr && rc.to_addr) {
      blockedEdges[`${rc.from_addr}_${rc.to_addr}`] = { rc, from: rc.from_addr, to: rc.to_addr };
    }
  }
  mapData.blockedEdges = blockedEdges;
  const activeCount = Object.keys(blockedEdges).length;

  // 사이드바 로그 (활성 ABNORMAL만 표시 — NORMAL은 맵에서 사라지는 것으로 충분)
  const activeOnly = rcs.filter(rc => {
    const st = (rc.state || '').toUpperCase().trim();
    // 정확히 일치로 필터 (ABNORMAL이 NORMAL을 포함하는 버그 방지!)
    return !(st === 'NORMAL' || st === 'RESTORE' || st === 'CLEAR' || st === 'RECOVER');
  });
  const rcLog = document.getElementById('railcut-log');
  if (activeOnly.length > 0) {
    let html = `<div style="color:#ef4444;font-weight:bold;margin-bottom:4px;">⛔ 활성 차단: ${activeCount}개 edge</div>`;
    html += activeOnly.slice(-5).reverse().map(rc => {
      return `<div class="ev-item" style="color:#ef4444;">⛔ ${rc.from_addr}→${rc.to_addr} [${rc.state}] (${rc.affect_count||0})</div>`;
    }).join('');
    rcLog.innerHTML = html;
  } else {
    rcLog.innerHTML = '<div style="color:#6b7280;font-size:11px;">없음</div>';
  }

  // 차량 위치 보간 업데이트
  const newVehicles = d.vehicles || [];
  for (const v of newVehicles) {
    if (vehicleDisplay[v.vid]) {
      vehicleDisplay[v.vid].tx = v.x;
      vehicleDisplay[v.vid].ty = v.y;
      vehicleDisplay[v.vid].state = v.state;
      vehicleDisplay[v.vid].isFull = v.isFull;
      vehicleDisplay[v.vid].velocity = v.velocity;
      vehicleDisplay[v.vid].currentNode = v.currentNode;
      vehicleDisplay[v.vid].nextNode = v.nextNode;
      vehicleDisplay[v.vid].destination = v.destination;
      vehicleDisplay[v.vid].ratio = v.ratio;
    } else {
      vehicleDisplay[v.vid] = {
        dx:v.x, dy:v.y, tx:v.x, ty:v.y,
        state:v.state, isFull:v.isFull, velocity:v.velocity,
        currentNode:v.currentNode, nextNode:v.nextNode,
        destination:v.destination, ratio:v.ratio
      };
    }
  }
  const activeIds = new Set(newVehicles.map(v=>v.vid));
  for (const vid in vehicleDisplay) {
    if (!activeIds.has(vid)) {
      vehicleDisplay[vid]._gone = (vehicleDisplay[vid]._gone||0) + 1;
      if (vehicleDisplay[vid]._gone > 5) delete vehicleDisplay[vid];
    } else {
      vehicleDisplay[vid]._gone = 0;
    }
  }
  mapData.vehicles = newVehicles;
  mapData.zoneCounts = d.zoneCounts || {};
  mapData.hotspots = d.hotspots || [];

  // 데드락 핫스팟 사이드바 갱신 + 데드락 위험 배지 오버라이드
  updateHotspotPanel(mapData.hotspots);

  // ===== 상단 통계 배지 업데이트 (공차운행/적재/OBS/정지/레일차단) =====
  let cntIdle = 0, cntLoaded = 0, cntObs = 0, cntStopped = 0, cntJam = 0;
  for (const v of newVehicles) {
    if (v.state === 6) cntObs++;
    else if (v.state === 7) cntJam++;
    else if (v.state === 2 || v.state === 8 || v.state === 9) cntStopped++;
    else if (v.isFull) cntLoaded++;
    else cntIdle++;
  }
  setText('cnt-idle', cntIdle);
  setText('cnt-loaded', cntLoaded);
  setText('cnt-obs', cntObs);
  setText('cnt-stopped', cntStopped + cntJam);  // 정지 + JAM 합산
  setText('cnt-railcut', activeCount);
}

let vehicleDisplay = {};

// 보간 애니메이션 루프 (60fps)
let lastAnimTime = 0;
function interpolateLoop(now) {
  const dt = now - lastAnimTime;
  lastAnimTime = now;
  const lerp = Math.min(1, dt * 0.008);
  let moved = false;
  for (const vid in vehicleDisplay) {
    const v = vehicleDisplay[vid];
    const ddx = v.tx - v.dx, ddy = v.ty - v.dy;
    const dist = Math.sqrt(ddx*ddx + ddy*ddy);
    if (dist > 200) { v.dx = v.tx; v.dy = v.ty; }
    else if (dist > 0.05) { v.dx += ddx * lerp; v.dy += ddy * lerp; moved = true; }
    else { v.dx = v.tx; v.dy = v.ty; }
  }
  // 차단 edge가 있으면 깜빡임 위해 계속 그리기
  const hasBlocked = showRailCut && mapData.blockedEdges && Object.keys(mapData.blockedEdges).length > 0;
  if (moved || hasBlocked) drawMap();
  requestAnimationFrame(interpolateLoop);
}
requestAnimationFrame(interpolateLoop);

async function loadRailGraph() {
  try {
    const resp = await fetch('/api/layout-graph');
    railGraph = await resp.json();
    mapData.bounds = railGraph.bounds;
    railCacheCanvas = null;
    console.log('레일 그래프 로드:', Object.keys(railGraph.nodes).length, '노드,', railGraph.edges.length, '엣지');
    drawMap();
  } catch(e) { console.error('레일 그래프 로드 실패:', e); }
}

function drawMap() {
  const rect = mapCanvas.parentElement.getBoundingClientRect();
  const w = Math.max(100, Math.floor(rect.width));
  const h = Math.max(100, Math.floor(rect.height));
  if (mapCanvas.width !== w || mapCanvas.height !== h) {
    mapCanvas.width = w; mapCanvas.height = h;
    railCacheCanvas = null;
  }

  const b = mapData.bounds || (railGraph && railGraph.bounds);
  if (!b) {
    mapCtx.fillStyle = '#0a0e17'; mapCtx.fillRect(0,0,w,h);
    mapCtx.fillStyle = '#374151'; mapCtx.font = '14px sans-serif'; mapCtx.textAlign = 'center';
    mapCtx.fillText('날짜를 선택하고 재생을 누르세요', w/2, h/2);
    return;
  }

  // 매 프레임 메인 캔버스 초기화 — pan 으로 drawImage 가 캔버스 전체를 덮지 못할 때 잔상(HID 라벨 고스트) 발생 방지.
  mapCtx.fillStyle = '#0a0e17';
  mapCtx.fillRect(0, 0, w, h);

  const dataW = b.max_x-b.min_x||1, dataH = b.max_y-b.min_y||1;
  const sc = Math.min((w-20)/dataW, (h-20)/dataH) * mapZoom;
  const ox = 10 + mapPanX + (w-20-dataW*sc)/2;
  const oy = 10 + mapPanY + (h-20-dataH*sc)/2;
  const toS = (x,y) => [ox+(x-b.min_x)*sc, oy+(y-b.min_y)*sc];

  // 1. 레일 배경 (캐시) — pan 은 캐시 key 에서 제외. 드래그 중에도 캐시 재사용.
  //    캐시 빌드 시 pan offset 만큼 여유 viewport 를 렌더 (드래그 여유)
  const CACHE_PAD = 800;
  const cacheKey = `${w}_${h}_${mapZoom.toFixed(3)}_${showZone?1:0}`;
  // pan 이 캐시 빌드 시점에서 많이 벗어나면 캐시 범위 밖이 비어 보이므로 재빌드
  const panDrift = railCacheCanvas
    ? Math.max(Math.abs(mapPanX - railCachePanX), Math.abs(mapPanY - railCachePanY))
    : 0;
  if (!railCacheCanvas || railCacheKey !== cacheKey || panDrift > CACHE_PAD * 0.5) {
    railCacheCanvas = document.createElement('canvas');
    // 캐시 캔버스는 viewport 보다 상하좌우 CACHE_PAD 만큼 크게 — 드래그로 노출되는 여유 영역까지 미리 렌더.
    railCacheCanvas.width = w + CACHE_PAD * 2;
    railCacheCanvas.height = h + CACHE_PAD * 2;
    railCachePanX = mapPanX;
    railCachePanY = mapPanY;
    const rc = railCacheCanvas.getContext('2d');
    rc.fillStyle = mapSettings.bgColor; rc.fillRect(0, 0, railCacheCanvas.width, railCacheCanvas.height);
    // 이후 그리기 좌표는 viewport 기준 → (+CACHE_PAD) 만큼 이동.
    rc.translate(CACHE_PAD, CACHE_PAD);

    if (railGraph) {
      const nodes = railGraph.nodes, edges = railGraph.edges;
      rc.strokeStyle = mapSettings.edgeColor;
      rc.lineWidth = Math.max(0.8, mapSettings.edgeWidth*mapZoom);
      rc.lineCap = 'round';
      rc.beginPath();
      for (let i=0; i<edges.length; i++) {
        const from=nodes[edges[i][0]], to=nodes[edges[i][1]];
        if (!from||!to) continue;
        const [fx,fy]=toS(from[0],from[1]), [tx,ty]=toS(to[0],to[1]);
        // 드래그 여유를 위해 컬링 범위를 CACHE_PAD 만큼 넓힘
        if ((fx<-CACHE_PAD&&tx<-CACHE_PAD)||(fx>w+CACHE_PAD&&tx>w+CACHE_PAD)||(fy<-CACHE_PAD&&ty<-CACHE_PAD)||(fy>h+CACHE_PAD&&ty>h+CACHE_PAD)) continue;
        rc.moveTo(fx,fy); rc.lineTo(tx,ty);
      }
      rc.stroke();

      if (mapZoom >= 3) {
        rc.fillStyle = mapSettings.nodeColor;
        const ns = Math.max(0.5, mapSettings.nodeSize);
        const half = ns / 2;
        for (const [nid, c] of Object.entries(nodes)) {
          const [sx,sy]=toS(c[0],c[1]);
          if (sx<-CACHE_PAD||sx>w+CACHE_PAD||sy<-CACHE_PAD||sy>h+CACHE_PAD) continue;
          rc.fillRect(sx-half, sy-half, ns, ns);
        }
      }

      if (showZone && railGraph.zones) {
        const nodes = railGraph.nodes;
        const zoneLineW = Math.max(1.5, 2*mapZoom);

        const visibleZones = [];
        for (const z of railGraph.zones) {
          const [cx,cy] = toS(z.cx, z.cy);
          if (cx >= -CACHE_PAD && cx <= w+CACHE_PAD && cy >= -CACHE_PAD && cy <= h+CACHE_PAD) {
            visibleZones.push({...z, _sx: cx, _sy: cy});
          }
        }

        if (mapZoom >= 2) {
          for (const z of visibleZones) {
            rc.strokeStyle = '#00ff88';
            rc.lineWidth = zoneLineW;
            rc.globalAlpha = 0.4;
            rc.setLineDash([]);
            for (const lane of z.inLanes) {
              const from=nodes[lane.from], to=nodes[lane.to];
              if (!from||!to) continue;
              const [fx,fy]=toS(from[0],from[1]), [tx,ty]=toS(to[0],to[1]);
              rc.beginPath(); rc.moveTo(fx,fy); rc.lineTo(tx,ty); rc.stroke();
            }
            rc.setLineDash([3*mapZoom, 2*mapZoom]);
            for (const lane of z.outLanes) {
              const from=nodes[lane.from], to=nodes[lane.to];
              if (!from||!to) continue;
              const [fx,fy]=toS(from[0],from[1]), [tx,ty]=toS(to[0],to[1]);
              rc.beginPath(); rc.moveTo(fx,fy); rc.lineTo(tx,ty); rc.stroke();
            }
          }
          rc.setLineDash([]);
          rc.globalAlpha = 1.0;
        }

        const fontSize = Math.max(7, Math.min(11, 5*mapZoom));
        rc.font = `bold ${fontSize}px sans-serif`;
        rc.textAlign = 'center';
        for (const z of visibleZones) {
          const cx = z._sx, cy = z._sy;
          const label = `HID ${z.id}`;
          const tw = rc.measureText(label).width;
          rc.fillStyle = 'rgba(0,0,0,0.7)';
          rc.fillRect(cx-tw/2-2, cy-fontSize/2-1, tw+4, fontSize+2);
          rc.fillStyle = '#00ffcc';
          rc.fillText(label, cx, cy + fontSize*0.3);
        }
      }
    }
    railCacheKey = cacheKey;
  }
  // 캐시는 railCachePanX/Y 시점 좌표계 + (CACHE_PAD, CACHE_PAD) 오프셋 으로 그려져 있음.
  const cacheDX = mapPanX - railCachePanX;
  const cacheDY = mapPanY - railCachePanY;
  mapCtx.drawImage(railCacheCanvas, cacheDX - CACHE_PAD, cacheDY - CACHE_PAD);

  // ===== 1.5. 레일 차단 edge 렌더링 (기존 모양 유지, 작고 투명하게) =====
  const blockedEdges = mapData.blockedEdges || {};
  const blockedKeys = Object.keys(blockedEdges);
  if (showRailCut && blockedKeys.length > 0 && railGraph) {
    const nodes = railGraph.nodes;
    const blink = (Date.now() / 600) % 2 < 1;

    // (A) 차단 edge — 빨간 점선 (얇게 + 투명)
    mapCtx.strokeStyle = '#ef4444';
    mapCtx.lineWidth = Math.max(1, 1.2 * mapZoom);
    mapCtx.setLineDash([5 * mapZoom, 3 * mapZoom]);
    mapCtx.globalAlpha = blink ? 0.45 : 0.18;
    mapCtx.lineCap = 'round';
    mapCtx.beginPath();
    const visibleBlocked = [];
    for (const key of blockedKeys) {
      const be = blockedEdges[key];
      const from = nodes[be.from], to = nodes[be.to];
      if (!from || !to) continue;
      const [fx, fy] = toS(from[0], from[1]);
      const [tx, ty] = toS(to[0], to[1]);
      if ((fx<-50 && tx<-50) || (fx>w+50 && tx>w+50) || (fy<-50 && ty<-50) || (fy>h+50 && ty>h+50)) continue;
      mapCtx.moveTo(fx, fy); mapCtx.lineTo(tx, ty);
      visibleBlocked.push({ be, fx, fy, tx, ty });
    }
    mapCtx.stroke();
    mapCtx.setLineDash([]);

    // (B) edge 중앙에 ⛔ 마커 — 빨간 원 + 흰 테두리 + 흰 X (작고 투명하게)
    const markerR = Math.max(2.5, 3 * mapZoom);
    const xSize = Math.max(1.8, 2 * mapZoom);
    for (const vb of visibleBlocked) {
      const mx = (vb.fx + vb.tx) / 2, my = (vb.fy + vb.ty) / 2;

      // 빨간 원 배경 (깜빡임, 투명)
      mapCtx.globalAlpha = blink ? 0.6 : 0.25;
      mapCtx.fillStyle = '#ef4444';
      mapCtx.beginPath();
      mapCtx.arc(mx, my, markerR, 0, 6.283);
      mapCtx.fill();

      // 흰 테두리 (얇게)
      mapCtx.strokeStyle = '#fff';
      mapCtx.lineWidth = 0.8;
      mapCtx.stroke();

      // 흰 X 마커 (얇게)
      mapCtx.globalAlpha = blink ? 0.75 : 0.3;
      mapCtx.strokeStyle = '#fff';
      mapCtx.lineWidth = Math.max(0.8, 1 * mapZoom);
      mapCtx.beginPath();
      mapCtx.moveTo(mx - xSize, my - xSize); mapCtx.lineTo(mx + xSize, my + xSize);
      mapCtx.moveTo(mx + xSize, my - xSize); mapCtx.lineTo(mx - xSize, my + xSize);
      mapCtx.stroke();
    }
    mapCtx.globalAlpha = 1.0;
  }

  // 2. 차량 (보간된 위치) — 카테고리별 색상 + 도형 모양 (설정 가능: ⚙ 설정)
  const r = Math.max(1.5, Math.min(8, mapSettings.vehicleRadius*mapZoom));
  const shape = mapSettings.vehicleShape || 'circle';
  for (const vid in vehicleDisplay) {
    const v = vehicleDisplay[vid];
    const [sx,sy] = toS(v.dx, v.dy);
    if (sx<-5||sx>w+5||sy<-5||sy>h+5) continue;

    mapCtx.beginPath();
    if (shape === 'triangle') {
      // 위로 향한 정삼각형 (외접원 반경 r)
      mapCtx.moveTo(sx,        sy - r);
      mapCtx.lineTo(sx + r*0.866, sy + r*0.5);
      mapCtx.lineTo(sx - r*0.866, sy + r*0.5);
      mapCtx.closePath();
    } else if (shape === 'square') {
      // 변 길이를 원 시각면적과 비슷하게: 한 변 ≈ r*sqrt(π) ≈ 1.77r
      const s = r * 1.77;
      mapCtx.rect(sx - s/2, sy - s/2, s, s);
    } else if (shape === 'diamond') {
      mapCtx.moveTo(sx,     sy - r);
      mapCtx.lineTo(sx + r, sy);
      mapCtx.lineTo(sx,     sy + r);
      mapCtx.lineTo(sx - r, sy);
      mapCtx.closePath();
    } else {
      // circle (기본)
      mapCtx.arc(sx, sy, r, 0, 6.283);
    }

    let color;
    if (v.state === 6) color = mapSettings.colorObs;                                                   // OBS
    else if (v.state === 7) color = mapSettings.colorJam;                                              // JAM
    else if (v.state === 2 || v.state === 8 || v.state === 9) color = mapSettings.colorStop;           // 정지
    else if (v.isFull) color = mapSettings.colorLoaded;                                                // 적재
    else color = mapSettings.colorEmpty;                                                               // 공차
    mapCtx.fillStyle = color;
    mapCtx.fill();
  }

  // 2.4. 선택된 OHT 강조 (노란 링 + 라벨)
  if (selectedOHT && vehicleDisplay[selectedOHT]) {
    const v = vehicleDisplay[selectedOHT];
    const [sx,sy] = toS(v.dx, v.dy);
    const ringR = Math.max(8, r * 4);
    mapCtx.beginPath();
    mapCtx.arc(sx, sy, ringR, 0, 6.283);
    mapCtx.strokeStyle = '#ffff00';
    mapCtx.lineWidth = 2.5;
    mapCtx.setLineDash([6, 3]);
    mapCtx.stroke();
    mapCtx.setLineDash([]);
    // 라벨
    const labelFS = Math.max(11, 12*Math.min(mapZoom, 2));
    mapCtx.font = `bold ${labelFS}px sans-serif`;
    const label = selectedOHT;
    const tw = mapCtx.measureText(label).width;
    const lx = sx, ly = sy - ringR - 8;
    mapCtx.fillStyle = 'rgba(0,0,0,0.85)';
    mapCtx.fillRect(lx-tw/2-5, ly-labelFS-2, tw+10, labelFS+6);
    mapCtx.fillStyle = '#ffff00';
    mapCtx.textAlign = 'center';

const OBS_HISTORY_MAX = 60;       // 최근 60샘플 보관
const OBS_SPIKE_SIGMA = 1.5;      // 평균 + 1.5σ를 spike로 판정
let obsHistory = [];              // [{time, obs}, ...]
let jamEvents = [];               // [{time, obs, jam}] — 재생 누적 (호환성용)
let fullObsHistory = [];          // 전체 데이터 시계열 [{time, obs, jam}, ...] — 데이터 로드 시 1회 채움
let currentPlayTime = '';         // 현재 재생 시각 (HH:MM:SS) — 차트의 노란 세로선 위치
let _lastObsSampleTime = '';
let _lastJamCount = 0;

function trackOBSHistory(obs, timeStr, jamCount) {
  if (!timeStr || timeStr === _lastObsSampleTime) return;
  _lastObsSampleTime = timeStr;
  obsHistory.push({ time: timeStr, obs: obs, jam: jamCount });
  if (obsHistory.length > OBS_HISTORY_MAX) obsHistory.shift();

  // JAM 발생 (state=7 차량이 0→1 이상으로 새로 등장한 시점만)
  if (jamCount > 0 && _lastJamCount === 0) {
    jamEvents.push({ time: timeStr, obs: obs, jam: jamCount });
    if (jamEvents.length > 50) jamEvents.shift();
  }
  _lastJamCount = jamCount;
}

function updateSpikePanel() {
  // 현재 OBS = 재생 시점 (최신)
  const cur = obsHistory.length ? obsHistory[obsHistory.length - 1].obs : '-';

// 위험 레벨 이력 모달 (정상 제외)
// ============================================================
// macro_predictor 임계값과 동일
const RISK_THRESHOLDS = { WARNING: 180, DANGER: 200, CRITICAL: 300 };

function computeRiskLevel(avg_obs, trend) {
  if (avg_obs >= RISK_THRESHOLDS.CRITICAL) return 'CRITICAL';
  if (avg_obs >= RISK_THRESHOLDS.DANGER) return 'DANGER';
  if (avg_obs >= RISK_THRESHOLDS.WARNING) return 'WARNING';
  if (trend > 10) return 'WATCH';
  return 'NORMAL';
}

function buildRiskHistory() {
  // fullObsHistory 분당 데이터에서 위험 레벨 시계열 생성
  // 같은 레벨이 연속되면 한 구간으로 묶음
  if (!fullObsHistory.length) return [];
  const segments = [];
  let curSeg = null;
  for (let i = 0; i < fullObsHistory.length; i++) {
    const recent = fullObsHistory.slice(Math.max(0, i - 9), i + 1);
    const avg = recent.reduce((s, h) => s + h.obs, 0) / recent.length;
    let trend = 0;
    if (recent.length >= 5) {
      const half = Math.floor(recent.length / 2);
      const f = recent.slice(0, half).reduce((s, h) => s + h.obs, 0) / half;
      const sec = recent.slice(half).reduce((s, h) => s + h.obs, 0) / (recent.length - half);
      trend = sec - f;
    }
    const level = computeRiskLevel(avg, trend);
    if (level === 'NORMAL') {
      if (curSeg) { segments.push(curSeg); curSeg = null; }
    } else {
      if (!curSeg || curSeg.level !== level) {
        if (curSeg) segments.push(curSeg);
        curSeg = { level: level, start: fullObsHistory[i].time, end: fullObsHistory[i].time, peak: fullObsHistory[i].obs };
      } else {
        curSeg.end = fullObsHistory[i].time;
        if (fullObsHistory[i].obs > curSeg.peak) curSeg.peak = fullObsHistory[i].obs;
      }
    }
  }
  if (curSeg) segments.push(curSeg);
  return segments;
}

    peak: e.peak,
    kind: e.kind,
    addrs: e.addrs,
    peakTime: e.peakTime,
    active: !!activeHotspotEvents[e.key],
  }));

  // ===== 2차 소스(보조): 기존 OBS 기반 segments (호환성 — WATCH 전용으로 사용) =====
  const segments = buildRiskHistory();
  for (const s of segments) {
    mapped.push({ level: s.level, start: s.start, end: s.end, peak: s.peak, kind: 'obs', addrs: [], active: false });
  }

  mapped.forEach(s => { if (counts[s.level] !== undefined) counts[s.level]++; });
  setText('r-cnt-watch', counts.WATCH);
  setText('r-cnt-warning', counts.WARNING);
  setText('r-cnt-danger', counts.DANGER);
  setText('r-cnt-critical', counts.CRITICAL);

  const lvlMeta = {
    WATCH:    { label: '관찰',  color: '#60a5fa', icon: '👁️' },
    WARNING:  { label: '주의',  color: '#f59e0b', icon: '⚠️' },
    DANGER:   { label: '위험',  color: '#ef4444', icon: '🔴' },
    CRITICAL: { label: '심각',  color: '#dc2626', icon: '🚨' },
  };

  const logEl = document.getElementById('risk-log');
  if (mapped.length === 0) {
    logEl.innerHTML = '<span style="color:#22c55e;">정상 외 이벤트 없음 (전 시간 안정)</span>';
    return;
  }
  // 시간순 정렬 — 빠른 시간 먼저 (start 시각 오름차순)
  const timeToSec = (s) => {
    const p = (s || '').split(':');
    return (parseInt(p[0])||0)*3600 + (parseInt(p[1])||0)*60 + (parseInt(p[2])||0);
  };
  const sorted = mapped.slice().sort((a, b) => timeToSec(a.start) - timeToSec(b.start));

  logEl.innerHTML = sorted.map(s => {
    const m = lvlMeta[s.level] || { label: s.level, color: '#888', icon: '•' };
    const isHotspot = s.kind === 'cluster' || s.kind === 'chain';
    const timeSpan = s.start === s.end ? s.start : `${s.start}~${s.end}`;
    const live = s.active ? ' 🔴진행중' : '';
    if (isHotspot) {
      const addrs = s.addrs.length > 4
        ? `${s.addrs[0]}~${s.addrs[s.addrs.length-1]}`
        : s.addrs.join(',');
      const icon = s.kind === 'chain' ? '🔗' : '●';
      return `<div class="jam-entry" style="padding:6px 8px;margin-bottom:4px;background:#1a1a2e;border-left:3px solid ${m.color};border-radius:3px;">${m.icon} <b style="color:${m.color};">${m.label}</b>${live} — ${icon} addr ${addrs} peak ${s.peak}대 · ⏱ ${timeSpan}${s.peakTime ? ` (peak@${s.peakTime})` : ''}</div>`;
    } else {
      // OBS 기반 구간
      return `<div class="jam-entry" title="${m.label}(${s.level}) — 최대 OBS ${s.peak}">${m.icon} <b style="color:${m.color};">${m.label}</b> ${timeSpan} (OBS 피크 ${s.peak})</div>`;
    }
  }).join('');
}

function renderSpikeModal() {
  // 전체 데이터 우선 사용. 없으면 재생 누적 fallback.
  const src = fullObsHistory.length ? fullObsHistory : obsHistory;
  if (!src.length) return;
  const vals = src.map(h => h.obs);
  const cur = vals[vals.length - 1];
  const avg = Math.round(vals.reduce((a, b) => a + b, 0) / vals.length);
  const max = Math.max(...vals);
  // JAM 발생 시점 (전체 데이터 기준)
  const allJam = src.filter(h => h.jam && h.jam > 0);
  setText('m-obs-now', cur);
  setText('m-obs-avg', avg);
  setText('m-obs-max', max);
  setText('m-spike-cnt', allJam.length);

  // JAM 이벤트 표시 (시간 내림차순)
  const logEl = document.getElementById('spike-log');
  const timeToSec = (s) => {
    const p = (s || '').split(':');
    return (parseInt(p[0])||0)*3600 + (parseInt(p[1])||0)*60 + (parseInt(p[2])||0);
  };

  // 연속 구간(인접 분) 그룹핑: jam>=1 인 분들을 이어붙여 세그먼트 구성.
  // 각 세그먼트에서 durationMin(지속 분 수), maxJam(최대 JAM 수),
  // sustained10min(연속해서 jam>=10 이 유지된 최대 분 수) 을 계산.
  const chrono = src.slice().sort((a, b) => timeToSec(a.time) - timeToSec(b.time));
  const segments = [];
  let curSeg = null;
  for (const h of chrono) {
    const hSec = timeToSec(h.time);
    if (h.jam && h.jam > 0) {
      if (curSeg && (hSec - curSeg.lastSec) <= 75) {
        curSeg.entries.push(h);
        curSeg.maxJam = Math.max(curSeg.maxJam, h.jam);
        curSeg.lastSec = hSec;
      } else {
        curSeg = { entries: [h], maxJam: h.jam, lastSec: hSec };
        segments.push(curSeg);
      }
    } else {
      curSeg = null;
    }
  }
  const segByTime = {};
  for (const seg of segments) {
    seg.durationMin = seg.entries.length;
    let maxRun10 = 0, runLen = 0;
    for (const h of seg.entries) {
      if (h.jam >= 10) { runLen++; if (runLen > maxRun10) maxRun10 = runLen; }
      else runLen = 0;
    }
    seg.sustained10min = maxRun10;
    seg.startTime = seg.entries[0].time;
    seg.endTime = seg.entries[seg.entries.length - 1].time;
    for (const h of seg.entries) segByTime[h.time] = seg;
  }

  const sorted = allJam.slice().sort((a, b) => timeToSec(b.time) - timeToSec(a.time));
  if (sorted.length === 0) {
    logEl.innerHTML = '<span style="color:#22c55e;">JAM 발생 없음 (시스템 안정)</span>';
  } else {
    logEl.innerHTML = sorted.map(e => {
      const seg = segByTime[e.time] || { durationMin: 1, maxJam: e.jam, sustained10min: e.jam>=10?1:0, startTime: e.time, endTime: e.time };
      // 판정: 지속 시간 + 최대 JAM 수 + 10대 이상 유지 기반
      let status, statusColor, icon, tooltip;
      if (seg.sustained10min >= 4) {
        status = '데드락 발생';
        statusColor = '#ef4444';
        icon = '🚨';
        tooltip = `데드락 발생: ${seg.startTime}~${seg.endTime} 구간 ${seg.durationMin}분 지속, 최대 ${seg.maxJam}대. 10대 이상이 ${seg.sustained10min}분 연속 유지 = 풀리지 않는 잠금. 즉시 운영 개입 필요.`;
      } else if (seg.durationMin >= 2 || e.jam >= 5) {
        status = '주의 (다수 정지)';
        statusColor = '#f59e0b';
        icon = '⚠️';
        tooltip = `주의: ${seg.startTime}~${seg.endTime} 구간 ${seg.durationMin}분 지속, 최대 ${seg.maxJam}대. 다수 차량 동시 정지이나 자동 회복 범위. 회복 양상 확인 필요.`;
      } else {
        status = '데드락 미발생 (경고)';
        statusColor = '#fbbf24';
        icon = '🟡';
        tooltip = `데드락 미발생: 1분 단발 ${e.jam}대 JAM 후 즉시 회복. 단발성 경고. 시스템 정상.`;
      }
      return `<div class="jam-entry" title="${tooltip}">${icon} <b>${e.time}</b> — JAM (OBS ${e.obs}, ${e.jam}대) <span style="color:${statusColor};font-weight:bold;">— ${status}</span></div>`;
    }).join('');
  }
  drawSpikeChart();
}

function drawSpikeChart() {
  const canvas = document.getElementById('spike-chart');
  const ctx = canvas.getContext('2d');
  const dpr = window.devicePixelRatio || 1;
  const rect = canvas.getBoundingClientRect();
  const w = Math.max(400, Math.floor(rect.width));
  const h = Math.max(200, Math.floor(rect.height));
  canvas.width = w * dpr;
  canvas.height = h * dpr;
  canvas.style.width = w + 'px';
  canvas.style.height = h + 'px';
  ctx.scale(dpr, dpr);
  ctx.fillStyle = '#0a0e17';
  ctx.fillRect(0, 0, w, h);

  // 전체 데이터 우선. 없으면 재생 누적 fallback.
  const data = fullObsHistory.length ? fullObsHistory : obsHistory;
  if (data.length < 2) {
    ctx.fillStyle = '#6b7280';
    ctx.font = '12px sans-serif';
    ctx.textAlign = 'center';
    ctx.fillText('데이터 로드 대기 중...', w / 2, h / 2);
    return;
  }

  const vals = data.map(h => h.obs);
  const mean = vals.reduce((a, b) => a + b, 0) / vals.length;
  const variance = vals.reduce((s, v) => s + (v - mean) ** 2, 0) / vals.length;
  const threshold = mean + OBS_SPIKE_SIGMA * Math.sqrt(variance);
  const yMax = Math.max(...vals, threshold) * 1.1;
  const yMin = Math.min(...vals) * 0.9;

  const px = 40, py = 20;
  const pw = w - px - 10, ph = h - py - 30;

  // JAM 발생 시점 빨간 세로선 (진짜 데드락만 강조)
  data.forEach((h, i) => {
    if (h.jam && h.jam > 0) {
      const x = px + (i / Math.max(1, data.length - 1)) * pw;
      ctx.strokeStyle = '#ef4444';
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.moveTo(x, py);
      ctx.lineTo(x, py + ph);
      ctx.stroke();
      // JAM 라벨
      ctx.fillStyle = '#ef4444';
      ctx.font = 'bold 10px sans-serif';
      ctx.textAlign = 'center';
      ctx.fillText('JAM', x, py - 4);
    }
  });

  // 임계선
  const thY = py + ph - ((threshold - yMin) / (yMax - yMin)) * ph;
  ctx.strokeStyle = '#ef4444';
  ctx.setLineDash([5, 3]);
  ctx.lineWidth = 1;
  ctx.beginPath();
  ctx.moveTo(px, thY);
  ctx.lineTo(px + pw, thY);
  ctx.stroke();
  ctx.setLineDash([]);
  ctx.fillStyle = '#ef4444';
  ctx.font = '10px sans-serif';
  ctx.textAlign = 'left';
  ctx.fillText(`임계 ${Math.round(threshold)}`, px + 4, thY - 3);

  // 평균선
  const avgY = py + ph - ((mean - yMin) / (yMax - yMin)) * ph;
  ctx.strokeStyle = '#60a5fa';
  ctx.setLineDash([2, 2]);
  ctx.beginPath();
  ctx.moveTo(px, avgY);
  ctx.lineTo(px + pw, avgY);
  ctx.stroke();
  ctx.setLineDash([]);
  ctx.fillStyle = '#60a5fa';
  ctx.fillText(`평균 ${Math.round(mean)}`, px + 4, avgY - 3);

  // 시계열 라인 (전체 데이터)
  ctx.strokeStyle = '#22c55e';
  ctx.lineWidth = 1.5;
  ctx.beginPath();
  data.forEach((h, i) => {
    const x = px + (i / (data.length - 1)) * pw;
    const y = py + ph - ((h.obs - yMin) / (yMax - yMin)) * ph;
    if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
  });
  ctx.stroke();

  // 현재 PLAY 위치 표시 (노란 세로선)
  const curTime = (typeof currentPlayTime === 'string') ? currentPlayTime : '';
  if (curTime) {
    const curHM = curTime.substring(0, 5);  // "HH:MM"
    const curIdx = data.findIndex(h => h.time === curHM);
    if (curIdx >= 0) {
      const x = px + (curIdx / Math.max(1, data.length - 1)) * pw;
      ctx.strokeStyle = '#fbbf24';
      ctx.lineWidth = 2;
      ctx.setLineDash([3, 2]);
      ctx.beginPath();
      ctx.moveTo(x, py);
      ctx.lineTo(x, py + ph);
      ctx.stroke();
      ctx.setLineDash([]);
      ctx.fillStyle = '#fbbf24';
      ctx.font = 'bold 10px sans-serif';
      ctx.textAlign = 'center';
      ctx.fillText('▼ 현재', x, py + ph + 12);
    }
  }

  // Y축 라벨
  ctx.fillStyle = '#6b7280';
  ctx.font = '10px sans-serif';
  ctx.textAlign = 'right';
  ctx.fillText(Math.round(yMax), px - 4, py + 8);
  ctx.fillText(Math.round(yMin), px - 4, py + ph);

  // X축 (첫/끝 시간)
  ctx.textAlign = 'left';
  ctx.fillText(data[0].time, px, h - 8);
  ctx.textAlign = 'right';
  ctx.fillText(data[data.length - 1].time, px + pw, h - 8);
}

const stateNames = {1:'운행',2:'정지',3:'가속',4:'감속',5:'커브',6:'OBS',7:'JAM',8:'HT정지',9:'타임아웃'};
const stateClasses = {1:'running',2:'stopped',3:'running',4:'running',5:'running',6:'obs',7:'jam',8:'stopped',9:'stopped'};

function renderOHTList() {
  const el = document.getElementById('ohtList');
  const search = (document.getElementById('ohtSearch').value||'').toUpperCase();
  let vehicles = Object.values(vehicleDisplay);
  if (ohtFilter === 'running') vehicles = vehicles.filter(v => v.state === 1 && !v.isFull);
  else if (ohtFilter === 'loaded') vehicles = vehicles.filter(v => v.isFull);
  else if (ohtFilter === 'stopped') vehicles = vehicles.filter(v => v.state === 2 || v.state === 8);
  else if (ohtFilter === 'jam') vehicles = vehicles.filter(v => v.state === 7 || v.state === 6);
  if (search) vehicles = vehicles.filter(v => (v._vid||'').includes(search));
  const pri = {7:0, 6:1, 8:2, 9:2, 2:3};
  vehicles.sort((a,b) => (pri[a.state]??5) - (pri[b.state]??5));
  const max = 200;
  let html = '';
  for (let i = 0; i < Math.min(vehicles.length, max); i++) {
    const v = vehicles[i];
    const vid = v._vid || '?';
    const cls = stateClasses[v.state] || 'stopped';
    const sn = stateNames[v.state] || '?';
    const loadTxt = v.isFull ? ' [적재]' : '';
    const vel = (typeof v.velocity === 'number') ? v.velocity : null;
    let velTxt, velCls;
    if (vel === null) { velTxt = '- (측정 불가)'; velCls = '#6b7280'; }
    else if (vel === 0) { velTxt = '0 m/min'; velCls = '#ef4444'; }
    else if (vel < 30) { velTxt = vel.toFixed(1) + ' m/min'; velCls = '#f59e0b'; }
    else { velTxt = vel.toFixed(1) + ' m/min'; velCls = '#22c55e'; }
    const isSel = selectedOHT === vid;
    const sel = isSel ? ' style="background:#1a3a5e;border-left-color:#ffff00;"' : '';
    html += `<div class="ritem ${cls}" ondblclick="selectOHT('${vid}')"${sel}>`;
    html += `<div class="rh"><span class="rid">${vid}</span><span class="rst">${sn}${loadTxt}</span></div>`;
    html += `<div class="rv" style="color:${velCls};">속도: ${velTxt}</div>`;
    if (isSel) {
      // 선택 시 상세 정보 표시
      const x = (typeof v.tx === 'number') ? v.tx.toFixed(1) : '-';
      const y = (typeof v.ty === 'number') ? v.ty.toFixed(1) : '-';
      const cn = v.currentNode ?? '-';
      const nn = v.nextNode ?? '-';
      const dest = v.destination || '-';
      const rt = (typeof v.ratio === 'number') ? (v.ratio*100).toFixed(0)+'%' : '-';
      html += `<div class="rdetail">`;
      html += `<div>📍 좌표: (${x}, ${y})</div>`;
      html += `<div>🔗 Edge: ${cn} → ${nn} (${rt})</div>`;
      html += `<div>🎯 목적지: ${dest}</div>`;
      html += `</div>`;
    }
    html += `</div>`;
  }
  if (vehicles.length > max) html += `<div style="color:#6b7280;font-size:10px;padding:4px;">...외 ${vehicles.length-max}대</div>`;
  if (!vehicles.length) html = '<div style="color:#6b7280;padding:8px;">차량 없음</div>';
  el.innerHTML = html;
}

function renderZoneList() {
  const el = document.getElementById('zoneList');
  const search = (document.getElementById('zoneSearch').value||'').toUpperCase();
  if (!railGraph || !railGraph.zones) {
    el.innerHTML = '<div style="color:#6b7280;padding:8px;">데이터 로드 필요</div>';
    return;
  }
  const zc = mapData.zoneCounts || {};
  let zones = railGraph.zones.map(z => {
    const count = zc[z.id] || 0;
    const max = z.max || 37;
    const pct = Math.round(count / max * 100);
    const status = count >= max ? 'full' : pct >= 70 ? 'precaution' : 'normal';
    return { ...z, count, pct, status };
  });
  if (zoneFilter !== 'all') zones = zones.filter(z => z.status === zoneFilter);
  if (search) zones = zones.filter(z => (z.name||'').toUpperCase().includes(search) || String(z.id).includes(search));
  // 속도 정보 결합
  zones.forEach(z => {
    const sp = hidSpeeds[z.id] || hidSpeeds[String(z.id)];
    z.speed = sp ? sp.avg_speed : null;
  });
  // 정렬: 차량 수(count) 많은 순 = OHT 많이 지나는 Zone이 위로
  // 같은 count면 status 우선(포화→주의→정상)
  const pri = {full:0, precaution:1, normal:2};
  zones.sort((a,b) => {
    if (b.count !== a.count) return b.count - a.count;
    return (pri[a.status]??2) - (pri[b.status]??2);
  });
  const max = 200;
  let html = '';
  for (let i = 0; i < Math.min(zones.length, max); i++) {
    const z = zones[i];
    const stTxt = z.status==='full' ? '포화' : z.status==='precaution' ? '주의' : '정상';
    const stColor = z.status==='full' ? '#ef4444' : z.status==='precaution' ? '#f59e0b' : '#22c55e';
    const sel = selectedZone && selectedZone.id === z.id ? ' style="background:#1a3a5e;border-left-color:#ffff00;"' : '';
    // 속도 표시 (색상 구분)
    let spTxt, spColor;
    if (z.speed === null) { spTxt = '-'; spColor = '#6b7280'; }
    else if (z.speed === 0) { spTxt = '0 m/min'; spColor = '#ef4444'; }
    else if (z.speed < 30) { spTxt = z.speed.toFixed(1) + ' m/min'; spColor = '#f59e0b'; }
    else { spTxt = z.speed.toFixed(1) + ' m/min'; spColor = '#22c55e'; }
    html += `<div class="ritem zone-${z.status}" ondblclick="selectZone(${z.id})"${sel}>`;
    html += `<div class="rh"><span class="rid">HID ${z.id}</span><span class="rst" style="background:${stColor};color:#000;">${z.count}/${z.max} ${stTxt}</span></div>`;
    html += `<div style="color:#9ca3af;font-size:10px;margin-top:2px;">${z.name} | ${z.bay}</div>`;
    html += `<div class="rv" style="color:${spColor};">속도: ${spTxt}</div>`;
    html += `<div class="zone-bar"><div class="zone-bar-fill" style="width:${z.pct}%;"></div></div>`;
    html += `</div>`;
  }
  if (zones.length > max) html += `<div style="color:#6b7280;font-size:10px;padding:4px;">...외 ${zones.length-max}개</div>`;
  if (!zones.length) html = '<div style="color:#6b7280;padding:8px;">Zone 없음</div>';
  el.innerHTML = html;
}

    const dataW = b.max_x - b.min_x, dataH = b.max_y - b.min_y;

}

function selectZone(zoneId) {
  if (!railGraph || !railGraph.zones) return;
  if (selectedZone && selectedZone.id === zoneId) {
    selectedZone = null;
    railCacheCanvas = null;
    drawMap();
    renderZoneList();
    return;
  }
  const z = railGraph.zones.find(zz => zz.id === zoneId);
  if (!z) return;
  selectedZone = z;
  const nodes = railGraph.nodes;
  let minX=Infinity, maxX=-Infinity, minY=Infinity, maxY=-Infinity;
  for (const lane of [...z.inLanes, ...z.outLanes]) {
    const from=nodes[lane.from], to=nodes[lane.to];
    if (from) { minX=Math.min(minX,from[0]); maxX=Math.max(maxX,from[0]); minY=Math.min(minY,from[1]); maxY=Math.max(maxY,from[1]); }
    if (to) { minX=Math.min(minX,to[0]); maxX=Math.max(maxX,to[0]); minY=Math.min(minY,to[1]); maxY=Math.max(maxY,to[1]); }
  }
  if (minX < Infinity) {
    const b = mapData.bounds || railGraph.bounds;
    const rect = mapCanvas.parentElement.getBoundingClientRect();
    const cw = Math.max(100, Math.floor(rect.width));
    const ch = Math.max(100, Math.floor(rect.height));
    const zoneCx = (minX + maxX) / 2, zoneCy = (minY + maxY) / 2;
    const zoneW = Math.max(maxX - minX, 50) + 100;
    const zoneH = Math.max(maxY - minY, 50) + 100;
    const dataW = b.max_x - b.min_x, dataH = b.max_y - b.min_y;
    const baseScale = Math.min((cw-20)/dataW, (ch-20)/dataH);
    mapZoom = Math.max(2, Math.min(12, Math.min(cw*0.6/zoneW, ch*0.6/zoneH) / baseScale));
    const sc = baseScale * mapZoom;
    mapPanX = (cw/2) - 10 - (cw-20-dataW*sc)/2 - (zoneCx - b.min_x)*sc;
    mapPanY = (ch/2) - 10 - (ch-20-dataH*sc)/2 - (zoneCy - b.min_y)*sc;
  }
  railCacheCanvas = null;
  drawMap();
  renderZoneList();
}

// 데드락 핫스팟 패널 업데이트 + 이력 로그 + 위험 배지 오버라이드

let hotspotHistory = [];          // 전체 재생 누적 이벤트 (분 단위 요약)
let activeHotspotEvents = {};     // 현재 진행 중인 이벤트 (key → event)
let _lastHotspotLogTime = '';

function _hotspotKey(hs) {
  // 같은 핫스팟을 추적하는 key — 주소 범위 기반
  const a = hs.addrs || [];
  if (a.length === 0) return hs.kind + ':unknown';
  return hs.kind + ':' + Math.min(...a) + '-' + Math.max(...a);
}

function _logHotspots(hotspots, timeStr) {
  if (!timeStr || timeStr === _lastHotspotLogTime) return;
  _lastHotspotLogTime = timeStr;

  const now = Date.now();
  const currentKeys = new Set();

  for (const hs of hotspots) {
    const k = _hotspotKey(hs);
    currentKeys.add(k);
    if (activeHotspotEvents[k]) {
      // 진행 중 이벤트 갱신 — peak 업데이트
      const ev = activeHotspotEvents[k];
      ev.end = timeStr;
      if (hs.stopped > ev.peak) { ev.peak = hs.stopped; ev.peakTime = timeStr; }
      const sevRank = { WARN:1, DANGER:2, CRITICAL:3 };
      if ((sevRank[hs.severity]||0) > (sevRank[ev.maxSev]||0)) ev.maxSev = hs.severity;
    } else {
      // 새 이벤트 시작
      activeHotspotEvents[k] = {
        key: k,
        kind: hs.kind,
        addrs: hs.addrs.slice(),
        start: timeStr,
        end: timeStr,
        peak: hs.stopped,
        peakTime: timeStr,
        maxSev: hs.severity,
        x: hs.x, y: hs.y,
      };
    }
  }

  // 사라진 이벤트 종료 → history로 이동
  for (const k in activeHotspotEvents) {
    if (!currentKeys.has(k)) {
      hotspotHistory.push(activeHotspotEvents[k]);
      delete activeHotspotEvents[k];
    }
  }
}

function _allHotspotEvents() {
  // 진행 중 + 종료된 모든 이벤트
  return [...Object.values(activeHotspotEvents), ...hotspotHistory];
}

function updateHotspotPanel(hotspots) {
  const el = document.getElementById('p-hotspots');
  if (!el) return;

  // 이력 로그 먼저 (현재 재생 시각 기준)
  _logHotspots(hotspots || [], currentPlayTime);

  hotspots = hotspots || [];

  // 심각도별 카운트 (현재 활성)
  const sevCount = { CRITICAL: 0, DANGER: 0, WARN: 0 };
  for (const hs of hotspots) sevCount[hs.severity] = (sevCount[hs.severity] || 0) + 1;

  // 위험 배지 오버라이드 (STAR 기반 예측 결과 덮어쓰기)
  const dlBadge = document.getElementById('p-dl-badge');
  if (dlBadge) {
    if (sevCount.CRITICAL > 0) {
      dlBadge.textContent = '심각 (' + sevCount.CRITICAL + '개)';
      dlBadge.className = 'risk risk-CRITICAL';
    } else if (sevCount.DANGER > 0) {
      dlBadge.textContent = '위험 (' + sevCount.DANGER + '개)';
      dlBadge.className = 'risk risk-DANGER';
    } else if (sevCount.WARN > 0) {
      dlBadge.textContent = '주의 (' + sevCount.WARN + '개)';
      dlBadge.className = 'risk risk-WARNING';
    }
    const msg = document.getElementById('p-dl-msg');
    if (msg) {
      const top = hotspots[0];
      msg.textContent = `최상위: addr ${top.addrs.join(',')} 에 ${top.stopped}대 정지`;
    }
  }

  // ===== 렌더: 현재 활성 + 지난 이벤트(WARN 이상) 누적 =====
  const sevRank = { WARN:1, DANGER:2, CRITICAL:3 };
  // 현재 활성 이벤트 (activeHotspotEvents)
  const activeEvs = Object.values(activeHotspotEvents);
  // 종료된 이벤트 중 WARN 이상만 남김 (관찰/노이즈는 제외)
  const pastEvs = hotspotHistory.filter(e => (sevRank[e.maxSev]||0) >= 1);

  // 이벤트를 심각도 순 → peak 내림차순
  const byScore = (a, b) => (sevRank[b.maxSev]||0) - (sevRank[a.maxSev]||0) || b.peak - a.peak;
  activeEvs.sort(byScore);
  pastEvs.sort((a,b) => byScore(a,b));

  const sevColor = s => s === 'CRITICAL' ? '#ef4444' : s === 'DANGER' ? '#f59e0b' : '#eab308';
  const renderEvent = (ev, active) => {
    const color = sevColor(ev.maxSev);
    const icon = ev.kind === 'chain' ? '🔗' : '●';
    const addrs = ev.addrs.length > 4
      ? `${ev.addrs[0]}~${ev.addrs[ev.addrs.length-1]} (${ev.addrs.length}개)`
      : ev.addrs.join(',');
    const timeSpan = ev.start === ev.end ? ev.start : `${ev.start}~${ev.end}`;
    const opacity = active ? '1' : '0.68';
    const bg = active ? '#1a1a2e' : '#0f1220';
    const liveTag = active ? '<span style="font-size:9px;color:#22c55e;font-weight:bold;">🔴 진행중</span>' : `<span style="font-size:9px;color:#9ca3af;">종료</span>`;
    let html = `<div style="padding:5px 7px;margin-bottom:3px;background:${bg};border-left:3px solid ${color};border-radius:3px;cursor:pointer;opacity:${opacity};" onclick="jumpToHotspot(${ev.x || 0}, ${ev.y || 0})">`;
    html += `<div style="display:flex;justify-content:space-between;align-items:center;">`;
    html += `<b style="color:${color};">${icon} peak ${ev.peak}대 [${ev.maxSev}]</b>${liveTag}</div>`;
    html += `<div style="color:#cbd5e1;">addr ${addrs}</div>`;
    html += `<div style="font-size:9px;color:#6b7280;">⏱ ${timeSpan}${ev.peakTime ? ` · peak@${ev.peakTime}` : ''}</div>`;
    html += `</div>`;
    return html;
  };

  let html = '';
  html += `<div style="color:#9ca3af;margin-bottom:4px;font-size:10px;">`;
  html += `진행중 ${activeEvs.length}개 · 이력 ${pastEvs.length}개 (WARN+)`;
  html += `</div>`;

  if (activeEvs.length === 0 && pastEvs.length === 0) {
    html += '<div style="color:#6b7280;">감지 없음</div>';
  } else {
    if (activeEvs.length > 0) {
      html += `<div style="font-size:10px;color:#22c55e;margin:4px 0 2px;font-weight:bold;">▶ 진행 중</div>`;
      const maxA = 15;
      for (let i = 0; i < Math.min(activeEvs.length, maxA); i++) html += renderEvent(activeEvs[i], true);
      if (activeEvs.length > maxA) html += `<div style="color:#6b7280;font-size:9px;">...외 ${activeEvs.length - maxA}개</div>`;
    }
    if (pastEvs.length > 0) {
      html += `<div style="font-size:10px;color:#9ca3af;margin:6px 0 2px;font-weight:bold;">⏸ 지난 이력</div>`;
      const maxP = 30;
      for (let i = 0; i < Math.min(pastEvs.length, maxP); i++) html += renderEvent(pastEvs[i], false);
      if (pastEvs.length > maxP) html += `<div style="color:#6b7280;font-size:9px;">...외 ${pastEvs.length - maxP}개</div>`;
    }
  }
  el.innerHTML = html;
}

}

async function applyFab(fab, prefix) {
  const loadingMsg = document.getElementById('loading-msg');
  loadingMsg.style.display = 'inline';
  loadingMsg.textContent = `${fab}/${prefix} 레이아웃 로딩...`;

  // 이전 데이터 리셋
  obsHistory = []; jamEvents = []; _lastObsSampleTime = ''; _lastJamCount = 0;
  fullObsHistory = [];
  bottleneckCache = null;
  vehicleDisplay = {};
  mapData.vehicles = [];
  mapData.blockedEdges = {};
  starHistory = [];
  railGraph = null;
  railCacheCanvas = null;

  try {
    const resp = await fetch('/api/fab/select', {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify({fab, prefix})
    });
    if (!resp.ok) {
      const err = await resp.json().catch(() => ({}));
      throw new Error(err.error || `HTTP ${resp.status}`);
    }
    const data = await resp.json();
    currentFab = data.fab;
    currentPrefix = data.prefix;
    rebuildLayoutSelect();
    syncLogpressoTable();   // FAB/prefix 변경 시 테이블명 input 자동 갱신

    // 맵 재로드
    await loadRailGraph();
    drawQueueChart();
    renderZoneList();
    console.log(`FAB 전환: ${currentFab}/${currentPrefix} (노드 ${data.nodes}, 엣지 ${data.edges}, Zone ${data.zones})`);
  } catch(e) {
    console.error('FAB 전환 실패:', e);
    alert(`FAB 전환 실패: ${e.message || e}`);
  } finally {
    loadingMsg.style.display = 'none';
    loadingMsg.textContent = '로딩중...';
  }
}

async function openBottleneckModal() {
  document.getElementById('bottleneck-modal').style.display = 'flex';
  const tbl = document.getElementById('bn-table');
  if (!bottleneckCache) {
    tbl.innerHTML = '<span style="color:#9ca3af;">분석 중...</span>';
    try {
      const resp = await fetch('/api/bottleneck-analysis');
      bottleneckCache = await resp.json();
    } catch (e) {
      tbl.innerHTML = '<span style="color:#ef4444;">불러오기 실패: ' + e + '</span>';
      return;
    }
  }
  renderBottleneckModal(bottleneckCache);
  drawBottleneckHeatmap(bottleneckCache);
}

function renderBottleneckModal(data) {
  const s = data.summary || {};
  setText('bn-section-cnt', s.section_count ?? '-');
  setText('bn-bottleneck-cnt', s.bottleneck_count ?? '-');
  setText('bn-external-pct', (s.external_pct ?? 0) + '%');
  setText('bn-internal-pct', (s.internal_pct ?? 0) + '%');

  const tbl = document.getElementById('bn-table');
  const sections = data.sections || [];
  if (!sections.length) {
    // 진단 정보 표시 — "정말 없는 건지 / 데이터 자체가 없는 건지" 구분
    let diag = '';
    if (s.hid_events === 0) {
      diag = `<div style="color:#ef4444;font-weight:bold;">❌ HID_INOUT 데이터 없음</div>
              <div style="color:#9ca3af;margin-top:4px;">이 날짜 폴더에 hid_inout CSV 파일이 없거나 비어있습니다.</div>`;
    } else if (s.sections_above_threshold === 0 && s.sections_raw > 0) {
      diag = `<div style="color:#fbbf24;font-weight:bold;">⚠️ 사용률 임계 미달</div>
              <div style="color:#9ca3af;margin-top:4px;">HID 이벤트는 ${s.hid_events}건 로드됐고 구간 ${s.sections_raw}개 집계됐지만, 모두 사용률 P95 &lt; 10% 로 병목 후보 아님. 해당 날짜는 전반적으로 여유 운영.</div>`;
    } else if (s.reason) {
      diag = `<div style="color:#f59e0b;font-weight:bold;">⚠️ ${s.reason}</div>`;
    } else {
      diag = `<div style="color:#22c55e;">✅ 병목 구간 없음 (모든 구간 여유)</div>`;
    }
    const meta = `<div style="color:#64748b;font-size:10px;margin-top:8px;">
      진단 — HID 이벤트: ${s.hid_events ?? 0}건 · ts 이벤트: ${s.ts_events ?? 0}건 ·
      분석 구간: ${s.sections_raw ?? 0}개 · limit 있는 구간: ${s.sections_with_limit ?? 0}개 ·
      임계 통과: ${s.sections_above_threshold ?? 0}개</div>`;
    tbl.innerHTML = diag + meta;
    return;
  }

  const badge = (cause) => {
    if (cause === 'external') return '<span style="background:#1e3a8a;color:#93c5fd;padding:2px 6px;border-radius:3px;font-size:10px;">외부</span>';
    if (cause === 'internal') return '<span style="background:#7f1d1d;color:#fca5a5;padding:2px 6px;border-radius:3px;font-size:10px;">내부</span>';
    return '<span style="background:#713f12;color:#fcd34d;padding:2px 6px;border-radius:3px;font-size:10px;">혼합</span>';
  };

  let html = '<table style="width:100%;border-collapse:collapse;font-size:11px;">';
  html += '<thead><tr style="border-bottom:1px solid #334155;color:#94a3b8;">';
  html += '<th style="text-align:left;padding:4px;">#</th>';
  html += '<th style="text-align:left;padding:4px;">구간</th>';
  html += '<th style="text-align:right;padding:4px;">사용률 avg</th>';
  html += '<th style="text-align:right;padding:4px;">사용률 peak</th>';
  html += '<th style="text-align:right;padding:4px;">체류(초)</th>';
  html += '<th style="text-align:right;padding:4px;">score</th>';
  html += '<th style="text-align:right;padding:4px;">피크시각</th>';
  html += '<th style="text-align:right;padding:4px;">상관</th>';
  html += '<th style="text-align:center;padding:4px;">원인</th>';
  html += '</tr></thead><tbody>';

  sections.forEach((sec, i) => {
    html += '<tr style="border-bottom:1px solid #1e293b;">';
    html += `<td style="padding:4px;color:#64748b;">${i + 1}</td>`;
    html += `<td style="padding:4px;">${sec.zone_name} <span style="color:#64748b;">(${sec.from_hid}→${sec.to_hid})</span></td>`;
    html += `<td style="padding:4px;text-align:right;">${sec.usage_pct_avg.toFixed(1)}%</td>`;
    html += `<td style="padding:4px;text-align:right;color:${sec.usage_pct_peak >= 80 ? '#f87171' : '#fbbf24'};">${sec.usage_pct_peak.toFixed(1)}%</td>`;
    html += `<td style="padding:4px;text-align:right;">${sec.dwell_avg_sec.toFixed(1)}</td>`;
    html += `<td style="padding:4px;text-align:right;color:#a78bfa;font-weight:bold;">${sec.bottleneck_score.toFixed(3)}</td>`;
    html += `<td style="padding:4px;text-align:right;">${sec.peak_time}</td>`;
    html += `<td style="padding:4px;text-align:right;">${sec.ts_correlation.toFixed(2)}</td>`;
    html += `<td style="padding:4px;text-align:center;">${badge(sec.cause)}</td>`;
    html += '</tr>';
  });
  html += '</tbody></table>';
  tbl.innerHTML = html;
}

function drawBottleneckHeatmap(data) {
  const canvas = document.getElementById('bottleneck-heatmap');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  const dpr = window.devicePixelRatio || 1;
  const rect = canvas.getBoundingClientRect();
  const w = Math.max(400, Math.floor(rect.width));
  const h = Math.max(200, Math.floor(rect.height));
  canvas.width = w * dpr;
  canvas.height = h * dpr;
  canvas.style.width = w + 'px';
  canvas.style.height = h + 'px';
  ctx.scale(dpr, dpr);
  ctx.fillStyle = '#0a0e17';
  ctx.fillRect(0, 0, w, h);

  const sections = (data.sections || []).filter(s => s.cx && s.cy);
  if (!sections.length || !mapData || !mapData.bounds) {
    ctx.fillStyle = '#64748b';
    ctx.font = '12px sans-serif';
    ctx.fillText('히트맵 데이터 부족', 12, 20);
    return;
  }

  // 기존 맵 bounds 재사용
  const b = mapData.bounds;
  const pad = 12;
  const scX = (w - pad * 2) / Math.max(1, b.max_x - b.min_x);
  const scY = (h - pad * 2) / Math.max(1, b.max_y - b.min_y);
  const sc = Math.min(scX, scY);
  const ox = pad + ((w - pad * 2) - (b.max_x - b.min_x) * sc) / 2;
  const oy = pad + ((h - pad * 2) - (b.max_y - b.min_y) * sc) / 2;
  const toS = (x, y) => [ox + (x - b.min_x) * sc, oy + (y - b.min_y) * sc];

  // 배경: 레이아웃 엣지 (흰색으로 또렷하게)
  if (railGraph && railGraph.edges && railGraph.nodes) {
    ctx.strokeStyle = 'rgba(255,255,255,0.85)';
    ctx.lineWidth = 1.8;
    ctx.lineCap = 'round';
    ctx.beginPath();
    for (const e of railGraph.edges) {
      const a = railGraph.nodes[e[0]], bn = railGraph.nodes[e[1]];
      if (!a || !bn) continue;
      const [x1, y1] = toS(a[0], a[1]);
      const [x2, y2] = toS(bn[0], bn[1]);
      ctx.moveTo(x1, y1);
      ctx.lineTo(x2, y2);
    }
    ctx.stroke();
  }

  // 구간별 원
  sections.forEach(s => {
    const [sx, sy] = toS(s.cx, s.cy);
    const score = Math.max(0, Math.min(1, s.bottleneck_score));
    let color;
    if (score >= 0.66) color = '#ef4444';
    else if (score >= 0.33) color = '#fbbf24';
    else color = '#22c55e';
    const radius = 4 + score * 14;
    ctx.fillStyle = color + 'cc';
    ctx.beginPath();
    ctx.arc(sx, sy, radius, 0, Math.PI * 2);
    ctx.fill();
    ctx.strokeStyle = color;
    ctx.lineWidth = 1;
    ctx.stroke();
  });

  // 범례
  ctx.fillStyle = '#94a3b8';
  ctx.font = '11px sans-serif';
  ctx.fillText('● 초록=여유  ● 노랑=주의  ● 빨강=병목', 12, h - 8);
}
