#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
발동이벤트_요약.py — M16 HUBROOM 발동이벤트.csv(분당 1행, 하루 1440행)를
'위험 구간 묶음'(수~수십 행)으로 압축한다.

목적: 발동이벤트 원본은 1440행 × 100여 열이라 LLM 컨텍스트(128K)를 넘김.
      이 스크립트가 샌드박스에서 미리 요약 → 데모스 LLM 은 작은 요약표만 보고
      하루 보고서를 작성. (데모스 스크립트 실행 기능으로 자동 호출)

입력: 발동이벤트.csv (16열 축약본도, 131열 원본도 모두 OK — 헤더 이름으로 읽음)
출력: <outdir>/발동이벤트_요약.csv  (주의 이상 위험 구간을 시간순으로 묶은 표)

규칙:
- '주의' 이상(점수 >=120) 인 분(minute)만 위험으로 보고, 30분 이내로 가까운
  위험 분들은 한 구간으로 병합(짧은 진동 흡수). 이게 하루 보고서의 시간대 표가 됨.
- 구간마다: 시간대 / 지속 / 최고위험레벨 / 최고점수 / 진원지 / 대표 전파경로 / 발동룰 / 특이신호
- stdlib 만 사용 (pandas 불필요).
"""
import sys
import os
import csv
import re
import argparse
from collections import Counter
from datetime import datetime

LEVEL_ORDER = {"정상": 0, "관심": 1, "주의": 2, "경계": 3, "위험": 4, "발동": 5}
RISK_LEVEL_MIN = 1          # '관심' 이상(100점↑)을 활동 구간으로 — 조용한 날도 하루 전체가 잡히게


def _norm_key(k):
    return (k or "").lstrip("﻿").strip()


def _level_from(level_str, score):
    """레벨 문자열 우선, 없으면 점수로 등급 산출."""
    lv = (level_str or "").strip()
    if lv in LEVEL_ORDER:
        return lv
    try:
        s = float(score)
    except (TypeError, ValueError):
        return "정상"
    if s >= 220: return "발동"
    if s >= 160: return "위험"
    if s >= 130: return "경계"
    if s >= 120: return "주의"
    if s >= 100: return "관심"
    return "정상"


def _parse_dt(row):
    """datetime 컬럼(YYYY-MM-DD HH:MM) 우선, 없으면 date+time 조합."""
    dt = (row.get("datetime") or "").strip()
    for fmt in ("%Y-%m-%d %H:%M:%S", "%Y-%m-%d %H:%M"):
        try:
            return datetime.strptime(dt, fmt)
        except ValueError:
            pass
    d = (row.get("date") or "").strip()
    t = (row.get("time") or "").strip()
    for fmt in ("%Y-%m-%d %H:%M:%S", "%Y-%m-%d %H:%M"):
        try:
            return datetime.strptime(f"{d} {t}", fmt)
        except ValueError:
            pass
    return None


# propagation_chain 예: "M16(23:05,RB+RB_fast) → M16HUB(23:07,RC+RD)"
_RULE_RE = re.compile(r"\(([^)]*?)\)")
_RULE_TOKEN_RE = re.compile(r"R[A-D](?:_sus)?|RB_fast|SLA|SORT|MAXCAPA\*?\d*|RC|RD|RA")


def _rules_in_chain(chain):
    rules = []
    for inside in _RULE_RE.findall(chain or ""):
        # inside 예: "23:07,RC+RD"  → 룰 토큰만
        parts = inside.split(",", 1)
        if len(parts) == 2:
            for tok in re.split(r"[+\s]+", parts[1]):
                tok = tok.strip()
                if tok:
                    rules.append(tok)
    return rules


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("input", help="발동이벤트.csv 경로")
    ap.add_argument("-o", "--outdir", default=".", help="출력 폴더")
    args = ap.parse_args()

    # 입력 읽기 (utf-8-sig 로 BOM 흡수)
    with open(args.input, "r", encoding="utf-8-sig", newline="") as f:
        reader = csv.DictReader(f)
        reader.fieldnames = [_norm_key(k) for k in (reader.fieldnames or [])]
        rows = []
        for r in reader:
            rows.append({_norm_key(k): v for k, v in r.items()})

    # 분 단위 정보 추출 + 시간순 정렬
    minutes = []
    for r in rows:
        dt = _parse_dt(r)
        if dt is None:
            continue
        try:
            score = float(r.get("unified_risk_score") or 0)
        except ValueError:
            score = 0.0
        level = _level_from(r.get("unified_risk_level"), score)
        minutes.append({
            "dt": dt,
            "time": (r.get("time") or dt.strftime("%H:%M")).strip(),
            "score": score,
            "level": level,
            "lvl_n": LEVEL_ORDER.get(level, 0),
            "hot_area": (r.get("hot_area") or "").strip(),
            "chain": (r.get("propagation_chain") or "").strip(),
            "flow": (r.get("flow_signals") or "").strip(),
            "maxcapa": (r.get("maxcapa_signals") or "").strip(),
            "affected": (r.get("affected_areas") or "").strip(),
        })
    minutes.sort(key=lambda m: m["dt"])

    # 1) 시간(hour) 버킷으로 묶고, 각 시간의 '최고 등급'을 구한다
    from collections import defaultdict
    hour_buckets = defaultdict(list)
    for m in minutes:
        hour_buckets[m["dt"].strftime("%Y-%m-%d %H")].append(m)

    hourly = []
    for hkey in sorted(hour_buckets):
        hm = hour_buckets[hkey]
        peak = max(hm, key=lambda x: x["score"])
        if peak["lvl_n"] < RISK_LEVEL_MIN:        # 그 시간의 최고가 '관심' 미만이면 조용한 시간 → 스킵
            continue
        # 시간키의 시작 datetime (병합 시 연속 시간 판정용)
        hdt = datetime.strptime(hkey, "%Y-%m-%d %H")
        hourly.append({"hdt": hdt, "lvl_n": peak["lvl_n"], "mins": hm})

    # 2) 연속된 시간 + 같은 최고등급끼리 병합 → 에피소드 구간
    windows = []
    for h in hourly:
        prev = windows[-1] if windows else None
        contiguous = prev and (h["hdt"] - prev["_last_hdt"]).total_seconds() <= 3600 + 1
        if prev and contiguous and h["lvl_n"] == prev["_lvl_n"]:
            prev["mins"].extend(h["mins"])
            prev["_last_hdt"] = h["hdt"]
        else:
            windows.append({"mins": list(h["mins"]), "_lvl_n": h["lvl_n"], "_last_hdt": h["hdt"]})

    # 구간별 요약 행 생성
    out_rows = []
    for w in windows:
        mins = w["mins"]
        # 대표 분 = '최고 등급' 우선, 같으면 최고 점수 (레벨·점수 일관성)
        peak = max(mins, key=lambda x: (x["lvl_n"], x["score"]))
        start = mins[0]
        end = mins[-1]
        dur = int((end["dt"] - start["dt"]).total_seconds() / 60.0) + 1
        hot = Counter(m["hot_area"] for m in mins if m["hot_area"]).most_common(1)
        hot_area = hot[0][0] if hot else (peak["hot_area"] or "-")
        # 발동 룰 집계 (전 구간 전파경로에서)
        rule_ct = Counter()
        for m in mins:
            for rr in _rules_in_chain(m["chain"]):
                rule_ct[rr] += 1
        top_rules = ", ".join(f"{r}" for r, _ in rule_ct.most_common(6)) or "-"
        # 특이신호 (flow/maxcapa 가 잡힌 게 있으면 대표 하나)
        flow = next((m["flow"] for m in mins if m["flow"]), "")
        maxcapa = next((m["maxcapa"] for m in mins if m["maxcapa"]), "")
        sig = "; ".join(s for s in [flow and f"흐름:{flow}", maxcapa and f"CAPA:{maxcapa}"] if s) or "-"
        out_rows.append({
            "시간대": f"{start['time']}~{end['time']}",
            "지속분": dur,
            "최고위험레벨": peak["level"],
            "최고점수": int(peak["score"]),
            "진원지": hot_area,
            "대표_전파경로": peak["chain"] or "-",
            "발동룰": top_rules,
            "특이신호": sig,
        })

    # 출력
    os.makedirs(args.outdir, exist_ok=True)
    out_path = os.path.join(args.outdir, "발동이벤트_요약.csv")
    cols = ["시간대", "지속분", "최고위험레벨", "최고점수", "진원지",
            "대표_전파경로", "발동룰", "특이신호"]
    with open(out_path, "w", encoding="utf-8-sig", newline="") as f:
        w = csv.DictWriter(f, fieldnames=cols)
        w.writeheader()
        for r in out_rows:
            w.writerow(r)

    # 콘솔 요약(데모스가 csv 없을 때 stdout 도 읽음)
    n_risk = sum(1 for m in minutes if m["lvl_n"] >= RISK_LEVEL_MIN)
    peak_all = max(minutes, key=lambda x: x["score"]) if minutes else None
    print(f"[발동이벤트 요약] 총 {len(minutes)}분 중 위험(주의↑) {n_risk}분 "
          f"→ {len(out_rows)}개 구간으로 압축")
    if peak_all:
        print(f"하루 최고: {peak_all['time']} {peak_all['level']} "
              f"{int(peak_all['score'])}점 (진원지 {peak_all['hot_area']})")
    print(f"출력: {out_path}")


if __name__ == "__main__":
    main()
