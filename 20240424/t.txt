<script>
const $ = id => document.getElementById(id);

async function api(path, opts) {
  const r = await fetch(path, Object.assign({headers:{'Content-Type':'application/json'}}, opts||{}));
  if (!r.ok) throw new Error(r.status + ' ' + r.statusText);
  return r.json();
}

function fmtNum(n){ return (n||0).toLocaleString(); }
function fmtBytes(b){
  if (b < 1024) return b + ' B';
  if (b < 1024*1024) return (b/1024).toFixed(1) + ' KB';
  return (b/1024/1024).toFixed(2) + ' MB';
}
function fmtSec(s){
  s = Math.floor(s||0);
  const h = Math.floor(s/3600), m = Math.floor((s%3600)/60), x = s%60;
  return (h?h+'시간 ':'') + (m?m+'분 ':'') + x + '초';
}

async function refresh() {
  try {
    const s = await api('/api/status');
    if (!$('host-input').dataset.touched && s.udp_host) {
      $('host-input').value = s.udp_host;
      $('host-input').dataset.touched = '1';
    }
    document.querySelectorAll('.speed-preset').forEach(b => {
      b.classList.toggle('active', parseFloat(b.dataset.s) === s.speed);
    });
    $('csv-path').textContent   = s.csv;
    $('csv-exists').textContent = s.csv_exists ? '있음' : '없음 ⚠️';
    $('csv-exists').style.color = s.csv_exists ? '#86efac' : '#fca5a5';
    $('file-size').textContent  = fmtBytes(s.file_bytes||0);
    $('time-col').textContent   = s.time_col || '⚠ 미감지';
    $('time-col').style.color   = s.time_col ? '#cbd5e1' : '#fca5a5';
    $('first-ts').textContent   = s.rows_first_ts || '—';

    $('tx-packets').textContent = fmtNum(s.tx_packets);
    $('tx-rows').textContent    = fmtNum(s.tx_rows);
    $('tx-bytes').textContent   = fmtBytes(s.tx_bytes);
    $('cur-ts').textContent     = s.cur_ts || '—';
    $('cycle').textContent      = s.cycle;
    $('elapsed').textContent    = fmtSec(s.elapsed_real_sec);

    const tot = s.file_bytes || 1, cur = s.cur_byte || 0;
    const pct = Math.min(100, (cur/tot*100));
    $('bar').style.width = pct + '%';
    $('pos-cur').textContent = fmtBytes(cur);
    $('pos-tot').textContent = fmtBytes(tot);
    $('pos-pct').textContent = pct.toFixed(1) + '%';

    const pill = $('state');
    if (s.running && !s.paused) { pill.className='status-pill pill-on'; pill.textContent='송신 중'; }
    else if (s.running && s.paused) { pill.className='status-pill pill-pause'; pill.textContent='일시정지'; }
    else { pill.className='status-pill pill-off'; pill.textContent='대기'; }

    $('err').textContent = s.last_error ? ('⚠ ' + s.last_error) : '';
  } catch (e) {
    $('err').textContent = '⚠ ' + e.message;
  }
}

$('btn-start').onclick  = () => api('/api/start',  {method:'POST'}).then(refresh);
$('btn-pause').onclick  = () => api('/api/pause',  {method:'POST'}).then(refresh);
$('btn-resume').onclick = () => api('/api/resume', {method:'POST'}).then(refresh);
$('btn-stop').onclick   = () => api('/api/stop',   {method:'POST'}).then(refresh);
$('btn-speed').onclick  = () => api('/api/speed',  {method:'POST', body: JSON.stringify({speed:parseFloat($('speed-num').value)})}).then(refresh);
$('btn-host').onclick   = () => api('/api/host',   {method:'POST', body: JSON.stringify({host:$('host-input').value.trim()})}).then(refresh);
document.querySelectorAll('.speed-preset').forEach(b => {
  b.onclick = async () => {
    const s = parseFloat(b.dataset.s);
    $('speed-num').value = s;
    await api('/api/speed', {method:'POST', body: JSON.stringify({speed:s})});
    refresh();
  };
});
$('loop-cb').onchange   = () => api('/api/loop',   {method:'POST', body: JSON.stringify({loop:$('loop-cb').checked})});
$('btn-seek').onclick   = () => api('/api/seek',   {method:'POST', body: JSON.stringify({ratio:parseInt($('seek-range').value)/100})}).then(refresh);
$('btn-test').onclick   = async () => {
  $('test-result').textContent = '...';
  try {
    const r = await api('/api/test', {method:'POST', body:'{}'});
    $('test-result').textContent = r.ok
      ? `✅ 테스트 패킷 송신 (${r.bytes} byte → ${r.to}). 수신측에서 확인.`
      : `⚠ ${r.error || 'fail'}`;
    $('test-result').style.color = r.ok ? '#86efac' : '#fca5a5';
  } catch(e) { $('test-result').textContent = '⚠ ' + e.message; }
  refresh();
};

refresh();
setInterval(refresh, 1000);
</script>
