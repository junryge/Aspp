import requests, json

with open("token.txt", "r") as f:
    token = f.read().strip()

BASE = "http://dev.hcp.llm.skhynix.com"
headers = {
    "Authorization": f"Bearer {token}",
    "Content-Type": "application/json"
}

# 모델 목록 조회
resp = requests.get(f"{BASE}/v1/models", headers=headers, verify=False)
print("Status:", resp.status_code)

models = resp.json()
if isinstance(models, dict):
    models = models.get("data", [models])

print(f"\n총 {len(models)}개 모델:")
for m in models:
    if isinstance(m, dict):
        mid = m.get("id") or m.get("model") or m.get("name") or "?"
        print(f"  - {mid}")
    else:
        print(f"  - {m}")

# 397B 만 필터
print("\n397B/72B 관련:")
for m in models:
    s = json.dumps(m, ensure_ascii=False) if isinstance(m, dict) else str(m)
    if "397B" in s or "72B" in s:
        print(f"  ★ {s}")
