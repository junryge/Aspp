<script>
const $ = id => document.getElementById(id);

let currentFab = 'M14A';
let layoutData = null;          // {nodes, edges, bounds}
const layers = { edges:true, nodes:false, trail:true, id:false };

const STATE_COLOR = {
  1:'#10b981',  // RUN
  2:'#f59e0b',  // WAIT
  3:'#ef4444',  // OBS_BZ_STOP
  4:'#6b7280',  // MAINT
  5:'#ef4444',  // STOP
  6:'#ef4444',  // OBS_STOP
  7:'#a78bfa',  // EMERG
  8:'#f59e0b',  // STANDBY
};

// ── 뷰포트 (zoom/pan) ─────────────────────────────
const view = { x:0, y:0, scale:1 };
const svg = $('map');

function resetView() {
  if (!layoutData || !layoutData.bounds || layoutData.bounds.length<4) {
    view.x=0; view.y=0; view.scale=1; render(); return;
  }
  const [xmin, ymin, xmax, ymax] = layoutData.bounds;
  const W = svg.clientWidth, H = svg.clientHeight;
  const w = xmax-xmin, h = ymax-ymin;
  if (w<=0 || h<=0) return;
  const m = 40;
  view.scale = Math.min((W-m*2)/w, (H-m*2)/h);
  view.x = -xmin*view.scale + m + (W-m*2 - w*view.scale)/2;
  view.y = -ymin*view.scale + m + (H-m*2 - h*view.scale)/2;
  render();
}

svg.addEventListener('wheel', (e) => {
  e.preventDefault();
  const k = e.deltaY < 0 ? 1.15 : 1/1.15;
  const r = svg.getBoundingClientRect();
  const cx = e.clientX - r.left, cy = e.clientY - r.top;
  view.x = cx - (cx - view.x)*k;
  view.y = cy - (cy - view.y)*k;
  view.scale *= k;
  render();
}, { passive:false });

let drag = null;
svg.addEventListener('mousedown', (e) => { drag = { x:e.clientX-view.x, y:e.clientY-view.y }; });
window.addEventListener('mousemove', (e) => {
  if (!drag) return;
  view.x = e.clientX - drag.x;
  view.y = e.clientY - drag.y;
  render();
});
window.addEventListener('mouseup', () => { drag = null; });
svg.addEventListener('dblclick', resetView);

// ── 좌표 변환 ────────────────────────────────────
const px = x => x*view.scale + view.x;
const py = y => y*view.scale + view.y;

// ── 잔상 ─────────────────────────────────────────
const trail = new Map();   // vid -> [{x,y,t}, ...] (max 8)

// ── 렌더 (SVG DOM 직접 조작 — innerHTML 안 씀) ───
const SVG_NS = "http://www.w3.org/2000/svg";
function svgEl(tag, attrs) {
  const el = document.createElementNS(SVG_NS, tag);
  for (const k in attrs) el.setAttribute(k, attrs[k]);
  return el;
}

let lastVehicles = [];
let layoutError = null;

function render() {
  const W = svg.clientWidth || svg.parentElement.clientWidth || 800;
  const H = svg.clientHeight || svg.parentElement.clientHeight || 600;

  // bg (흰색)
  while (svg.firstChild) svg.removeChild(svg.firstChild);
  svg.appendChild(svgEl('rect', {x:0, y:0, width:W, height:H, fill:'#ffffff'}));

  if (layoutError) {
    const t = svgEl('text', {x:W/2, y:H/2, fill:'#fca5a5', 'font-size':14, 'text-anchor':'middle'});
    t.textContent = '⚠ 레이아웃 로드 실패: ' + layoutError;
    svg.appendChild(t);
    const t2 = svgEl('text', {x:W/2, y:H/2+22, fill:'#9ca3af', 'font-size':11, 'text-anchor':'middle'});
    t2.textContent = 'UDP_SIM/cache/ 또는 OHT_MAP/cache/ 의 layout_cache.json 확인';
    svg.appendChild(t2);
    return;
  }
  if (!layoutData) {
    const t = svgEl('text', {x:W/2, y:H/2, fill:'#6b7280', 'font-size':14, 'text-anchor':'middle'});
    t.textContent = '레이아웃 로딩...';
    svg.appendChild(t);
    return;
  }

  // edges (한 path 로 묶음)
  if (layers.edges) {
    let d = '';
    for (const [a,b] of layoutData.edges) {
      const A = layoutData.nodes[a], B = layoutData.nodes[b];
      if (!A || !B) continue;
      d += `M${px(A[0]).toFixed(1)},${py(A[1]).toFixed(1)}L${px(B[0]).toFixed(1)},${py(B[1]).toFixed(1)}`;
    }
    if (d) svg.appendChild(svgEl('path', {d, stroke:'#334155', 'stroke-width':0.8, fill:'none', opacity:0.9}));
  }

  // nodes
  if (layers.nodes && view.scale > 0.05) {
    const r = Math.max(0.6, 1.2*view.scale);
    const g = svgEl('g', {fill:'#1f2937', opacity:0.7});
    for (const k in layoutData.nodes) {
      const [x,y] = layoutData.nodes[k];
      g.appendChild(svgEl('circle', {cx:px(x), cy:py(y), r}));
    }
    svg.appendChild(g);
  }

  // trails
  if (layers.trail) {
    for (const [vid, pts] of trail) {
      if (pts.length<2) continue;
      let d = `M${px(pts[0].x).toFixed(1)},${py(pts[0].y).toFixed(1)}`;
      for (let i=1;i<pts.length;i++) d += `L${px(pts[i].x).toFixed(1)},${py(pts[i].y).toFixed(1)}`;
      svg.appendChild(svgEl('path', {d, stroke:'#ea580c', 'stroke-width':1.2, fill:'none', opacity:0.5}));
    }
  }

  // vehicles
  for (const v of lastVehicles) {
    if (v.x == null || v.y == null) continue;
    const cx = px(v.x), cy = py(v.y);
    const color  = STATE_COLOR[v.state] || '#6b7280';
    const stroke = v.fab==='M14A' ? '#22c55e' : '#3b82f6';
    svg.appendChild(svgEl('circle', {cx, cy, r:4.5, fill:color, stroke, 'stroke-width':1.5}));
    if (layers.id) {
      const t = svgEl('text', {x:cx+6, y:cy-6, fill:'#1f2937', 'font-size':9});
      t.textContent = v.vid;
      svg.appendChild(t);
    }
  }
}

// ── FAB / 레이어 토글 ──────────────────────────
async function onFabChange() {
  const f = $('fab-select').value;
  currentFab = f;
  $('cur-fab').textContent = f;
  trail.clear();
  await loadLayout(f);
}
function toggleLayer(k) {
  layers[k] = !layers[k];
  $('btn-'+k).classList.toggle('on', layers[k]);
  render();
}

async function loadLayout(fab) {
  layoutError = null;
  layoutData  = null;
  $('lo-nodes').textContent = '...';
  $('lo-edges').textContent = '...';
  render();
  try {
    const r = await fetch('/api/layout?fab='+fab);
    if (!r.ok) {
      let msg = `HTTP ${r.status}`;
      try { const j = await r.json(); if (j.error) msg = j.error; } catch(e){}
      layoutError = msg;
      $('lo-nodes').textContent = '?';
      $('lo-edges').textContent = '?';
      render();
      return;
    }
    const d = await r.json();
    if (!d.nodes || Object.keys(d.nodes).length === 0) {
      layoutError = '레이아웃 노드 0개 — JSON 캐시 비었음';
      render();
      return;
    }
    layoutData = d;
    $('lo-nodes').textContent = Object.keys(d.nodes).length.toLocaleString();
    $('lo-edges').textContent = d.edges.length.toLocaleString();
    requestAnimationFrame(() => { resetView(); render(); });
  } catch(e) {
    layoutError = (e && e.message) || String(e);
    render();
  }
}

// ── WebSocket ───────────────────────────────────
let ws = null;
let lastWsAt = 0;
function connectWs() {
  const proto = location.protocol === 'https:' ? 'wss' : 'ws';
  ws = new WebSocket(`${proto}://${location.host}/ws`);
  ws.onopen = () => {
    $('ws-state').textContent = 'WS: 연결됨';
    $('ws-state').style.color = '#10b981';
    $('disconn').style.display = 'none';
  };
  ws.onclose = () => {
    $('ws-state').textContent = 'WS: 끊김 (재연결 시도)';
    $('ws-state').style.color = '#ef4444';
    $('disconn').style.display = 'block';
    setTimeout(connectWs, 1500);
  };
  ws.onerror = () => { try{ws.close();}catch(e){} };
  ws.onmessage = (ev) => {
    try {
      const d = JSON.parse(ev.data);
      lastWsAt = Date.now();
      onPush(d);
    } catch(e){}
  };
  // ping (keep-alive)
  setInterval(() => { try { ws && ws.readyState===1 && ws.send('ping'); } catch(e){} }, 5000);
}

function onPush(d) {
  // 차량 (현재 FAB 만)
  const list = (d.vehicles || []).filter(v => v.fab === currentFab);
  // 잔상 갱신
  for (const v of list) {
    if (v.x==null) continue;
    const arr = trail.get(v.vid) || [];
    const last = arr[arr.length-1];
    if (!last || (last.x-v.x)*(last.x-v.x)+(last.y-v.y)*(last.y-v.y) > 1) {
      arr.push({x:v.x, y:v.y});
      if (arr.length > 8) arr.shift();
      trail.set(v.vid, arr);
    }
  }
  lastVehicles = list;

  // UDP 통계 갱신
  updateStats(d);
  render();
}

function fmtBytes(b){ if(b<1024)return b+'B'; if(b<1048576)return (b/1024).toFixed(1)+'KB'; return (b/1048576).toFixed(2)+'MB'; }
function fmtNum(n){ return (n||0).toLocaleString(); }
function shortTs(s){ if(!s) return '—'; const m = s.match(/T(\d{2}:\d{2}:\d{2})/); return m?m[1]:s.slice(-12); }

function updateStats(d) {
  const now = Date.now();
  for (const fab of ['M14A','M16A_BR']) {
    const u = d.udp[fab] || {};
    const dotId = fab==='M14A' ? 'udp-m14a-dot' : 'udp-m16a-dot';
    const numId = fab==='M14A' ? 'udp-m14a-num' : 'udp-m16a-num';
    const dot = $(dotId);
    const lastRx = u.last_at ? new Date(u.last_at).getTime() : 0;
    if (!u.bound) { dot.className = 'udp-dot off'; }
    else if (lastRx && (now - lastRx) < 2500) { dot.className = 'udp-dot live'; }
    else if (u.rx_packets > 0) { dot.className = 'udp-dot stale'; }
    else { dot.className = 'udp-dot off'; }
    $(numId).textContent = fmtNum(u.rx_packets);
  }

  // 사이드 패널
  const m14 = d.udp.M14A || {}, m16 = d.udp.M16A_BR || {};
  $('m14-pkt').textContent  = fmtNum(m14.rx_packets);
  $('m14-rows').textContent = fmtNum(m14.rx_rows);
  $('m14-bytes').textContent= fmtBytes(m14.rx_bytes||0);
  $('m14-ts').textContent   = shortTs(m14.last_ts);
  $('m14-rx').textContent   = shortTs(m14.last_at);
  $('m14-from').textContent = m14.last_addr ? (m14.last_addr[0]+':'+m14.last_addr[1]) : '—';
  $('m16-pkt').textContent  = fmtNum(m16.rx_packets);
  $('m16-rows').textContent = fmtNum(m16.rx_rows);
  $('m16-bytes').textContent= fmtBytes(m16.rx_bytes||0);
  $('m16-ts').textContent   = shortTs(m16.last_ts);
  $('m16-rx').textContent   = shortTs(m16.last_at);
  $('m16-from').textContent = m16.last_addr ? (m16.last_addr[0]+':'+m16.last_addr[1]) : '—';

  // 차량 카운트 (현재 FAB)
  const live = (d.vehicles||[]).filter(v=>v.fab===currentFab && v.x!=null).length;
  const total= (d.vehicles||[]).filter(v=>v.fab===currentFab).length;
  $('vh-live').textContent  = live;
  $('vh-total').textContent = total;

  // 상태 분포
  const buckets = {};
  for (const v of (d.vehicles||[])) {
    if (v.fab !== currentFab) continue;
    const name = v.stateName || ('S'+v.state);
    buckets[name] = (buckets[name]||0) + 1;
  }
  let html = '';
  const max = Math.max(1, ...Object.values(buckets));
  for (const k of Object.keys(buckets).sort()) {
    const n = buckets[k];
    const w = Math.round(n/max*100);
    html += `<div style="margin:3px 0;">
      <div style="display:flex;justify-content:space-between;font-size:11px;color:#9ca3af">
        <span>${escapeHtml(k)}</span><span style="color:#cbd5e1">${n}</span></div>
      <div style="height:5px;background:#1f2937;border-radius:3px;overflow:hidden;">
        <div style="height:100%;background:#60a5fa;width:${w}%;"></div></div></div>`;
  }
  $('state-bars').innerHTML = html || '<div style="font-size:11px;color:#6b7280">데이터 대기...</div>';
}

window.addEventListener('resize', render);
loadLayout(currentFab);
connectWs();
</script>
