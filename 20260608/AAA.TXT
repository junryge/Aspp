# -*- coding: utf-8 -*-
"""
llama-server GGUF Chat (tkinter, Vulkan) - Qwen3.5용
- llama.cpp 본체 prebuilt(llama-server.exe)를 파이썬이 자동 실행
- OpenAI 호환 API로 채팅, 스트리밍 출력
- 필요: pip install requests
- 준비: llama.cpp Releases에서 llama-*-bin-win-vulkan-x64.zip 받아 압축 해제
        https://github.com/ggml-org/llama.cpp/releases
"""

import os
import time
import socket
import subprocess
import threading
import queue
import json
import tkinter as tk
from tkinter import filedialog, messagebox

import requests


class _StreamThinkFilter:
    """스트리밍 중 사고과정(<think>) 실시간 제거. 토큰 경계로 태그가 쪼개져도 안전.
    implicit=True(qwen3.x): 템플릿이 여는 <think>를 프롬프트에 미리 넣어 모델 출력이
    사고로 바로 시작하고 닫는 </think>만 나온다 → 첫 </think> 전까지 전부 버린다."""
    OPEN = "<think>"
    CLOSE = "</think>"

    def __init__(self, implicit=False):
        self.buf = ""
        self.in_think = False
        self.implicit = implicit
        self.done = False

    def feed(self, tok):
        if self.done:
            return tok
        self.buf += tok
        if self.implicit and not self.in_think:
            i = self.buf.find(self.CLOSE)
            if i != -1:
                after = self.buf[i + len(self.CLOSE):]
                self.buf = ""
                self.done = True
                return after.lstrip("\n")
            return ""
        out = ""
        while self.buf:
            if not self.in_think:
                i = self.buf.find(self.OPEN)
                if i == -1:
                    cut = self._safe_emit_len(self.buf, self.OPEN)
                    out += self.buf[:cut]
                    self.buf = self.buf[cut:]
                    break
                out += self.buf[:i]
                self.buf = self.buf[i + len(self.OPEN):]
                self.in_think = True
            else:
                j = self.buf.find(self.CLOSE)
                if j == -1:
                    keep = self._tail_partial_len(self.buf, self.CLOSE)
                    self.buf = self.buf[len(self.buf) - keep:] if keep else ""
                    break
                self.buf = self.buf[j + len(self.CLOSE):]
                self.in_think = False
        return out

    def flush(self):
        out = "" if self.in_think else self.buf
        self.buf = ""
        return out

    @staticmethod
    def _safe_emit_len(s, tag):
        for k in range(min(len(tag) - 1, len(s)), 0, -1):
            if s.endswith(tag[:k]):
                return len(s) - k
        return len(s)

    @staticmethod
    def _tail_partial_len(s, tag):
        for k in range(min(len(tag) - 1, len(s)), 0, -1):
            if s.endswith(tag[:k]):
                return k
        return 0


# ====== 경로 설정 (네 환경에 맞게 수정) ======
LLAMA_SERVER_EXE = r"D:\llama\llama-server.exe"          # 압축 푼 위치
DEFAULT_GGUF = r"D:\models\Qwen3.5-2B-UD-Q8_K_XL.gguf"   # GGUF 위치
PORT = 8080
N_GPU_LAYERS = 99
N_CTX = 8192
# =============================================

BASE_URL = f"http://127.0.0.1:{PORT}"


def port_open():
    try:
        with socket.create_connection(("127.0.0.1", PORT), timeout=1):
            return True
    except OSError:
        return False


class ChatApp:
    def __init__(self, root):
        self.root = root
        root.title("Qwen3.5 Chat (llama-server / Vulkan)")
        root.geometry("760x660")

        self.proc = None
        self.history = []
        self.q = queue.Queue()
        self.busy = False

        self._build_ui()
        self._poll_queue()
        root.protocol("WM_DELETE_WINDOW", self._on_close)

    # ---------- UI ----------
    def _build_ui(self):
        top = tk.Frame(self.root)
        top.pack(fill="x", padx=8, pady=6)
        tk.Label(top, text="GGUF:").pack(side="left")
        self.gguf_var = tk.StringVar(value=DEFAULT_GGUF)
        tk.Entry(top, textvariable=self.gguf_var).pack(
            side="left", fill="x", expand=True, padx=4)
        tk.Button(top, text="찾기", command=self._pick_file).pack(side="left", padx=2)
        tk.Button(top, text="서버 시작", command=self._start_server).pack(side="left", padx=2)

        opt = tk.Frame(self.root)
        opt.pack(fill="x", padx=8)
        tk.Label(opt, text="temp:").pack(side="left")
        self.temp_var = tk.StringVar(value="0.7")
        tk.Entry(opt, textvariable=self.temp_var, width=6).pack(side="left", padx=4)
        self.think_var = tk.BooleanVar(value=False)   # 기본 OFF = /no_think
        tk.Checkbutton(opt, text="사고모드(think)", variable=self.think_var).pack(side="left", padx=8)
        tk.Button(opt, text="대화 초기화", command=self._reset).pack(side="left", padx=8)

        self.chat = tk.Text(self.root, wrap="word", state="disabled",
                            font=("Malgun Gothic", 11))
        self.chat.pack(fill="both", expand=True, padx=8, pady=6)
        self.chat.tag_config("user", foreground="#1565c0",
                             font=("Malgun Gothic", 11, "bold"))
        self.chat.tag_config("bot", foreground="#2e7d32",
                             font=("Malgun Gothic", 11, "bold"))
        self.chat.tag_config("sys", foreground="#9e9e9e")

        bottom = tk.Frame(self.root)
        bottom.pack(fill="x", padx=8, pady=6)
        self.entry = tk.Text(bottom, height=3, font=("Malgun Gothic", 11))
        self.entry.pack(side="left", fill="x", expand=True)
        self.entry.bind("<Return>", self._on_enter)
        self.send_btn = tk.Button(bottom, text="전송", width=8,
                                  command=self._send, state="disabled")
        self.send_btn.pack(side="left", padx=4)

        self.status = tk.Label(self.root, text="'서버 시작'을 누르세요",
                               anchor="w", fg="#666")
        self.status.pack(fill="x", padx=8)

    def _append(self, text, tag=None):
        self.chat.config(state="normal")
        self.chat.insert("end", text, tag if tag else ())
        self.chat.see("end")
        self.chat.config(state="disabled")

    def _pick_file(self):
        path = filedialog.askopenfilename(
            title="GGUF 선택", filetypes=[("GGUF", "*.gguf"), ("All", "*.*")])
        if path:
            self.gguf_var.set(path)

    def _reset(self):
        self.history.clear()
        self.chat.config(state="normal")
        self.chat.delete("1.0", "end")
        self.chat.config(state="disabled")
        self._append("[대화 초기화됨]\n\n", "sys")

    # ---------- 서버 ----------
    def _start_server(self):
        if port_open():
            self._append("[이미 떠 있는 서버에 연결]\n\n", "sys")
            self.send_btn.config(state="normal")
            self.status.config(text="준비됨")
            return
        gguf = self.gguf_var.get().strip()
        if not os.path.exists(LLAMA_SERVER_EXE):
            messagebox.showerror("오류",
                f"llama-server.exe 없음:\n{LLAMA_SERVER_EXE}\n\n"
                "코드 상단 LLAMA_SERVER_EXE 경로 수정하세요.")
            return
        if not os.path.exists(gguf):
            messagebox.showerror("오류", f"GGUF 파일 없음:\n{gguf}")
            return

        self.status.config(text="서버 시작 중... (모델 로딩, 수십 초)")
        threading.Thread(target=self._server_worker, args=(gguf,), daemon=True).start()

    def _server_worker(self, gguf):
        try:
            cmd = [
                LLAMA_SERVER_EXE,
                "-m", gguf,
                "-ngl", str(N_GPU_LAYERS),
                "-c", str(N_CTX),
                "--port", str(PORT),
            ]
            # 콘솔 창 같이 띄움 -> Vulkan 잡혔는지 로그로 확인 가능
            self.proc = subprocess.Popen(cmd)
        except Exception as e:
            self.q.put(("status", f"서버 실행 실패: {e}"))
            return

        # 뜰 때까지 대기 (모델 로딩 포함 최대 120초)
        for _ in range(240):
            if port_open():
                try:
                    r = requests.get(f"{BASE_URL}/health", timeout=2)
                    if r.status_code == 200:
                        self.q.put(("sys", "[서버 준비 완료 — 콘솔에서 Vulkan 확인]\n\n"))
                        self.q.put(("ready", None))
                        return
                except requests.RequestException:
                    pass
            time.sleep(0.5)
        self.q.put(("status", "서버가 시간 내에 안 떴습니다. 콘솔 로그 확인"))

    # ---------- 채팅 ----------
    def _on_enter(self, event):
        if event.state & 0x0001:
            return
        self._send()
        return "break"

    def _send(self):
        if self.busy:
            return
        msg = self.entry.get("1.0", "end").strip()
        if not msg:
            return
        self.entry.delete("1.0", "end")

        # qwen3.x: /no_think 로 사고 비활성화 (DEMOS 동일 패턴)
        send_msg = msg
        if not self.think_var.get():
            if "/no_think" not in send_msg:
                send_msg = send_msg.rstrip() + " /no_think"

        self.history.append({"role": "user", "content": send_msg})
        self._append("나: ", "user")
        self._append(msg + "\n\n")   # 화면엔 /no_think 빼고 표시
        self._append("AI: ", "bot")

        self.busy = True
        self.send_btn.config(state="disabled")
        self.status.config(text="생성 중...")
        threading.Thread(target=self._chat_worker, daemon=True).start()

    def _chat_worker(self):
        try:
            temp = float(self.temp_var.get())
        except ValueError:
            temp = 0.7
        full = ""
        # qwen3.x 는 닫는 </think> 만 나오므로 implicit 모드
        tf = _StreamThinkFilter(implicit=True)
        try:
            r = requests.post(
                f"{BASE_URL}/v1/chat/completions",
                json={
                    "model": "qwen3.5-2b",
                    "messages": self.history,
                    "temperature": temp,
                    "stream": True,
                },
                stream=True,
                timeout=300,
            )
            for line in r.iter_lines():
                if not line:
                    continue
                line = line.decode("utf-8")
                if not line.startswith("data: "):
                    continue
                payload = line[6:]
                if payload.strip() == "[DONE]":
                    break
                try:
                    obj = json.loads(payload)
                    piece = obj["choices"][0]["delta"].get("content", "")
                    if piece:
                        vis = tf.feed(piece)        # 사고과정 실시간 제거
                        if vis:
                            full += vis
                            self.q.put(("token", vis))
                except (json.JSONDecodeError, KeyError, IndexError):
                    continue
            tail = tf.flush()
            if tail:
                full += tail
                self.q.put(("token", tail))
            self.history.append({"role": "assistant", "content": full})
            self.q.put(("done", None))
        except Exception as e:
            self.q.put(("token", f"\n[오류] {e}\n"))
            self.q.put(("done", None))

    # ---------- queue ----------
    def _poll_queue(self):
        try:
            while True:
                kind, val = self.q.get_nowait()
                if kind == "token":
                    self._append(val)
                elif kind == "sys":
                    self._append(val, "sys")
                elif kind == "status":
                    self.status.config(text=val)
                elif kind == "ready":
                    self.send_btn.config(state="normal")
                    self.status.config(text="준비됨 — 입력하세요")
                elif kind == "done":
                    self._append("\n\n")
                    self.busy = False
                    self.send_btn.config(state="normal")
                    self.status.config(text="준비됨")
        except queue.Empty:
            pass
        self.root.after(40, self._poll_queue)

    def _on_close(self):
        # 내가 띄운 서버면 같이 종료
        if self.proc and self.proc.poll() is None:
            self.proc.terminate()
        self.root.destroy()


if __name__ == "__main__":
    root = tk.Tk()
    ChatApp(root)
    root.mainloop()
