// ============ 시스템 프롬프트 조립 ============
function memorySnapshot(){
  const mem=get(LS.mem), usr=get(LS.usr); let s='';
  if(mem.length) s += '=== 기억된 사실 (MEMORY) ===\n' + mem.map(t=>'- '+t).join('\n') + '\n\n';
  if(usr.length) s += '=== 사용자 선호 (USER) ===\n' + usr.map(t=>'- '+t).join('\n') + '\n\n';
  return s.trim();
}
function skillIndex(){ const sk=get(LS.sk); return sk.length ? '=== 배운 스킬 (이름: 언제) ===\n'+sk.map(s=>`- ${s.name}: ${s.when}`).join('\n') : ''; }
function recallSkills(q,k=2){
  const sk=get(LS.sk), ql=(q||'').toLowerCase();
  const scored = sk.map(s=>({s, h: ((s.name+' '+s.when).toLowerCase().split(/[\s-]+/).filter(w=>w&&ql.includes(w)).length)}))
                   .filter(x=>x.h>0).sort((a,b)=>b.h-a.h);
  return scored.slice(0,k).map(x=>x.s);
}
function recallBuiltin(q,k=2){
  const ql=(q||'').toLowerCase();
  return Object.entries(BUILTIN).map(([n,s])=>({n,s,h:s.keywords.filter(kw=>ql.includes(kw.toLowerCase())).length}))
    .filter(x=>x.h>0).sort((a,b)=>b.h-a.h).slice(0,k);
}
function buildSystemPrompt(query){
  const parts=[];
  const mem=memorySnapshot(); if(mem) parts.push(mem);
  const idx=skillIndex(); if(idx) parts.push(idx);
  const rs=recallSkills(query); if(rs.length) parts.push('=== 관련 개인 스킬 본문 ===\n'+rs.map(s=>`--- 스킬: ${s.name} ---\n${s.body}`).join('\n\n'));
  const bi=recallBuiltin(query); if(bi.length) parts.push('=== 권장 작업 방식 (헤르메스 빌트인 스킬) ===\n'+bi.map(x=>`[${x.n}] ${x.s.desc}\n${x.s.body}`).join('\n\n'));
  parts.push(PROTOCOL_GUIDE);
  return parts.join('\n\n');
}

// ============ 블록 파싱 (hermes/protocol.py 와 동일) ============
const FENCE = /```[ \t]*hermes:(memory|skill|ask)[ \t]*\n([\s\S]*?)```/gi;
function parseBlocks(text){
  const blocks=[];
  const clean = (text||'').replace(FENCE, (m,kind,body)=>{ blocks.push({kind:kind.toLowerCase(), body}); return ''; });
  return { clean: clean.trim(), blocks };
}
function parseKV(body){
  const out={}; const lines=body.split('\n'); const multi=['body','text'];
  for(let i=0;i<lines.length;i++){
    const m=lines[i].match(/^([a-zA-Z_]+)\s*:\s*(.*)$/); if(!m) continue;
    const key=m[1].toLowerCase(), val=m[2];
    if(multi.includes(key) && ['','|','|-','>'].includes(val.trim())){
      out[key]=lines.slice(i+1).map(c=>c.startsWith('  ')?c.slice(2):c).join('\n').trim(); break;
    }
    out[key]=val.trim();
  }
  return out;
}
function parseAsk(body){
  const qs=[]; for(const ln of body.split('\n')){ const s=ln.trim(); const m=s.match(/^[-*•]\s+(.+)$/);
    if(m) qs.push(m[1].trim()); else if(s && !qs.length) qs.push(s); }
  return qs.filter(Boolean).slice(0,5);
}

// ============ 블록 적용 (memory 저장 / skill 승인대기 / ask) ============
function applyBlocks(blocks){
  const memResults=[], pending=[], questions=[];
  for(const b of blocks){
    if(b.kind==='memory'){
      const kv=parseKV(b.body); const store=(kv.store==='user')?LS.usr:LS.mem; const arr=get(store);
      const act=(kv.action||'add').toLowerCase();
      if(act==='add' && kv.text){ if(!arr.includes(kv.text)){ arr.push(kv.text); set(store,arr); memResults.push((store===LS.usr?'USER':'MEMORY')+' + '+kv.text); } }
      else if(act==='remove' && kv.target){ const n=arr.filter(t=>!t.includes(kv.target)); set(store,n); memResults.push('제거: '+kv.target); }
      else if(act==='replace' && kv.target && kv.text){ const n=arr.map(t=>t.includes(kv.target)?kv.text:t); set(store,n); memResults.push('교체: '+kv.text); }
    } else if(b.kind==='skill'){
      const kv=parseKV(b.body);
      if(kv.name && kv.body){ pending.push({name:kv.name, when:kv.when||'', body:kv.body}); }
    } else if(b.kind==='ask'){
      parseAsk(b.body).forEach(q=>questions.push(q));
    }
  }
  return { memResults, pending, questions };
}

// ============ LLM API 호출 (OpenAI 호환, 스트리밍) ============
async function callLLM(messages, onToken){
  const cfg=getCfg();
  if(!cfg.base || !cfg.model){ throw new Error('먼저 API 주소·모델을 설정/저장하세요.'); }
  let msgs = messages;
  if(cfg.fold){ // gemma 등 system 미지원 → system을 첫 user 앞에 합침
    const sys=msgs.filter(m=>m.role==='system').map(m=>m.content).join('\n\n');
    const rest=msgs.filter(m=>m.role!=='system'); let done=false;
    msgs=rest.map(m=>{ if(!done && m.role==='user'){ done=true; return {role:'user', content:sys+'\n\n'+m.content}; } return m; });
    if(!done) msgs=[{role:'user',content:sys}].concat(rest);
  }
  const url = cfg.base.replace(/\/+$/,'') + '/chat/completions';
  const headers={'Content-Type':'application/json'}; if(cfg.key) headers['Authorization']='Bearer '+cfg.key;
  const body={model:cfg.model, messages:msgs, max_tokens:2048};
  if(cfg.stream) body.stream=true;
  let resp;
  try{
    resp=await fetch(url,{method:'POST',headers,body:JSON.stringify(body)});
  }catch(e){ // 네트워크/CORS 실패 (Failed to fetch)
    throw new Error('네트워크/CORS 차단 — 브라우저가 사내 API에 직접 접근 못 함. proxy.py 를 켜고 주소를 http://localhost:8000/v1 로 바꾸세요. ('+(e.message||e)+')');
  }
  if(!resp.ok){ const t=await resp.text().catch(()=> ''); throw new Error('HTTP '+resp.status+(t?(' '+t.slice(0,300)):'')); }
  // 비스트리밍 (기본, 당신 Python 과 동일)
  if(!cfg.stream){
    const j=await resp.json();
    const acc=(j.choices&&j.choices[0]&&(j.choices[0].message?.content||j.choices[0].text))||'';
    onToken&&onToken(acc);
    return acc;
  }
  // 스트리밍 (SSE)
  if(!resp.body){ throw new Error('스트리밍 응답 없음 — 스트리밍 체크를 끄세요.'); }
  const reader=resp.body.getReader(), dec=new TextDecoder(); let buf='', acc='';
  while(true){
    const {done,value}=await reader.read(); if(done) break;
    buf+=dec.decode(value,{stream:true}); const lines=buf.split('\n'); buf=lines.pop()||'';
    for(const ln of lines){ const s=ln.trim(); if(!s.startsWith('data:')) continue;
      const pl=s.slice(5).trim(); if(pl==='[DONE]') continue;
      try{ const j=JSON.parse(pl); const tok=j.choices&&j.choices[0]&&(j.choices[0].delta?.content||''); if(tok){ acc+=tok; onToken&&onToken(acc); } }catch(_){}
    }
  }
  return acc;
}

// 모델 목록 불러오기 (/v1/models) → datalist 채움
async function loadModels(){
  const cfg=getCfg(); if(!cfg.base){ saveCfg(); }
  const b=($('cfgBase').value.trim()||'').replace(/\/+$/,''); const key=$('cfgKey').value.trim();
  if(!b){ $('cfgMsg').innerHTML='<span class="warn">먼저 API 주소를 입력하세요.</span>'; return; }
  $('cfgMsg').textContent='모델 목록 불러오는 중…';
  try{
    const h={}; if(key) h['Authorization']='Bearer '+key;
    const r=await fetch(b+'/models',{headers:h}); const j=await r.json();
    const arr=(j.data||j||[]).map(m=>m.id||m).filter(Boolean);
    $('modelList').innerHTML=arr.map(n=>`<option value="${esc(n)}">`).join('');
    if(arr.length && !$('cfgModel').value) $('cfgModel').value=arr[0];
    $('cfgMsg').innerHTML='<span class="ok">✅ 모델 '+arr.length+'개 — 입력칸에서 선택</span>';
  }catch(e){ $('cfgMsg').innerHTML='<span class="warn">목록 실패(CORS 가능): '+esc(e.message||e)+' → proxy.py 사용</span>'; }
}

const $=id=>document.getElementById(id);
function esc(s){ return String(s==null?'':s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }
function addMsg(role, text){ const d=document.createElement('div'); d.className='msg '+role; d.textContent=text; $('chat').appendChild(d); $('chat').scrollTop=$('chat').scrollHeight; return d; }
function stripThink(t){ return String(t||'').replace(/<think>[\s\S]*?<\/think>\s*/gi,'').replace(/<think>[\s\S]*$/i,'').trim(); }

function renderSide(){

  $('memCount').textContent = mem.length?`(${mem.length})`:'';
  $('skCount').textContent = sk.length?`(${sk.length})`:'';

  $('usrList').innerHTML=li(usr,LS.usr,'usr');

}

// 스킬 승인 카드
function showApproval(pending){
  const box=$('approveBox');
  pending.forEach(spec=>{
    const d=document.createElement('div'); d.className='approve';
    d.innerHTML=`💡 새 스킬 제안: <span class="nm">${esc(spec.name)}</span><div class="muted">${esc(spec.when)}</div>`+
      `<details><summary>본문 보기</summary><pre>${esc(spec.body)}</pre></details>`+
      `<button onclick='this.closest(".approve").remove(); approveSkill(${JSON.stringify(spec).replace(/'/g,"&#39;")})'>✅ 저장</button> `+
      `<button class="sec" onclick='this.closest(".approve").remove()'>건너뛰기</button>`;
    box.appendChild(d);
  });
}
function approveSkill(spec){ const a=get(LS.sk); if(!a.find(s=>s.name===spec.name)){ a.push(spec); set(LS.sk,a); renderSide(); addMsg('sys','📚 스킬 저장됨: '+spec.name); } }

// ============ 전송 ============
let busy=false;
async function onSend(){
  if(busy) return;
  const text=$('input').value.trim(); if(!text) return;
  const cfg=getCfg(); if(!cfg.base||!cfg.model){ $('cfgMsg').innerHTML='<span class="warn">먼저 API를 설정/저장하세요.</span>'; return; }
  $('input').value=''; addMsg('user',text);
  busy=true; $('sendBtn').disabled=true; $('status').textContent='상태: 생성 중…';
  const sys=buildSystemPrompt(text);
  const msgs=[{role:'system',content:sys}].concat(history).concat([{role:'user',content:text}]);
  const bubble=addMsg('assistant','…'); let acc='';
  try{
    acc=await callLLM(msgs,(p)=>{ bubble.textContent=stripThink(p)||'…'; $('chat').scrollTop=$('chat').scrollHeight; });
    const {clean,blocks}=parseBlocks(stripThink(acc));
    const {memResults,pending,questions}=applyBlocks(blocks);
    let shown=clean;
    if(questions.length) shown += (shown?'\n\n':'')+'❓ '+questions.join('\n❓ ');
    bubble.textContent = shown || '(빈 응답)';
    history.push({role:'user',content:text}); history.push({role:'assistant',content:clean});
    if(history.length>16) history=history.slice(-16);
    set(LS.hist,history);
    if(memResults.length){ addMsg('sys','🧠 기억 갱신: '+memResults.join(' · ')); renderSide(); }
    if(pending.length) showApproval(pending);
    $('status').textContent='상태: 완료'+(blocks.length?` · 블록 ${blocks.length}개 처리`:'');
  }catch(e){
    bubble.textContent='❌ '+(e.message||e); $('status').textContent='상태: 오류';
  }finally{ busy=false; $('sendBtn').disabled=false; }
}
