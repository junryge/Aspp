MD_COLORS = {
    "h1": "#00d4ff", "h2": "#a78bfa", "h3": "#34d399",
    "bold": "#f0f0ff", "code_fg": "#fb923c", "code_bg": "#1a1a2e",
    "block_fg": "#ccccdd", "block_bg": "#12121e",
    "list_bullet": "#00d4ff", "list_text": "#d0d0e0",
    "link": "#00d4ff", "hr": "#333355",
    "normal": "#c8c8dd", "dim": "#8888a8",
}

def render_md(textbox, raw_text):
    """Parse markdown and insert formatted text into a CTkTextbox."""
    tb = textbox._textbox if hasattr(textbox, '_textbox') else textbox

    # Configure tags
    base_size = 11
    tb.tag_configure("normal", foreground=MD_COLORS["normal"], font=("Noto Sans KR", base_size))
    tb.tag_configure("h1", foreground=MD_COLORS["h1"], font=("Noto Sans KR", base_size+5, "bold"))
    tb.tag_configure("h2", foreground=MD_COLORS["h2"], font=("Noto Sans KR", base_size+3, "bold"))
    tb.tag_configure("h3", foreground=MD_COLORS["h3"], font=("Noto Sans KR", base_size+1, "bold"))
    tb.tag_configure("bold", foreground=MD_COLORS["bold"], font=("Noto Sans KR", base_size, "bold"))
    tb.tag_configure("italic", foreground=MD_COLORS["dim"], font=("Noto Sans KR", base_size, "italic"))
    tb.tag_configure("code", foreground=MD_COLORS["code_fg"], background=MD_COLORS["code_bg"],
                     font=("JetBrains Mono", base_size-1))
    tb.tag_configure("codeblock", foreground=MD_COLORS["block_fg"], background=MD_COLORS["block_bg"],
                     font=("JetBrains Mono", base_size-1), lmargin1=20, lmargin2=20, rmargin=20)
    tb.tag_configure("codelang", foreground=MD_COLORS["dim"], background=MD_COLORS["block_bg"],
                     font=("JetBrains Mono", base_size-2))
    tb.tag_configure("bullet", foreground=MD_COLORS["list_bullet"], font=("Noto Sans KR", base_size, "bold"))
    tb.tag_configure("list_text", foreground=MD_COLORS["list_text"], font=("Noto Sans KR", base_size))
    tb.tag_configure("hr", foreground=MD_COLORS["hr"])
    tb.tag_configure("blockquote", foreground=MD_COLORS["dim"], font=("Noto Sans KR", base_size, "italic"),
                     lmargin1=16, lmargin2=16)

    tb.configure(state="normal")
    tb.delete("1.0", "end")

    lines = raw_text.split("\n")
    i = 0
    while i < len(lines):
        line = lines[i]

        # Code block ```
        if line.strip().startswith("```"):
            lang = line.strip()[3:].strip()
            if lang:
                tb.insert("end", f"  {lang}\n", "codelang")
            block_lines = []
            i += 1
            while i < len(lines) and not lines[i].strip().startswith("```"):
                block_lines.append(lines[i])
                i += 1
            if block_lines:
                tb.insert("end", "\n".join(block_lines) + "\n", "codeblock")
            i += 1
            continue

        # Horizontal rule
        if re.match(r'^-{3,}$|^\*{3,}$|^_{3,}$', line.strip()):
            tb.insert("end", "─" * 50 + "\n", "hr")
            i += 1; continue

        # Headers
        m = re.match(r'^(#{1,3})\s+(.*)', line)
        if m:
            level = len(m.group(1))
            tag = f"h{level}"
            tb.insert("end", m.group(2) + "\n", tag)
            i += 1; continue

        # Blockquote
        if line.strip().startswith(">"):
            text = line.strip().lstrip(">").strip()
            tb.insert("end", f"  │ {text}\n", "blockquote")
            i += 1; continue

        # List items (- or * or numbered)
        m = re.match(r'^(\s*)([-*]|\d+\.)\s+(.*)', line)
        if m:
            indent = len(m.group(1))
            prefix = "  " * (indent // 2) + "  ● " if m.group(2) in ["-", "*"] else f"  {m.group(2)} "
            tb.insert("end", prefix, "bullet")
            _insert_inline(tb, m.group(3) + "\n", "list_text")
            i += 1; continue

        # Normal line with inline formatting
        if line.strip():
            _insert_inline(tb, line + "\n", "normal")
        else:
            tb.insert("end", "\n")
        i += 1

    tb.configure(state="disabled")


def _insert_inline(tb, text, default_tag):
    """Handle inline markdown: **bold**, *italic*, `code`, [link](url)."""
    pattern = re.compile(r'(\*\*(.+?)\*\*|`(.+?)`|\*(.+?)\*|\[(.+?)\]\((.+?)\))')
    pos = 0
    for m in pattern.finditer(text):
        # Insert text before match
        if m.start() > pos:
            tb.insert("end", text[pos:m.start()], default_tag)
        if m.group(2):  # **bold**
            tb.insert("end", m.group(2), "bold")
        elif m.group(3):  # `code`
            tb.insert("end", m.group(3), "code")
        elif m.group(4):  # *italic*
            tb.insert("end", m.group(4), "italic")
        elif m.group(5):  # [link](url)
            tb.insert("end", m.group(5), "bold")
        pos = m.end()
    if pos < len(text):
        tb.insert("end", text[pos:], default_tag)

class ChatDisplay(ctk.CTkScrollableFrame):
    def __init__(self, master, **kw):
        super().__init__(master, fg_color=C["bg"], corner_radius=0, **kw)
        self.widgets = []

    def _make_header(self, parent, role, ts):
        icons = {"user":("👤","You",C["text2"]),"assistant":("🤖","AI 비서",C["dev"]),"tool":("🔧","도구",C["orange"])}
        ico,nm,clr = icons.get(role,("💬",role,C["text2"]))
        hdr = ctk.CTkFrame(parent, fg_color="transparent"); hdr.pack(fill="x")
        ctk.CTkLabel(hdr, text=ico, font=("",14), text_color=clr).pack(side="left", padx=(0,4))
        ctk.CTkLabel(hdr, text=nm, font=("JetBrains Mono",11,"bold"), text_color=clr).pack(side="left")
        ctk.CTkLabel(hdr, text=ts, font=("JetBrains Mono",9), text_color=C["text3"]).pack(side="left", padx=(8,0))

    def add_msg(self, role, text, ts=""):
        ts = ts or datetime.now().strftime("%H:%M")
        f = ctk.CTkFrame(self, fg_color="transparent"); f.pack(fill="x", padx=16, pady=(5,1))
        self._make_header(f, role, ts)

        if role == "assistant":
            # Markdown rendered textbox
            line_count = text.count("\n") + 1
            height = min(max(line_count * 20 + 10, 40), 600)
            body = ctk.CTkTextbox(f, height=height, font=("Noto Sans KR",11),
                                  fg_color="transparent", text_color=C["text"],
                                  border_width=0, wrap="word", activate_scrollbars=False)
            body.pack(fill="x", padx=(20,8), pady=(2,0))
            render_md(body, text)
            # Auto-resize after render
            self.after(50, lambda b=body: self._auto_height(b))
        else:
            # User messages: plain label
            body = ctk.CTkLabel(f, text=text, font=("Noto Sans KR",12), text_color=C["text"],
                                wraplength=700, justify="left", anchor="w")
            body.pack(fill="x", padx=(24,8), pady=(2,0), anchor="w")

        self.widgets.append(f); self.after(80, self._bottom)
        return body

    def _auto_height(self, tb):
        """Auto-resize textbox to fit content."""
        try:
            inner = tb._textbox if hasattr(tb, '_textbox') else tb
            inner.update_idletasks()
            # Count visible lines
            last = inner.index("end-1c")
            row = int(last.split(".")[0])
            new_h = min(max(row * 20, 40), 800)
            tb.configure(height=new_h)
        except: pass

    def add_stream(self):
        f = ctk.CTkFrame(self, fg_color="transparent"); f.pack(fill="x", padx=16, pady=(5,1))
        self._make_header(f, "assistant", datetime.now().strftime("%H:%M"))
        body = ctk.CTkLabel(f, text="⏳ 생성 중...", font=("Noto Sans KR",12),
                            text_color=C["text2"], wraplength=700, justify="left", anchor="w")
        body.pack(fill="x", padx=(24,8), pady=(2,0), anchor="w")
        self.widgets.append(f); self.after(50, self._bottom); return body

    def add_tool_log(self, name, status):
        f = ctk.CTkFrame(self, fg_color="transparent"); f.pack(fill="x", padx=40, pady=(1,0))
        ctk.CTkLabel(f, text=f"  🔧 {name}: {status}", font=("JetBrains Mono",10),
                     text_color=C["orange"], wraplength=660, justify="left", anchor="w").pack(fill="x", anchor="w")
        self.widgets.append(f); self.after(50, self._bottom)

    def clear(self):
        for w in self.widgets: w.destroy()
        self.widgets.clear()

    def _bottom(self):
        self._parent_canvas.yview_moveto(1.0)

    def _on_enter(self, e):
        if e.state & 0x1: return
        self._send(); return "break"

    def _send(self):
        if self.generating: self.llm.cancel(); self._set_gen(False); return
        text=self.inp.get("1.0","end").strip()
        if not text: return
        self.inp.delete("1.0","end")
        if text=="/tools": self.chat.clear(); self.chat.add_msg("assistant", tool_summary()); return
        if not self.api_key: self._api_dialog(); return
        if not self.chat_id: self._new_chat()
        ch=self._cur(); ch["model"]=self.cur_model; ts=datetime.now().strftime("%H:%M")
        ch["msgs"].append({"role":"user","content":text,"time":ts})
        if len(ch["msgs"])==1: ch["title"]=text[:32]+("..." if len(text)>32 else ""); self._refresh_history()
        if len(ch["msgs"])==1: self.chat.clear()
        self.chat.add_msg("user", text, ts)
        use_tools = needs_tools(text) if self.settings["auto_tools"] else False
        self._set_mode("🔧 도구 모드" if use_tools else "💬 채팅 모드")
        self._set_gen(True); self._call_llm(ch, use_tools)

    def _quick(self, text):
        if not self.chat_id: self._new_chat()
        self.inp.delete("1.0","end"); self.inp.insert("1.0",text); self._send()

    def _call_llm(self, ch, use_tools):
        model=MODELS[self.cur_model]; msgs=[]
        if self.settings["sys_prompt"] and self.settings["sys_msg"]:
            sm=self.settings["sys_msg"]
            if use_tools: sm+=f"\n\n[도구 모드] 제공된 도구를 적극 사용하세요. 작업 폴더: {self.settings['work_dir']}"
            msgs.append({"role":"system","content":sm})
        ctx=[m for m in ch["msgs"] if m["role"] in ("user","assistant")][-20:]
        msgs.extend({"role":m["role"],"content":m["content"]} for m in ctx)
        tools=ALL_TOOLS if use_tools else None
        self.stream_buf=""; self.stream_lbl=self.chat.add_stream()
        def on_chunk(d): self.stream_buf+=d; self.after(0, lambda: self._upd(self.stream_buf))
        def on_done(full):
            ts=datetime.now().strftime("%H:%M"); final=full or self.stream_buf
            ch["msgs"].append({"role":"assistant","content":final,"time":ts})
            self.after(0, lambda: (self._set_gen(False), self._render(ch), self._save_history()))
        def on_error(err):
            ts=datetime.now().strftime("%H:%M"); ch["msgs"].append({"role":"assistant","content":f"⚠️ 오류: {err}\n\n모델: {model['name']}","time":ts})
            self.after(0, lambda: (self._set_gen(False), self._render(ch)))
        def on_tool_log(name, args, result):
            self.after(0, lambda: self.chat.add_tool_log(name, result[:150]))
            self.after(0, lambda: self._upd(f"🔧 {name} 완료, 분석 중..."))
        self.llm.api_key=self.api_key
        self.llm.chat_threaded(url=model["url"],model=model["model"],messages=msgs,max_tokens=model["max_tokens"],temperature=self.settings["temperature"],stream=self.settings["stream"],tools=tools,tool_executor=call_tool if use_tools else None,on_chunk=on_chunk,on_done=on_done,on_error=on_error,on_tool_log=on_tool_log)
