// ════════════ State ════════════
let allSkills = [];

let currentTab = "skills";

const LS_HISTORY = "hermes_history_v1";
const LS_SID = "hermes_sid_v1";
const LS_RAW = "hermes_raw_v1";
const LS_TAB = "hermes_tab_v1";

// ════════════ Theme ════════════
function applyTheme(t) {
  document.documentElement.setAttribute("data-theme", t);
  localStorage.setItem("hermes_theme", t);
}
function toggleTheme() {
  const cur = document.documentElement.getAttribute("data-theme") || "light";
  applyTheme(cur === "dark" ? "light" : "dark");
}
applyTheme(localStorage.getItem("hermes_theme") || "dark");

// ════════════ Drawer (모바일) ════════════
function toggleDrawer(force) {
  const side = document.querySelector(".side");
  const bg = document.getElementById("drawerBg");
  const open = (typeof force === "boolean") ? force : !side.classList.contains("open");
  side.classList.toggle("open", open);
  bg.classList.toggle("open", open);
}

// ════════════ Tab 전환 ════════════
function switchTab(name) {
  currentTab = name;
  localStorage.setItem(LS_TAB, name);
  document.querySelectorAll(".side-tab").forEach(b => {
    b.classList.toggle("active", b.dataset.tab === name);
  });
  document.querySelectorAll(".side-panel").forEach(p => {
    p.classList.toggle("hidden", p.id !== "panel-" + name);
  });
  if (name === "sessions") loadSessions();
  else if (name === "memory") loadMemory();
}

// ════════════ RAW 모드 ════════════
function toggleRaw() {
  rawMode = document.getElementById("rawChk").checked;
  localStorage.setItem(LS_RAW, rawMode ? "1" : "0");
  document.getElementById("rawToggle").classList.toggle("on", rawMode);
}

// ════════════ Session ID ════════════
function setSessionId(sid) {
  currentSessionId = sid || null;
  if (sid) localStorage.setItem(LS_SID, sid);
  else localStorage.removeItem(LS_SID);
  const bar = document.getElementById("sessionBar");
  const span = document.getElementById("sessionSid");
  if (sid) {
    bar.style.display = "inline-flex";
    span.textContent = sid.length > 12 ? sid.slice(0, 10) + "…" : sid;
    span.title = sid;
  } else {
    bar.style.display = "none";
  }
}
function newSession() {
  setSessionId(null);
  history = [];
  totalTokens = 0;
  updateTokens(0);
  saveHistory();
  document.getElementById("msgs").innerHTML =
    '<div class="msg assistant"><div class="who">HERMES</div>' +
    '<div class="bubble">새 대화를 시작했어요.</div></div>';
}
function clearChat() {
  if (!confirm("대화 히스토리를 모두 지울까요?")) return;
  history = [];
  saveHistory();
  setSessionId(null);
  totalTokens = 0;
  updateTokens(0);
  document.getElementById("msgs").innerHTML = "";
}

// ════════════ History 영구화 ════════════
function saveHistory() {
  try {
    localStorage.setItem(LS_HISTORY, JSON.stringify(history.slice(-50)));
  } catch (_) {}
}
function loadHistory() {
  try {
    const raw = localStorage.getItem(LS_HISTORY);
    if (!raw) return;
    const arr = JSON.parse(raw);
    if (!Array.isArray(arr)) return;
    history = arr;
    const msgs = document.getElementById("msgs");
    msgs.innerHTML = "";
    arr.forEach(m => {
      if (m.role === "user") addMsg("user", m.content);
      else if (m.role === "assistant") {
        const b = addMsg("assistant", "");
        renderBubble(b, m.content);
        attachCodeCopy(b);
      }
    });
  } catch (_) {}
}

// ════════════ Health ════════════
async function refreshHealth() {
  try {
    const r = await fetch("/api/health");
    const d = await r.json();
    document.getElementById("envBadge").textContent = d.env_mode;
    const envDot = document.getElementById("envDot");
    envDot.classList.remove("ok", "bad");
    envDot.classList.add(d.skills_dir_exists ? "ok" : "bad");
    document.getElementById("topSub").textContent =
      `${d.env_mode}  ·  ${d.skills_dir_exists ? "SKILLS READY" : "SKILLS MISSING"}`;
    const pDot = document.getElementById("proxyDot");
    pDot.classList.remove("ok", "bad");
    pDot.classList.add(d.proxy_alive ? "ok" : "bad");
    document.getElementById("proxyText").textContent =
      "PROXY " + (d.proxy_alive ? "ONLINE" : "OFFLINE");
  } catch (e) {
    document.getElementById("topSub").textContent = "CONNECTION ERROR";
  }
}

// ════════════ Skills ════════════
async function loadSkills() {
  try {
    const r = await fetch("/api/skills");
    const d = await r.json();
    allSkills = d.skills || [];
    populateCategories(d.categories || [], allSkills);
    document.getElementById("skillCount").textContent = `${allSkills.length} SKILLS`;
    renderSkills();
  } catch (e) {
    document.getElementById("skillList").innerHTML =
      '<div style="padding:24px;color:var(--text-mute);font-size:11px;text-align:center;">SKILLS 로드 실패</div>';
  }
}
function populateCategories(cats, skills) {
  const sel = document.getElementById("categorySel");
  if (!sel || !cats.length) return;
  const counts = {};
  skills.forEach(s => { counts[s.category] = (counts[s.category] || 0) + 1; });
  const total = skills.length;
  const cur = sel.value || "all";
  sel.innerHTML = cats.map(c => {
    const n = c.key === "all" ? total : (counts[c.key] || 0);
    return `<option value="${c.key}">${c.name} (${n})</option>`;
  }).join("");

function escapeHtml(s) {
  return (s || "").replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
}

function renderBubble(el, text) {

  }
}

async function resumeSession(sid) {
  setSessionId(sid);
  // 세션 상세 받아서 화면에 표시
  try {
    const r = await fetch("/api/sessions/" + encodeURIComponent(sid));
    const d = await r.json();
    if (d.error) {
      alert("세션 로드 실패: " + d.error);
      return;
    }
    const messages = d.messages || d.history || [];
    history = messages.filter(m => m.role === "user" || m.role === "assistant").map(m => ({
      role: m.role,
      content: typeof m.content === "string" ? m.content : JSON.stringify(m.content),
    }));
    saveHistory();
    // 다시 렌더
    const msgs = document.getElementById("msgs");
    msgs.innerHTML = "";
    history.forEach(m => {
      if (m.role === "user") addMsg("user", m.content);
      else {
        const b = addMsg("assistant", "");
        renderBubble(b, m.content);
        attachCodeCopy(b);
      }
    });
  } catch (e) {
    alert("세션 로드 에러: " + e.message);
  }
  if (window.innerWidth <= 900) toggleDrawer(false);
}

// 도구 호출 인라인 박스 (raw 출력에서 패턴 매치해 박스로 변환)
const TOOL_PATTERNS = [
  { re: /^Q?\s*┊?\s*✍️\s+(?:write\s+)?(\S+)/i, icon: "✍️", name: "write_file", get: m => `path: ${m[1]}` },
  { re: /^Q?\s*┊?\s*💻\s+\$\s+(.+)/, icon: "💻", name: "terminal", get: m => `$ ${m[1]}` },
  { re: /^Q?\s*┊?\s*🔍\s+(.+)/, icon: "🔍", name: "search", get: m => m[1] },
  { re: /^Q?\s*┊?\s*📖\s+read\s+(\S+)/i, icon: "📖", name: "read_file", get: m => `path: ${m[1]}` },
];

function renderToolCall(bubble, line) {
  for (const p of TOOL_PATTERNS) {
    const m = line.match(p.re);
    if (m) {
      const statusMatch = line.match(/\[(ok|error|done|fail)\]|\b(\d+(?:\.\d+)?)s\b/i);
      let status = "running";
      if (/\[error\]|\[fail\]/i.test(line)) status = "err";
      else if (/\[ok\]|\[done\]/i.test(line)) status = "ok";
      const div = document.createElement("div");
      div.className = "tool-call";
      div.innerHTML = `
        <div class="tool-call-header">
          <span class="tool-call-icon">${p.icon}</span>
          <span class="tool-call-name">${p.name}</span>
          <span class="tool-call-status ${status}">${status.toUpperCase()}</span>
        </div>
        <div class="tool-call-body">${escapeHtml(p.get(m))}</div>
      `;
      bubble.appendChild(div);
      return true;
    }
  }
  return false;
}

async function send() {
  const ta = document.getElementById("msg");
  const text = (ta.value || "").trim();
  if (!text || sending) return;
  ta.value = ""; ta.style.height = "40px";
  history.push({ role: "user", content: text });
  saveHistory();
  addMsg("user", text);
  const skillIds = Array.from(selected);
  setSending(true);
  const safetyTimer = setTimeout(() => {
    if (sending) {
      if (currentAbort) { try { currentAbort.abort(); } catch(_) {} }
      setSending(false);
      bubble.classList.remove("typing");
      if (!acc.trim()) bubble.textContent = "⏱ 응답 시간 초과 (5분)";
    }
  }, 300000);
  const bubble = addMsg("assistant", "");
  bubble.classList.add("typing");
  let acc = "";
  let hadError = false;
  currentAbort = new AbortController();

  try {
    const resp = await fetch("/api/chat", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      signal: currentAbort.signal,
      body: JSON.stringify({
        messages: history,
        skills: skillIds,
        session_id: currentSessionId,
        raw: rawMode,
      }),
    });
    if (!resp.ok || !resp.body) throw new Error("SSE 시작 실패: " + resp.status);
    const reader = resp.body.getReader();
    const decoder = new TextDecoder();
    let buf = "";
    outer: while (true) {
      const { value, done } = await reader.read();
      if (done) break outer;
      buf += decoder.decode(value, { stream: true });
      let nl;
      while ((nl = buf.indexOf("\n\n")) !== -1) {
        const evt = buf.slice(0, nl);
        buf = buf.slice(nl + 2);
        if (!evt.startsWith("data:")) continue;
        const payload = evt.slice(5).trim();
        if (payload === "[DONE]") {
          try { await reader.cancel(); } catch(_) {}
          break outer;
        }
        try {
          const obj = JSON.parse(payload);
          if (obj.type === "meta") {
            const wrap = bubble.parentElement;
            const m = wrap.querySelector(".meta-line") || (() => {
              const d = document.createElement("div"); d.className = "meta-line";
              wrap.appendChild(d); return d;
            })();
            m.textContent = (obj.info || "") + (obj.raw ? "  ·  RAW" : "");
          } else if (obj.type === "token") {
            const tok = obj.t || "";
            // 도구 호출 패턴이면 박스로
            const isTool = tok.split("\n").some(l => TOOL_PATTERNS.some(p => p.re.test(l)));
            if (isTool && !rawMode) {
              tok.split("\n").forEach(line => {
                if (!line.trim()) return;
                if (!renderToolCall(bubble, line)) {
                  acc += line + "\n";
                }
              });
              renderBubble(bubble, acc);
              bubble.classList.remove("typing");
            } else {
              acc += tok;
              renderBubble(bubble, acc);
              bubble.classList.remove("typing");
            }
            scrollToBottom();
          } else if (obj.type === "error") {
            hadError = true;
            bubble.classList.remove("typing");
            bubble.parentElement.classList.add("error");
            bubble.textContent = "✦ " + (obj.error || "오류") +
              (obj.detail ? "\n→ " + obj.detail : "");
            acc = bubble.textContent;
          } else if (obj.type === "end") {
            bubble.classList.remove("typing");
            const stats = obj.stats || {};
            const m = bubble.parentElement.querySelector(".meta-line");
            if (m) m.textContent += `  ·  ${stats.emitted ?? 0} LINES`;
            if (obj.hermes_session_id) setSessionId(obj.hermes_session_id);
            if (obj.usage && obj.usage.total_tokens) updateTokens(obj.usage.total_tokens);
          }
        } catch (e) {}
      }
    }
    bubble.classList.remove("typing");
    attachCodeCopy(bubble);
    if (!hadError && !acc.trim() && !bubble.querySelector(".tool-call")) {
      bubble.textContent = rawMode
        ? "(hermes 가 빈 출력)"
        : "응답이 비어있어요. RAW 토글 켜고 다시 보내봐.";
    }
    if (acc.trim() && !hadError) history.push({ role: "assistant", content: acc });
    saveHistory();
  } catch (e) {
    bubble.classList.remove("typing");
    if (e.name === "AbortError") {
      if (!acc.trim()) bubble.textContent = "(중단됨)";
      else attachCodeCopy(bubble);
    } else {
      bubble.parentElement.classList.add("error");
      bubble.textContent = "✦ 전송 실패: " + e.message;
    }
  } finally {
    clearTimeout(safetyTimer);
    setSending(false);
    bubble.classList.remove("typing");
    currentAbort = null;
  }
}

// ════════════ 코드 복사 ════════════
function attachCodeCopy(bubble) {
  if (!bubble) return;
  bubble.querySelectorAll("pre").forEach(pre => {
    if (pre.querySelector(".code-copy")) return;
    const btn = document.createElement("button");
    btn.className = "code-copy";
    btn.textContent = "COPY";
    btn.onclick = (e) => {
      e.preventDefault(); e.stopPropagation();
      const code = pre.querySelector("code");
      const text = code ? code.textContent : pre.textContent;
      navigator.clipboard.writeText(text).then(() => {
        btn.textContent = "COPIED"; btn.classList.add("done");
        setTimeout(() => { btn.textContent = "COPY"; btn.classList.remove("done"); }, 1500);
      });
    };
    pre.appendChild(btn);
  });
}

        <span class="icon">${icon}</span>
        <span class="name">${escapeHtml(it.name)}</span>
        <span class="size">${size}</span>
      </div>`;
    }).join("");
  } catch (e) {
    list.innerHTML = '<div style="padding:20px 14px;color:#c66;font-size:11px;">' + escapeHtml(e.message) + '</div>';
  }
}
function formatBytes(n) {
  if (n < 1024) return n + "B";
  if (n < 1024*1024) return (n/1024).toFixed(1) + "K";
  if (n < 1024*1024*1024) return (n/1024/1024).toFixed(1) + "M";
  return (n/1024/1024/1024).toFixed(1) + "G";
}

function updateTokens(n) {
  totalTokens = n;
  const el = document.getElementById("tokenCount");
  if (!el) return;
  if (n > 0) {
    const pct = Math.min(100, (n / 128000) * 100);  // 128k context
    const ring = el.parentElement;
    if (ring && ring.classList.contains("token-ring")) {
      // SVG 링 갱신
      let svg = ring.querySelector("svg");
      if (!svg) {
        ring.insertAdjacentHTML("afterbegin",
          '<svg viewBox="0 0 24 24"><circle class="ring-bg" cx="12" cy="12" r="9"/>' +
          '<circle class="ring-fg" cx="12" cy="12" r="9" stroke-dasharray="0 100"/></svg>');
        svg = ring.querySelector("svg");
      }
      const c = svg.querySelector(".ring-fg");
      const circumference = 2 * Math.PI * 9;
      const dash = (pct / 100) * circumference;
      c.setAttribute("stroke-dasharray", `${dash} ${circumference - dash}`);
    }
    el.textContent = n.toLocaleString() + ` (${pct.toFixed(0)}%)`;
  } else {
    el.textContent = "—";
  }
}
