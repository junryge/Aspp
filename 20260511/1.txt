# -*- coding: utf-8 -*-
"""
M16A HUBROOM 수집 + 예측 동시 실행
- 수집기 스레드 시작
- 0.1초 뒤 예측기 watch 시작 (메인)
- Ctrl+C 한 번으로 같이 종료
"""

import threading
import time

import aws_idc_realtime_collector as collector
import hubroom_predictor as predictor

# 수집기 (백그라운드 스레드, 데몬 — 메인 종료 시 같이 죽음)
threading.Thread(target=collector.main, daemon=True).start()

# 0.1초 후 예측기 watch (메인 스레드)
time.sleep(0.1)
predictor.watch(str(predictor.DEFAULT_INPUT), str(predictor.DEFAULT_OUTDIR))
