<script>
let allSkills = [];
let selected = new Set();
let history = [];
let sending = false;

// ─── Theme ───
function applyTheme(t) {
  document.documentElement.setAttribute("data-theme", t);
  localStorage.setItem("hermes_theme", t);
}
function toggleTheme() {
  const cur = document.documentElement.getAttribute("data-theme") || "light";
  applyTheme(cur === "dark" ? "light" : "dark");
}
applyTheme(localStorage.getItem("hermes_theme") || "light");

// ─── Health ───
async function refreshHealth() {
  try {
    const r = await fetch("/api/health");
    const d = await r.json();
    document.getElementById("envBadge").textContent = d.env_mode;
    const envDot = document.getElementById("envDot");
    envDot.classList.remove("ok", "bad");
    envDot.classList.add(d.skills_dir_exists ? "ok" : "bad");

    document.getElementById("topSub").textContent =

    const pDot = document.getElementById("proxyDot");
    pDot.classList.remove("ok", "bad");
    pDot.classList.add(d.proxy_alive ? "ok" : "bad");
    document.getElementById("proxyText").textContent =
      "PROXY " + (d.proxy_alive ? "ONLINE" : "OFFLINE");
  } catch (e) {
    document.getElementById("topSub").textContent = "CONNECTION ERROR";
  }
}

// ─── Skills ───
async function loadSkills() {
  try {
    const r = await fetch("/api/skills");
    const d = await r.json();
    allSkills = d.skills || [];
    populateCategories(d.categories || [], allSkills);
    document.getElementById("skillCount").textContent = `${allSkills.length} SKILLS`;
    renderSkills();
  } catch (e) {
    document.getElementById("skillList").innerHTML =
      '<div style="padding:24px;color:var(--text-mute);font-size:11px;text-align:center;letter-spacing:0.1em;">SKILLS 로드 실패</div>';
  }
}
function populateCategories(cats, skills) {
  const sel = document.getElementById("categorySel");
  if (!sel || !cats.length) return;
  // 카테고리별 카운트
  const counts = {};
  skills.forEach(s => { counts[s.category] = (counts[s.category] || 0) + 1; });
  const total = skills.length;
  const cur = sel.value || "all";
  sel.innerHTML = cats.map(c => {
    const n = c.key === "all" ? total : (counts[c.key] || 0);
    return `<option value="${c.key}">${c.name} (${n})</option>`;
  }).join("");
  sel.value = cats.some(c => c.key === cur) ? cur : "all";
}
function renderSkills() {
  const q = document.getElementById("search").value.trim().toLowerCase();
  const catSel = document.getElementById("categorySel");
  const cat = catSel ? catSel.value : "all";
  const list = document.getElementById("skillList");
  let filtered = allSkills;
  if (cat && cat !== "all") {
    filtered = filtered.filter(s => (s.category || "etc") === cat);
  }
  if (q) {
    filtered = filtered.filter(s =>
      (s.id || "").toLowerCase().includes(q) ||
      (s.name || "").toLowerCase().includes(q) ||
      (s.description || "").toLowerCase().includes(q) ||
      (s.description_kr || "").toLowerCase().includes(q));
  }
  list.innerHTML = filtered.map(s => {
    const id = s.id, name = s.name || id;
    // 한글 캐시 있으면 우선 사용, 없으면 영문
    const desc = (s.description_kr && s.description_kr.trim()) ? s.description_kr : (s.description || "");
    const checked = selected.has(id) ? "checked" : "";
    const klass = selected.has(id) ? "skill checked" : "skill";
    return `<label class="${klass}">
      <input type="checkbox" ${checked} onchange="toggleSkill('${id.replace(/'/g, "\\'")}', this.checked, this)">
      <div class="skill-body">
        <div class="skill-name">${escapeHtml(name)}</div>
        ${desc ? `<div class="skill-desc">${escapeHtml(desc)}</div>` : ''}
      </div>
    </label>`;
  }).join("");
  updateSelectedCount();
}
function escapeHtml(s) {
  return (s || "").replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
}

// ─── 마크다운 → 안전한 HTML 렌더링 ───
// marked.js + DOMPurify 사용. 표/헤더/코드블록/리스트 다 처리.
if (window.marked) {
  marked.setOptions({
    gfm: true,           // GitHub Flavored Markdown (표 등)
    breaks: true,        // 줄바꿈 그대로
    headerIds: false,
    mangle: false,
  });
}
function renderBubble(el, text) {
  if (!text) { el.textContent = ""; return; }
  try {
    const html = window.marked ? marked.parse(text) : escapeHtml(text);
    const safe = window.DOMPurify ? DOMPurify.sanitize(html) : html;
    el.innerHTML = safe;
  } catch (e) {
    el.textContent = text;
  }
}
function toggleSkill(id, checked, el) {
  if (checked) selected.add(id); else selected.delete(id);
  if (el) el.closest(".skill").classList.toggle("checked", checked);
  updateSelectedCount();
}
function updateSelectedCount() {
  document.getElementById("selectedCount").textContent = `${selected.size} SELECTED`;
}
function clearSkills() { selected.clear(); renderSkills(); }
function reloadSkills() { loadSkills(); }
document.getElementById("search").addEventListener("input", renderSkills);

// ─── 한글 번역 (배치 반복) ───
let translating = false;
async function translateSkills() {
  if (translating) return;
  const btn = document.getElementById("trBtn");
  const prog = document.getElementById("trProgress");
  translating = true;
  btn.disabled = true;
  prog.style.display = "block";
  let totalDone = 0;
  let cumStart = 0;
  try {
    while (true) {
      const r = await fetch("/api/translate-skills", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ batch_size: 10, only_missing: true })
      });
      const d = await r.json();
      if (d.error) { prog.textContent = "에러: " + d.error; break; }
      totalDone += (d.translated || 0);
      const cachedTotal = d.cached_total || 0;
      const total = d.total || allSkills.length;
      prog.textContent = `번역 중… ${cachedTotal} / ${total}  (이번 배치: ${d.translated}/${d.batch_size})`;
      if (d.status === "complete" || !d.remaining) break;
      // UI 업데이트 (중간중간 한글 반영)
      await loadSkills();
    }
    prog.textContent = `완료 — 총 ${totalDone} 항목 번역됨. 새로고침 적용 완료.`;
    setTimeout(() => { prog.style.display = "none"; }, 4000);
    await loadSkills();
  } catch (e) {
    prog.textContent = "실패: " + e.message;
  } finally {
    translating = false;
    btn.disabled = false;
  }
}

// ─── Messages ───
function addMsg(role, content, opts) {
  opts = opts || {};
  const wrap = document.createElement("div");
  wrap.className = "msg " + role + (opts.error ? " error" : "");
  const who = document.createElement("div"); who.className = "who";
  who.textContent = role === "user" ? "YOU" : "HERMES";
  const bub = document.createElement("div"); bub.className = "bubble";
  bub.textContent = content;
  wrap.appendChild(who); wrap.appendChild(bub);
  if (opts.meta) {
    const m = document.createElement("div"); m.className = "meta-line";
    m.textContent = opts.meta; wrap.appendChild(m);
  }
  document.getElementById("msgs").appendChild(wrap);
  const c = document.getElementById("msgs");
  c.scrollTop = c.scrollHeight;
  return bub;
}

// ─── Send ───
function setSending(v) {
  // 무조건 READY — sending 잠금 안 함
  sending = false;
  const dot = document.getElementById("sendDot");
  const txt = document.getElementById("sendText");

function forceReset() {

  setSending(false);
}
window.forceReset = forceReset;
setSending(false);  // 초기 상태

let currentAbort = null;

async function send() {
  console.log("[hermes] send() called, sending=" + sending);
  // 이전 요청이 아직 sending 이면 강제 취소 후 진행
  if (sending) {
    console.warn("[hermes] 이전 요청 중인데 새 요청 → 이전 abort 후 진행");
    if (currentAbort) { try { currentAbort.abort(); } catch(_) {} }
    setSending(false);
  }
  const ta = document.getElementById("msg");
  const text = (ta.value || "").trim();
  if (!text) return;
  ta.value = ""; ta.style.height = "44px";
  history.push({ role: "user", content: text });
  addMsg("user", text);

  const skillIds = Array.from(selected);
  setSending(true);

  // 안전망: 5분 후에도 끝 안 나면 강제 해제
  const safetyTimer = setTimeout(() => {
    if (sending) {
      console.warn("[hermes] 5분 타임아웃 — 강제 해제");
      setSending(false);
      bubble.classList.remove("typing");
      if (!acc.trim()) bubble.textContent = "⏱ 응답 시간 초과 (5분)";
    }
  }, 300000);

  const bubble = addMsg("assistant", "");
  bubble.classList.add("typing");
  let acc = "";

  currentAbort = new AbortController();

  try {
    const resp = await fetch("/api/chat", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      signal: currentAbort.signal,
      body: JSON.stringify({ messages: history, skills: skillIds }),
    });
    if (!resp.ok || !resp.body) throw new Error("SSE start failed: " + resp.status);

    const reader = resp.body.getReader();
    const decoder = new TextDecoder();
    let buf = "";

    outer: while (true) {
      const { value, done } = await reader.read();
      if (done) break outer;
      buf += decoder.decode(value, { stream: true });
      let nl;
      while ((nl = buf.indexOf("\n\n")) !== -1) {
        const evt = buf.slice(0, nl);
        buf = buf.slice(nl + 2);
        if (!evt.startsWith("data:")) continue;
        const payload = evt.slice(5).trim();
        if (payload === "[DONE]") {
          console.log("[hermes] [DONE] received, breaking outer");
          try { await reader.cancel(); } catch(_) {}
          break outer;
        }
        try {
          const obj = JSON.parse(payload);
          if (obj.type === "meta") {
            const wrap = bubble.parentElement;
            const m = wrap.querySelector(".meta-line") || (() => {
              const d = document.createElement("div"); d.className = "meta-line";
              wrap.appendChild(d); return d;
            })();
            m.textContent = obj.info || "";
          } else if (obj.type === "token") {
            acc += obj.t || "";
            renderBubble(bubble, acc);
            bubble.classList.remove("typing");
            const c = document.getElementById("msgs");
            c.scrollTop = c.scrollHeight;
          } else if (obj.type === "error") {
            bubble.classList.remove("typing");
            bubble.parentElement.classList.add("error");
            bubble.textContent = "✦ " + (obj.error || "오류") +
              (obj.detail ? "\n→ " + obj.detail : "");
            acc = bubble.textContent;
          } else if (obj.type === "end") {
            bubble.classList.remove("typing");
            const stats = obj.stats || {};
            const m = bubble.parentElement.querySelector(".meta-line");
            if (m) m.textContent += `  ·  ${stats.emitted ?? 0} LINES`;
          }
        } catch (e) {}
      }
    }
    bubble.classList.remove("typing");
    if (acc.trim()) history.push({ role: "assistant", content: acc });
  } catch (e) {
    bubble.classList.remove("typing");
    bubble.parentElement.classList.add("error");
    bubble.textContent = "✦ 전송 실패: " + e.message;
  } finally {
    clearTimeout(safetyTimer);
    setSending(false);
    bubble.classList.remove("typing");
    console.log("[hermes] send() finally — sending=false, button enabled");
  }
}

// ─── Input ───
const ta = document.getElementById("msg");
ta.addEventListener("keydown", e => {
  if (e.key === "Enter" && !e.shiftKey) {
    e.preventDefault();
    // sending 상태로 막혀있으면 강제 풀고 보냄
    if (sending) {
      console.warn("[hermes] Enter 입력 — sending=true 자동 해제 후 진행");
      forceReset();
    }
    send();
  }
});
ta.addEventListener("input", () => {
  ta.style.height = "44px";
  ta.style.height = Math.min(ta.scrollHeight, 200) + "px";
});
