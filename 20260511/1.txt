찾았다. .hermes_history 가 옛 sid 기억하는 범인. 그리고 state.db-shm / state.db-wal 도 같이 안 지우면 SQLite 가 옛 상태로 복원함.

:: hermes 죽이고
taskkill /F /IM hermes.exe 2>nul
taskkill /F /IM python.exe 2>nul

:: SQLite 3종 세트 한 방에
del /F /Q "%USERPROFILE%\.hermes\state.db" "%USERPROFILE%\.hermes\state.db-shm" "%USERPROFILE%\.hermes\state.db-wal" 2>nul

:: 옛 sid 기억하는 history 파일
del /F /Q "%USERPROFILE%\.hermes\.hermes_history" 2>nul

:: snapshot 캐시도 (안전)
del /F /Q "%USERPROFILE%\.hermes\.skills_prompt_snapshot.json" 2>nul

:: 새 세션 시작
hermes chat -q "안녕"

type "%USERPROFILE%\.hermes\.env"
