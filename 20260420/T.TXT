"""
SK Hynix LLM-CHAT
실행하면 먼저 모델 고르고 대화 시작
"""
import requests
import json
import base64
import time


# ============================================================
# 토큰 (F12 → Network → authorization 에서 복사)
# ============================================================
TOKEN = "여기에_토큰_붙여넣기"


# ============================================================
# 사용 가능한 모델 10개
# ============================================================
MODELS = [
    ("Qwen/Qwen3.5-397B-A17B-FP8",       "Qwen3.5-397B-A17B"),
    ("Qwen/Qwen3.5-397B-A17B-thinking",  "Qwen3.5-397B-A17B-Thinking"),
    ("zai-org/GLM-5-thinking",           "GLM-5-Thinking"),
    ("Kimi-K2.5",                        "Kimi-K2.5"),
    ("Kimi-K2.5-thinking",               "Kimi-K2.5-Thinking"),
]


# ============================================================
# 설정
# ============================================================
URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com/api/chat/completions"
SYSTEM = "당신은 SK하이닉스 반도체 FAB 물류(AMHS/OHT) 전문가입니다. 한국어로 답변합니다."


# ============================================================
# JWT 만료 확인
# ============================================================
def check_token(token):
    try:
        parts = token.split(".")
        payload = parts[1] + "=" * (4 - len(parts[1]) % 4)
        data = json.loads(base64.urlsafe_b64decode(payload))
        remaining = data.get("exp", 0) - int(time.time())
        return remaining // 60, data
    except Exception:
        return None, None


def show_status(token):
    remaining, data = check_token(token)
    if remaining is None:
        return False
    if remaining <= 0:
        print(f"❌ 토큰 만료됨 ({-remaining}분 지남)")
        return False
    h, m = remaining // 60, remaining % 60
    time_str = f"{h}시간 {m}분" if h else f"{m}분"
    print(f"✓ 토큰 유효 (남은시간: {time_str})")
    return True


# ============================================================
# 모델 선택 화면
# ============================================================
def pick_model():
    print("\n━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━")
    print("  사용할 모델을 선택하세요")
    print("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━")
    for i, (_, label) in enumerate(MODELS, 1):
        print(f"  [{i:>2}] {label}")
    print("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━")

    while True:
        choice = input("\n번호 선택 > ").strip()
        if choice.isdigit() and 1 <= int(choice) <= len(MODELS):
            model_id, label = MODELS[int(choice) - 1]
            print(f"✓ 선택됨: {label}")
            return model_id, label
        print("❌ 1~10 숫자만 입력하세요")


# ============================================================
# 챗봇
# ============================================================
def ask(history, question, model):
    history.append({"role": "user", "content": question})
    r = requests.post(
        URL,
        headers={"Authorization": f"Bearer {TOKEN}", "Content-Type": "application/json"},
        json={"model": model, "messages": history, "stream": False},
        timeout=300,
    )
    r.raise_for_status()
    answer = r.json()["choices"][0]["message"]["content"]
    history.append({"role": "assistant", "content": answer})
    return answer


def main():
    if TOKEN == "여기에_토큰_붙여넣기":
        print("❌ TOKEN에 JWT 붙여넣으세요!")
        return

    print("=" * 50)
    print("  SK Hynix LLM-CHAT")
    print("=" * 50)

    if not show_status(TOKEN):
        return

    # 모델 먼저 선택
    model, model_label = pick_model()

    print()
    print("=" * 50)
    print(f"  🤖 {model_label}")
    print("  명령어: /model(변경) /reset(초기화) /exit(종료)")
    print("=" * 50)

    history = [{"role": "system", "content": SYSTEM}]

    while True:
        try:
            q = input("\n🧑 > ").strip()
        except (KeyboardInterrupt, EOFError):
            break

        if not q:
            continue
        if q in ("/exit", "/quit", "/종료", "exit", "quit"):
            break
        if q in ("/reset", "/초기화"):
            history = [history[0]]
            print("✓ 대화 초기화")
            continue
        if q in ("/model", "/모델"):
            model, model_label = pick_model()
            history = [history[0]]  # 모델 바꾸면 대화도 초기화
            print("✓ 대화 초기화됨")
            continue

        try:
            print("\n🤖 ", end="", flush=True)
            answer = ask(history, q, model)
            print(answer)
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                print("\n❌ 토큰 만료! 새 토큰 받아서 교체 후 재실행")
                break
            print(f"\n❌ HTTP {e.response.status_code}: {e.response.text[:200]}")
            history.pop()
        except Exception as e:
            print(f"\n❌ {e}")
            history.pop()


if __name__ == "__main__":
    main()
