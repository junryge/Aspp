"""
사용 가능한 모델 목록 조회
"""
import requests
import json

# 토큰 붙여넣기
TOKEN = "여기에_토큰_붙여넣기"

BASE = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com"

headers = {
    "Authorization": f"Bearer {TOKEN}",
    "Content-Type": "application/json",
}

# 모델 목록 조회 (여러 엔드포인트 시도)
endpoints = [
    "/api/models",
    "/api/v1/models",
    "/openai/models",
    "/api/tags",
]

for ep in endpoints:
    url = f"{BASE}{ep}"
    print(f"\n▶ GET {url}")
    try:
        r = requests.get(url, headers=headers, timeout=10)
        print(f"  Status: {r.status_code}")
        if r.status_code == 200:
            data = r.json()
            print(f"  응답:\n{json.dumps(data, ensure_ascii=False, indent=2)[:2000]}")
            break
    except Exception as e:
        print(f"  에러: {e}")
