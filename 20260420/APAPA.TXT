import uuid
import time

# ============================================================

#       authorization: Bearer <여기부분> 복사해서 교체)

TOKEN = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpZCI6IjkwMjkzZTI2LTA2ZDYtNGFhOS04Mzc1LTg2YmM1ZTkzZjA5MyIsImV4cCI6MTc3NjgyNjA5Miwicm9sZSI6InBhcnRuZXIifQ.E6zNWLsODq-bTyuT6q5IU8XyqtVeDs82k1YG5NglgbE"


class LLMChatClient:
    BASE_URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com"
    ENDPOINT = "/api/chat/completed"
    DEFAULT_MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"
    FRONTEND_ORIGIN = "http://llm-chat-frontend-basic-prd.api.aipp01.skhynix.com"

    def __init__(self, bearer_token: str, session_id: str = None):
        self.token = bearer_token
        self.session_id = session_id or self._gen_session_id()
        self.chat_id = str(uuid.uuid4())
        self.history = []

    @staticmethod
    def _gen_session_id(length: int = 20) -> str:
        chars = string.ascii_letters + string.digits
        return ''.join(random.choices(chars, k=length))

    def _headers(self) -> dict:
        return {
            "Content-Type": "application/json",
            "Accept": "application/json",
            "Accept-Language": "ko-KR,ko;q=0.9,en-US;q=0.8,en;q=0.7",
            "Authorization": f"Bearer {self.token}",
            "Origin": self.FRONTEND_ORIGIN,
            "Referer": f"{self.FRONTEND_ORIGIN}/",
            "User-Agent": (
                "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                "AppleWebKit/537.36 (KHTML, like Gecko) "
                "Chrome/147.0.0.0 Safari/537.36"
            ),
        }

    def chat(self, user_message: str, model: str = None,
             keep_history: bool = True, timeout: int = 120) -> dict:
        """
        메시지 전송 → 응답 JSON 반환
        keep_history=True 면 대화 맥락 유지
        """
        user_msg = {
            "id": str(uuid.uuid4()),
            "role": "user",
            "content": user_message,
            "timestamp": int(time.time()),
        }

        if keep_history:
            self.history.append(user_msg)
            messages = self.history
        else:
            messages = [user_msg]

        payload = {
            "model": model or self.DEFAULT_MODEL,
            "chat_id": self.chat_id,
            "id": str(uuid.uuid4()),
            "session_id": self.session_id,
            "messages": messages,
        }

        url = f"{self.BASE_URL}{self.ENDPOINT}"
        print(f"[REQUEST] POST {url}")
        print(f"[PAYLOAD] {json.dumps(payload, ensure_ascii=False)[:300]}...")

        resp = requests.post(url, headers=self._headers(),
                             json=payload, timeout=timeout)

        print(f"[STATUS] {resp.status_code}")

        if resp.status_code == 401:
            raise RuntimeError(
                "401 Unauthorized - JWT 토큰 만료됨.\n"
                "브라우저 F12 → Network → completed 요청 → "
                "Request Headers 의 authorization 값을 새로 복사해서 "
                "TOKEN 변수 교체하세요."
            )
        resp.raise_for_status()
        data = resp.json()

        # 어시스턴트 응답을 히스토리에 추가
        if keep_history:
            content = self._extract_content(data)
            self.history.append({
                "id": str(uuid.uuid4()),
                "role": "assistant",
                "content": content,
                "timestamp": int(time.time()),
            })

        return data

    @staticmethod
    def _extract_content(response: dict) -> str:
        """응답 JSON에서 어시스턴트 텍스트 추출 (여러 포맷 대응)"""
        # OpenAI 스타일
        if isinstance(response.get("choices"), list) and response["choices"]:
            choice = response["choices"][0]
            if isinstance(choice.get("message"), dict):
                return choice["message"].get("content", "")
            if "text" in choice:
                return choice["text"]
        # Ollama/Open WebUI 스타일
        if isinstance(response.get("message"), dict):
            return response["message"].get("content", "")
        # Plain
        if "content" in response:
            return response["content"]
        if "response" in response:
            return response["response"]
        # fallback - 전체 JSON
        return json.dumps(response, ensure_ascii=False)

# 실행 예시
# ============================================================
if __name__ == "__main__":
    client = LLMChatClient(bearer_token=TOKEN)

    print("=" * 60)
    print("LLM-CHAT 테스트")
    print("=" * 60)

    try:
        result = client.chat("안녕하세요, 당신은 누구인가요?")

        print("\n[FULL RESPONSE JSON]")
        print(json.dumps(result, ensure_ascii=False, indent=2)[:2000])

        print("\n[EXTRACTED CONTENT]")
        print(client._extract_content(result))

        # 2번째 턴 - 맥락 유지 테스트
        print("\n" + "=" * 60)
        print("2번째 턴 (맥락 유지 테스트)")
        print("=" * 60)
        result2 = client.chat("방금 내가 뭐라고 물어봤지?")
        print(client._extract_content(result2))

    except RuntimeError as e:
        print(f"\n[ERROR] {e}")
    except requests.exceptions.ConnectionError:
        print("\n[ERROR] 서버 접속 불가 - 폐쇄망 내부에서만 동작합니다")
    except requests.exceptions.HTTPError as e:
        print(f"\n[HTTP ERROR] {e}")
        print(f"Response: {e.response.text[:500]}")
