"""
SK Hynix LLM-CHAT 자동 토큰 버전
====================================
브라우저에서 한 번만 로그인해두면, 실행 시마다 자동으로 토큰을 가져옴.

필요 라이브러리:
    pip install requests browser-cookie3

동작 원리:
    1. 브라우저의 쿠키/localStorage에 저장된 JWT를 읽음
    2. 읽은 토큰으로 API 호출
    3. 401이면 브라우저로 다시 로그인하라고 안내

사용 전 준비:
    - Chrome 또는 Edge 로그인 상태 유지
    - LLM-CHAT 페이지 한 번 이상 방문 (쿠키 심어두기)
"""
import requests
import json
import os
import sys


# ============================================================
# 설정
# ============================================================
URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com/api/chat/completions"
DOMAIN = "llm-chat-frontend-basic-prd.api.aipp01.skhynix.com"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"
SYSTEM_PROMPT = "당신은 SK하이닉스 반도체 FAB 물류(AMHS/OHT) 전문가입니다. 한국어로 친절하고 정확하게 답변합니다."

# 토큰 캐시 파일 (수동 토큰도 여기 저장)
TOKEN_CACHE = os.path.join(os.path.dirname(__file__), ".llm_token")


# ============================================================
# 토큰 자동 추출 방법들
# ============================================================

def get_token_from_browser_cookies():
    """방법 A: 브라우저 쿠키에서 직접 추출 (Chrome/Edge/Firefox)"""
    try:
        import browser_cookie3
    except ImportError:
        return None

    # 여러 브라우저 시도
    for browser_fn in [browser_cookie3.chrome, browser_cookie3.edge, browser_cookie3.firefox]:
        try:
            cj = browser_fn(domain_name=DOMAIN)
            for cookie in cj:
                if cookie.name.lower() in ("token", "jwt", "access_token", "auth_token"):
                    if cookie.value and cookie.value.startswith("eyJ"):
                        print(f"✓ 토큰 발견: {browser_fn.__name__} 브라우저 쿠키")
                        return cookie.value
        except Exception:
            continue
    return None


def get_token_from_localstorage():
    """방법 B: Chrome의 localStorage에서 추출 (LevelDB 직접 읽기)"""
    try:
        import plyvel  # pip install plyvel
    except ImportError:
        return None

    # Chrome localStorage 경로 (Windows 기준)
    chrome_leveldb = os.path.expanduser(
        r"~\AppData\Local\Google\Chrome\User Data\Default\Local Storage\leveldb"
    )
    if not os.path.exists(chrome_leveldb):
        return None

    try:
        # Chrome이 켜져있으면 락 걸림 → 임시 복사해서 읽기
        import shutil, tempfile
        tmp = tempfile.mkdtemp()
        tmp_db = os.path.join(tmp, "leveldb")
        shutil.copytree(chrome_leveldb, tmp_db)

        db = plyvel.DB(tmp_db)
        for key, value in db:
            if DOMAIN.encode() in key and b"token" in key.lower():
                val = value.decode(errors="ignore")
                if "eyJ" in val:
                    import re
                    m = re.search(r"eyJ[A-Za-z0-9_\-\.]+", val)
                    if m:
                        print("✓ 토큰 발견: Chrome localStorage")
                        return m.group(0)
        db.close()
        shutil.rmtree(tmp)
    except Exception as e:
        print(f"  localStorage 읽기 실패: {e}")
    return None


def get_token_from_cache():
    """방법 C: 로컬 캐시 파일에서 읽기"""
    if os.path.exists(TOKEN_CACHE):
        with open(TOKEN_CACHE, "r") as f:
            token = f.read().strip()
            if token and token.startswith("eyJ"):
                print("✓ 토큰 발견: 캐시 파일")
                return token
    return None


def save_token_to_cache(token: str):
    """토큰을 캐시 파일에 저장"""
    with open(TOKEN_CACHE, "w") as f:
        f.write(token)
    print(f"✓ 토큰 캐시 저장: {TOKEN_CACHE}")


def get_token_manually():
    """방법 D: 수동 입력"""
    print("\n" + "=" * 60)
    print("  토큰을 자동으로 가져올 수 없습니다.")
    print("=" * 60)
    print("  수동 입력 방법:")
    print("    1. 브라우저에서 LLM-CHAT 페이지 접속")
    print("    2. F12 → Network 탭 → 'chat' 필터")
    print("    3. 아무 채팅 전송")
    print("    4. 요청 클릭 → Request Headers")
    print("    5. 'authorization: Bearer eyJ...' 에서 eyJ 시작 부분 복사")
    print("=" * 60)
    token = input("\n🔑 토큰 붙여넣기 > ").strip()
    if token.startswith("Bearer "):
        token = token[7:]
    if not token.startswith("eyJ"):
        print("❌ 유효하지 않은 토큰입니다.")
        return None
    save_token_to_cache(token)
    return token


def auto_get_token(force_refresh=False):
    """토큰 자동 획득 - 여러 방법 순차 시도"""
    print("\n🔍 토큰 자동 획득 중...")

    if not force_refresh:
        # 1) 캐시 우선
        token = get_token_from_cache()
        if token and is_token_valid(token):
            return token

    # 2) 브라우저 쿠키
    token = get_token_from_browser_cookies()
    if token:
        save_token_to_cache(token)
        return token

    # 3) Chrome localStorage
    token = get_token_from_localstorage()
    if token:
        save_token_to_cache(token)
        return token

    # 4) 수동 입력
    return get_token_manually()


def is_token_valid(token: str) -> bool:
    """JWT 만료시간 체크 (서명 검증 X, exp만 확인)"""
    try:
        import base64, time
        parts = token.split(".")
        if len(parts) != 3:
            return False
        # JWT payload는 base64url 인코딩
        payload = parts[1] + "=" * (4 - len(parts[1]) % 4)
        decoded = json.loads(base64.urlsafe_b64decode(payload))
        exp = decoded.get("exp", 0)
        now = int(time.time())
        remaining = exp - now
        if remaining > 0:
            print(f"  토큰 남은 시간: {remaining // 60}분")
            return True
        else:
            print(f"  토큰 만료됨 ({-remaining // 60}분 지남)")
            return False
    except Exception:
        return False


# ============================================================
# LLM API 호출
# ============================================================

def chat(messages: list, token: str) -> str:
    resp = requests.post(
        URL,
        headers={
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json",
        },
        json={
            "model": MODEL,
            "messages": messages,
            "stream": False,
            "temperature": 0.7,
        },
        timeout=180,
    )
    resp.raise_for_status()
    return resp.json()["choices"][0]["message"]["content"]


# ============================================================
# 메인 대화 루프
# ============================================================

def main():
    print("=" * 60)
    print("  SK Hynix LLM-CHAT (자동 토큰)")
    print(f"  Model: {MODEL}")
    print("=" * 60)

    token = auto_get_token()
    if not token:
        print("❌ 토큰 획득 실패")
        sys.exit(1)

    print("\n사용법:")
    print("  - 'exit': 종료  |  'reset': 초기화  |  'token': 토큰 갱신")
    print("=" * 60)

    history = [{"role": "system", "content": SYSTEM_PROMPT}]

    while True:
        try:
            q = input("\n🧑 당신> ").strip()
        except (KeyboardInterrupt, EOFError):
            print("\n👋 종료")
            break

        if not q:
            continue
        if q.lower() in ("exit", "quit", "종료"):
            print("👋 종료")
            break
        if q.lower() in ("reset", "clear", "초기화"):
            history = [{"role": "system", "content": SYSTEM_PROMPT}]
            print("✓ 초기화")
            continue
        if q.lower() in ("token", "토큰"):
            token = auto_get_token(force_refresh=True)
            if not token:
                break
            continue

        history.append({"role": "user", "content": q})
        try:
            print("\n🤖 AI>", end=" ", flush=True)
            answer = chat(history, token)
            print(answer)
            history.append({"role": "assistant", "content": answer})
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                print("\n⚠️  토큰 만료 - 자동 갱신 시도")
                token = auto_get_token(force_refresh=True)
                if not token:
                    break
                history.pop()  # 실패한 질문 제거하고 재시도 유도
            else:
                print(f"\n❌ HTTP {e.response.status_code}")
                history.pop()
        except Exception as e:
            print(f"\n❌ {e}")
            history.pop()


if __name__ == "__main__":
    main()
