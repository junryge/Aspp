"""
SK Hynix LLM-CHAT
==================
사용법:
  1. 아래 TOKEN에 JWT 붙여넣기
  2. python chat.py

기능:
  - 실행 시 토큰 남은 시간 자동 표시
  - 만료 30분 전 경고
  - 만료되면 교체 안내
"""
import requests
import json
import base64
import time


# ============================================================
# 토큰 (만료되면 아래 안내대로 교체)
# ============================================================
TOKEN = "여기에_토큰_붙여넣기"


# ============================================================
# 설정
# ============================================================
URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com/api/chat/completions"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"
SYSTEM = "당신은 SK하이닉스 반도체 FAB 물류 전문가입니다. 한국어로 답변합니다."


# ============================================================
# 토큰 만료 시간 확인
# ============================================================
def check_token_expiry(token):
    """JWT 파싱해서 남은 시간(분) 반환"""
    try:
        parts = token.split(".")
        payload = parts[1] + "=" * (4 - len(parts[1]) % 4)
        data = json.loads(base64.urlsafe_b64decode(payload))
        exp = data.get("exp", 0)
        remaining_sec = exp - int(time.time())
        remaining_min = remaining_sec // 60
        return remaining_min, data
    except Exception:
        return None, None


def print_token_status(token):
    """토큰 상태 예쁘게 출력"""
    remaining, data = check_token_expiry(token)

    if remaining is None:
        print("⚠ 토큰 형식 확인 불가")
        return False

    # 사용자 정보
    user_id = data.get("id", "?")
    role = data.get("role", "?")

    print(f"👤 사용자: {user_id} ({role})")

    # 만료 시간
    if remaining <= 0:
        print(f"❌ 토큰 만료됨 ({-remaining}분 지남)")
        print_howto_refresh()
        return False
    elif remaining < 30:
        print(f"⚠ 토큰 만료 임박! 남은 시간: {remaining}분")
        print("  미리 새 토큰 준비하세요.")
        return True
    else:
        hours = remaining // 60
        mins = remaining % 60
        if hours > 0:
            print(f"✓ 토큰 유효 (남은 시간: {hours}시간 {mins}분)")
        else:
            print(f"✓ 토큰 유효 (남은 시간: {mins}분)")
        return True


def print_howto_refresh():
    """토큰 갱신 방법 출력"""
    print()
    print("━" * 60)
    print("  🔑 토큰 갱신 방법")
    print("━" * 60)
    print("  1. 브라우저에서 LLM-CHAT 페이지 접속")
    print("  2. F12 누르기 → 상단에서 'Network' 탭 클릭")
    print("  3. 상단 필터박스에 입력:  completions")
    print("  4. LLM-CHAT 채팅창에 아무거나 전송 (예: 'a')")
    print("  5. 새로 뜬 요청 클릭 → 오른쪽 'Headers' 탭")
    print("  6. 'Request Headers' 안에서 이거 찾기:")
    print("     authorization: Bearer eyJ...")
    print("  7. 'Bearer ' 뒤의 eyJ로 시작하는 긴 문자열 드래그 복사")
    print("  8. chat.py 열어서 TOKEN = \"...\" 안쪽 교체")
    print("  9. 저장 후 재실행")
    print("━" * 60)


# ============================================================
# 챗봇
# ============================================================
def ask(history, question):
    history.append({"role": "user", "content": question})
    r = requests.post(
        URL,
        headers={"Authorization": f"Bearer {TOKEN}", "Content-Type": "application/json"},
        json={"model": MODEL, "messages": history, "stream": False},
        timeout=180,
    )
    r.raise_for_status()
    answer = r.json()["choices"][0]["message"]["content"]
    history.append({"role": "assistant", "content": answer})
    return answer


def main():
    print("=" * 60)
    print("  SK Hynix LLM-CHAT")
    print("=" * 60)

    # 토큰 체크
    if TOKEN == "여기에_토큰_붙여넣기":
        print("❌ 먼저 TOKEN에 JWT 붙여넣으세요!")
        print_howto_refresh()
        return

    # 토큰 상태 확인
    if not print_token_status(TOKEN):
        return

    print()
    print("  명령어: exit(종료) | reset(초기화) | status(토큰상태)")
    print("=" * 60)

    history = [{"role": "system", "content": SYSTEM}]

    while True:
        try:
            q = input("\n🧑 > ").strip()
        except (KeyboardInterrupt, EOFError):
            break
        if not q:
            continue
        if q.lower() in ("exit", "quit", "종료"):
            break
        if q.lower() in ("reset", "초기화"):
            history = [history[0]]
            print("✓ 대화 초기화")
            continue
        if q.lower() in ("status", "상태"):
            print_token_status(TOKEN)
            continue

        # 보내기 전에 토큰 만료 체크
        remaining, _ = check_token_expiry(TOKEN)
        if remaining is not None and remaining <= 0:
            print("\n❌ 토큰이 만료되었습니다!")
            print_howto_refresh()
            break

        try:
            print(f"\n🤖 {ask(history, q)}")

            # 대화 중에도 만료 임박 알림
            remaining, _ = check_token_expiry(TOKEN)
            if remaining is not None and 0 < remaining < 10:
                print(f"\n⚠ 토큰 만료까지 {remaining}분 남음! 곧 교체하세요.")

        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                print("\n❌ 토큰 만료됨!")
                print_howto_refresh()
                break
            print(f"\n❌ 에러: {e}")
            history.pop()
        except Exception as e:
            print(f"\n❌ 에러: {e}")
            history.pop()


if __name__ == "__main__":
    main()
