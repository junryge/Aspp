"""
SK Hynix LLM-CHAT 완전 자동화 (Playwright)
============================================
처음 한 번만 설정하면 됩니다.

[처음 1번만]
    pip install playwright requests
    playwright install chromium

[그 다음부터]
    python auto_chat.py   ← 이것만 실행

동작:
    1. 브라우저 창 자동으로 열림
    2. 처음 실행시: HCP SSO 로그인 하세요 (1번만)
    3. 로그인되면 토큰 자동 획득
    4. 브라우저는 백그라운드에 남고 챗봇 시작
    5. 다음 실행부터는 로그인 세션 기억해서 자동 로그인
"""
import requests
import json
import os
import sys
import time


# ============================================================
# 설정
# ============================================================
CHAT_URL_FRONTEND = "http://llm-chat.skhynix.com/apps/llm-chat"
CHAT_URL_BACKEND = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com/api/chat/completions"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"
SYSTEM_PROMPT = "당신은 SK하이닉스 반도체 FAB 물류 전문가입니다. 한국어로 답변합니다."

# 브라우저 세션 저장 폴더 (로그인 유지)
PROFILE_DIR = os.path.join(os.path.dirname(os.path.abspath(__file__)), ".browser_profile")


# ============================================================
# Playwright로 토큰 자동 획득
# ============================================================

def get_token_auto():
    """브라우저 자동으로 띄워서 토큰 가져오기"""
    try:
        from playwright.sync_api import sync_playwright
    except ImportError:
        print("❌ 라이브러리 설치 필요:")
        print("   pip install playwright requests")
        print("   playwright install chromium")
        sys.exit(1)

    print("🌐 브라우저 실행 중...")

    with sync_playwright() as p:
        # 영구 프로파일로 실행 (로그인 세션 유지)
        ctx = p.chromium.launch_persistent_context(
            user_data_dir=PROFILE_DIR,
            headless=False,  # 브라우저 창 보이기
            args=["--disable-blink-features=AutomationControlled"],
        )

        page = ctx.new_page()

        # 네트워크 요청 가로채서 토큰 추출
        token_container = {"token": None}

        def on_request(request):
            """모든 요청의 Authorization 헤더 감시"""
            auth = request.headers.get("authorization", "")
            if auth.startswith("Bearer ey") and not token_container["token"]:
                token = auth[7:]  # "Bearer " 제거
                token_container["token"] = token
                print(f"✓ 토큰 자동 획득 ({len(token)}자)")

        page.on("request", on_request)

        # LLM-CHAT 페이지 열기
        print(f"📍 접속: {CHAT_URL_FRONTEND}")
        try:
            page.goto(CHAT_URL_FRONTEND, timeout=60000)
        except Exception as e:
            print(f"⚠ 페이지 로드 지연 (계속 진행): {e}")

        # 토큰 획득까지 대기 (최대 3분)
        print("\n" + "=" * 60)
        print("  브라우저에서 로그인 해주세요 (처음 1번만)")
        print("  로그인되면 자동으로 토큰 획득합니다")
        print("=" * 60)

        for i in range(180):  # 최대 180초 대기
            if token_container["token"]:
                break
            time.sleep(1)
            if i % 10 == 9:
                print(f"  ... 대기 중 ({i+1}초)")

        if not token_container["token"]:
            # 직접 JS 실행으로 시도
            print("\n🔍 토큰 재검색 중...")
            try:
                # iframe 내부 확인
                for frame in page.frames:
                    try:
                        t = frame.evaluate("() => localStorage.getItem('token') || localStorage.getItem('authToken') || ''")
                        if t and t.startswith("eyJ"):
                            token_container["token"] = t
                            print(f"✓ iframe localStorage에서 획득")
                            break
                    except Exception:
                        continue
            except Exception as e:
                print(f"  재검색 실패: {e}")

        ctx.close()

        if not token_container["token"]:
            print("\n❌ 토큰 획득 실패")
            print("   브라우저에서 로그인이 완료되었는지 확인하세요")
            return None

        # 토큰 캐시 저장
        cache_file = os.path.join(os.path.dirname(__file__), ".token_cache")
        with open(cache_file, "w") as f:
            f.write(token_container["token"])
        print(f"💾 토큰 저장: {cache_file}")

        return token_container["token"]


def get_cached_token():
    """캐시된 토큰 확인 (만료 체크)"""
    cache_file = os.path.join(os.path.dirname(__file__), ".token_cache")
    if not os.path.exists(cache_file):
        return None

    with open(cache_file, "r") as f:
        token = f.read().strip()

    if not token.startswith("eyJ"):
        return None

    # JWT 만료 확인
    try:
        import base64
        parts = token.split(".")
        payload = parts[1] + "=" * (4 - len(parts[1]) % 4)
        data = json.loads(base64.urlsafe_b64decode(payload))
        exp = data.get("exp", 0)
        remaining = exp - int(time.time())
        if remaining > 120:  # 2분 이상 남음
            print(f"✓ 캐시 토큰 사용 (남은 시간: {remaining // 60}분)")
            return token
        else:
            print("⚠ 캐시 토큰 만료됨")
            return None
    except Exception:
        return None


# ============================================================
# LLM 호출
# ============================================================

def chat_api(messages, token):
    r = requests.post(
        CHAT_URL_BACKEND,
        headers={
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json",
        },
        json={
            "model": MODEL,
            "messages": messages,
            "stream": False,
            "temperature": 0.7,
        },
        timeout=180,
    )
    r.raise_for_status()
    return r.json()["choices"][0]["message"]["content"]


# ============================================================
# 메인
# ============================================================

def main():
    print("=" * 60)
    print("  SK Hynix LLM-CHAT 자동 버전")
    print("=" * 60)

    # 1. 캐시 확인
    token = get_cached_token()

    # 2. 없으면 브라우저로 자동 획득
    if not token:
        token = get_token_auto()

    if not token:
        print("\n❌ 토큰 획득 실패")
        sys.exit(1)

    # 3. 챗봇 시작
    print(f"\n{'=' * 60}")
    print("  대화 시작 (exit: 종료, reset: 초기화)")
    print("=" * 60)

    history = [{"role": "system", "content": SYSTEM_PROMPT}]

    while True:
        try:
            q = input("\n🧑 > ").strip()
        except (KeyboardInterrupt, EOFError):
            print("\n👋")
            break

        if not q:
            continue
        if q.lower() in ("exit", "quit", "종료"):
            break
        if q.lower() in ("reset", "초기화"):
            history = [history[0]]
            print("✓ 초기화")
            continue

        history.append({"role": "user", "content": q})
        try:
            print("\n🤖 ", end="", flush=True)
            answer = chat_api(history, token)
            print(answer)
            history.append({"role": "assistant", "content": answer})
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                print("\n⚠ 토큰 만료 → 브라우저 재열림")
                history.pop()
                # 캐시 삭제 후 재획득
                cache_file = os.path.join(os.path.dirname(__file__), ".token_cache")
                if os.path.exists(cache_file):
                    os.remove(cache_file)
                token = get_token_auto()
                if not token:
                    break
                print("✓ 토큰 갱신! 다시 질문하세요")
            else:
                print(f"\n❌ HTTP {e.response.status_code}")
                history.pop()
        except Exception as e:
            print(f"\n❌ {e}")
            history.pop()


if __name__ == "__main__":
    main()
