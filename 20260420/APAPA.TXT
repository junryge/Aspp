따라서 표준 OpenAI 호환 엔드포인트 사용:
    POST /api/chat/completions

사용법:
    1. TOKEN 에 최신 JWT 입력
    2. python llm_chat_client_v2.py
"""

TOKEN = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpZCI6IjkwMjkzZTI2LTA2ZDYtNGFhOS04Mzc1LTg2YmM1ZTkzZjA5MyIsImV4cCI6MTc3NjgyNjA5Miwicm9sZSI6InBhcnRuZXIifQ.E6zNWLsODq-bTyuT6q5IU8XyqtVeDs82k1YG5NglgbE"

BASE_URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"


class LLMChat:
    """Open WebUI 표준 API 클라이언트"""

    def __init__(self, token: str, base_url: str = BASE_URL, model: str = MODEL):
        self.token = token
        self.base_url = base_url
        self.model = model
        self.history = []

    def _headers(self) -> dict:
        return {
            "Authorization": f"Bearer {self.token}",
            "Content-Type": "application/json",
            "Accept": "application/json",
            "Origin": "http://llm-chat-frontend-basic-prd.api.aipp01.skhynix.com",
            "Referer": "http://llm-chat-frontend-basic-prd.api.aipp01.skhynix.com/",
        }

    def ask(self, question: str, system: str = None,
            keep_history: bool = True, temperature: float = 0.7) -> str:
        """질문 → 답변 반환"""
        messages = []

        if system:
            messages.append({"role": "system", "content": system})

        if keep_history:
            messages.extend(self.history)

        messages.append({"role": "user", "content": question})

        payload = {
            "model": self.model,
            "messages": messages,
            "stream": False,
            "temperature": temperature,
        }

        # 시도할 엔드포인트들 (우선순위 순)
        endpoints = [
            "/api/chat/completions",     # Open WebUI 표준
            "/openai/chat/completions",  # OpenAI 프록시
            "/ollama/v1/chat/completions",  # Ollama 프록시
            "/api/v1/chat/completions",
        ]

        last_error = None
        for ep in endpoints:
            url = f"{self.base_url}{ep}"
            try:
                print(f"[TRY] POST {url}")
                resp = requests.post(url, headers=self._headers(),
                                     json=payload, timeout=180)
                print(f"  → STATUS {resp.status_code}")

                if resp.status_code == 404:
                    last_error = f"404 at {ep}"
                    continue
                if resp.status_code == 401:
                    raise RuntimeError("토큰 만료됨 - 새로 발급받아 TOKEN 교체")

                resp.raise_for_status()
                data = resp.json()

                # 답변 추출
                answer = self._extract(data)

                # 답변이 있으면 성공
                if answer and answer.strip() and not self._is_echo(answer, question):
                    print(f"  ✓ SUCCESS (엔드포인트: {ep})")
                    if keep_history:
                        self.history.append({"role": "user", "content": question})
                        self.history.append({"role": "assistant", "content": answer})
                    return answer
                else:
                    last_error = f"빈 응답 or 에코 at {ep}"
                    print(f"  ✗ 에코 응답 (계속 시도)")
                    continue

            except requests.exceptions.HTTPError as e:
                last_error = f"HTTP {e.response.status_code} at {ep}"
                print(f"  ✗ {last_error}")
                continue
            except requests.exceptions.ConnectionError:
                raise RuntimeError("서버 접속 불가 (폐쇄망 확인)")

        raise RuntimeError(f"모든 엔드포인트 실패. 마지막: {last_error}")

    @staticmethod
    def _extract(response: dict) -> str:
        """응답 JSON에서 답변 텍스트 추출"""
        # OpenAI 표준: choices[0].message.content
        if isinstance(response.get("choices"), list) and response["choices"]:
            choice = response["choices"][0]
            if isinstance(choice.get("message"), dict):
                return choice["message"].get("content", "")
            if "text" in choice:
                return choice["text"]
            if "delta" in choice:
                return choice["delta"].get("content", "")

        # Ollama 스타일
        if isinstance(response.get("message"), dict):
            return response["message"].get("content", "")

        # 기타
        if "content" in response:
            return response["content"]
        if "response" in response:
            return response["response"]

        return ""

    @staticmethod
    def _is_echo(answer: str, question: str) -> bool:
        """응답이 질문 에코인지 판별"""
        if not answer:
            return True
        # 응답에 model/messages 키가 들어있으면 JSON 에코
        if '"messages"' in answer and '"role"' in answer:
            return True
        return False

    def reset(self):
        """대화 히스토리 초기화"""
        self.history = []
