"""

[원리]
    Chrome을 '디버깅 모드'로 실행하면 F12(DevTools)에
    파이썬이 직접 접속할 수 있음 (Chrome DevTools Protocol).
    
    F12에서 Network 탭 볼 수 있듯이, 파이썬도 똑같이
    모든 네트워크 요청을 볼 수 있어서 Authorization 헤더에서
    JWT를 자동으로 긁어옴.

[준비 - 처음 1번만]
    Chrome 바로가기에 디버깅 포트 추가:
    
    1. Chrome 바로가기 우클릭 → 속성
    2. '대상' 필드 끝에 추가:
       --remote-debugging-port=9222
    3. 예시:
       "C:\Program Files\Google\Chrome\Application\chrome.exe" --remote-debugging-port=9222
    4. 저장

[사용]
    1. 위 바로가기로 Chrome 실행
    2. http://llm-chat.skhynix.com 접속 & 로그인
    3. python chrome_devtools.py 실행
    4. 끝!
"""
import json
import time
import sys
import requests
import websocket  # pip install websocket-client (또는 표준라이브러리 socket)
import threading


# ============================================================
# 설정
# ============================================================
CHROME_DEBUG_PORT = 9222
LLM_URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com/api/chat/completions"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"
SYSTEM_PROMPT = "당신은 SK하이닉스 반도체 FAB 물류 전문가입니다. 한국어로 답변합니다."


# ============================================================
# Chrome DevTools 접속
# ============================================================

def find_llm_chat_tab():
    """열려있는 Chrome 탭 중 LLM-CHAT 탭 찾기"""
    try:
        resp = requests.get(f"http://127.0.0.1:{CHROME_DEBUG_PORT}/json", timeout=5)
        tabs = resp.json()
    except Exception as e:
        print(f"❌ Chrome 디버깅 포트({CHROME_DEBUG_PORT})에 접속 못함")
        print(f"   에러: {e}")
        print()
        print("해결 방법:")
        print(f"  1. Chrome을 디버깅 모드로 실행해야 함")
        print(f'  2. cmd에서 실행:')
        print(f'     chrome.exe --remote-debugging-port={CHROME_DEBUG_PORT}')
        print(f"  3. 또는 Chrome 바로가기 속성 → 대상 끝에 추가:")
        print(f"     --remote-debugging-port={CHROME_DEBUG_PORT}")
        return None

    # LLM-CHAT 탭 찾기
    for tab in tabs:
        url = tab.get("url", "")
        if "llm-chat" in url or "skhynix.com" in url:
            print(f"✓ LLM-CHAT 탭 발견: {tab.get('title', '')}")
            return tab

    print("❌ LLM-CHAT 탭이 안 열려있음")
    print(f"   Chrome에서 http://llm-chat.skhynix.com 접속하세요")
    print(f"\n현재 열려있는 탭:")
    for tab in tabs:
        print(f"   - {tab.get('title', '?')}: {tab.get('url', '?')[:60]}")
    return None


# ============================================================
# 방법 A: localStorage / sessionStorage 직접 읽기
# ============================================================

def get_token_via_storage(tab):
    """JavaScript 실행해서 localStorage/sessionStorage/cookie 전부 뒤짐"""
    ws_url = tab["webSocketDebuggerUrl"]
    print(f"🔌 DevTools 연결: {ws_url[:50]}...")

    ws = websocket.create_connection(ws_url, timeout=10)

    # Runtime.evaluate 로 JavaScript 실행
    js_code = """
    (() => {
        // 1. localStorage 전부 뒤지기
        for (let k of Object.keys(localStorage)) {
            const v = localStorage.getItem(k);
            if (v && v.startsWith('eyJ')) return {source: 'localStorage.' + k, token: v};
        }
        // 2. sessionStorage 전부 뒤지기
        for (let k of Object.keys(sessionStorage)) {
            const v = sessionStorage.getItem(k);
            if (v && v.startsWith('eyJ')) return {source: 'sessionStorage.' + k, token: v};
        }
        // 3. 쿠키에서 뒤지기
        const cookies = document.cookie.split(';');
        for (let c of cookies) {
            const m = c.match(/=(eyJ[A-Za-z0-9_\\-\\.]+)/);
            if (m) return {source: 'cookie', token: m[1]};
        }
        // 4. iframe 내부도 확인
        try {
            const frames = document.querySelectorAll('iframe');
            for (let frame of frames) {
                try {
                    const fdoc = frame.contentWindow;
                    for (let k of Object.keys(fdoc.localStorage || {})) {
                        const v = fdoc.localStorage.getItem(k);
                        if (v && v.startsWith('eyJ')) return {source: 'iframe.localStorage.' + k, token: v};
                    }
                } catch(e) {}
            }
        } catch(e) {}
        return null;
    })()
    """

    ws.send(json.dumps({
        "id": 1,
        "method": "Runtime.evaluate",
        "params": {
            "expression": js_code,
            "returnByValue": True
        }
    }))

    result = json.loads(ws.recv())
    ws.close()

    value = result.get("result", {}).get("result", {}).get("value")
    if value and "token" in value:
        print(f"✓ 토큰 발견: {value['source']}")
        return value["token"]

    print("⚠ storage/cookie에 토큰 없음 → Network 감시 모드로 전환")
    return None


# ============================================================
# 방법 B: Network 탭 감시 (F12 Network와 동일)
# ============================================================

def get_token_via_network(tab, timeout=60):
    """F12 Network 탭처럼 모든 요청 감시해서 Authorization 헤더 캡처"""
    ws_url = tab["webSocketDebuggerUrl"]
    ws = websocket.create_connection(ws_url, timeout=10)

    # Network 도메인 활성화 (F12 Network 탭 켜는 것과 동일)
    ws.send(json.dumps({"id": 1, "method": "Network.enable"}))
    ws.recv()

    print(f"\n📡 Network 감시 중... ({timeout}초)")
    print(f"   브라우저에서 채팅 한 번 보내주세요!")

    token = None
    start = time.time()

    while time.time() - start < timeout:
        try:
            ws.settimeout(1)
            msg = json.loads(ws.recv())
        except (websocket.WebSocketTimeoutException, TimeoutError):
            continue
        except Exception:
            break

        # Network.requestWillBeSent 이벤트에서 Authorization 헤더 추출
        if msg.get("method") == "Network.requestWillBeSent":
            headers = msg.get("params", {}).get("request", {}).get("headers", {})
            auth = headers.get("Authorization", "") or headers.get("authorization", "")
            if auth.startswith("Bearer ey"):
                token = auth[7:]
                print(f"✓ Network에서 토큰 캡처 ({len(token)}자)")
                break

    ws.close()
    return token


# ============================================================
# 통합
# ============================================================

def get_token():
    """토큰 자동 획득"""
    # 1. Chrome 탭 찾기
    tab = find_llm_chat_tab()
    if not tab:
        return None

    # 2. 먼저 storage에서 시도 (빠름)
    token = get_token_via_storage(tab)
    if token:
        return token

    # 3. 안되면 Network 감시 (채팅 한번 보내라고 요청)
    token = get_token_via_network(tab)
    return token


# ============================================================
# LLM 호출
# ============================================================

def chat_api(messages, token):
    r = requests.post(
        LLM_URL,
        headers={
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json",
        },
        json={"model": MODEL, "messages": messages, "stream": False},
        timeout=180,
    )
    r.raise_for_status()
    return r.json()["choices"][0]["message"]["content"]


# ============================================================
# 메인
# ============================================================

def main():
    print("=" * 60)
    print("  SK Hynix LLM-CHAT")
    print("  (Chrome DevTools 직접 접속 - F12 파이썬 버전)")
    print("=" * 60)

    # websocket-client 체크
    try:
        import websocket
    except ImportError:
        print("\n❌ websocket-client 필요")
        print("   pip install websocket-client")
        sys.exit(1)

    token = get_token()
    if not token:
        print("\n❌ 토큰 획득 실패")
        sys.exit(1)

    print(f"\n{'=' * 60}")
    print("  대화 시작 (exit: 종료, reset: 초기화, token: 토큰 재획득)")
    print("=" * 60)

    history = [{"role": "system", "content": SYSTEM_PROMPT}]

    while True:
        try:
            q = input("\n🧑 > ").strip()
        except (KeyboardInterrupt, EOFError):
            break

        if not q:
            continue
        if q.lower() in ("exit", "quit", "종료"):
            break
        if q.lower() in ("reset", "초기화"):
            history = [history[0]]
            print("✓ 초기화")
            continue
        if q.lower() in ("token", "토큰"):
            t = get_token()
            if t:
                token = t
            continue

        history.append({"role": "user", "content": q})
        try:
            print("\n🤖 ", end="", flush=True)
            answer = chat_api(history, token)
            print(answer)
            history.append({"role": "assistant", "content": answer})
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                print("\n⚠ 토큰 만료 → F12 자동 재획득")
                history.pop()
                t = get_token()
                if t:
                    token = t
                    print("✓ 갱신 완료! 다시 질문하세요")
                else:
                    break
            else:
                print(f"\n❌ HTTP {e.response.status_code}")
                history.pop()
        except Exception as e:
            print(f"\n❌ {e}")
            history.pop()


if __name__ == "__main__":
    main()
