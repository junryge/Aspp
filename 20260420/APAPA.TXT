URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com/api/chat/completions"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"

# 시스템 프롬프트 (LLM 역할 설정 - 자유롭게 수정 가능)
SYSTEM_PROMPT = "당신은 SK하이닉스 반도체 FAB 물류(AMHS/OHT) 전문가입니다. 한국어로 친절하고 정확하게 답변합니다."


def chat(messages: list) -> str:
    """LLM API 호출"""
    resp = requests.post(
        URL,
        headers={
            "Authorization": f"Bearer {TOKEN}",
            "Content-Type": "application/json",
        },
        json={
            "model": MODEL,
            "messages": messages,
            "stream": False,
            "temperature": 0.7,
        },
        timeout=180,
    )

    if resp.status_code == 401:
        print("\n❌ 토큰 만료됨!")
        print("   1. 브라우저에서 LLM-CHAT 새로고침")
        print("   2. F12 → Network → authorization 값 복사")
        print("   3. 코드 상단 TOKEN = \"...\" 교체")
        exit(1)

    resp.raise_for_status()
    return resp.json()["choices"][0]["message"]["content"]


def main():
    print("=" * 60)
    print("  SK Hynix LLM-CHAT 터미널")
    print(f"  Model: {MODEL}")
    print("=" * 60)
    print("  사용법:")
    print("    - 질문 입력 후 엔터")
    print("    - 'exit' 또는 'quit' 입력 시 종료")
    print("    - 'reset' 입력 시 대화 초기화")
    print("    - 'history' 입력 시 지금까지 대화 확인")
    print("=" * 60)

    # 대화 기록
    history = [
        {"role": "system", "content": SYSTEM_PROMPT}
    ]

    while True:
        # ─ 사용자 입력 받기 ────────────────────
        try:
            question = input("\n🧑 당신> ").strip()
        except (KeyboardInterrupt, EOFError):
            print("\n\n👋 종료합니다.")
            break

        # 빈 입력 무시
        if not question:
            continue

        # 종료 명령
        if question.lower() in ("exit", "quit", "종료"):
            print("\n👋 종료합니다.")
            break

        # 대화 초기화
        if question.lower() in ("reset", "clear", "초기화"):
            history = [{"role": "system", "content": SYSTEM_PROMPT}]
            print("✓ 대화 기록을 초기화했습니다.")
            continue

        # 대화 히스토리 보기
        if question.lower() in ("history", "기록"):
            print("\n─── 대화 기록 ───")
            for msg in history:
                role_emoji = {"system": "⚙️", "user": "🧑", "assistant": "🤖"}
                emoji = role_emoji.get(msg["role"], "?")
                content = msg["content"][:100] + ("..." if len(msg["content"]) > 100 else "")
                print(f"{emoji} [{msg['role']}] {content}")
            print("─────────────────")
            continue

        # ─ LLM 호출 ────────────────────────
        history.append({"role": "user", "content": question})

        try:
            print("\n🤖 AI> ", end="", flush=True)
            answer = chat(history)
            print(answer)

            history.append({"role": "assistant", "content": answer})

        except requests.exceptions.ConnectionError:
            print("\n❌ 서버 접속 실패 (폐쇄망 확인)")
            history.pop()  # 실패한 질문 제거
        except requests.exceptions.HTTPError as e:
            print(f"\n❌ HTTP 에러: {e.response.status_code}")
            print(f"   {e.response.text[:200]}")
            history.pop()
        except Exception as e:
            print(f"\n❌ 오류: {e}")
            history.pop()


if __name__ == "__main__":
    main()
