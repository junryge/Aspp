"""
SK Hynix LLM-CHAT - HCP SSO 자동 토큰 발급
============================================
원리:
    프론트엔드 코드 분석 결과, /auths/hcp/signin 엔드포인트가
    HCP SSO 쿠키를 받아서 JWT를 자동 발급해줌.
    
    브라우저에 HCP 로그인만 되어있으면:
    1. 브라우저 쿠키 긁어오기
    2. /auths/hcp/signin 호출 → JWT 자동 발급
    3. LLM API 호출

필요 라이브러리:
    pip install requests browser-cookie3
"""
import requests
import json
import sys
import os


# ============================================================
# 엔드포인트
# ============================================================
BACKEND = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com"
FRONTEND_DOMAIN = "llm-chat-frontend-basic-prd.api.aipp01.skhynix.com"

SIGNIN_URL = f"{BACKEND}/api/v1/auths/hcp/signin"  # 후보 1
# SIGNIN_URL = f"{BACKEND}/auths/hcp/signin"       # 후보 2
# SIGNIN_URL = f"{BACKEND}/api/auths/hcp/signin"   # 후보 3

CHAT_URL = f"{BACKEND}/api/chat/completions"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"
SYSTEM_PROMPT = "당신은 SK하이닉스 반도체 FAB 물류 전문가입니다."


# ============================================================
# HCP SSO → JWT 자동 발급
# ============================================================

def get_browser_cookies():
    """브라우저에서 HCP 관련 쿠키 전부 긁어오기"""
    try:
        import browser_cookie3
    except ImportError:
        print("❌ pip install browser-cookie3 필요")
        sys.exit(1)

    # HCP SSO 쿠키가 있을 법한 도메인들
    target_domains = [
        FRONTEND_DOMAIN,
        "skhynix.com",
        "hcpp01.skhynix.com",
        "aipp01.skhynix.com",
        "llm-chat-backend-basic-prd.api.hcpp01.skhynix.com",
    ]

    all_cookies = {}
    browsers = [
        ("Chrome", lambda: __import__("browser_cookie3").chrome()),
        ("Edge", lambda: __import__("browser_cookie3").edge()),
        ("Firefox", lambda: __import__("browser_cookie3").firefox()),
    ]

    for name, fn in browsers:
        try:
            cj = fn()
            for cookie in cj:
                if any(d in cookie.domain for d in target_domains):
                    all_cookies[cookie.name] = cookie.value
            if all_cookies:
                print(f"✓ {name} 쿠키 발견: {len(all_cookies)}개")
                print(f"  쿠키 이름들: {list(all_cookies.keys())}")
                return all_cookies, name
        except Exception as e:
            continue

    return None, None


def issue_jwt_via_hcp_sso(cookies: dict) -> str:
    """HCP SSO 쿠키로 JWT 발급 요청"""
    # 여러 엔드포인트 후보 시도
    endpoints = [
        f"{BACKEND}/api/v1/auths/hcp/signin",
        f"{BACKEND}/auths/hcp/signin",
        f"{BACKEND}/api/auths/hcp/signin",
    ]

    headers = {
        "Content-Type": "application/json",
        "Origin": f"http://{FRONTEND_DOMAIN}",
        "Referer": f"http://{FRONTEND_DOMAIN}/",
    }

    for url in endpoints:
        try:
            print(f"  시도: {url}")
            resp = requests.post(
                url,
                headers=headers,
                cookies=cookies,
                json={},  # 빈 body
                timeout=30,
            )
            print(f"    → {resp.status_code}")

            if resp.status_code == 200:
                data = resp.json()
                token = data.get("token")
                if token and token.startswith("eyJ"):
                    print(f"✓ JWT 발급 성공!")
                    print(f"  사용자: {data.get('name', '?')} ({data.get('email', '?')})")
                    return token
                else:
                    print(f"    응답에 token 없음: {data}")
            elif resp.status_code in (401, 403):
                print(f"    인증 실패 - HCP SSO 로그인 확인 필요")
            elif resp.status_code == 404:
                continue  # 다음 URL 시도
        except Exception as e:
            print(f"    에러: {e}")
            continue

    return None


def auto_get_token():
    """전체 프로세스: 쿠키 → HCP SSO → JWT"""
    print("🔍 브라우저 쿠키 스캔 중...")
    cookies, browser_name = get_browser_cookies()

    if not cookies:
        print("❌ HCP SSO 쿠키를 찾을 수 없습니다.")
        print("   브라우저에서 LLM-CHAT 페이지에 로그인되어 있는지 확인하세요.")
        return None

    print(f"\n🔑 HCP SSO → JWT 발급 요청 중... ({browser_name} 쿠키 사용)")
    token = issue_jwt_via_hcp_sso(cookies)

    if not token:
        print("\n❌ JWT 자동 발급 실패")
        print("   브라우저에서 LLM-CHAT 페이지 한 번 접속 후 재시도")
        return None

    return token


# ============================================================
# LLM API 호출
# ============================================================

def chat(messages: list, token: str) -> str:
    resp = requests.post(
        CHAT_URL,
        headers={
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json",
        },
        json={
            "model": MODEL,
            "messages": messages,
            "stream": False,
            "temperature": 0.7,
        },
        timeout=180,
    )
    resp.raise_for_status()
    return resp.json()["choices"][0]["message"]["content"]


# ============================================================
# 메인
# ============================================================

def main():
    print("=" * 60)
    print("  SK Hynix LLM-CHAT - HCP SSO 자동 로그인")
    print("=" * 60)

    token = auto_get_token()
    if not token:
        sys.exit(1)

    print(f"\n{'=' * 60}")
    print("  대화 시작 (exit: 종료, reset: 초기화, token: 토큰 갱신)")
    print("=" * 60)

    history = [{"role": "system", "content": SYSTEM_PROMPT}]

    while True:
        try:
            q = input("\n🧑 당신> ").strip()
        except (KeyboardInterrupt, EOFError):
            print("\n👋 종료")
            break

        if not q:
            continue
        if q.lower() in ("exit", "quit", "종료"):
            break
        if q.lower() in ("reset", "초기화"):
            history = [{"role": "system", "content": SYSTEM_PROMPT}]
            print("✓ 초기화")
            continue
        if q.lower() in ("token", "토큰"):
            new_token = auto_get_token()
            if new_token:
                token = new_token
            continue

        history.append({"role": "user", "content": q})
        try:
            print("\n🤖 AI>", end=" ", flush=True)
            answer = chat(history, token)
            print(answer)
            history.append({"role": "assistant", "content": answer})
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                print("\n⚠️ 토큰 만료 → 자동 재발급")
                token = auto_get_token()
                if not token:
                    break
                history.pop()
                print("✓ 다시 질문해주세요")
            else:
                print(f"\n❌ HTTP {e.response.status_code}: {e.response.text[:200]}")
                history.pop()


if __name__ == "__main__":
    main()
