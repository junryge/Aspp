SK Hynix LLM-CHAT - 클립보드 자동 읽기 버전
=============================================
사용 순서:
    1. 브라우저에서 북마크 '🔑토큰복사' 클릭 → 토큰이 클립보드에 복사됨
    2. python chat_clipboard.py 실행
    3. 파이썬이 클립보드에서 자동으로 토큰 읽어서 시작

클립보드에 토큰이 없거나 만료됐을 때만 수동 안내.
토큰은 .llm_token 파일에 캐시되어 다음 실행때는 자동 재사용.

필요 라이브러리:
    pip install requests pyperclip
"""
import requests
import json
import os
import sys
import time
import base64


# ============================================================
# 설정
# ============================================================
URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com/api/chat/completions"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"
SYSTEM_PROMPT = "당신은 SK하이닉스 반도체 FAB 물류(AMHS/OHT) 전문가입니다."

TOKEN_CACHE = os.path.join(os.path.dirname(os.path.abspath(__file__)), ".llm_token")


# ============================================================
# JWT 유효성 체크
# ============================================================

def is_token_valid(token: str) -> bool:
    """JWT 만료시간 확인"""
    if not token or not token.startswith("eyJ"):
        return False
    try:
        parts = token.split(".")
        if len(parts) != 3:
            return False
        payload = parts[1] + "=" * (4 - len(parts[1]) % 4)
        decoded = json.loads(base64.urlsafe_b64decode(payload))
        exp = decoded.get("exp", 0)
        now = int(time.time())
        remaining = exp - now
        if remaining > 60:  # 1분 이상 남아야 유효
            mins = remaining // 60
            print(f"  ✓ 토큰 유효 (남은 시간: {mins}분)")
            return True
        else:
            print(f"  ⚠ 토큰 만료됨")
            return False
    except Exception as e:
        print(f"  ⚠ 토큰 파싱 실패: {e}")
        return False


# ============================================================
# 토큰 소스
# ============================================================

def token_from_cache():
    """캐시 파일에서 읽기"""
    if os.path.exists(TOKEN_CACHE):
        with open(TOKEN_CACHE, "r") as f:
            t = f.read().strip()
            if is_token_valid(t):
                print("✓ 캐시에서 토큰 복구")
                return t
    return None


def token_from_clipboard():
    """클립보드에서 읽기"""
    try:
        import pyperclip
    except ImportError:
        print("  (pyperclip 미설치 - 클립보드 읽기 건너뜀)")
        return None

    try:
        t = pyperclip.paste().strip()
        if t.startswith("Bearer "):
            t = t[7:]
        if is_token_valid(t):
            print("✓ 클립보드에서 토큰 획득")
            return t
    except Exception:
        pass
    return None


def token_from_input():
    """수동 입력"""
    print("\n" + "=" * 60)
    print("  토큰을 찾을 수 없습니다.")
    print("=" * 60)
    print("  방법 1 (북마크릿 사용 - 추천):")
    print("    브라우저에서 북마크 '🔑토큰복사' 클릭 후 재실행")
    print()
    print("  방법 2 (수동):")
    print("    1. LLM-CHAT 페이지에서 F12 → Console")
    print("    2. copy(localStorage.token) 입력 후 엔터")
    print("    3. 아래에 Ctrl+V")
    print("=" * 60)
    t = input("\n🔑 토큰 > ").strip()
    if t.startswith("Bearer "):
        t = t[7:]
    if not is_token_valid(t):
        return None
    return t


def save_token(token: str):
    """토큰 캐시"""
    with open(TOKEN_CACHE, "w") as f:
        f.write(token)


def get_token(force=False):
    """토큰 획득 - 캐시→클립보드→수동 순"""
    print("\n🔍 토큰 찾는 중...")

    if not force:
        t = token_from_cache()
        if t:
            return t

    t = token_from_clipboard()
    if t:
        save_token(t)
        return t

    t = token_from_input()
    if t:
        save_token(t)
        return t

    return None


# ============================================================
# LLM 호출
# ============================================================

def chat(messages, token):
    resp = requests.post(
        URL,
        headers={
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json",
        },
        json={
            "model": MODEL,
            "messages": messages,
            "stream": False,
            "temperature": 0.7,
        },
        timeout=180,
    )
    resp.raise_for_status()
    return resp.json()["choices"][0]["message"]["content"]


# ============================================================
# 메인
# ============================================================

def main():
    print("=" * 60)
    print("  SK Hynix LLM-CHAT (클립보드 자동)")
    print("=" * 60)

    token = get_token()
    if not token:
        print("❌ 토큰 획득 실패")
        sys.exit(1)

    print(f"\n{'=' * 60}")
    print("  대화 시작 - 명령어: exit | reset | token")
    print("=" * 60)

    history = [{"role": "system", "content": SYSTEM_PROMPT}]

    while True:
        try:
            q = input("\n🧑 > ").strip()
        except (KeyboardInterrupt, EOFError):
            print("\n👋")
            break

        if not q:
            continue
        if q.lower() in ("exit", "quit", "종료"):
            break
        if q.lower() in ("reset", "초기화"):
            history = [{"role": "system", "content": SYSTEM_PROMPT}]
            print("✓ 초기화됨")
            continue
        if q.lower() in ("token", "토큰"):
            new = get_token(force=True)
            if new:
                token = new
            continue

        history.append({"role": "user", "content": q})
        try:
            print("\n🤖 ", end="", flush=True)
            answer = chat(history, token)
            print(answer)
            history.append({"role": "assistant", "content": answer})
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                print("\n⚠ 토큰 만료 → 클립보드 확인")
                print("  북마크 '🔑토큰복사' 누르고 엔터치세요...")
                input("  완료 후 엔터 > ")
                new = get_token(force=True)
                if new:
                    token = new
                    history.pop()
                    print("✓ 토큰 갱신! 다시 질문 입력하세요.")
                else:
                    break
            else:
                print(f"\n❌ HTTP {e.response.status_code}")
                history.pop()
        except Exception as e:
            print(f"\n❌ {e}")
            history.pop()


if __name__ == "__main__":
    main()
