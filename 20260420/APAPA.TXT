"""
브라우저 쿠키 진단 스크립트
=============================
어느 브라우저에 어떤 쿠키가 있는지 전부 확인해서
자동 토큰 가져오기가 가능한지 판단.
"""
import sys

try:
    import browser_cookie3
except ImportError:
    print("❌ browser-cookie3 미설치")
    print("   실행: pip install browser-cookie3")
    sys.exit(1)


# 관심있는 도메인 키워드
TARGET_KEYWORDS = [
    "skhynix",
    "hcpp01",
    "aipp01",
    "llm-chat",
    "hcp",
]


def scan_browser(name, func):
    """특정 브라우저의 쿠키 스캔"""
    print(f"\n{'=' * 60}")
    print(f"  {name} 브라우저 스캔")
    print("=" * 60)

    try:
        cj = func()
        cookies = list(cj)
        print(f"✓ 전체 쿠키 개수: {len(cookies)}")
    except Exception as e:
        print(f"❌ 접근 실패: {e}")
        print(f"   → {name} 브라우저가 열려있으면 닫고 재시도")
        return []

    # SK하이닉스 관련 쿠키 필터링
    relevant = []
    for cookie in cookies:
        domain_lower = cookie.domain.lower()
        if any(kw in domain_lower for kw in TARGET_KEYWORDS):
            relevant.append(cookie)

    if not relevant:
        print(f"⚠️  SK하이닉스 관련 쿠키 없음")
        # 전체 도메인 통계만 보여주기
        domains = set(c.domain for c in cookies)
        skh_domains = [d for d in domains if "skhynix" in d.lower() or "hcp" in d.lower()]
        if skh_domains:
            print(f"   skhynix/hcp 관련 도메인 발견: {skh_domains}")
        return []

    print(f"\n🎯 SK하이닉스 관련 쿠키 {len(relevant)}개 발견:")
    for c in relevant:
        val_preview = c.value[:40] + "..." if len(c.value) > 40 else c.value
        print(f"  • [{c.domain}]")
        print(f"    {c.name} = {val_preview}")

    return relevant


def main():
    print("=" * 60)
    print("  브라우저 쿠키 진단")
    print("=" * 60)

    browsers = [
        ("Chrome", browser_cookie3.chrome),
        ("Edge", browser_cookie3.edge),
        ("Firefox", browser_cookie3.firefox),
        ("Brave", browser_cookie3.brave),
        ("Opera", browser_cookie3.opera),
    ]

    found_any = False
    for name, func in browsers:
        try:
            cookies = scan_browser(name, func)
            if cookies:
                found_any = True
        except AttributeError:
            # 일부 브라우저는 지원 안됨
            continue
        except Exception as e:
            print(f"\n  {name}: 스캔 실패 ({type(e).__name__})")

    print(f"\n{'=' * 60}")
    print("  결론")
    print("=" * 60)
    if found_any:
        print("✓ 자동 토큰 가져오기 가능!")
        print("  → chat_hcp_sso.py 실행")
    else:
        print("❌ 어느 브라우저에서도 SK하이닉스 쿠키를 못 찾음")
        print()
        print("원인 후보:")
        print("  1. 브라우저에 LLM-CHAT 로그인 안 되어있음")
        print("     → 브라우저 열고 로그인 시도")
        print()
        print("  2. 브라우저가 열려있어서 쿠키 DB 락 걸림")
        print("     → 브라우저 닫고 재시도")
        print()
        print("  3. Chrome의 새 쿠키 암호화 (v127+) 이슈")
        print("     → 다른 브라우저(Firefox) 써보기")
        print("     → 또는 JS 스니펫 방식 사용 (아래 참조)")
        print()
        print("대안: F12에서 수동으로 토큰 가져오기")
        print("  1. 브라우저 F12 → Console 탭")
        print("  2. 붙여넣기: copy(localStorage.token)")
        print("     (또는 copy(document.cookie))")
        print("  3. 코드에 붙여넣기")


if __name__ == "__main__":
    main()
