"""
SK Hynix LLM-CHAT
==================
사용법:
  1. 아래 TOKEN에 JWT 붙여넣기
  2. python chat.py

토큰 얻는 법:
  1. 브라우저에서 LLM-CHAT 접속
  2. F12 → Network 탭
  3. 필터박스에 "completions" 입력
  4. 아무 메시지나 전송
  5. 요청 클릭 → Headers → authorization 줄 찾기
  6. "Bearer " 뒤의 eyJ... 부분 복사
  7. 아래 TOKEN = "..." 안에 붙여넣기
"""
import requests


# ============================================================
# 토큰 (2시간마다 교체)
# ============================================================
TOKEN = "여기에_토큰_붙여넣기"


# ============================================================
# 설정
# ============================================================
URL = "http://llm-chat-backend-basic-prd.api.hcpp01.skhynix.com/api/chat/completions"
MODEL = "Qwen/Qwen3.5-397B-A17B-FP8"
SYSTEM = "당신은 SK하이닉스 반도체 FAB 물류 전문가입니다. 한국어로 답변합니다."


# ============================================================
# 챗봇
# ============================================================
def ask(history, question):
    history.append({"role": "user", "content": question})
    r = requests.post(
        URL,
        headers={"Authorization": f"Bearer {TOKEN}", "Content-Type": "application/json"},
        json={"model": MODEL, "messages": history, "stream": False},
        timeout=180,
    )
    r.raise_for_status()
    answer = r.json()["choices"][0]["message"]["content"]
    history.append({"role": "assistant", "content": answer})
    return answer


def main():
    if TOKEN == "여기에_토큰_붙여넣기":
        print("❌ 먼저 TOKEN에 JWT 붙여넣으세요!")
        return

    print("=" * 50)
    print("  SK Hynix LLM-CHAT")
    print("  (exit: 종료, reset: 초기화)")
    print("=" * 50)

    history = [{"role": "system", "content": SYSTEM}]

    while True:
        try:
            q = input("\n🧑 > ").strip()
        except (KeyboardInterrupt, EOFError):
            break
        if not q:
            continue
        if q.lower() in ("exit", "quit", "종료"):
            break
        if q.lower() in ("reset", "초기화"):
            history = [history[0]]
            print("✓ 초기화")
            continue

        try:
            print(f"\n🤖 {ask(history, q)}")
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 401:
                print("\n❌ 토큰 만료! 새 토큰 받아서 TOKEN 교체 후 재실행")
                break
            print(f"\n❌ 에러: {e}")
            history.pop()
        except Exception as e:
            print(f"\n❌ 에러: {e}")
            history.pop()


if __name__ == "__main__":
    main()
