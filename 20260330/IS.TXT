def parse_xml_to_json(xml_path, output_path=DEFAULT_CACHE):
    node_xy, connections = {}, []
    cur_addr, cur_params, in_next, next_params = None, {}, False, {}
    print(f"  XML 파싱: {xml_path}")
    with open(xml_path, 'r', encoding='utf-8') as f:
        for i, line in enumerate(f, 1):
            if i % 500000 == 0: print(f"    {i:,} 라인...")
            line = line.strip()
            if '<group name="Addr' in line and 'address.Addr"' in line:
                if cur_addr and 'address' in cur_params:
                    no = int(cur_params.get('address', 0))
                    if no > 0: node_xy[no] = (float(cur_params.get('draw-x',0)), float(cur_params.get('draw-y',0)))
                cur_params, cur_addr, in_next = {}, line, False; continue
            if '<group name="NextAddr' in line and 'NextAddr"' in line:
                in_next, next_params = True, {}; continue
            if in_next and '</group>' in line:
                if 'address' in cur_params and 'next-address' in next_params:
                    try:
                        fa, ta = int(cur_params.get('address',0)), int(next_params.get('next-address','0'))
                        if fa > 0 and ta > 0: connections.append((fa, ta))
                    except ValueError: pass
                in_next = False; continue
            if '<param ' in line and 'key="' in line:
                km, vm = re.search(r'key="([^"]+)"', line), re.search(r'value="([^"]*)"', line)
                if km and vm:
                    if in_next: next_params[km.group(1)] = vm.group(1)
                    elif cur_addr: cur_params[km.group(1)] = vm.group(1)
    if cur_addr and 'address' in cur_params:
        no = int(cur_params.get('address', 0))
        if no > 0: node_xy[no] = (float(cur_params.get('draw-x',0)), float(cur_params.get('draw-y',0)))
    adj, edge_dist = {}, {}
    for f, t in connections:
        adj.setdefault(f, []).append(t)
        if f in node_xy and t in node_xy:
            dx, dy = node_xy[t][0]-node_xy[f][0], node_xy[t][1]-node_xy[f][1]
            edge_dist[(f,t)] = max(int(math.sqrt(dx*dx+dy*dy)*10), 10)
    data = {"nodes":{str(k):list(v) for k,v in node_xy.items()}, "adj":{str(k):v for k,v in adj.items()}, "edges":{f"{k[0]},{k[1]}":v for k,v in edge_dist.items()}}
    with open(output_path, "w", encoding="utf-8") as f: json.dump(data, f)
    print(f"  노드: {len(node_xy)}개  엣지: {len(edge_dist)}개 → {output_path}")

def ensure_cache():
    if os.path.exists(DEFAULT_CACHE): return
    print("layout_cache.json 생성 중...")
    if os.path.exists(LAYOUT_XML): parse_xml_to_json(LAYOUT_XML); return
    if os.path.exists(LAYOUT_ZIP):
        with zipfile.ZipFile(LAYOUT_ZIP,'r') as z: z.extractall(os.path.join(MAP_DIR,FAB_NAME,f"{LAYOUT_PREFIX}.layout"))
        if os.path.exists(LAYOUT_XML): parse_xml_to_json(LAYOUT_XML); return
    if os.path.exists(MAP_ZIP):
        with zipfile.ZipFile(MAP_ZIP,'r') as z: z.extractall(MAP_DIR)
        if os.path.exists(LAYOUT_XML): parse_xml_to_json(LAYOUT_XML); return
    print(f"\n오류: layout_cache.json 이 없습니다!")
    print(f"  이 폴더에 layout_cache.json을 넣으세요: {SCRIPT_DIR}")

def calc_position(nodes, edges, cur, dist, nxt):
    cur_s, nxt_s = str(cur), str(nxt)
    if cur_s not in nodes: return None, None
    if nxt_s not in nodes: return nodes[cur_s][0], nodes[cur_s][1]
    x1, y1 = nodes[cur_s]
    x2, y2 = nodes[nxt_s]
    key = f"{cur},{nxt}"
    el = edges.get(key, 0)
    if el <= 0:
        el = math.sqrt((x2-x1)**2 + (y2-y1)**2) * 10
        if el <= 0: return x1, y1
    r = min(dist / el, 1.0)
    return round(x1 + (x2-x1)*r, 2), round(y1 + (y2-y1)*r, 2)


STATE_MAP = {'1':'RUN','2':'STOP','3':'ABNORMAL','4':'MANUAL','6':'OBS_STOP','7':'JAM','8':'HT_STOP','9':'E84_TIMEOUT'}

def parse_and_calc(messages_text):
    """메시지 텍스트 → 차량 위치 계산 → vehicles_state 업데이트"""
    results = []
    for line in messages_text.split("\n"):
        line = line.strip().strip('"').rstrip(',').strip('"')
        if "2,OHT," not in line: continue
        line = line[line.index("2,OHT,"):]
        f = line.split(",")
        if len(f) < 10 or f[1] != "OHT": continue

        cur, dist, nxt = int(f[7]), int(f[8]), int(f[9])
        x, y = calc_position(layout_data["nodes"], layout_data["edges"], cur, dist, nxt)
        if x is None: continue

        v = {
            "vehicleId": f[2], "x": x, "y": y,
            "state": f[3], "stateName": STATE_MAP.get(f[3], f[3]),
            "isFull": f[4], "currentNode": cur, "distance": dist, "nextNode": nxt,
            "carrierId": f[12] if len(f) > 12 else "",
            "destPort": f[17] if len(f) > 17 else "",
        }
        results.append(v)
        with vehicles_lock:
            vehicles_state[v["vehicleId"]] = v

    return results

class Handler(BaseHTTPRequestHandler):
    def do_GET(self):
        if self.path == "/api/state":
            with vehicles_lock:
                data = list(vehicles_state.values())
            self.send_response(200)
            self.send_header("Content-Type", "application/json")
            self.send_header("Access-Control-Allow-Origin", "*")
            self.end_headers()
            self.wfile.write(json.dumps(data).encode())
        elif self.path == "/api/layout":
            self.send_response(200)
            self.send_header("Content-Type", "application/json")
            self.end_headers()
            self.wfile.write(json.dumps(layout_data).encode())
        else:
            self.send_response(200)
            self.send_header("Content-Type", "text/html; charset=utf-8")
            self.end_headers()
            self.wfile.write(HTML.encode("utf-8"))

    def do_POST(self):
        if self.path == "/api/send":
            length = int(self.headers.get("Content-Length", 0))
            body = json.loads(self.rfile.read(length))
            text = body.get("messages", "")
            results = parse_and_calc(text)
            self.send_response(200)
            self.send_header("Content-Type", "application/json")
            self.send_header("Access-Control-Allow-Origin", "*")
            self.end_headers()
            self.wfile.write(json.dumps({"count": len(results), "vehicles": [r["vehicleId"] for r in results]}).encode())
        else:
            self.send_response(404)
            self.end_headers()

    def do_OPTIONS(self):
        self.send_response(200)
        self.send_header("Access-Control-Allow-Origin", "*")
        self.send_header("Access-Control-Allow-Methods", "POST, GET, OPTIONS")
        self.send_header("Access-Control-Allow-Headers", "Content-Type")
        self.end_headers()

    def log_message(self, *a): pass

// === 레이아웃 로드 ===
async function loadLayout() {
    const resp = await fetch('/api/layout');
    const data = await resp.json();

    for (const [no, xy] of Object.entries(data.nodes)) {
        nodes[parseInt(no)] = {x: xy[0], y: xy[1]};
        if (xy[0] < minX) minX = xy[0]; if (xy[0] > maxX) maxX = xy[0];
        if (xy[1] < minY) minY = xy[1]; if (xy[1] > maxY) maxY = xy[1];
    }
    nodeCount = Object.keys(nodes).length;

    // edges from adj
    for (const [from, tos] of Object.entries(data.adj)) {
        for (const to of tos) {
            if (nodes[parseInt(from)] && nodes[to]) {
                edges.push({from: parseInt(from), to});
            }
        }
    }
    edgeCount = edges.length;

    document.getElementById('info').textContent = `노드 ${nodeCount.toLocaleString()} | 엣지 ${edgeCount.toLocaleString()}`;
    fitView();
}

// === 전송 ===
async function sendData() {
    const text = document.getElementById('input').value.trim();
    if (!text) return;

    const resp = await fetch('/api/send', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({messages: text})
    });
    const r = await resp.json();
    document.getElementById('status').innerHTML =
        `<span class="ok">전송 완료</span> <span class="cnt">${r.count}대</span> (${r.vehicles.join(', ')})`;

    // 상태 갱신
    await refreshState();
}

async function refreshState() {
    const resp = await fetch('/api/state');
    const data = await resp.json();
    for (const v of data) {
        if (!vehicles[v.vehicleId]) {
            vehicles[v.vehicleId] = {...v, dispX: v.x, dispY: v.y};
        } else {
            Object.assign(vehicles[v.vehicleId], v);
        }
    }
    render();
}

function resize() {
    const area = document.getElementById('mapArea');
    canvas.width = area.clientWidth;
    canvas.height = area.clientHeight;

function worldToScreen(wx, wy) {
    return {
        x: (wx - offsetX) * scale + canvas.width / 2,
        y: (wy - offsetY) * scale + canvas.height / 2
    };
}

function screenToWorld(sx, sy) {
    return {
        x: (sx - canvas.width / 2) / scale + offsetX,
        y: (sy - canvas.height / 2) / scale + offsetY
    };
}

const STATE_COLORS = {'1':'#00cc44','2':'#cccc00','3':'#cc4400','4':'#cc8800','6':'#cc8800','7':'#cc0000','8':'#cc4400','9':'#cc0000'};
const STATE_NAMES = {'1':'RUN','2':'STOP','3':'ABNORMAL','4':'MANUAL','6':'OBS_STOP','7':'JAM','8':'HT_STOP','9':'E84_TIMEOUT'};

function render() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // 엣지
    ctx.strokeStyle = '#1a2a4a';
    ctx.lineWidth = 1;
    for (const e of edges) {
        const n1 = nodes[e.from], n2 = nodes[e.to];
        if (!n1 || !n2) continue;
        const p1 = worldToScreen(n1.x, n1.y);
        const p2 = worldToScreen(n2.x, n2.y);
        ctx.beginPath();
        ctx.moveTo(p1.x, p1.y);
        ctx.lineTo(p2.x, p2.y);
        ctx.stroke();
    }

    // 차량
    const vehSize = Math.max(4, 8 / Math.sqrt(scale > 0.01 ? scale : 0.01));
    for (const v of Object.values(vehicles)) {
        // 부드러운 보간
        v.dispX = v.dispX + (v.x - v.dispX) * 0.2;
        v.dispY = v.dispY + (v.y - v.dispY) * 0.2;

        const p = worldToScreen(v.dispX, v.dispY);
        const color = STATE_COLORS[v.state] || '#888';

        // 테두리
        ctx.beginPath();
        ctx.arc(p.x, p.y, vehSize + 2, 0, Math.PI * 2);
        ctx.fillStyle = 'rgba(0,0,0,0.5)';
        ctx.fill();

        // 차량
        ctx.beginPath();
        ctx.arc(p.x, p.y, vehSize, 0, Math.PI * 2);
        ctx.fillStyle = color;
        ctx.fill();

        // 라벨
        if (scale > 0.03) {
            ctx.fillStyle = '#fff';
            ctx.font = `${Math.max(9, 11/Math.sqrt(scale > 0.01 ? scale : 0.01))}px sans-serif`;
            ctx.textAlign = 'center';
            ctx.fillText(v.vehicleId, p.x, p.y - vehSize - 4);
        }
    }
}

// === 뷰 컨트롤 ===
function fitView() {
    if (nodeCount === 0) return;
    const pad = 50;
    const w = maxX - minX || 1;
    const h = maxY - minY || 1;
    const sx = (canvas.width - pad*2) / w;
    const sy = (canvas.height - pad*2) / h;
    scale = Math.min(sx, sy);
    offsetX = (minX + maxX) / 2;
    offsetY = (minY + maxY) / 2;
    render();
}

function zoomIn() { scale *= 1.3; render(); }
function zoomOut() { scale /= 1.3; render(); }

// === 마우스 이벤트 ===
canvas.addEventListener('wheel', e => {

    const factor = e.deltaY < 0 ? 1.15 : 1/1.15;

    const my = e.clientY - rect.top;
    const before = screenToWorld(mx, my);
    scale *= factor;
    const after = screenToWorld(mx, my);
    offsetX -= (after.x - before.x);
    offsetY -= (after.y - before.y);
    render();
});

    if len(sys.argv) > 1 and sys.argv[1] == "convert":
        if len(sys.argv) < 3:
            print("사용법: python oht_map.py convert layout.xml")
            return
        fp = sys.argv[2]
        if fp.endswith('.zip'):
            with zipfile.ZipFile(fp,'r') as z:
                xn = [n for n in z.namelist() if n.endswith('layout.xml')]
                if xn: z.extractall(os.path.dirname(fp) or ".")
                for x in xn:
                    if os.path.exists(x): parse_xml_to_json(x); return
        else:
            parse_xml_to_json(fp)
        return

    ensure_cache()

    global layout_data
    with open(DEFAULT_CACHE, "r", encoding="utf-8") as f:
        layout_data = json.load(f)

    nc = len(layout_data["nodes"])
    ec = len(layout_data["edges"])
    print(f"노드 {nc}개, 엣지 {ec}개 로드")
    print(f"\nOHT 맵 뷰어: http://localhost:{PORT}")

    webbrowser.open(f"http://localhost:{PORT}")

    server = HTTPServer(("0.0.0.0", PORT), Handler)
    server.serve_forever()
