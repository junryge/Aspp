#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
oht_position.py - OHT 위치 계산기

실행하면 OHT 메시지를 붙여넣기만 하면 자동으로 위치 계산됩니다.
layout_cache.json 없으면 layout.xml에서 자동 생성합니다.

사용법:
    python oht_position.py  → OHT 메시지 붙여넣기 → 위치 계산 결과 출력

코드에서 사용:
    from oht_position import OHTPosition
    pos = OHTPosition()
    x, y = pos.calc(12340, 14, 12341)
"""

import json
import math
import os
import re
import sys
import zipfile

SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__))
DEFAULT_CACHE = os.path.join(SCRIPT_DIR, "layout_cache.json")

# layout.xml 찾을 경로들
FAB_NAME = "M14A"
LAYOUT_PREFIX = "A"
MAP_DIR = os.path.join(SCRIPT_DIR, "MAP")
LAYOUT_XML = os.path.join(MAP_DIR, FAB_NAME, f"{LAYOUT_PREFIX}.layout", "layout.xml")
LAYOUT_ZIP = os.path.join(MAP_DIR, FAB_NAME, f"{LAYOUT_PREFIX}.layout.zip")
MAP_ZIP = os.path.join(SCRIPT_DIR, "MAP.zip")

STATE_MAP = {
    "1": "RUN", "2": "STOP", "3": "ABNORMAL", "4": "MANUAL",
    "6": "OBS_STOP", "7": "JAM", "8": "HT_STOP", "9": "E84_TIMEOUT",
}


class OHTPosition:
    """OHT 위치 계산기"""

    def __init__(self, cache_path=DEFAULT_CACHE):
        if not os.path.exists(cache_path):
            raise FileNotFoundError(f"layout_cache.json 없음: {cache_path}")

        with open(cache_path, "r", encoding="utf-8") as f:
            data = json.load(f)

        self.nodes = {int(k): tuple(v) for k, v in data["nodes"].items()}
        self.adj = {int(k): v for k, v in data["adj"].items()}
        self.edge_dist = {}
        for k, v in data["edges"].items():
            parts = k.split(",")
            self.edge_dist[(int(parts[0]), int(parts[1]))] = v

    def calc(self, current, distance, next_node):
        """현재번지, 거리, 다음번지 → (x, y)"""
        if current not in self.nodes:
            return None, None
        if next_node not in self.nodes:
            return self.nodes[current]

        x1, y1 = self.nodes[current]
        x2, y2 = self.nodes[next_node]

        edge_len = self.edge_dist.get((current, next_node), 0)
        if edge_len <= 0:
            edge_len = math.sqrt((x2-x1)**2 + (y2-y1)**2) * 10
            if edge_len <= 0:
                return x1, y1

        ratio = min(distance / edge_len, 1.0)
        return round(x1 + (x2-x1) * ratio, 2), round(y1 + (y2-y1) * ratio, 2)

    def calc_from_message(self, message):
        """OHT raw 메시지 → 위치 계산 결과"""
        line = message.strip()
        if "2,OHT," in line:
            line = line[line.index("2,OHT,"):]
        line = line.strip('"').rstrip(',').strip('"')

        fields = line.split(",")
        if len(fields) < 10 or fields[1] != "OHT":
            return None

        cur = int(fields[7])
        dist = int(fields[8])
        nxt = int(fields[9])
        x, y = self.calc(cur, dist, nxt)

        edge_len = self.edge_dist.get((cur, nxt), 0)
        ratio = min(dist / edge_len, 1.0) * 100 if edge_len > 0 else 0

        return {
            "vehicleId": fields[2],
            "state": STATE_MAP.get(fields[3], fields[3]),
            "isFull": "적재" if fields[4] == "1" else "빈차",
            "currentNode": cur,
            "currentXY": self.nodes.get(cur),
            "distance": dist,
            "distance_m": round(dist * 0.1, 1),
            "nextNode": nxt,
            "nextXY": self.nodes.get(nxt),
            "edgeLength": edge_len,
            "edgeLength_m": round(edge_len * 0.1, 1),
            "ratio": round(ratio, 1),
            "x": x,
            "y": y,
            "carrierId": fields[12] if len(fields) > 12 else "",
            "sourcePort": fields[16] if len(fields) > 16 else "",
            "destPort": fields[17] if len(fields) > 17 else "",
        }


# ============================================================
# XML → JSON 변환
# ============================================================
def parse_xml_to_json(xml_path, output_path=DEFAULT_CACHE):
    """layout.xml → layout_cache.json"""
    node_xy = {}
    connections = []
    current_addr = None
    current_addr_params = {}
    in_next_addr = False
    next_addr_params = {}

    print(f"  XML 파싱: {xml_path}")
    with open(xml_path, 'r', encoding='utf-8') as f:
        lc = 0
        for line in f:
            lc += 1
            if lc % 500000 == 0:
                print(f"    {lc:,} 라인...")
            line = line.strip()

            if '<group name="Addr' in line and 'class=' in line and 'address.Addr"' in line:
                if current_addr is not None and 'address' in current_addr_params:
                    no = int(current_addr_params.get('address', 0))
                    if no > 0:
                        node_xy[no] = (float(current_addr_params.get('draw-x', 0)),
                                       float(current_addr_params.get('draw-y', 0)))
                current_addr_params = {}
                current_addr = line
                in_next_addr = False
                continue
            if '<group name="NextAddr' in line and 'class=' in line and 'NextAddr"' in line:
                in_next_addr = True
                next_addr_params = {}
                continue
            if in_next_addr and '</group>' in line:
                if 'address' in current_addr_params and 'next-address' in next_addr_params:
                    f_a = int(current_addr_params.get('address', 0))
                    try:
                        t_a = int(next_addr_params.get('next-address', '0'))
                        if f_a > 0 and t_a > 0:
                            connections.append((f_a, t_a))
                    except ValueError:
                        pass
                in_next_addr = False
                continue
            if '<param ' in line and 'key="' in line and 'value="' in line:
                km = re.search(r'key="([^"]+)"', line)
                vm = re.search(r'value="([^"]*)"', line)
                if km and vm:
                    if in_next_addr:
                        next_addr_params[km.group(1)] = vm.group(1)
                    elif current_addr is not None:
                        current_addr_params[km.group(1)] = vm.group(1)

    if current_addr is not None and 'address' in current_addr_params:
        no = int(current_addr_params.get('address', 0))
        if no > 0:
            node_xy[no] = (float(current_addr_params.get('draw-x', 0)),
                           float(current_addr_params.get('draw-y', 0)))

    adj = {}
    edge_dist = {}
    for f, t in connections:
        adj.setdefault(f, []).append(t)
        if f in node_xy and t in node_xy:
            dx = node_xy[t][0] - node_xy[f][0]
            dy = node_xy[t][1] - node_xy[f][1]
            edge_dist[(f, t)] = max(int(math.sqrt(dx*dx + dy*dy) * 10), 10)

    data = {
        "nodes": {str(k): list(v) for k, v in node_xy.items()},
        "adj": {str(k): v for k, v in adj.items()},
        "edges": {f"{k[0]},{k[1]}": v for k, v in edge_dist.items()},
    }
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(data, f)
    print(f"  노드: {len(node_xy)}개  엣지: {len(edge_dist)}개")
    print(f"  → {output_path}")
    return len(node_xy), len(edge_dist)


def ensure_cache():
    """layout_cache.json 없으면 자동 생성"""
    if os.path.exists(DEFAULT_CACHE):
        return

    print("layout_cache.json 없음 → 자동 생성")

    # XML 직접
    if os.path.exists(LAYOUT_XML):
        parse_xml_to_json(LAYOUT_XML)
        return

    # ZIP에서 추출
    if os.path.exists(LAYOUT_ZIP):
        extract_dir = os.path.join(MAP_DIR, FAB_NAME, f"{LAYOUT_PREFIX}.layout")
        with zipfile.ZipFile(LAYOUT_ZIP, 'r') as z:
            z.extractall(extract_dir)
        if os.path.exists(LAYOUT_XML):
            parse_xml_to_json(LAYOUT_XML)
            return

    # MAP.zip 풀기
    if os.path.exists(MAP_ZIP):
        print(f"  MAP.zip 추출 중...")
        with zipfile.ZipFile(MAP_ZIP, 'r') as z:
            z.extractall(MAP_DIR)
        if os.path.exists(LAYOUT_XML):
            parse_xml_to_json(LAYOUT_XML)
            return

    print("오류: layout.xml을 찾을 수 없습니다")
    sys.exit(1)


def clean_line(line):
    """따옴표, 쉼표 등 정리"""
    line = line.strip()
    if line.startswith('"'):
        line = line.lstrip('"')
    if line.endswith('",') or line.endswith('"'):
        line = line.rstrip(',').rstrip('"')
    return line.strip()


def print_result(r):
    """계산 결과 출력"""
    cur_xy = r['currentXY']
    nxt_xy = r['nextXY']
    print(f"  ┌─ {r['vehicleId']}  [{r['state']}]  {r['isFull']}")
    print(f"  │  현재번지: {r['currentNode']:>6d}  ({cur_xy[0]:.1f}, {cur_xy[1]:.1f})")
    print(f"  │  다음번지: {r['nextNode']:>6d}  ({nxt_xy[0]:.1f}, {nxt_xy[1]:.1f})")
    print(f"  │  엣지거리: {r['edgeLength_m']}m   이동거리: {r['distance_m']}m   비율: {r['ratio']}%")
    if r['destPort']:
        print(f"  │  목적지:   {r['destPort']}")
    if r['carrierId']:
        print(f"  │  캐리어:   {r['carrierId']}")
    print(f"  └─ 계산 위치: ({r['x']}, {r['y']})")
    print()


# ============================================================
# 메인
# ============================================================
def main():
    # convert 모드
    if len(sys.argv) > 1 and sys.argv[1] == "convert":
        if len(sys.argv) < 3:
            print("사용법: python oht_position.py convert layout.xml")
            return
        filepath = sys.argv[2]
        if filepath.endswith('.zip'):
            with zipfile.ZipFile(filepath, 'r') as z:
                xml_names = [n for n in z.namelist() if n.endswith('layout.xml')]
                if not xml_names:
                    print("오류: ZIP 안에 layout.xml 없음")
                    return
                # 추출 후 파싱
                z.extractall(os.path.dirname(filepath) or ".")
                for xn in xml_names:
                    if os.path.exists(xn):
                        parse_xml_to_json(xn)
                        return
        else:
            parse_xml_to_json(filepath)
        return

    # 메인: OHT 메시지 입력 → 위치 계산
    print("=" * 55)
    print("  OHT 위치 계산기")
    print("=" * 55)

    # layout_cache.json 자동 생성
    ensure_cache()
    pos = OHTPosition()
    print(f"  레이아웃 로드: 노드 {len(pos.nodes)}개, 엣지 {len(pos.edge_dist)}개")
    print()
    print("  OHT 메시지를 붙여넣고 빈 줄에서 엔터 → 계산")
    print("  q = 종료")
    print("-" * 55)

    count = 0
    while True:
        print(f"\n[{count+1}] OHT 메시지 입력:")

        lines = []
        while True:
            try:
                line = input()
            except EOFError:
                return

            if line.strip().lower() == "q":
                return

            cleaned = clean_line(line)
            if not cleaned:
                break
            lines.append(cleaned)

        if not lines:
            continue

        # 파싱 + 계산
        results = []
        for line in lines:
            if "2,OHT," not in line:
                continue
            r = pos.calc_from_message(line)
            if r:
                results.append(r)

        if not results:
            print("  OHT 메시지를 인식할 수 없습니다.")
            print("  예시: 2,OHT,V00795,1,1,0000,1,12340,14,12341,4,4,...")
            continue

        count += 1
        print(f"\n  === {len(results)}대 위치 계산 결과 ===\n")
        for r in results:
            print_result(r)


if __name__ == "__main__":
    try:
        main()
    except KeyboardInterrupt:
        print("\n종료.")
