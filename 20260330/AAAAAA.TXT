# token.txt를 스크립트와 같은 폴더에서 자동 탐색
def _find_token_path():
    """스크립트 위치 기준으로 token.txt 자동 탐색"""
    candidates = []
    # 1) __file__ 기준 (외부 스크립트 실행 시)
    try:
        candidates.append(os.path.join(os.path.dirname(os.path.abspath(__file__)), "token.txt"))
    except:
        pass
    # 2) Blender 텍스트 에디터에서 실행 시 → 저장된 경로 확인
    try:
        for text in bpy.data.texts:
            if text.filepath:
                candidates.append(os.path.join(os.path.dirname(bpy.path.abspath(text.filepath)), "token.txt"))
    except:
        pass
    # 3) .blend 파일 위치 기준
    if bpy.data.filepath:
        candidates.append(os.path.join(os.path.dirname(bpy.data.filepath), "token.txt"))
    # 4) 폴백
    candidates.append("C:/token.txt")

    for p in candidates:
        if os.path.isfile(p):
            return p
    return candidates[0] if candidates else "token.txt"

DEFAULT_TOKEN_PATH = _find_token_path()

def load_token(token_path):
    """token.txt에서 JWT 토큰 읽기"""
    try:
        with open(token_path, "r") as f:
            return f.read().strip()
    except FileNotFoundError:
        raise Exception(f"토큰 파일 없음: {token_path}\n파일 경로를 확인하세요.")
    except Exception as e:
        raise Exception(f"토큰 읽기 실패: {e}")

# 씬 컨텍스트 수집
# ============================================================
def get_scene_context():
    """현재 Blender 씬 상태를 텍스트로 반환"""
    lines = []
    objs = bpy.data.objects
    if len(objs) == 0:
        lines.append("씬: 비어있음")
    else:
        obj_list = []
        for o in objs:
            loc = tuple(round(v, 1) for v in o.location)
            info = f"{o.name}(type={o.type}, loc={loc})"
            obj_list.append(info)
        lines.append(f"씬 오브젝트({len(objs)}개): {', '.join(obj_list[:30])}")
        if len(objs) > 30:
            lines.append(f"  ... 외 {len(objs)-30}개")

    selected = bpy.context.selected_objects
    active = bpy.context.active_object
    if selected:
        lines.append(f"선택된 오브젝트: {[s.name for s in selected]}")
    if active:
        lines.append(f"활성 오브젝트: {active.name} (type={active.type})")
    if bpy.context.object:
        lines.append(f"현재 모드: {bpy.context.object.mode}")
    return "\n".join(lines)


# ============================================================
# LLM API 호출 (urllib만 사용, 외부 패키지 불필요)
# ============================================================
def call_llm_api(user_message, scene_context, api_base, model, token):
    """내부 LLM API 호출"""

    full_message = user_message
    if scene_context:
        full_message = f"[현재 씬 상태]\n{scene_context}\n\n[요청]\n{user_message}"

    payload = {
        "model": model,
        "messages": [
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": full_message}
        ],
        "temperature": TEMPERATURE,
        "max_tokens": MAX_TOKENS,
        "top_p": 0.95
    }

    headers = {
        "Content-Type": "application/json",
        "Authorization": f"Bearer {token}"
    }

    url = f"{api_base}/v1/chat/completions"
    data = json.dumps(payload).encode("utf-8")
    req = urllib.request.Request(url, data=data, headers=headers, method="POST")

    try:
        with urllib.request.urlopen(req, timeout=120) as resp:
            result = json.loads(resp.read().decode("utf-8"))
            return result["choices"][0]["message"]["content"]
    except urllib.error.HTTPError as e:
        body = e.read().decode("utf-8", errors="replace")
        raise Exception(f"HTTP {e.code}: {body[:200]}")
    except urllib.error.URLError as e:
        raise Exception(f"연결 실패: {e.reason}")
    except Exception as e:
        raise Exception(f"API 에러: {e}")

# 코드 추출 / 검증
# ============================================================
def check_safe_eval(code):
    """위험 모듈 차단"""
    unsafe_libs = ["shutil", "subprocess", "ctypes", "pickle",
                   "http", "socket", "pathlib", "glob"]
    dangerous_calls = ["os.system", "os.remove", "os.rmdir",
                       "os.unlink", "os.popen", "os.exec", "os.kill"]
    for line in code.split('\n'):
        s = line.strip()
        if s.startswith('#'):
            continue
        for lib in unsafe_libs:
            if f"import {lib}" in s or f"from {lib}" in s:
                raise Exception(f"차단된 모듈: {lib}")
        for func in dangerous_calls:
            if func in s:
                raise Exception(f"차단된 호출: {func}")

def extract_code(text):
    """LLM 응답에서 Python 코드 추출"""

    # 패턴1: ```python ... ```
    match = re.search(r'```python\s*\n(.*?)```', text, re.DOTALL)
    if match:
        return match.group(1).strip()

    # 패턴2: ``` ... ```
    match = re.search(r'```\s*\n(.*?)```', text, re.DOTALL)
    if match:
        code = match.group(1).strip()
        if "bpy" in code:
            return code

    # 패턴3: 코드블록 없이 import bpy로 시작하는 부분 추출
    lines = text.strip().split('\n')
    code_lines = []
    in_code = False
    for line in lines:
        stripped = line.strip()
        if stripped.startswith("import bpy"):
            in_code = True
        if in_code:
            code_lines.append(line)
    if code_lines:
        return '\n'.join(code_lines).strip()

    return ""

def preprocess_code(text):
    """코드 추출 + 문법검증"""
    code = extract_code(text)
    if not code:
        raise Exception("코드 추출 실패: 응답에서 Python 코드를 못 찾음")
    code = code.replace('\t', '    ')
    code = textwrap.dedent(code)
    check_safe_eval(code)
    try:
        ast.parse(code)
    except SyntaxError as e:
        raise Exception(f"문법 에러 (line {e.lineno}): {e.msg}")
    return code


# ============================================================
# 메인: 코드 생성 + 실행
# ============================================================
def gen_code(user_prompt, api_base, model, token):
    """자연어 → 코드 생성 → 실행"""

    print(f"\n{'='*50}")
    print(f"[LLM] 프롬프트: {user_prompt}")
    print(f"[LLM] 모델: {model}")
    print(f"{'='*50}")

    scene_ctx = get_scene_context()
    print(f"[씬 상태]\n{scene_ctx}\n")

    # 첫 호출
    try:
        output = call_llm_api(user_prompt, scene_ctx, api_base, model, token)
    except Exception as e:
        print(f"[에러] {e}")
        return f"API 실패: {e}"

    print(f"[LLM 응답]\n{output}\n")

    # 실행 (재시도 루프)
    code = ""
    last_error = ""
    for attempt in range(MAX_RETRY):
        try:
            code = preprocess_code(output)
            print(f"[실행 시도 {attempt+1}]\n{code}\n")
            exec(code)
            print("[성공]")
            return "성공"
        except Exception as e:
            last_error = str(e)
            print(f"[에러] 시도 {attempt+1}: {last_error}")
            if attempt < MAX_RETRY - 1:
                fix_prompt = f"""에러 발생. 수정해주세요.
에러: {last_error}
코드:
```python
{code if code else '(추출실패)'}
```"""
                try:
                    output = call_llm_api(fix_prompt, scene_ctx, api_base, model, token)
                    print(f"[수정 응답]\n{output}\n")
                except Exception as api_err:
                    print(f"[수정 API 실패] {api_err}")
                    break

    return f"실패({MAX_RETRY}회): {last_error}"


# ============================================================
# 모델 목록 조회
# ============================================================
def fetch_model_list(api_base, token):
    """서버에서 사용 가능한 모델 목록 가져오기"""
    headers = {
        "Authorization": f"Bearer {token}",
        "Content-Type": "application/json"
    }
    url = f"{api_base}/v1/models"
    req = urllib.request.Request(url, headers=headers)
    try:
        with urllib.request.urlopen(req, timeout=30) as resp:
            result = json.loads(resp.read().decode("utf-8"))
            if isinstance(result, dict):
                result = result.get("data", [])
            models = []
            for m in result:
                name = m.get("id", m) if isinstance(m, dict) else str(m)
                models.append(name)
            return models
    except Exception as e:
        print(f"[모델 목록 조회 실패] {e}")
        return []

class LLM_PT_MainPanel(bpy.types.Panel):
    bl_label = "LLM 3D Assistant"
    bl_idname = "LLM_PT_main_panel"
    bl_space_type = 'VIEW_3D'
    bl_region_type = 'UI'
    bl_category = "LLM Assistant"

    def draw(self, context):
        layout = self.layout
        scene = context.scene

        # 설정
        box = layout.box()
        box.label(text="설정", icon='PREFERENCES')
        box.prop(scene, "llm_api_base", text="API Base")
        box.prop(scene, "llm_token_path", text="Token 경로")
        box.prop(scene, "llm_model_name", text="모델")
        box.operator("llm.refresh_models", text="모델 목록 새로고침", icon='FILE_REFRESH')

        layout.separator()

        # 프롬프트
        box = layout.box()
        box.label(text="자연어 명령", icon='TEXT')
        box.prop(scene, "llm_user_prompt", text="")
        row = box.row(align=True)
        row.scale_y = 1.5
        row.operator("llm.submit_prompt", text="실행", icon='PLAY')

        layout.separator()

        # 상태
        box = layout.box()
        box.label(text="상태", icon='INFO')
        status = scene.llm_status
        # 긴 메시지 줄바꿈
        while status:
            box.label(text=status[:70])
            status = status[70:]

        # 히스토리
        if scene.llm_history:
            box = layout.box()
            box.label(text="최근 명령", icon='TIME')
            for line in scene.llm_history.split("\n")[-5:]:
                if line.strip():
                    box.label(text=line[:80])


class LLM_OT_SubmitPrompt(bpy.types.Operator):
    bl_label = "Submit Prompt"
    bl_idname = "llm.submit_prompt"
    bl_description = "자연어 명령을 LLM에 전달하여 실행"

    def execute(self, context):
        scene = context.scene
        prompt = scene.llm_user_prompt.strip()
        if not prompt:
            self.report({'WARNING'}, "프롬프트를 입력하세요")
            return {'CANCELLED'}

        api_base = scene.llm_api_base.strip() or DEFAULT_API_BASE
        model = scene.llm_model_name.strip() or DEFAULT_MODEL
        token_path = scene.llm_token_path.strip() or DEFAULT_TOKEN_PATH

        # 토큰 로드
        try:
            token = load_token(token_path)
        except Exception as e:
            scene.llm_status = str(e)
            self.report({'ERROR'}, str(e))
            return {'CANCELLED'}

        scene.llm_status = "처리 중..."
        result = gen_code(prompt, api_base, model, token)
        scene.llm_status = result

        # 히스토리
        h = f"▸ {prompt} → {result}"
        scene.llm_history = (scene.llm_history + "\n" + h) if scene.llm_history else h
        scene.llm_user_prompt = ""

        self.report({'INFO'}, result)
        return {'FINISHED'}


class LLM_OT_RefreshModels(bpy.types.Operator):
    bl_label = "Refresh Models"
    bl_idname = "llm.refresh_models"
    bl_description = "서버에서 사용 가능한 모델 목록 조회"

    def execute(self, context):
        scene = context.scene
        api_base = scene.llm_api_base.strip() or DEFAULT_API_BASE
        token_path = scene.llm_token_path.strip() or DEFAULT_TOKEN_PATH

        try:
            token = load_token(token_path)
        except Exception as e:
            self.report({'ERROR'}, str(e))
            return {'CANCELLED'}

        models = fetch_model_list(api_base, token)
        if models:
            # 콘솔에 출력
            print("\n=== 사용 가능 모델 ===")
            for i, m in enumerate(models):
                print(f"  [{i}] {m}")
            # 코더 모델 자동 선택
            for m in models:
                if "Coder" in m:
                    scene.llm_model_name = m
                    break
            scene.llm_status = f"모델 {len(models)}개 조회됨 (콘솔 확인)"
            self.report({'INFO'}, f"모델 {len(models)}개 조회 완료")
        else:
            scene.llm_status = "모델 조회 실패 (토큰/URL 확인)"
            self.report({'WARNING'}, "모델 조회 실패")
        return {'FINISHED'}

# 등록 / 해제
# ============================================================
classes = [LLM_PT_MainPanel, LLM_OT_SubmitPrompt, LLM_OT_RefreshModels]

def register():
    for cls in classes:
        bpy.utils.register_class(cls)

    bpy.types.Scene.llm_api_base = bpy.props.StringProperty(
        name="API Base", default=DEFAULT_API_BASE
    )
    bpy.types.Scene.llm_token_path = bpy.props.StringProperty(
        name="Token Path", default=DEFAULT_TOKEN_PATH, subtype='FILE_PATH'
    )
    bpy.types.Scene.llm_model_name = bpy.props.StringProperty(
        name="Model", default=DEFAULT_MODEL
    )
    bpy.types.Scene.llm_user_prompt = bpy.props.StringProperty(
        name="Prompt", default=""
    )
    bpy.types.Scene.llm_status = bpy.props.StringProperty(
        name="Status", default="대기 중"
    )
    bpy.types.Scene.llm_history = bpy.props.StringProperty(
        name="History", default=""
    )

def unregister():
    for cls in reversed(classes):
        bpy.utils.unregister_class(cls)
    del bpy.types.Scene.llm_api_base
    del bpy.types.Scene.llm_token_path
    del bpy.types.Scene.llm_model_name
    del bpy.types.Scene.llm_user_prompt
    del bpy.types.Scene.llm_status
    del bpy.types.Scene.llm_history

if __name__ == "__main__":
    register()
