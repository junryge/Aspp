class OHTPosition:
    def __init__(self, cache_path=DEFAULT_CACHE):
        with open(cache_path, "r", encoding="utf-8") as f:
            data = json.load(f)
        self.nodes = {int(k): tuple(v) for k, v in data["nodes"].items()}
        self.adj = {int(k): v for k, v in data["adj"].items()}
        self.edge_dist = {}
        for k, v in data["edges"].items():
            parts = k.split(",")
            self.edge_dist[(int(parts[0]), int(parts[1]))] = v

    def calc(self, current, distance, next_node):
        if current not in self.nodes: return None, None
        if next_node not in self.nodes: return self.nodes[current]
        x1, y1 = self.nodes[current]
        x2, y2 = self.nodes[next_node]
        edge_len = self.edge_dist.get((current, next_node), 0)
        if edge_len <= 0:
            edge_len = math.sqrt((x2-x1)**2 + (y2-y1)**2) * 10
            if edge_len <= 0: return x1, y1
        ratio = min(distance / edge_len, 1.0)
        return round(x1 + (x2-x1)*ratio, 2), round(y1 + (y2-y1)*ratio, 2)

def parse_xml_to_json(xml_path, output_path=DEFAULT_CACHE):
    node_xy, connections = {}, []
    cur_addr, cur_params, in_next, next_params = None, {}, False, {}
    print(f"  XML 파싱: {xml_path}")
    with open(xml_path, 'r', encoding='utf-8') as f:
        for i, line in enumerate(f, 1):
            if i % 500000 == 0: print(f"    {i:,} 라인...")
            line = line.strip()
            if '<group name="Addr' in line and 'address.Addr"' in line:
                if cur_addr and 'address' in cur_params:
                    no = int(cur_params.get('address', 0))
                    if no > 0: node_xy[no] = (float(cur_params.get('draw-x',0)), float(cur_params.get('draw-y',0)))
                cur_params, cur_addr, in_next = {}, line, False
                continue
            if '<group name="NextAddr' in line and 'NextAddr"' in line:
                in_next, next_params = True, {}; continue
            if in_next and '</group>' in line:
                if 'address' in cur_params and 'next-address' in next_params:
                    fa, ta = int(cur_params.get('address',0)), int(next_params.get('next-address','0'))
                    if fa > 0 and ta > 0: connections.append((fa, ta))
                in_next = False; continue
            if '<param ' in line and 'key="' in line:
                km, vm = re.search(r'key="([^"]+)"', line), re.search(r'value="([^"]*)"', line)
                if km and vm:
                    if in_next: next_params[km.group(1)] = vm.group(1)
                    elif cur_addr: cur_params[km.group(1)] = vm.group(1)
    if cur_addr and 'address' in cur_params:
        no = int(cur_params.get('address', 0))
        if no > 0: node_xy[no] = (float(cur_params.get('draw-x',0)), float(cur_params.get('draw-y',0)))
    adj, edge_dist = {}, {}
    for f, t in connections:
        adj.setdefault(f, []).append(t)
        if f in node_xy and t in node_xy:
            dx, dy = node_xy[t][0]-node_xy[f][0], node_xy[t][1]-node_xy[f][1]
            edge_dist[(f,t)] = max(int(math.sqrt(dx*dx+dy*dy)*10), 10)
    data = {"nodes":{str(k):list(v) for k,v in node_xy.items()}, "adj":{str(k):v for k,v in adj.items()}, "edges":{f"{k[0]},{k[1]}":v for k,v in edge_dist.items()}}
    with open(output_path, "w", encoding="utf-8") as f: json.dump(data, f)
    print(f"  노드: {len(node_xy)}개  엣지: {len(edge_dist)}개 → {output_path}")

def ensure_cache():
    if os.path.exists(DEFAULT_CACHE): return
    print("layout_cache.json 없음 → 자동 생성")
    if os.path.exists(LAYOUT_XML): parse_xml_to_json(LAYOUT_XML); return
    if os.path.exists(LAYOUT_ZIP):
        with zipfile.ZipFile(LAYOUT_ZIP,'r') as z: z.extractall(os.path.join(MAP_DIR,FAB_NAME,f"{LAYOUT_PREFIX}.layout"))
        if os.path.exists(LAYOUT_XML): parse_xml_to_json(LAYOUT_XML); return
    if os.path.exists(MAP_ZIP):
        with zipfile.ZipFile(MAP_ZIP,'r') as z: z.extractall(MAP_DIR)
        if os.path.exists(LAYOUT_XML): parse_xml_to_json(LAYOUT_XML); return
    print("\n오류: layout_cache.json 이 없습니다!")
    print(f"  찾은 위치: {DEFAULT_CACHE}")
    print()
    print("  해결 방법:")
    print(f"  1. layout_cache.json 파일을 이 폴더에 넣으세요: {SCRIPT_DIR}")
    print(f"  2. 또는 layout.xml로 생성: python oht_position.py convert layout.xml")
    print()
    input("엔터를 누르면 종료...")
    sys.exit(1)

class Handler(BaseHTTPRequestHandler):
    html_content = ""
    def do_GET(self):
        self.send_response(200)
        self.send_header("Content-Type", "text/html; charset=utf-8")
        self.end_headers()
        self.wfile.write(self.html_content.encode("utf-8"))
    def log_message(self, *a): pass


# ============================================================
# main
# ============================================================
def main():
    if len(sys.argv) > 1 and sys.argv[1] == "convert":
        if len(sys.argv) < 3:
            print("사용법: python oht_position.py convert layout.xml")
            return
        fp = sys.argv[2]
        if fp.endswith('.zip'):
            with zipfile.ZipFile(fp,'r') as z:
                xn = [n for n in z.namelist() if n.endswith('layout.xml')]
                if not xn: print("ZIP에 layout.xml 없음"); return
                z.extractall(os.path.dirname(fp) or ".")
                for x in xn:
                    if os.path.exists(x): parse_xml_to_json(x); return
        else:
            parse_xml_to_json(fp)
        return

    # layout_cache.json 준비
    ensure_cache()

    # JSON 읽기 → HTML에 내장
    with open(DEFAULT_CACHE, "r", encoding="utf-8") as f:
        layout_json = f.read()

    data = json.loads(layout_json)
    nc = len(data["nodes"])
    ec = len(data["edges"])

    Handler.html_content = get_html(layout_json)

    print(f"노드 {nc}개, 엣지 {ec}개 로드")
    print(f"\nOHT 위치 계산기: http://localhost:{PORT}")
    print("브라우저가 자동으로 열립니다. Ctrl+C 종료")

    webbrowser.open(f"http://localhost:{PORT}")
    HTTPServer(("0.0.0.0", PORT), Handler).serve_forever()


if __name__ == "__main__":
    try:
        main()
    except KeyboardInterrupt:
        print("\n종료.")
