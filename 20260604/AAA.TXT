# -*- coding: utf-8 -*-
"""
token_count.py - /tokenize 기반 토큰 수 계산
사용: python token_count.py "텍스트"
      python token_count.py -f prompt.txt
"""
import sys
import requests

with open("token.txt", "r") as f:
    token = f.read().strip()

#    "http://dev.hcp.llm.skhynix.com"
#    "http://common.llm.skhynix.com"
BASE = "http://dev.hcp.llm.skhynix.com"

headers = {
    "Authorization": f"Bearer {token}",
    "Content-Type": "application/json"
}

# 모델 목록
models = requests.get(f"{BASE}/v1/models", headers=headers).json()
if isinstance(models, dict):
    models = models.get("data", models)
MODEL = models[0]["id"] if models else "Qwen3-Coder-480B"


def count_text(text, model=MODEL):
    """단순 텍스트 토큰 수"""
    r = requests.post(f"{BASE}/tokenize", headers=headers,
                      json={"model": model, "prompt": text}, timeout=30)
    if r.status_code in (403, 404, 405):
        return _fallback([{"role": "user", "content": text}], model)
    r.raise_for_status()
    return r.json()  # {"count": N, "max_model_len": M, "tokens": [...]}


def count_messages(messages, model=MODEL):
    """chat 형식 토큰 수 (chat template 적용 = 실제 요청과 동일)"""
    r = requests.post(f"{BASE}/tokenize", headers=headers,
                      json={"model": model,
                            "messages": messages,
                            "add_generation_prompt": True,
                            "add_special_tokens": False}, timeout=30)
    if r.status_code in (403, 404, 405):
        return _fallback(messages, model)
    r.raise_for_status()
    return r.json()


def _fallback(messages, model):
    """/tokenize 차단 시: max_tokens=1 요청으로 usage.prompt_tokens 확인"""
    r = requests.post(f"{BASE}/v1/chat/completions", headers=headers,
                      json={"model": model, "messages": messages,
                            "max_tokens": 1}, timeout=60)
    r.raise_for_status()
    return {"count": r.json()["usage"]["prompt_tokens"], "method": "fallback"}


def detokenize(tokens, model=MODEL):
    r = requests.post(f"{BASE}/detokenize", headers=headers,
                      json={"model": model, "tokens": tokens}, timeout=30)
    r.raise_for_status()
    return r.json()["prompt"]


if __name__ == "__main__":
    args = sys.argv[1:]
    if args and args[0] == "-f":
        with open(args[1], "r", encoding="utf-8") as f:
            text = f.read()
    elif args:
        text = " ".join(args)
    else:
        text = sys.stdin.read()

    print(f"사용 가능 모델: {[m['id'] for m in models]}")
    print(f"선택 모델     : {MODEL}\n")

    r = count_text(text)
    print(f"토큰 수       : {r['count']}")
    if "max_model_len" in r:
        print(f"max_model_len : {r['max_model_len']}")
        print(f"남은 여유     : {r['max_model_len'] - r['count']}")
    if r.get("method") == "fallback":
        print("(주의: /tokenize 차단됨 -> usage.prompt_tokens로 계산)")
